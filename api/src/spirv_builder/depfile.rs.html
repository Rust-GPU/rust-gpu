4use raw_string::{RawStr, RawString};
5use std::fs::File;
6use std::io::{BufRead, BufReader, Error, ErrorKind, Read};
7use std::mem::{replace, take};
8use std::path::Path;
9
10/// Read a Makfile-style dependency file.
11///
12/// `f` is called for every target. The first argument is the target, the
13/// second is the list of dependencies.
14pub fn read_deps_file(
15    file_name: &Path,
16    f: impl FnMut(RawString, Vec<RawString>) -> Result<(), Error>,
17) -> Result<(), Error> {
18    let file = File::open(file_name)
19        .map_err(|e| Error::new(e.kind(), format!("Unable to read {file_name:?}: {e}")))?;
20    read_deps_file_from(file, f)
21}

23/// Read a Makfile-style dependency file.
24///
25/// `f` is called for every target. The first argument is the target, the
26/// second is the list of dependencies.
27pub fn read_deps_file_from(
28    file: impl Read,
29    mut f: impl FnMut(RawString, Vec<RawString>) -> Result<(), Error>,
30) -> Result<(), Error> {
31    let mut file = BufReader::new(file);
32
33    let mut state = State::default();
34
35    let mut line = RawString::new();
36
37    loop {
38        line.clear();
39        if file.read_until(b'\n', line.as_mut_bytes())? == 0 {
40            break;
41        }
42
43        if line.last() == Some(b'\n') {
44            line.pop();
45        }
46
47        if cfg!(windows) && line.last() == Some(b'\r') {
48            line.pop();
49        }
50
51        let mut write_offset = 0;
52        let mut read_offset = 0;
53
54        loop {
55            match memchr::memchr2(b' ', b'\\', line[read_offset..].as_bytes())
56                .map(|i| i + read_offset)
57            {
58                Some(i) if line[i] == b'\\' && i + 1 == line.len() => {
59                    // Backslash at the end of the line
60                    state.add_part(&line[write_offset..i]);
61                    state.finish_path()?;
62                    break;
63                }
64                Some(i) if line[i] == b'\\' => {
65                    // Backslash before character.
66                    let c = line[i + 1];
67                    match c {
68                        b' ' | b'\\' | b'#' | b'*' | b'[' | b']' | b'|' => {
69                            // Escaped character. Drop the '\'.
70                            state.add_part(&line[write_offset..i]);
71                            write_offset = i + 1;
72                        }
73                        _ => (), // Keep the '\'.
74                    }
75                    read_offset = i + 2;
76                }
77                Some(i) => {
78                    // A space.
79                    debug_assert_eq!(line[i], b' ');
80                    state.add_part(&line[write_offset..i]);
81                    state.finish_path()?;
82                    write_offset = i + 1;
83                    read_offset = i + 1;
84                }
85                None => {
86                    // End of the line.
87                    state.add_part(&line[write_offset..]);
88                    state.finish_deps(&mut f)?;
89                    break;
90                }
91            }
92        }
93    }
94
95    if state.target.is_none() {
96        Ok(())
97    } else {
98        Err(Error::new(ErrorKind::InvalidData, "Unexpected end of file"))
99    }
100}
101
102#[derive(Default)]
103struct State {
104    /// The (incomplete) path we're currently reading.
105    path: RawString,
106    /// The target, once we've finished reading it.
107    target: Option<RawString>,
108    /// The rest of the paths we've finished reading.
109    deps: Vec<RawString>,
110}
111
112impl State {
113    fn add_part(&mut self, s: &RawStr) {
114        self.path.push_str(s);
115    }
116    fn finish_path(&mut self) -> Result<(), Error> {
117        if !self.path.is_empty() {
118            let mut path = replace(&mut self.path, RawString::new());
119            if self.target.is_none() && path.last() == Some(b':') {
120                path.pop();
121                self.target = Some(path);
122            } else if self.target.is_none() {
123                return Err(Error::new(
124                    ErrorKind::InvalidData,
125                    "Rule in dependency file has multiple outputs",
126                ));
127            } else {
128                self.deps.push(path);
129            }
130        }
131        Ok(())
132    }
133    fn finish_deps(
134        &mut self,
135        f: &mut impl FnMut(RawString, Vec<RawString>) -> Result<(), Error>,
136    ) -> Result<(), Error> {
137        self.finish_path()?;
138        if let Some(target) = self.target.take() {
139            f(target, take(&mut self.deps))?;
140        }
141        Ok(())
142    }
143}
