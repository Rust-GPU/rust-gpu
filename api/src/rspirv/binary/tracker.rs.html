12// TODO: Add support for other types.
13#[derive(Clone, Copy, Debug, Eq, PartialEq)]
14pub enum Type {
15    /// Integer type (size, signed).
16    Integer(u32, bool),
17    Float(u32),
18}
19
20/// Tracks ids to their types.
21///
22/// If the type of an id cannot be resolved due to some reason, this will
23/// silently ignore that id instead of erroring out.
24#[derive(Debug)]
25pub struct TypeTracker {
26    /// Mapping from an id to its type.
27    ///
28    /// Ids for both defining and using types are all kept here.
29    types: collections::HashMap<spirv::Word, Type>,
30}
31
32impl TypeTracker {
33    pub fn new() -> TypeTracker {
34        TypeTracker {
35            types: collections::HashMap::new(),
36        }
37    }
38
39    pub fn track(&mut self, inst: &dr::Instruction) {
40        if let Some(rid) = inst.result_id {
41            if grammar::reflect::is_type(inst.class.opcode) {
42                match inst.class.opcode {
43                    spirv::Op::TypeInt => {
44                        if let (
45                            &dr::Operand::LiteralBit32(bits),
46                            &dr::Operand::LiteralBit32(sign),
47                        ) = (&inst.operands[0], &inst.operands[1])
48                        {
49                            self.types.insert(rid, Type::Integer(bits, sign == 1));
50                        }
51                    }
52                    spirv::Op::TypeFloat => {
53                        if let dr::Operand::LiteralBit32(bits) = inst.operands[0] {
54                            self.types.insert(rid, Type::Float(bits));
55                        }
56                    }
57                    // TODO: handle the other types here.
58                    _ => (),
59                }
60            } else {
61                inst.result_type
62                    .and_then(|t| self.resolve(t))
63                    .map(|t| self.types.insert(rid, t));
64            }
65        }
66    }
67
68    pub fn resolve(&self, id: spirv::Word) -> Option<Type> {
69        self.types.get(&id).cloned()
70    }
71}

73#[allow(clippy::upper_case_acronyms)]
74enum ExtInstSet {
75    GlslStd450,
76    OpenCLStd100,
77}

79/// Struct for tracking extended instruction sets.
80///
81/// If a given extended instruction set is not supported, it will just be
82/// silently ignored.
83pub struct ExtInstSetTracker {
84    sets: collections::HashMap<spirv::Word, ExtInstSet>,
85}
86
87impl ExtInstSetTracker {
88    pub fn new() -> ExtInstSetTracker {
89        ExtInstSetTracker {
90            sets: collections::HashMap::new(),
91        }
92    }
93
94    /// Tracks the extended instruction set declared by the given `inst`.
95    ///
96    /// If the given extended instruction set is not recognized, it will
97    /// be silently ignored.
98    pub fn track(&mut self, inst: &dr::Instruction) {
99        if inst.class.opcode != spirv::Op::ExtInstImport
100            || inst.result_id.is_none()
101            || inst.operands.is_empty()
102        {
103            return;
104        }
105        if let dr::Operand::LiteralString(ref s) = inst.operands[0] {
106            if s == "GLSL.std.450" {
107                self.sets
108                    .insert(inst.result_id.unwrap(), ExtInstSet::GlslStd450);
109            } else if s == "OpenCL.std" {
110                self.sets
111                    .insert(inst.result_id.unwrap(), ExtInstSet::OpenCLStd100);
112            }
113        }
114    }
115
116    /// Returns true if the given extended instruction `set` has been
117    /// recognized thus tracked.
118    pub fn have(&self, set: spirv::Word) -> bool {
119        self.sets.get(&set).is_some()
120    }
121
122    /// Resolves the extended instruction with `opcode` in set `set`.
123    ///
124    /// This method will return `None` for both untracked instruction
125    /// sets and unknown opcode in tracked instruction sets.
126    pub fn resolve(&self, set: spirv::Word, opcode: spirv::Word) -> Option<GExtInstRef> {
127        if let Some(ext_inst_set) = self.sets.get(&set) {
128            match *ext_inst_set {
129                ExtInstSet::GlslStd450 => GGlInstTable::lookup_opcode(opcode),
130                ExtInstSet::OpenCLStd100 => GClInstTable::lookup_opcode(opcode),
131            }
132        } else {
133            None
134        }
135    }
136}
