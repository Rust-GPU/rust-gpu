21/// To compute the arity (param and result counts) of "variable-arity"
22/// operators, the operator_arity macro needs information about the
23/// module's types and the current control stack. The ModuleArity
24/// trait exposes this information.
25pub trait ModuleArity {
26    /// Type with given index
27    fn sub_type_at(&self, type_idx: u32) -> Option<&SubType>;
28
29    /// Arity (param and result counts) of tag with given index
30    fn tag_type_arity(&self, at: u32) -> Option<(u32, u32)>;
31
32    /// Type index of function with given index
33    fn type_index_of_function(&self, function_idx: u32) -> Option<u32>;
34
35    /// Function type for a given continuation type
36    fn func_type_of_cont_type(&self, c: &ContType) -> Option<&FuncType>;
37
38    /// Sub type for a given reference type
39    fn sub_type_of_ref_type(&self, rt: &RefType) -> Option<&SubType>;
40
41    /// Current height of control stack
42    fn control_stack_height(&self) -> u32;
43
44    /// BlockType and FrameKind of label with given index
45    fn label_block(&self, depth: u32) -> Option<(BlockType, FrameKind)>;
46
47    /// Computes arity of given SubType
48    fn sub_type_arity(&self, t: &SubType) -> Option<(u32, u32)> {
49        match &t.composite_type.inner {
50            CompositeInnerType::Func(f) => {
51                Some((f.params().len() as u32, f.results().len() as u32))
52            }
53            CompositeInnerType::Struct(s) => Some((s.fields.len() as u32, s.fields.len() as u32)),
54            CompositeInnerType::Array(_) => None,
55            CompositeInnerType::Cont(c) => {
56                let f = self.func_type_of_cont_type(c)?;
57                Some((f.params().len() as u32, f.results().len() as u32))
58            }
59        }
60    }
61
62    /// Computes arity of given BlockType
63    fn block_type_arity(&self, ty: BlockType) -> Option<(u32, u32)> {
64        match ty {
65            BlockType::Empty => Some((0, 0)),
66            BlockType::Type(_) => Some((0, 1)),
67            BlockType::FuncType(t) => self.sub_type_arity(self.sub_type_at(t)?),
68        }
69    }
70}

72impl OperatorsReader<'_> {
73    /// Read the next operator and compute its arity (param and result counts)
74    pub fn operator_arity(&self, module: &impl ModuleArity) -> Result<(u32, u32)> {
75        self.clone().read()?.operator_arity(module).ok_or_else(|| {
76            BinaryReaderError::new("operator arity is unknown", self.original_position())
77        })
78    }
79}

81/// The operator_arity macro interprets the annotations in the for_each_operator macro
82/// to compute the arity of each operator. It needs access to a ModuleArity implementation.
83#[cfg_attr(not(feature = "simd"), allow(unused_macro_rules))]
84macro_rules! operator_arity {
85    (arity $self:ident $({ $($arg:ident: $argty:ty),* })? arity $($ann:tt)*) => {
86	{
87	    let params  = (|| -> Option<(i32, i32)> { operator_arity!(params  $self { $($($arg: $argty),*)? } $($ann)*) })();
88	    let results = (|| -> Option<(i32, i32)> { operator_arity!(results $self { $($($arg: $argty),*)? } $($ann)*) })();
89	    match (params, results) {
90		(Some((a,_)), Some((_,d))) if a >= 0 && d >= 0  => (Some((a as u32, d as u32))),
91		_ => None,
92	    }
93	}
94    };
95
96    (arity $self:ident $({ $($arg:ident: $argty:ty),* })? $cat:ident $($ann:tt)*) => {
97	    Some(operator_arity!(fixed $cat $($ann)*))
98    };
99
100    (params  $self:ident { $($arg:ident: $argty:ty),* } ~ $cat:ident $($tokens:tt)*) => { { let (a, b) = operator_arity!(count $self { $($arg: $argty),* } $cat)?;
101                                                                                            let (c, d) = operator_arity!(params $self { $($arg: $argty),* } $($tokens)*)?;
102                                                                                            Some((b as i32 + c as i32, a as i32 + d as i32)) } };
103    (params  $self:ident { $($arg:ident: $argty:ty),* } $val:literal $($tokens:tt)*) => { { let rest = operator_arity!(params $self { $($arg: $argty),* } $($tokens)*)?;
104											    Some(($val + rest.0, $val + rest.1)) } };
105    (params  $self:ident { $($arg:ident: $argty:ty),* } $cat:ident   $($tokens:tt)*) => { { let (a, b) = operator_arity!(count $self { $($arg: $argty),* } $cat)?;
106											    let (c, d) = operator_arity!(params $self { $($arg: $argty),* } $($tokens)*)?;
107											    Some((a as i32 + c as i32, b as i32 + d as i32)) } };
108    (params  $self:ident { $($arg:ident: $argty:ty),* } ->           $($tokens:tt)*) => { Some((0, 0)) };
109    (params  $self:ident { $($arg:ident: $argty:ty),* })                             => { Some((0, 0)) };
110
111    (results $self:ident { $($arg:ident: $argty:ty),* } ~            $($tokens:tt)*) => { operator_arity!(results $self { $($arg: $argty),* } $($tokens)*) };
112    (results $self:ident { $($arg:ident: $argty:ty),* } $val:literal $($tokens:tt)*) => { operator_arity!(results $self { $($arg: $argty),* } $($tokens)*) };
113    (results $self:ident { $($arg:ident: $argty:ty),* } $cat:ident   $($tokens:tt)*) => { operator_arity!(results $self { $($arg: $argty),* } $($tokens)*) };
114    (results $self:ident { $($arg:ident: $argty:ty),* } ->           $($tokens:tt)*) => { operator_arity!(params  $self { $($arg: $argty),* } $($tokens)*) };
115
116    (count $self:ident { $tag_index:ident: $_:ty } tag) => {{
117	operator_arity!(tag_index $tag_index);
118	$self.tag_type_arity($tag_index)
119    }};
120
121    (count $self:ident { $_1:ident: $_2:ty, $tag_index:ident: $($_3:tt)* } tag) => { operator_arity!(count $self { $tag_index: _ } tag) };
122
123    (count $self:ident { $func_index:ident: $_:ty } func) => {{
124	operator_arity!(func_index $func_index);
125	$self.sub_type_arity($self.sub_type_at($self.type_index_of_function($func_index)?)?)
126    }};
127
128    (count $self:ident { $type_index:ident: $($_:tt)* } type) => {{
129	operator_arity!(type_index $type_index);
130	$self.sub_type_arity($self.sub_type_at($type_index)?)
131    }};
132
133    (count $self:ident { $type_index:ident: $($_:tt)* } switch) => {{
134	operator_arity!(type_index $type_index);
135	let st = &$self.sub_type_at($type_index)?.composite_type.inner;
136	if let CompositeInnerType::Cont(ct) = &st {
137            let last_param = $self.func_type_of_cont_type(ct)?.params().last()?;
138            $self.sub_type_arity($self.sub_type_of_ref_type(&last_param.as_reference_type()?)?)
139	} else {
140	    None
141	}
142    }};
143
144    (count $self:ident { $type1_index:ident: $t1:ty, $type2_index:ident: $t2:ty } type_diff) => {{
145	operator_arity!(type_index $type1_index);
146	operator_arity!(type_index $type2_index);
147	let a = $self.sub_type_arity($self.sub_type_at($type1_index)?)?;
148	let b = $self.sub_type_arity($self.sub_type_at($type2_index)?)?;
149	Some((a.0.checked_sub(b.0)?, a.1.checked_sub(b.1)?))
150    }};
151
152    (count $self:ident { $arg1:ident: $argty:ty, $size:ident: $sizety:ty } size) => {{
153        operator_arity!(size_value $size);
154        Some(($size, $size))
155    }};
156
157    (count $self:ident { $depth:ident: $($_:tt)* } br) => {{
158        operator_arity!(depth $depth);
159	let (ty, kind) = $self.label_block($depth)?;
160	let (params, results) = $self.block_type_arity(ty)?;
161	let n = match kind {
162	    FrameKind::Loop => params,
163	    _ => results,
164	};
165	Some((n, n))
166    }};
167
168    (count $self:ident { $($_:ident: $__:ty),* } ret) => {{
169	let (ty, _) = $self.control_stack_height().checked_sub(1)
170	    .and_then(|x| $self.label_block(x))?;
171	$self.block_type_arity(ty)
172    }};
173
174    (count $self:ident { $blockty:ident: $($_:tt)* } block) => {{
175        operator_arity!(blockty $blockty);
176        $self.block_type_arity($blockty)
177    }};
178
179    (count $self:ident {} implicit_else) => {{
180	let (ty, kind) = $self.label_block(0)?;
181	let (params, results) = $self.block_type_arity(ty)?;
182	Some(match kind {
183	    FrameKind::If => (results, params),
184	    _ => (0, 0),
185	})
186    }};
187
188    (count $self:ident { $($_: ident: $__:ty),* } end) => {{
189	let (ty, _) = $self.label_block(0)?;
190	$self.block_type_arity(ty)
191    }};
192
193    (count $self:ident { $try_table:ident: $($_:tt)* } try_table) => {{
194        operator_arity!(try_table $try_table);
195        $self.block_type_arity($try_table.ty)
196    }};
197
198    (count $self:ident { $br_table:ident: $($_:tt)* } br_table) => {{
199        operator_arity!(br_table $br_table);
200	let relative_depth: u32 = $br_table.default();
201	operator_arity!(count $self { relative_depth: u32 } br)
202    }};
203
204    (count $self:ident { $select_tys:ident: $($_:tt)* } select) => {{
205        operator_arity!(select_tys $select_tys);
206	let result_count = $select_tys.len();
207	Some((1 + 2 * result_count, result_count))
208    }};
209
210    (tag_index tag_index $($_:tt)*) => {};
211    (func_index function_index $($_:tt)*) => {};
212    (type_index type_index $($_:tt)*) => {};
213    (type_index struct_type_index $($_:tt)*) => {};
214    (type_index argument_index $($_:tt)*) => {};
215    (type_index result_index $($_:tt)*) => {};
216    (type_index cont_type_index $($_:tt)*) => {};
217    (size_value array_size $($_:tt)*) => {};
218    (depth relative_depth $($_:tt)*) => {};
219    (blockty blockty $($_:tt)*) => {};
220    (try_table try_table $($_:tt)*) => {};
221    (br_table targets $($_:tt)*) => {};
222    (select_tys tys $($_:tt)*) => {};
223
224    (fixed load lane $($_:tt)*)        => {(2, 1)};
225    (fixed load $($_:tt)*)             => {(1, 1)};
226    (fixed store $($_:tt)*)            => {(2, 0)};
227    (fixed test $($_:tt)*)             => {(1, 1)};
228    (fixed unary $($_:tt)*)            => {(1, 1)};
229    (fixed binary $($_:tt)*)           => {(2, 1)};
230    (fixed cmp $($_:tt)*)              => {(2, 1)};
231    (fixed shift $($_:tt)*)            => {(2, 1)};
232    (fixed splat $($_:tt)*)            => {(1, 1)};
233    (fixed ternary $($_:tt)*)          => {(3, 1)};
234    (fixed conversion $($_:tt)*)       => {(1, 1)};
235    (fixed push $($_:tt)*)             => {(0, 1)};
236    (fixed extract $($_:tt)*)          => {(1, 1)};
237    (fixed replace $($_:tt)*)          => {(2, 1)};
238    (fixed atomic rmw array $($_:tt)*) => {(3, 1)};
239    (fixed atomic rmw $($_:tt)*)       => {(2, 1)};
240    (fixed atomic cmpxchg $($_:tt)*)   => {(3, 1)};
241}

243impl Operator<'_> {
244    /// Compute the arity (param and result counts) of the operator, given
245    /// an impl ModuleArity, which stores the necessary module state.
246    pub fn operator_arity(&self, module: &impl ModuleArity) -> Option<(u32, u32)> {
247        macro_rules! define_arity {
248            ($(@$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*) )*) => (
249                match self.clone() {
250                    $(
251                        Operator::$op $({ $($arg),* })? => {
252                            $(
253                                $(let _ = $arg;)*
254                            )?
255                            operator_arity!(arity module $({ $($arg: $argty),* })? $($ann)*)
256                        }
257                    )*
258                }
259            );
260        }
261        crate::for_each_operator!(define_arity)
262    }
263}
