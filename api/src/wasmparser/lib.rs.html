16//! A simple event-driven library for parsing WebAssembly binary files
17//! (or streams).
18//!
19//! The parser library reports events as they happen and only stores
20//! parsing information for a brief period of time, making it very fast
21//! and memory-efficient. The event-driven model, however, has some drawbacks.
22//! If you need random access to the entire WebAssembly data-structure,
23//! this is not the right library for you. You could however, build such
24//! a data-structure using this library.
25//!
26//! To get started, create a [`Parser`] using [`Parser::new`] and then follow
27//! the examples documented for [`Parser::parse`] or [`Parser::parse_all`].
28

30#![no_std]
31#![cfg_attr(docsrs, feature(doc_auto_cfg))]

932/// A helper macro to conveniently iterate over all opcodes recognized by this
933/// crate. This can be used to work with either the [`Operator`] enumeration or
934/// the [`VisitOperator`] trait if your use case uniformly handles all operators
935/// the same way.
936///
937/// It is also possible to specialize handling of operators depending on the
938/// Wasm proposal from which they are originating.
939///
940/// This is an "iterator macro" where this macro is invoked with the name of
941/// another macro, and then that macro is invoked with the list of all
942/// operators.
943///
944/// The list of specializable Wasm proposals is as follows:
945///
946/// - `@mvp`: Denoting a Wasm operator from the initial Wasm MVP version.
947/// - `@exceptions`: [Wasm `exception-handling` proposal]
948/// - `@tail_call`: [Wasm `tail-calls` proposal]
949/// - `@reference_types`: [Wasm `reference-types` proposal]
950/// - `@sign_extension`: [Wasm `sign-extension-ops` proposal]
951/// - `@saturating_float_to_int`: [Wasm `non_trapping_float-to-int-conversions` proposal]
952/// - `@bulk_memory `:[Wasm `bulk-memory` proposal]
953/// - `@simd`: [Wasm `simd` proposal]
954/// - `@relaxed_simd`: [Wasm `relaxed-simd` proposal]
955/// - `@threads`: [Wasm `threads` proposal]
956/// - `@gc`: [Wasm `gc` proposal]
957/// - `@stack_switching`: [Wasm `stack-switching` proposal]
958/// - `@wide_arithmetic`: [Wasm `wide-arithmetic` proposal]
959///
960/// [Wasm `exception-handling` proposal]:
961/// https://github.com/WebAssembly/exception-handling
962///
963/// [Wasm `tail-calls` proposal]:
964/// https://github.com/WebAssembly/tail-call
965///
966/// [Wasm `reference-types` proposal]:
967/// https://github.com/WebAssembly/reference-types
968///
969/// [Wasm `sign-extension-ops` proposal]:
970/// https://github.com/WebAssembly/sign-extension-ops
971///
972/// [Wasm `non_trapping_float-to-int-conversions` proposal]:
973/// https://github.com/WebAssembly/nontrapping-float-to-int-conversions
974///
975/// [Wasm `bulk-memory` proposal]:
976/// https://github.com/WebAssembly/bulk-memory-operations
977///
978/// [Wasm `simd` proposal]:
979/// https://github.com/webassembly/simd
980///
981/// [Wasm `relaxed-simd` proposal]:
982/// https://github.com/WebAssembly/relaxed-simd
983///
984/// [Wasm `threads` proposal]:
985/// https://github.com/webassembly/threads
986///
987/// [Wasm `gc` proposal]:
988/// https://github.com/WebAssembly/gc
989///
990/// [Wasm `stack-switching` proposal]:
991/// https://github.com/WebAssembly/stack-switching
992///
993/// [Wasm `wide-arithmetic` proposal]:
994/// https://github.com/WebAssembly/wide-arithmetic
995///
996/// ```
997/// fn do_nothing(op: &wasmparser::Operator) {
998///     macro_rules! define_match_operator {
999///         // The outer layer of repetition represents how all operators are
1000///         // provided to the macro at the same time.
1001///         //
1002///         // The `$proposal` identifier indicates the Wasm proposals from which
1003///         // the Wasm operator is originating.
1004///         // For example to specialize the macro match arm for Wasm SIMD proposal
1005///         // operators you could write `@simd` instead of `@$proposal:ident` to
1006///         // only catch those operators.
1007///         //
1008///         // The `$op` name is bound to the `Operator` variant name. The
1009///         // payload of the operator is optionally specified (the `$(...)?`
1010///         // clause) since not all instructions have payloads. Within the payload
1011///         // each argument is named and has its type specified.
1012///         //
1013///         // The `$visit` name is bound to the corresponding name in the
1014///         // `VisitOperator` trait that this corresponds to.
1015///         //
1016///         // The `$ann` annotations give information about the operator's type (e.g. binary i32 or arity 2 -> 1).
1017///         ($( @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*))*) => {
1018///             match op {
1019///                 $(
1020///                     wasmparser::Operator::$op $( { $($arg),* } )? => {
1021///                         // do nothing for this example
1022///                     }
1023///                 )*
1024///                 _ => unreachable!(), // required because `Operator` enum is non-exhaustive
1025///             }
1026///         }
1027///     }
1028///     wasmparser::for_each_operator!(define_match_operator);
1029/// }
1030/// ```
1031///
1032/// If you only wanted to visit the initial base set of wasm instructions, for
1033/// example, you could do:
1034///
1035/// ```
1036/// fn is_mvp_operator(op: &wasmparser::Operator) -> bool {
1037///     macro_rules! define_match_operator {
1038///         // delegate the macro invocation to sub-invocations of this macro to
1039///         // deal with each instruction on a case-by-case basis.
1040///         ($( @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*))*) => {
1041///             match op {
1042///                 $(
1043///                     wasmparser::Operator::$op $( { $($arg),* } )? => {
1044///                         define_match_operator!(impl_one @$proposal)
1045///                     }
1046///                 )*
1047///                 _ => unreachable!(), // required because `Operator` enum is non-exhaustive
1048///             }
1049///         };
1050///
1051///         (impl_one @mvp) => { true };
1052///         (impl_one @$proposal:ident) => { false };
1053///     }
1054///     wasmparser::for_each_operator!(define_match_operator)
1055/// }
1056/// ```
1057#[doc(inline)]
1058pub use _for_each_operator_impl as for_each_operator;

1060/// Used to implement the [`VisitOperator`] trait.
1061///
1062/// A helper macro to conveniently iterate over all opcodes recognized by this
1063/// crate. This can be used to work with either the [`Operator`] enumeration or
1064/// the [`VisitOperator`] trait if your use case uniformly handles all operators
1065/// the same way.
1066///
1067/// It is also possible to specialize handling of operators depending on the
1068/// Wasm proposal from which they are originating.
1069///
1070/// This is an "iterator macro" where this macro is invoked with the name of
1071/// another macro, and then that macro is invoked with the list of all
1072/// operators.
1073///
1074/// The list of specializable Wasm proposals is as follows:
1075///
1076/// - `@mvp`: Denoting a Wasm operator from the initial Wasm MVP version.
1077/// - `@exceptions`: [Wasm `exception-handling` proposal]
1078/// - `@tail_call`: [Wasm `tail-calls` proposal]
1079/// - `@reference_types`: [Wasm `reference-types` proposal]
1080/// - `@sign_extension`: [Wasm `sign-extension-ops` proposal]
1081/// - `@saturating_float_to_int`: [Wasm `non_trapping_float-to-int-conversions` proposal]
1082/// - `@bulk_memory `:[Wasm `bulk-memory` proposal]
1083/// - `@threads`: [Wasm `threads` proposal]
1084/// - `@gc`: [Wasm `gc` proposal]
1085/// - `@stack_switching`: [Wasm `stack-switching` proposal]
1086/// - `@wide_arithmetic`: [Wasm `wide-arithmetic` proposal]
1087///
1088/// Note that this macro does not iterate over the SIMD-related proposals. Those
1089/// are provided in [`VisitSimdOperator`] and [`for_each_visit_simd_operator`].
1090/// This macro only handles non-SIMD related operators and so users wanting to
1091/// handle the SIMD-class of operators need to use that trait/macro as well.
1092///
1093/// [Wasm `exception-handling` proposal]:
1094/// https://github.com/WebAssembly/exception-handling
1095///
1096/// [Wasm `tail-calls` proposal]:
1097/// https://github.com/WebAssembly/tail-call
1098///
1099/// [Wasm `reference-types` proposal]:
1100/// https://github.com/WebAssembly/reference-types
1101///
1102/// [Wasm `sign-extension-ops` proposal]:
1103/// https://github.com/WebAssembly/sign-extension-ops
1104///
1105/// [Wasm `non_trapping_float-to-int-conversions` proposal]:
1106/// https://github.com/WebAssembly/nontrapping-float-to-int-conversions
1107///
1108/// [Wasm `bulk-memory` proposal]:
1109/// https://github.com/WebAssembly/bulk-memory-operations
1110/// [Wasm `simd` proposal]:
1111/// https://github.com/webassembly/simd
1112///
1113/// [Wasm `relaxed-simd` proposal]:
1114/// https://github.com/WebAssembly/relaxed-simd
1115///
1116/// [Wasm `threads` proposal]:
1117/// https://github.com/webassembly/threads
1118///
1119/// [Wasm `gc` proposal]:
1120/// https://github.com/WebAssembly/gc
1121///
1122/// [Wasm `stack-switching` proposal]:
1123/// https://github.com/WebAssembly/stack-switching
1124///
1125/// [Wasm `wide-arithmetic` proposal]:
1126/// https://github.com/WebAssembly/wide-arithmetic
1127///
1128/// ```
1129/// macro_rules! define_visit_operator {
1130///     // The outer layer of repetition represents how all operators are
1131///     // provided to the macro at the same time.
1132///     //
1133///     // The `$proposal` identifier indicates the Wasm proposals from which
1134///     // the Wasm operator is originating.
1135///     // For example to specialize the macro match arm for Wasm `gc` proposal
1136///     // operators you could write `@gc` instead of `@$proposal:ident` to
1137///     // only catch those operators.
1138///     //
1139///     // The `$op` name is bound to the `Operator` variant name. The
1140///     // payload of the operator is optionally specified (the `$(...)?`
1141///     // clause) since not all instructions have payloads. Within the payload
1142///     // each argument is named and has its type specified.
1143///     //
1144///     // The `$visit` name is bound to the corresponding name in the
1145///     // `VisitOperator` trait that this corresponds to.
1146///     //
1147///     // The `$ann` annotations give information about the operator's type (e.g. binary i32 or arity 2 -> 1).
1148///     ($( @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*))*) => {
1149///         $(
1150///             fn $visit(&mut self $($(,$arg: $argty)*)?) {
1151///                 // do nothing for this example
1152///             }
1153///         )*
1154///     }
1155/// }
1156///
1157/// pub struct VisitAndDoNothing;
1158///
1159/// impl<'a> wasmparser::VisitOperator<'a> for VisitAndDoNothing {
1160///     type Output = ();
1161///
1162///     wasmparser::for_each_visit_operator!(define_visit_operator);
1163/// }
1164/// ```
1165///
1166/// If you only wanted to visit the initial base set of wasm instructions, for
1167/// example, you could do:
1168///
1169/// ```
1170/// macro_rules! visit_only_mvp {
1171///     // delegate the macro invocation to sub-invocations of this macro to
1172///     // deal with each instruction on a case-by-case basis.
1173///     ($( @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*))*) => {
1174///         $(
1175///             visit_only_mvp!(visit_one @$proposal $op $({ $($arg: $argty),* })? => $visit);
1176///         )*
1177///     };
1178///
1179///     // MVP instructions are defined manually, so do nothing.
1180///     (visit_one @mvp $($rest:tt)*) => {};
1181///
1182///     // Non-MVP instructions all return `false` here. The exact type depends
1183///     // on `type Output` in the trait implementation below. You could change
1184///     // it to `Result<()>` for example and return an error here too.
1185///     (visit_one @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident) => {
1186///         fn $visit(&mut self $($(,$arg: $argty)*)?) -> bool {
1187///             false
1188///         }
1189///     }
1190/// }
1191/// # // to get this example to compile another macro is used here to define
1192/// # // visit methods for all mvp operators.
1193/// # macro_rules! visit_mvp {
1194/// #     ($( @$proposal:ident $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident ($($ann:tt)*))*) => {
1195/// #         $(
1196/// #             visit_mvp!(visit_one @$proposal $op $({ $($arg: $argty),* })? => $visit);
1197/// #         )*
1198/// #     };
1199/// #     (visit_one @mvp $op:ident $({ $($arg:ident: $argty:ty),* })? => $visit:ident) => {
1200/// #         fn $visit(&mut self $($(,$arg: $argty)*)?) -> bool {
1201/// #             true
1202/// #         }
1203/// #     };
1204/// #     (visit_one @$proposal:ident $($rest:tt)*) => {};
1205/// # }
1206///
1207/// pub struct VisitOnlyMvp;
1208///
1209/// impl<'a> wasmparser::VisitOperator<'a> for VisitOnlyMvp {
1210///     type Output = bool;
1211///
1212///     wasmparser::for_each_visit_operator!(visit_only_mvp);
1213/// #   wasmparser::for_each_visit_operator!(visit_mvp);
1214///
1215///     // manually define `visit_*` for all MVP operators here
1216/// }
1217/// ```
1218#[doc(inline)]
1219pub use _for_each_visit_operator_impl as for_each_visit_operator;
