44#[cfg(not(any(target_os = "redox", target_os = "wasi")))]
45#[inline]
46#[must_use]
47pub(crate) fn clock_getres(id: ClockId) -> Timespec {
48    // Old 32-bit version: libc has `clock_getres` but it is not y2038 safe by
49    // default. But there may be a `__clock_getres64` we can use.
50    #[cfg(fix_y2038)]
51    {
52        #[cfg(target_env = "gnu")]
53        if let Some(libc_clock_getres) = __clock_getres64.get() {
54            let mut timespec = MaybeUninit::<LibcTimespec>::uninit();
55            unsafe {
56                ret(libc_clock_getres(id as c::clockid_t, timespec.as_mut_ptr())).unwrap();
57                return timespec.assume_init().into();
58            }
59        }
60
61        clock_getres_old(id)
62    }
63
64    // Main version: libc is y2038 safe and has `clock_getres`.
65    #[cfg(not(fix_y2038))]
66    unsafe {
67        let mut timespec = MaybeUninit::<Timespec>::uninit();
68        let _ = c::clock_getres(id as c::clockid_t, as_libc_timespec_mut_ptr(&mut timespec));
69        timespec.assume_init()
70    }
71}

95#[must_use]
96pub(crate) fn clock_gettime(id: ClockId) -> Timespec {
97    // Old 32-bit version: libc has `clock_gettime` but it is not y2038 safe by
98    // default. But there may be a `__clock_gettime64` we can use.
99    #[cfg(fix_y2038)]
100    {
101        #[cfg(target_env = "gnu")]
102        if let Some(libc_clock_gettime) = __clock_gettime64.get() {
103            let mut timespec = MaybeUninit::<LibcTimespec>::uninit();
104            unsafe {
105                ret(libc_clock_gettime(
106                    id as c::clockid_t,
107                    timespec.as_mut_ptr(),
108                ))
109                .unwrap();
110                return timespec.assume_init().into();
111            }
112        }
113
114        clock_gettime_old(id)
115    }
116
117    // Use `.unwrap()` here because `clock_getres` can fail if the clock itself
118    // overflows a number of seconds, but if that happens, the monotonic clocks
119    // can't maintain their invariants, or the realtime clocks aren't properly
120    // configured.
121    #[cfg(not(fix_y2038))]
122    unsafe {
123        let mut timespec = MaybeUninit::<Timespec>::uninit();
124        ret(c::clock_gettime(
125            id as c::clockid_t,
126            as_libc_timespec_mut_ptr(&mut timespec),
127        ))
128        .unwrap();
129        let timespec = timespec.assume_init();
130        #[cfg(apple)]
131        let timespec = fix_negative_timespec_nsecs(timespec);
132        timespec
133    }
134}

156#[cfg(not(target_os = "wasi"))]
157#[inline]
158pub(crate) fn clock_gettime_dynamic(id: DynamicClockId<'_>) -> io::Result<Timespec> {
159    let id: c::clockid_t = match id {
160        DynamicClockId::Known(id) => id as c::clockid_t,
161
162        #[cfg(linux_kernel)]
163        DynamicClockId::Dynamic(fd) => {
164            use crate::fd::AsRawFd as _;
165            const CLOCKFD: i32 = 3;
166            (!fd.as_raw_fd() << 3) | CLOCKFD
167        }
168
169        #[cfg(not(linux_kernel))]
170        DynamicClockId::Dynamic(_fd) => {
171            // Dynamic clocks are not supported on this platform.
172            return Err(io::Errno::INVAL);
173        }
174
175        #[cfg(linux_kernel)]
176        DynamicClockId::RealtimeAlarm => c::CLOCK_REALTIME_ALARM,
177
178        #[cfg(linux_kernel)]
179        DynamicClockId::Tai => c::CLOCK_TAI,
180
181        #[cfg(any(
182            linux_kernel,
183            target_os = "freebsd",
184            target_os = "fuchsia",
185            target_os = "openbsd"
186        ))]
187        DynamicClockId::Boottime => c::CLOCK_BOOTTIME,
188
189        #[cfg(any(linux_kernel, target_os = "fuchsia"))]
190        DynamicClockId::BoottimeAlarm => c::CLOCK_BOOTTIME_ALARM,
191    };
192
193    // Old 32-bit version: libc has `clock_gettime` but it is not y2038
194    // safe by default. But there may be a `__clock_gettime64` we can use.
195    #[cfg(fix_y2038)]
196    {
197        #[cfg(target_env = "gnu")]
198        if let Some(libc_clock_gettime) = __clock_gettime64.get() {
199            let mut timespec = MaybeUninit::<LibcTimespec>::uninit();
200            unsafe {
201                ret(libc_clock_gettime(
202                    id as c::clockid_t,
203                    timespec.as_mut_ptr(),
204                ))?;
205
206                return Ok(timespec.assume_init().into());
207            }
208        }
209
210        clock_gettime_dynamic_old(id)
211    }
212
213    // Main version: libc is y2038 safe and has `clock_gettime`.
214    #[cfg(not(fix_y2038))]
215    unsafe {
216        let mut timespec = MaybeUninit::<Timespec>::uninit();
217
218        ret(c::clock_gettime(
219            id as c::clockid_t,
220            as_libc_timespec_mut_ptr(&mut timespec),
221        ))?;
222        let timespec = timespec.assume_init();
223        #[cfg(apple)]
224        let timespec = fix_negative_timespec_nsecs(timespec);
225        Ok(timespec)
226    }
227}

254#[inline]
255pub(crate) fn clock_settime(id: ClockId, timespec: Timespec) -> io::Result<()> {
256    // Old 32-bit version: libc has `clock_gettime` but it is not y2038 safe by
257    // default. But there may be a `__clock_settime64` we can use.
258    #[cfg(fix_y2038)]
259    {
260        #[cfg(target_env = "gnu")]
261        if let Some(libc_clock_settime) = __clock_settime64.get() {
262            unsafe {
263                let mut new_timespec = core::mem::zeroed::<LibcTimespec>();
264                new_timespec.tv_sec = timespec.tv_sec;
265                new_timespec.tv_nsec = timespec.tv_nsec as _;
266                return ret(libc_clock_settime(id as c::clockid_t, &new_timespec));
267            }
268        }
269
270        clock_settime_old(id, timespec)
271    }
272
273    // Main version: libc is y2038 safe and has `clock_settime`.
274    #[cfg(not(fix_y2038))]
275    unsafe {
276        ret(c::clock_settime(
277            id as c::clockid_t,
278            as_libc_timespec_ptr(&timespec),
279        ))
280    }
281}

306#[cfg(any(linux_kernel, target_os = "fuchsia"))]
307pub(crate) fn timerfd_settime(
308    fd: BorrowedFd<'_>,
309    flags: TimerfdTimerFlags,
310    new_value: &Itimerspec,
311) -> io::Result<Itimerspec> {
312    // Old 32-bit version: libc has `timerfd_settime` but it is not y2038 safe
313    // by default. But there may be a `__timerfd_settime64` we can use.
314    #[cfg(fix_y2038)]
315    {
316        #[cfg(target_env = "gnu")]
317        if let Some(libc_timerfd_settime) = __timerfd_settime64.get() {
318            let mut result = MaybeUninit::<LibcItimerspec>::uninit();
319            unsafe {
320                ret(libc_timerfd_settime(
321                    borrowed_fd(fd),
322                    bitflags_bits!(flags),
323                    &new_value.clone().into(),
324                    result.as_mut_ptr(),
325                ))?;
326                return Ok(result.assume_init().into());
327            }
328        }
329
330        timerfd_settime_old(fd, flags, new_value)
331    }
332
333    #[cfg(not(fix_y2038))]
334    unsafe {
335        use crate::backend::time::types::{as_libc_itimerspec_mut_ptr, as_libc_itimerspec_ptr};
336
337        let mut result = MaybeUninit::<LibcItimerspec>::uninit();
338        ret(c::timerfd_settime(
339            borrowed_fd(fd),
340            bitflags_bits!(flags),
341            as_libc_itimerspec_ptr(new_value),
342            as_libc_itimerspec_mut_ptr(&mut result),
343        ))?;
344        Ok(result.assume_init())
345    }
346}

349#[cfg(fix_y2038)]
350fn timerfd_settime_old(
351    fd: BorrowedFd<'_>,
352    flags: TimerfdTimerFlags,
353    new_value: &Itimerspec,
354) -> io::Result<Itimerspec> {
355    let mut old_result = MaybeUninit::<c::itimerspec>::uninit();
356
357    // Convert `new_value` to the old `itimerspec` format.
358    let old_new_value = c::itimerspec {
359        it_interval: c::timespec {
360            tv_sec: new_value
361                .it_interval
362                .tv_sec
363                .try_into()
364                .map_err(|_| io::Errno::OVERFLOW)?,
365            tv_nsec: new_value
366                .it_interval
367                .tv_nsec
368                .try_into()
369                .map_err(|_| io::Errno::INVAL)?,
370        },
371        it_value: c::timespec {
372            tv_sec: new_value
373                .it_value
374                .tv_sec
375                .try_into()
376                .map_err(|_| io::Errno::OVERFLOW)?,
377            tv_nsec: new_value
378                .it_value
379                .tv_nsec
380                .try_into()
381                .map_err(|_| io::Errno::INVAL)?,
382        },
383    };
384
385    let old_result = unsafe {
386        ret(c::timerfd_settime(
387            borrowed_fd(fd),
388            bitflags_bits!(flags),
389            &old_new_value,
390            old_result.as_mut_ptr(),
391        ))?;
392        old_result.assume_init()
393    };
394
395    Ok(Itimerspec {
396        it_interval: Timespec {
397            tv_sec: old_result
398                .it_interval
399                .tv_sec
400                .try_into()
401                .map_err(|_| io::Errno::OVERFLOW)?,
402            tv_nsec: old_result.it_interval.tv_nsec as _,
403        },
404        it_value: Timespec {
405            tv_sec: old_result
406                .it_interval
407                .tv_sec
408                .try_into()
409                .map_err(|_| io::Errno::OVERFLOW)?,
410            tv_nsec: old_result.it_interval.tv_nsec as _,
411        },
412    })
413}

476/// See [`crate::timespec::fix_negative_nsecs`] for details.
477#[cfg(apple)]
478#[cfg(not(fix_y2038))]
479fn fix_negative_timespec_nsecs(mut ts: Timespec) -> Timespec {
480    let (sec, nsec) = crate::timespec::fix_negative_nsecs(ts.tv_sec as _, ts.tv_nsec as _);
481    ts.tv_sec = sec as _;
482    ts.tv_nsec = nsec as _;
483    ts
