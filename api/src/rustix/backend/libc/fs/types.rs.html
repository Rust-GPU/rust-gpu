162    /// `Mode`.
163    #[inline]
164    pub const fn from_raw_mode(st_mode: RawMode) -> Self {
165        Self::from_bits_truncate(st_mode & !c::S_IFMT as RawMode)
166    }

168    /// Construct an `st_mode` value from a `Mode`.

171        self.bits()
172    }

177    /// Support conversions from raw mode values to `Mode`.

181    /// assert_eq!(Mode::from(0o700), Mode::RWXU);

477    /// `S_IFIFO`

479    #[doc(alias = "IFO")]

487    CharacterDevice = c::S_IFCHR as isize,
488
489    /// `S_IFBLK`
490    BlockDevice = c::S_IFBLK as isize,
491
492    /// An unknown filesystem object.
493    Unknown,
494}
495
496impl FileType {
497    /// Construct a `FileType` from the `S_IFMT` bits of the `st_mode` field of
498    /// a `Stat`.
499    #[inline]
500    pub const fn from_raw_mode(st_mode: RawMode) -> Self {
501        match (st_mode as c::mode_t) & c::S_IFMT {
502            c::S_IFREG => Self::RegularFile,
503            c::S_IFDIR => Self::Directory,
504            c::S_IFLNK => Self::Symlink,
505            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `S_IFIFO`.
506            c::S_IFIFO => Self::Fifo,
507            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `S_IFSOCK`.
508            c::S_IFSOCK => Self::Socket,
509            c::S_IFCHR => Self::CharacterDevice,
510            c::S_IFBLK => Self::BlockDevice,
511            _ => Self::Unknown,
512        }
513    }
514
515    /// Construct an `st_mode` value from a `FileType`.
516    #[inline]
517    pub const fn as_raw_mode(self) -> RawMode {
518        match self {
519            Self::RegularFile => c::S_IFREG as RawMode,
520            Self::Directory => c::S_IFDIR as RawMode,
521            Self::Symlink => c::S_IFLNK as RawMode,
522            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `S_IFIFO`.
523            Self::Fifo => c::S_IFIFO as RawMode,
524            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `S_IFSOCK`.
525            Self::Socket => c::S_IFSOCK as RawMode,
526            Self::CharacterDevice => c::S_IFCHR as RawMode,
527            Self::BlockDevice => c::S_IFBLK as RawMode,
528            Self::Unknown => c::S_IFMT as RawMode,
529        }
530    }
531
532    /// Construct a `FileType` from the `d_type` field of a `c::dirent`.
533    #[cfg(not(any(
534        solarish,
535        target_os = "aix",
536        target_os = "espidf",
537        target_os = "haiku",
538        target_os = "nto",
539        target_os = "redox",
540        target_os = "vita"
541    )))]
542    #[inline]
543    pub(crate) const fn from_dirent_d_type(d_type: u8) -> Self {
544        match d_type {
545            c::DT_REG => Self::RegularFile,
546            c::DT_DIR => Self::Directory,
547            c::DT_LNK => Self::Symlink,
548            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `DT_SOCK`.
549            c::DT_SOCK => Self::Socket,
550            #[cfg(not(target_os = "wasi"))] // TODO: Use WASI's `DT_FIFO`.
551            c::DT_FIFO => Self::Fifo,
552            c::DT_CHR => Self::CharacterDevice,
553            c::DT_BLK => Self::BlockDevice,
554            // c::DT_UNKNOWN |
555            _ => Self::Unknown,
556        }
557    }
