61/// A buffer type for use with [`vmsplice`].
62///
63/// It is guaranteed to be ABI compatible with the iovec type on Unix platforms
64/// and `WSABUF` on Windows. Unlike `IoSlice` and `IoSliceMut` it is
65/// semantically like a raw pointer, and therefore can be shared or mutated as
66/// needed.
67///
68/// [`vmsplice`]: crate::pipe::vmsplice
69#[cfg(linux_kernel)]
70#[repr(transparent)]
71pub struct IoSliceRaw<'a> {
72    _buf: c::iovec,
73    _lifetime: PhantomData<&'a ()>,
74}
75
76#[cfg(linux_kernel)]
77impl<'a> IoSliceRaw<'a> {
78    /// Creates a new `IoSlice` wrapping a byte slice.
79    pub fn from_slice(buf: &'a [u8]) -> Self {
80        IoSliceRaw {
81            _buf: c::iovec {
82                iov_base: buf.as_ptr() as *mut u8 as *mut c::c_void,
83                iov_len: buf.len() as _,
84            },
85            _lifetime: PhantomData,
86        }
87    }
88
89    /// Creates a new `IoSlice` wrapping a mutable byte slice.
90    pub fn from_slice_mut(buf: &'a mut [u8]) -> Self {
91        IoSliceRaw {
92            _buf: c::iovec {
93                iov_base: buf.as_mut_ptr() as *mut c::c_void,
94                iov_len: buf.len() as _,
95            },
96            _lifetime: PhantomData,
97        }
98    }
99}
