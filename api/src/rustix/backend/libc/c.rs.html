244#[cfg(target_os = "android")]
245mod readwrite_pv64 {
246    use super::*;
247
248    pub(in super::super) unsafe fn preadv64(
249        fd: libc::c_int,
250        iov: *const libc::iovec,
251        iovcnt: libc::c_int,
252        offset: libc::off64_t,
253    ) -> libc::ssize_t {
254        // Older Android libc lacks `preadv64`, so use the `weak!` mechanism to
255        // test for it, and call back to `libc::syscall`. We don't use
256        // `weak_or_syscall` here because we need to pass the 64-bit offset
257        // specially.
258        weak! {
259            fn preadv64(libc::c_int, *const libc::iovec, libc::c_int, libc::off64_t) -> libc::ssize_t
260        }
261        if let Some(fun) = preadv64.get() {
262            fun(fd, iov, iovcnt, offset)
263        } else {
264            // Unlike the plain "p" functions, the "pv" functions pass their
265            // offset in an endian-independent way, and always in two registers.
266            syscall! {
267                fn preadv(
268                    fd: libc::c_int,
269                    iov: *const libc::iovec,
270                    iovcnt: libc::c_int,
271                    offset_lo: usize,
272                    offset_hi: usize
273                ) via SYS_preadv -> libc::ssize_t
274            }
275            preadv(fd, iov, iovcnt, offset as usize, (offset >> 32) as usize)
276        }
277    }
278    pub(in super::super) unsafe fn pwritev64(
279        fd: libc::c_int,
280        iov: *const libc::iovec,
281        iovcnt: libc::c_int,
282        offset: libc::off64_t,
283    ) -> libc::ssize_t {
284        // See the comments in `preadv64`.
285        weak! {
286            fn pwritev64(libc::c_int, *const libc::iovec, libc::c_int, libc::off64_t) -> libc::ssize_t
287        }
288        if let Some(fun) = pwritev64.get() {
289            fun(fd, iov, iovcnt, offset)
290        } else {
291            // Unlike the plain "p" functions, the "pv" functions pass their
292            // offset in an endian-independent way, and always in two registers.
293            syscall! {
294                fn pwritev(
295                    fd: libc::c_int,
296                    iov: *const libc::iovec,
297                    iovcnt: libc::c_int,
298                    offset_lo: usize,
299                    offset_hi: usize
300                ) via SYS_pwritev -> libc::ssize_t
301            }
302            pwritev(fd, iov, iovcnt, offset as usize, (offset >> 32) as usize)
303        }
304    }
305}
