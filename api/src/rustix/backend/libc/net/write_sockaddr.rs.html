1//! The BSD sockets API requires us to read the `ss_family` field before we can
2//! interpret the rest of a `sockaddr` produced by the kernel.
3
4use super::addr::SocketAddrStorage;
5#[cfg(unix)]
6use super::addr::SocketAddrUnix;
7use super::ext::{in6_addr_new, in_addr_new, sockaddr_in6_new};
8use crate::backend::c;
9#[cfg(target_os = "linux")]
10use crate::net::xdp::SocketAddrXdp;
11use crate::net::{SocketAddrAny, SocketAddrV4, SocketAddrV6};
12use core::mem::size_of;
13
14pub(crate) unsafe fn write_sockaddr(
15    addr: &SocketAddrAny,
16    storage: *mut SocketAddrStorage,
17) -> usize {
18    match addr {
19        SocketAddrAny::V4(v4) => write_sockaddr_v4(v4, storage),
20        SocketAddrAny::V6(v6) => write_sockaddr_v6(v6, storage),
21        #[cfg(unix)]
22        SocketAddrAny::Unix(unix) => write_sockaddr_unix(unix, storage),
23        #[cfg(target_os = "linux")]
24        SocketAddrAny::Xdp(xdp) => write_sockaddr_xdp(xdp, storage),
25    }
26}
27
28pub(crate) fn encode_sockaddr_v4(v4: &SocketAddrV4) -> c::sockaddr_in {
29    c::sockaddr_in {
30        #[cfg(any(
31            bsd,
32            target_os = "aix",
33            target_os = "espidf",
34            target_os = "haiku",
35            target_os = "hurd",
36            target_os = "nto",
37            target_os = "vita",
38        ))]
39        sin_len: size_of::<c::sockaddr_in>() as _,
40        sin_family: c::AF_INET as _,
41        sin_port: u16::to_be(v4.port()),
42        sin_addr: in_addr_new(u32::from_ne_bytes(v4.ip().octets())),
43        #[cfg(not(any(target_os = "haiku", target_os = "vita")))]
44        sin_zero: [0; 8_usize],
45        #[cfg(target_os = "haiku")]
46        sin_zero: [0; 24_usize],
47        #[cfg(target_os = "vita")]
48        sin_zero: [0; 6_usize],
49        #[cfg(target_os = "vita")]
50        sin_vport: 0,
51    }
52}
53
54unsafe fn write_sockaddr_v4(v4: &SocketAddrV4, storage: *mut SocketAddrStorage) -> usize {
55    let encoded = encode_sockaddr_v4(v4);
56    core::ptr::write(storage.cast(), encoded);
57    size_of::<c::sockaddr_in>()
58}

60pub(crate) fn encode_sockaddr_v6(v6: &SocketAddrV6) -> c::sockaddr_in6 {
61    #[cfg(any(
62        bsd,
63        target_os = "aix",
64        target_os = "espidf",
65        target_os = "haiku",
66        target_os = "hurd",
67        target_os = "nto",
68        target_os = "vita"
69    ))]
70    {
71        sockaddr_in6_new(
72            size_of::<c::sockaddr_in6>() as _,
73            c::AF_INET6 as _,
74            u16::to_be(v6.port()),
75            u32::to_be(v6.flowinfo()),
76            in6_addr_new(v6.ip().octets()),
77            v6.scope_id(),
78        )
79    }
80    #[cfg(not(any(
81        bsd,
82        target_os = "aix",
83        target_os = "espidf",
84        target_os = "haiku",
85        target_os = "hurd",
86        target_os = "nto",
87        target_os = "vita"
88    )))]
89    {
90        sockaddr_in6_new(
91            c::AF_INET6 as _,
92            u16::to_be(v6.port()),
93            u32::to_be(v6.flowinfo()),
94            in6_addr_new(v6.ip().octets()),
95            v6.scope_id(),
96        )
97    }
98}

100unsafe fn write_sockaddr_v6(v6: &SocketAddrV6, storage: *mut SocketAddrStorage) -> usize {
101    let encoded = encode_sockaddr_v6(v6);
102    core::ptr::write(storage.cast(), encoded);
103    size_of::<c::sockaddr_in6>()
104}

106#[cfg(unix)]
107unsafe fn write_sockaddr_unix(unix: &SocketAddrUnix, storage: *mut SocketAddrStorage) -> usize {
108    core::ptr::write(storage.cast(), unix.unix);
109    unix.len()
110}

112#[cfg(target_os = "linux")]
113pub(crate) fn encode_sockaddr_xdp(xdp: &SocketAddrXdp) -> c::sockaddr_xdp {
114    c::sockaddr_xdp {
115        sxdp_family: c::AF_XDP as _,
116        sxdp_flags: xdp.flags().bits(),
117        sxdp_ifindex: xdp.interface_index(),
118        sxdp_queue_id: xdp.queue_id(),
119        sxdp_shared_umem_fd: xdp.shared_umem_fd(),
120    }
121}
