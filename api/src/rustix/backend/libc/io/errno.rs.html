11/// The error type for `rustix` APIs. This is similar to [`std::io::Error`],

22///  - [illumos]
23///  - [glibc]

1073impl Errno {
1074    /// Extract an `Errno` value from a `std::io::Error`.
1075    ///
1076    /// This isn't a `From` conversion because it's expected to be relatively
1077    /// uncommon.
1078    #[cfg(feature = "std")]
1079    #[inline]
1080    pub fn from_io_error(io_err: &std::io::Error) -> Option<Self> {
1081        io_err
1082            .raw_os_error()
1083            .and_then(|raw| if raw != 0 { Some(Self(raw)) } else { None })
1084    }
1085
1086    /// Extract the raw OS error number from this error.
1087    #[inline]
1088    pub const fn raw_os_error(self) -> i32 {
1089        self.0
1090    }
1091
1092    /// Construct an `Errno` from a raw OS error number.
1093    #[inline]
1094    pub const fn from_raw_os_error(raw: i32) -> Self {
1095        Self(raw)
1096    }
1097
1098    pub(crate) fn last_os_error() -> Self {
1099        Self(errno().0)
1100    }
