23fn attrs_to_spirv(attrs: &CodegenFnAttrs) -> FunctionControl {
24    let mut control = FunctionControl::NONE;
25    match attrs.inline {
26        InlineAttr::None => (),
27        InlineAttr::Hint | InlineAttr::Always | InlineAttr::Force { .. } => {
28            control.insert(FunctionControl::INLINE);
29        }
30        InlineAttr::Never => control.insert(FunctionControl::DONT_INLINE),
31    }
32    if attrs.flags.contains(CodegenFnAttrFlags::FFI_PURE) {
33        control.insert(FunctionControl::PURE);
34    }
35    if attrs.flags.contains(CodegenFnAttrFlags::FFI_CONST) {
36        control.insert(FunctionControl::CONST);
37    }
38    control
39}

41impl<'tcx> CodegenCx<'tcx> {
42    /// Returns a function if it already exists, or declares a header if it doesn't.
43    pub fn get_fn_ext(&self, instance: Instance<'tcx>) -> SpirvFunctionCursor {
44        assert!(!instance.args.has_infer());
45        assert!(!instance.args.has_escaping_bound_vars());
46
47        if let Some(&func) = self.fn_instances.borrow().get(&instance) {
48            return func;
49        }
50
51        // Because we've already declared everything with predefine_fn, if we hit this branch, we're guaranteed to be
52        // importing this function from elsewhere. So, slap an extern on it.
53        let linkage = Some(LinkageType::Import);
54        let llfn = self.declare_fn_ext(instance, linkage);
55
56        self.fn_instances.borrow_mut().insert(instance, llfn);
57
58        llfn
59    }
60
61    // The call graph of how this is reachable is a little tangled, so:
62    // MiscCodegenMethods::get_fn -> get_fn_ext -> declare_fn_ext
63    // MiscCodegenMethods::get_fn_addr -> get_fn_ext -> declare_fn_ext
64    // PreDefineCodegenMethods::predefine_fn -> declare_fn_ext
65    fn declare_fn_ext(
66        &self,
67        instance: Instance<'tcx>,
68        linkage: Option<LinkageType>,
69    ) -> SpirvFunctionCursor {
70        let def_id = instance.def_id();
71
72        let control = attrs_to_spirv(self.tcx.codegen_fn_attrs(def_id));
73        let fn_abi = self.fn_abi_of_instance(instance, ty::List::empty());
74        let span = self.tcx.def_span(def_id);
75        let function_type = fn_abi.spirv_type(span, self);
76        let (return_type, argument_types) = match self.lookup_type(function_type) {
77            SpirvType::Function {
78                return_type,
79                arguments,
80            } => (return_type, arguments),
81            other => bug!("fn_abi type {}", other.debug(function_type, self)),
82        };
83
84        let declared = {
85            let mut emit = self.emit_global();
86            let id = emit
87                .begin_function(return_type, None, control, function_type)
88                .unwrap();
89            let index_in_builder = emit.selected_function().unwrap();
90
91            // FIXME(eddyb) omitting `OpFunctionParameter` on imports might be
92            // illegal, this probably shouldn't be conditional at all.
93            if linkage != Some(LinkageType::Import) {
94                for &ty in argument_types {
95                    emit.function_parameter(ty).unwrap();
96                }
97            }
98            emit.end_function().unwrap();
99            SpirvFunctionCursor {
100                ty: function_type,
101                id,
102                index_in_builder,
103            }
104        };
105        let fn_id = declared.id;
106
107        // HACK(eddyb) this is a temporary workaround due to our use of `rspirv`,
108        // which prevents us from attaching `OpLine`s to `OpFunction` definitions,
109        // but we can use our custom `SrcLocDecoration` instead.
110        let src_loc_inst = SrcLocDecoration::from_rustc_span(
111            self.tcx.def_ident_span(def_id).unwrap_or(span),
112            &self.builder,
113        )
114        .map(|src_loc| src_loc.encode_to_inst(fn_id));
115        self.emit_global()
116            .module_mut()
117            .annotations
118            .extend(src_loc_inst);
119
120        // HACK(eddyb) this is a bit roundabout, but the easiest way to get a
121        // fully absolute path that contains at least as much information as
122        // `instance.to_string()` (at least with `-C symbol-mangling-version=v0`).
123        // While we could use the mangled symbol instead, like we do for linkage,
124        // `OpName` is more of a debugging aid, so not having to separately
125        // demangle the SPIR-V can help. However, if some tools assume `OpName`
126        // is always a valid identifier, we may have to offer the mangled name
127        // (as some sort of opt-in, or toggled based on the platform, etc.).
128        let symbol_name = self.tcx.symbol_name(instance).name;
129        let demangled_symbol_name = format!("{:#}", rustc_demangle::demangle(symbol_name));
130        self.emit_global().name(fn_id, &demangled_symbol_name);
131
132        if let Some(linkage) = linkage {
133            self.set_linkage(fn_id, symbol_name.to_owned(), linkage);
134        }
135
136        let attrs = AggregatedSpirvAttributes::parse(self, self.tcx.get_attrs_unchecked(def_id));
137        if let Some(entry) = attrs.entry.map(|attr| attr.value) {
138            // HACK(eddyb) early insert to let `shader_entry_stub` call this
139            // very function via `get_fn_addr`.
140            self.fn_instances.borrow_mut().insert(instance, declared);
141
142            let entry_name = entry
143                .name
144                .as_ref()
145                .map_or_else(|| instance.to_string(), ToString::to_string);
146            self.entry_stub(instance, fn_abi, entry_name, entry);
147        }
148
149        // FIXME(eddyb) should the maps exist at all, now that the `DefId` is known
150        // at `call` time, and presumably its high-level details can be looked up?
151        if attrs.buffer_load_intrinsic.is_some() {
152            let mode = &fn_abi.ret.mode;
153            self.buffer_load_intrinsics
154                .borrow_mut()
155                .insert(def_id, mode);
156        }
157        if attrs.buffer_store_intrinsic.is_some() {
158            let mode = &fn_abi.args.last().unwrap().mode;
159            self.buffer_store_intrinsics
160                .borrow_mut()
161                .insert(def_id, mode);
162        }
163
164        if self.tcx.crate_name(def_id.krate) == self.sym.libm {
165            let item_name = self.tcx.item_name(def_id);
166            let intrinsic = self.sym.libm_intrinsics.get(&item_name);
167            if self.tcx.visibility(def_id) == ty::Visibility::Public {
168                match intrinsic {
169                    Some(&intrinsic) => {
170                        self.libm_intrinsics.borrow_mut().insert(def_id, intrinsic);
171                    }
172                    None => {
173                        self.tcx.dcx().err(format!(
174                            "missing libm intrinsic {symbol_name}, which is {instance}"
175                        ));
176                    }
177                }
178            }
179        }
180
181        // Check if this is a From trait implementation
182        if let Some(impl_def_id) = self.tcx.impl_of_method(def_id)
183            && let Some(trait_ref) = self.tcx.impl_trait_ref(impl_def_id)
184        {
185            let trait_def_id = trait_ref.skip_binder().def_id;
186
187            // Check if this is the From trait.
188            let trait_path = self.tcx.def_path_str(trait_def_id);
189            if matches!(
190                trait_path.as_str(),
191                "core::convert::From" | "std::convert::From"
192            ) {
193                // Extract the source and target types from the trait substitutions
194                let trait_args = trait_ref.skip_binder().args;
195                if let (Some(target_ty), Some(source_ty)) =
196                    (trait_args.types().nth(0), trait_args.types().nth(1))
197                {
198                    self.from_trait_impls
199                        .borrow_mut()
200                        .insert(def_id, (source_ty, target_ty));
201                }
202            }
203        }
204
205        if [
206            self.tcx.lang_items().panic_fn(),
207            self.tcx.lang_items().panic_fmt(),
208            self.tcx.lang_items().panic_nounwind(),
209        ]
210        .contains(&Some(def_id))
211        {
212            self.panic_entry_points.borrow_mut().insert(def_id);
213        }
214
215        // HACK(eddyb) there is no good way to identify these definitions
216        // (e.g. no `#[lang = "..."]` attribute), but this works well enough.
217        if let Some("panic_nounwind_fmt" | "panic_explicit") =
218            demangled_symbol_name.strip_prefix("core::panicking::")
219        {
220            self.panic_entry_points.borrow_mut().insert(def_id);
221        }
222        if let Some(pieces_len) = demangled_symbol_name
223            .strip_prefix("<core::fmt::Arguments>::new_const::<")
224            .and_then(|s| s.strip_suffix(">"))
225        {
226            self.fmt_args_new_fn_ids
227                .borrow_mut()
228                .insert(fn_id, (pieces_len.parse().unwrap(), 0));
229        }
230        if let Some(generics) = demangled_symbol_name
231            .strip_prefix("<core::fmt::Arguments>::new_v1::<")
232            .and_then(|s| s.strip_suffix(">"))
233        {
234            let (pieces_len, rt_args_len) = generics.split_once(", ").unwrap();
235            self.fmt_args_new_fn_ids.borrow_mut().insert(
236                fn_id,
237                (pieces_len.parse().unwrap(), rt_args_len.parse().unwrap()),
238            );
239        }
240        if demangled_symbol_name == "<core::fmt::Arguments>::new_v1_formatted" {
241            // HACK(eddyb) `!0` used as a placeholder value to indicate "dynamic".
242            self.fmt_args_new_fn_ids
243                .borrow_mut()
244                .insert(fn_id, (!0, !0));
245        }
246
247        // HACK(eddyb) there is no good way to identify these definitions
248        // (e.g. no `#[lang = "..."]` attribute), but this works well enough.
249        if let Some(suffix) = demangled_symbol_name.strip_prefix("<core::fmt::rt::Argument>::new_")
250        {
251            let spec = suffix.split_once("::<").and_then(|(method_suffix, _)| {
252                Some(match method_suffix {
253                    "display" => ' ',
254                    "debug" => '?',
255                    "octal" => 'o',
256                    "lower_hex" => 'x',
257                    "upper_hex" => 'X',
258                    "pointer" => 'p',
259                    "binary" => 'b',
260                    "lower_exp" => 'e',
261                    "upper_exp" => 'E',
262                    _ => return None,
263                })
264            });
265            if let Some(spec) = spec
266                && let Some((ty,)) = instance.args.types().collect_tuple()
267            {
268                self.fmt_rt_arg_new_fn_ids_to_ty_and_spec
269                    .borrow_mut()
270                    .insert(fn_id, (ty, spec));
271            }
272        }
273
274        declared
275    }

277    pub fn get_static(&self, def_id: DefId) -> SpirvValue {
278        if let Some(&g) = self.statics.borrow().get(&def_id) {
279            return g;
280        }
281
282        let defined_in_current_codegen_unit = self
283            .codegen_unit
284            .items()
285            .contains_key(&MonoItem::Static(def_id));
286        assert!(
287            !defined_in_current_codegen_unit,
288            "get_static() should always hit the cache for statics defined in the same CGU, but did not for `{def_id:?}`"
289        );
290
291        let instance = Instance::mono(self.tcx, def_id);
292        let ty = instance.ty(self.tcx, TypingEnv::fully_monomorphized());
293        let sym = self.tcx.symbol_name(instance).name;
294        let span = self.tcx.def_span(def_id);
295        let g = self.declare_global(span, self.layout_of(ty).spirv_type(span, self));
296        self.statics.borrow_mut().insert(def_id, g);
297        self.set_linkage(g.def_cx(self), sym.to_string(), LinkageType::Import);
298        g
299    }
300
301    fn declare_global(&self, span: Span, ty: Word) -> SpirvValue {
302        let ptr_ty = SpirvType::Pointer { pointee: ty }.def(span, self);
303        // FIXME(eddyb) figure out what the correct storage class is.
304        let result = self
305            .emit_global()
306            .variable(ptr_ty, None, StorageClass::Private, None)
307            .with_type(ptr_ty);
308        // TODO: These should be StorageClass::Private, so just zombie for now.
309        // FIXME(eddyb) why zombie? this looks like it should just work nowadays.
310        self.zombie_with_span(result.def_cx(self), span, "globals are not supported yet");
311        result
312    }

315impl<'tcx> PreDefineCodegenMethods<'tcx> for CodegenCx<'tcx> {
316    fn predefine_static(
317        &mut self,
318        def_id: DefId,
319        linkage: Linkage,
320        _visibility: Visibility,
321        symbol_name: &str,
322    ) {
323        let instance = Instance::mono(self.tcx, def_id);
324        let ty = instance.ty(self.tcx, TypingEnv::fully_monomorphized());
325        let span = self.tcx.def_span(def_id);
326        let spvty = self.layout_of(ty).spirv_type(span, self);
327        let linkage = match linkage {
328            Linkage::External => Some(LinkageType::Export),
329            Linkage::Internal => None,
330            other => {
331                self.tcx.dcx().err(format!(
332                    "TODO: Linkage type {other:?} not supported yet for static var symbol {symbol_name}"
333                ));
334                None
335            }
336        };
337
338        let g = self.declare_global(span, spvty);
339
340        self.statics.borrow_mut().insert(def_id, g);
341        if let Some(linkage) = linkage {
342            self.set_linkage(g.def_cx(self), symbol_name.to_string(), linkage);
343        }
344    }
345
346    fn predefine_fn(
347        &mut self,
348        instance: Instance<'tcx>,
349        linkage: Linkage,
350        _visibility: Visibility,
351        symbol_name: &str,
352    ) {
353        let linkage2 = match linkage {
354            // super sketchy hack: memcpy, memmove, memset, memcmp, and bcmp in the
355            // compiler_builtins crate use the WeakAny linkage type. Treat it as actually External
356            // linkage because we know there's only one of them.
357            Linkage::External | Linkage::WeakAny => Some(LinkageType::Export),
358            Linkage::Internal => None,
359            other => {
360                self.tcx.dcx().err(format!(
361                    "TODO: Linkage type {other:?} not supported yet for function symbol {symbol_name}"
362                ));
363                None
364            }
365        };
366        let declared = self.declare_fn_ext(instance, linkage2);
367
368        self.fn_instances.borrow_mut().insert(instance, declared);
369    }
370}

372impl<'tcx> StaticCodegenMethods for CodegenCx<'tcx> {
373    fn static_addr_of(&self, cv: Self::Value, _align: Align, _kind: Option<&str>) -> Self::Value {
374        self.def_constant(
375            self.type_ptr_to(cv.ty),
376            SpirvConst::PtrTo {
377                pointee: cv.def_cx(self),
378            },
379        )
380    }
381
382    fn codegen_static(&mut self, def_id: DefId) {
383        let g = self.get_static(def_id);
384
385        let alloc = match self.tcx.eval_static_initializer(def_id) {
386            Ok(alloc) => alloc,
387            // Error has already been reported
388            Err(_) => return,
389        };
390        let value_ty = match self.lookup_type(g.ty) {
391            SpirvType::Pointer { pointee } => pointee,
392            other => self.tcx.dcx().fatal(format!(
393                "global had non-pointer type {}",
394                other.debug(g.ty, self)
395            )),
396        };
397        let v = self.create_const_alloc(alloc, value_ty);
398        assert_ty_eq!(self, value_ty, v.ty);
399        self.builder
400            .set_global_initializer(g.def_cx(self), v.def_cx(self));
401
402        let attrs = self.tcx.codegen_fn_attrs(def_id);
403
404        let alloc = alloc.inner();
405        let align_override =
406            Some(alloc.align).filter(|&align| align != self.lookup_type(value_ty).alignof(self));
407        if let Some(_align) = align_override {
408            // FIXME(eddyb) implement, or at least error.
409        }
410
411        if attrs.flags.contains(CodegenFnAttrFlags::THREAD_LOCAL) {
412            // FIXME(eddyb) implement, or at least error.
413        }
414
415        if let Some(_section) = attrs.link_section {
416            // FIXME(eddyb) implement, or at least error.
417        }
418
419        if attrs.flags.contains(CodegenFnAttrFlags::USED_COMPILER) {
420            // `USED` and `USED_LINKER` can't be used together.
421            assert!(!attrs.flags.contains(CodegenFnAttrFlags::USED_LINKER));
422
423            // The semantics of #[used] in Rust only require the symbol to make it into the
424            // object file. It is explicitly allowed for the linker to strip the symbol if it
425            // is dead, which means we are allowed to use `llvm.compiler.used` instead of
426            // `llvm.used` here.
427            //
428            // Additionally, https://reviews.llvm.org/D97448 in LLVM 13 started emitting unique
429            // sections with SHF_GNU_RETAIN flag for llvm.used symbols, which may trigger bugs
430            // in the handling of `.init_array` (the static constructor list) in versions of
431            // the gold linker (prior to the one released with binutils 2.36).
432            //
433            // That said, we only ever emit these when `#[used(compiler)]` is explicitly
434            // requested. This is to avoid similar breakage on other targets, in particular
435            // MachO targets have *their* static constructor lists broken if `llvm.compiler.used`
436            // is emitted rather than `llvm.used`. However, that check happens when assigning
437            // the `CodegenFnAttrFlags` in the `codegen_fn_attrs` query, so we don't need to
438            // take care of it here.
439            self.add_compiler_used_global(g);
440        }
441        if attrs.flags.contains(CodegenFnAttrFlags::USED_LINKER) {
442            // `USED` and `USED_LINKER` can't be used together.
443            assert!(!attrs.flags.contains(CodegenFnAttrFlags::USED_COMPILER));
444
445            self.add_used_global(g);
446        }
447    }
448}
