220    fn scalar_to_backend(
221        &self,
222        scalar: Scalar,
223        layout: abi::Scalar,
224        ty: Self::Type,
225    ) -> Self::Value {
226        match scalar {
227            Scalar::Int(int) => {
228                assert_eq!(int.size(), layout.primitive().size(self));
229                let data = int.to_uint(int.size());
230
231                if let Primitive::Pointer(_) = layout.primitive() {
232                    if data == 0 {
233                        self.constant_null(ty)
234                    } else {
235                        let result = self.undef(ty);
236                        self.zombie_no_span(
237                            result.def_cx(self),
238                            "pointer has non-null integer address",
239                        );
240                        result
241                    }
242                } else {
243                    self.def_constant(ty, SpirvConst::Scalar(data))
244                }
245            }
246            Scalar::Ptr(ptr, _) => {
247                let (prov, offset) = ptr.into_parts();
248                let alloc_id = prov.alloc_id();
249                let (base_addr, _base_addr_space) = match self.tcx.global_alloc(alloc_id) {
250                    GlobalAlloc::Memory(alloc) => {
251                        let pointee = match self.lookup_type(ty) {
252                            SpirvType::Pointer { pointee } => pointee,
253                            other => self.tcx.dcx().fatal(format!(
254                                "GlobalAlloc::Memory type not implemented: {}",
255                                other.debug(ty, self)
256                            )),
257                        };
258                        let init = self.create_const_alloc(alloc, pointee);
259                        let value = self.static_addr_of(init, alloc.inner().align, None);
260                        (value, AddressSpace::DATA)
261                    }
262                    GlobalAlloc::Function { instance } => (
263                        self.get_fn_addr(instance),
264                        self.data_layout().instruction_address_space,
265                    ),
266                    GlobalAlloc::VTable(vty, dyn_ty) => {
267                        let alloc = self
268                            .tcx
269                            .global_alloc(self.tcx.vtable_allocation((
270                                vty,
271                                dyn_ty.principal().map(|principal| {
272                                    self.tcx.instantiate_bound_regions_with_erased(principal)
273                                }),
274                            )))
275                            .unwrap_memory();
276                        let pointee = match self.lookup_type(ty) {
277                            SpirvType::Pointer { pointee } => pointee,
278                            other => self.tcx.dcx().fatal(format!(
279                                "GlobalAlloc::VTable type not implemented: {}",
280                                other.debug(ty, self)
281                            )),
282                        };
283                        let init = self.create_const_alloc(alloc, pointee);
284                        let value = self.static_addr_of(init, alloc.inner().align, None);
285                        (value, AddressSpace::DATA)
286                    }
287                    GlobalAlloc::Static(def_id) => {
288                        assert!(self.tcx.is_static(def_id));
289                        assert!(!self.tcx.is_thread_local_static(def_id));
290                        (self.get_static(def_id), AddressSpace::DATA)
291                    }
292                };
293                let value = if offset.bytes() == 0 {
294                    base_addr
295                } else {
296                    self.tcx
297                        .dcx()
298                        .fatal("Non-zero scalar_to_backend ptr.offset not supported")
299                    // let offset = self.constant_bit64(ptr.offset.bytes());
300                    // self.gep(base_addr, once(offset))
301                };
302                if let Primitive::Pointer(_) = layout.primitive() {
303                    assert_ty_eq!(self, value.ty, ty);
304                    value
305                } else {
306                    self.tcx
307                        .dcx()
308                        .fatal("Non-pointer-typed scalar_to_backend Scalar::Ptr not supported");
309                    // unsafe { llvm::LLVMConstPtrToInt(llval, llty) }
310                }
311            }
312        }
313    }

317    // to its final type (e.g. that will be loaded as).
318    // FIXME(eddyb) replace this with `qptr` handling of constant data.
319    fn const_data_from_alloc(&self, alloc: ConstAllocation<'_>) -> Self::Value {
320        // HACK(eddyb) the `ConstCodegenMethods` trait no longer guarantees the
321        // lifetime that `alloc` is interned for, but since it *is* interned,
322        // we can cheaply recover it (see also the `ty::Lift` infrastructure).
323        let alloc = self.tcx.lift(alloc).unwrap();
324
325        let void_type = SpirvType::Void.def(DUMMY_SP, self);
326        self.def_constant(void_type, SpirvConst::ConstDataFromAlloc(alloc))
327    }

342impl<'tcx> CodegenCx<'tcx> {
343    pub fn const_bitcast(&self, val: SpirvValue, ty: Word) -> SpirvValue {
344        // HACK(eddyb) special-case `const_data_from_alloc` + `static_addr_of`
345        // as the old `from_const_alloc` (now `OperandRef::from_const_alloc`).
346        if let SpirvValueKind::IllegalConst(_) = val.kind
347            && let Some(SpirvConst::PtrTo { pointee }) = self.builder.lookup_const(val)
348            && let Some(SpirvConst::ConstDataFromAlloc(alloc)) =
349                self.builder.lookup_const_by_id(pointee)
350            && let SpirvType::Pointer { pointee } = self.lookup_type(ty)
351        {
352            let mut offset = Size::ZERO;
353            let init = self.read_from_const_alloc(alloc, &mut offset, pointee);
354            return self.static_addr_of(init, alloc.inner().align, None);
355        }
356
357        if val.ty == ty {
358            val
359        } else {
360            // FIXME(eddyb) implement via `OpSpecConstantOp`.
361            // FIXME(eddyb) this zombies the original value without creating a new one.
362            let result = val.def_cx(self).with_type(ty);
363            self.zombie_no_span(result.def_cx(self), "const_bitcast");
364            result
365        }
366    }

370    pub fn primitive_to_scalar(&self, value: Primitive) -> abi::Scalar {
371        let bits = value.size(self.data_layout()).bits();
372        assert!(bits <= 128);
373        abi::Scalar::Initialized {
374            value,
375            valid_range: abi::WrappingRange {
376                start: 0,
377                end: (!0 >> (128 - bits)),
378            },
379        }
380    }

382    pub fn create_const_alloc(&self, alloc: ConstAllocation<'tcx>, ty: Word) -> SpirvValue {
383        tracing::trace!(
384            "Creating const alloc of type {} with {} bytes",
385            self.debug_type(ty),
386            alloc.inner().len()
387        );
388        let mut offset = Size::ZERO;
389        let result = self.read_from_const_alloc(alloc, &mut offset, ty);
390        assert_eq!(
391            offset.bytes_usize(),
392            alloc.inner().len(),
393            "create_const_alloc must consume all bytes of an Allocation"
394        );
395        tracing::trace!("Done creating alloc of type {}", self.debug_type(ty));
396        result
397    }

399    fn read_from_const_alloc(
400        &self,
401        alloc: ConstAllocation<'tcx>,
402        offset: &mut Size,
403        ty: Word,
404    ) -> SpirvValue {
405        let ty_concrete = self.lookup_type(ty);
406        *offset = offset.align_to(ty_concrete.alignof(self));
407        // these print statements are really useful for debugging, so leave them easily available
408        // println!("const at {}: {}", offset.bytes(), self.debug_type(ty));
409        match ty_concrete {
410            SpirvType::Void => self
411                .tcx
412                .dcx()
413                .fatal("cannot create const alloc of type void"),
414            SpirvType::Bool
415            | SpirvType::Integer(..)
416            | SpirvType::Float(_)
417            | SpirvType::Pointer { .. } => {
418                let size = ty_concrete.sizeof(self).unwrap();
419                let primitive = match ty_concrete {
420                    SpirvType::Bool => Primitive::Int(Integer::fit_unsigned(0), false),
421                    SpirvType::Integer(int_size, int_signedness) => Primitive::Int(
422                        match int_size {
423                            8 => Integer::I8,
424                            16 => Integer::I16,
425                            32 => Integer::I32,
426                            64 => Integer::I64,
427                            128 => Integer::I128,
428                            other => {
429                                self.tcx
430                                    .dcx()
431                                    .fatal(format!("invalid size for integer: {other}"));
432                            }
433                        },
434                        int_signedness,
435                    ),
436                    SpirvType::Float(float_size) => Primitive::Float(match float_size {
437                        16 => Float::F16,
438                        32 => Float::F32,
439                        64 => Float::F64,
440                        128 => Float::F128,
441                        other => {
442                            self.tcx
443                                .dcx()
444                                .fatal(format!("invalid size for float: {other}"));
445                        }
446                    }),
447                    SpirvType::Pointer { .. } => Primitive::Pointer(AddressSpace::DATA),
448                    unsupported_spirv_type => bug!(
449                        "invalid spirv type internal to create_alloc_const2: {:?}",
450                        unsupported_spirv_type
451                    ),
452                };
453                // alloc_id is not needed by read_scalar, so we just use 0. If the context
454                // refers to a pointer, read_scalar will find the actual alloc_id. It
455                // only uses the input alloc_id in the case that the scalar is uninitialized
456                // as part of the error output
457                // tldr, the pointer here is only needed for the offset
458                let value = match alloc.inner().read_scalar(
459                    self,
460                    alloc_range(*offset, size),
461                    matches!(primitive, Primitive::Pointer(_)),
462                ) {
463                    Ok(scalar) => {
464                        self.scalar_to_backend(scalar, self.primitive_to_scalar(primitive), ty)
465                    }
466                    _ => self.undef(ty),
467                };
468                *offset += size;
469                value
470            }
471            SpirvType::Adt {
472                size,
473                field_types,
474                field_offsets,
475                ..
476            } => {
477                let base = *offset;
478                let mut values = Vec::with_capacity(field_types.len());
479                let mut occupied_spaces = Vec::with_capacity(field_types.len());
480                for (&ty, &field_offset) in field_types.iter().zip(field_offsets.iter()) {
481                    let total_offset_start = base + field_offset;
482                    let mut total_offset_end = total_offset_start;
483                    values.push(
484                        self.read_from_const_alloc(alloc, &mut total_offset_end, ty)
485                            .def_cx(self),
486                    );
487                    occupied_spaces.push(total_offset_start..total_offset_end);
488                }
489                if let Some(size) = size {
490                    *offset += size;
491                } else {
492                    assert_eq!(
493                        offset.bytes_usize(),
494                        alloc.inner().len(),
495                        "create_const_alloc must consume all bytes of an Allocation after an unsized struct"
496                    );
497                }
498                self.constant_composite(ty, values.into_iter())
499            }
500            SpirvType::Array { element, count } => {
501                let count = self.builder.lookup_const_scalar(count).unwrap() as usize;
502                let values = (0..count).map(|_| {
503                    self.read_from_const_alloc(alloc, offset, element)
504                        .def_cx(self)
505                });
506                self.constant_composite(ty, values)
507            }
508            SpirvType::Vector { element, count } => {
509                let total_size = ty_concrete
510                    .sizeof(self)
511                    .expect("create_const_alloc: Vectors must be sized");
512                let final_offset = *offset + total_size;
513                let values = (0..count).map(|_| {
514                    self.read_from_const_alloc(alloc, offset, element)
515                        .def_cx(self)
516                });
517                let result = self.constant_composite(ty, values);
518                assert!(*offset <= final_offset);
519                // Vectors sometimes have padding at the end (e.g. vec3), skip over it.
520                *offset = final_offset;
521                result
522            }
523            SpirvType::Matrix { element, count } => {
524                let total_size = ty_concrete
525                    .sizeof(self)
526                    .expect("create_const_alloc: Matrices must be sized");
527                let final_offset = *offset + total_size;
528                let values = (0..count).map(|_| {
529                    self.read_from_const_alloc(alloc, offset, element)
530                        .def_cx(self)
531                });
532                let result = self.constant_composite(ty, values);
533                assert!(*offset <= final_offset);
534                // Matrices sometimes have padding at the end (e.g. Mat4x3), skip over it.
535                *offset = final_offset;
536                result
537            }
538            SpirvType::RuntimeArray { element } => {
539                let mut values = Vec::new();
540                while offset.bytes_usize() != alloc.inner().len() {
541                    values.push(
542                        self.read_from_const_alloc(alloc, offset, element)
543                            .def_cx(self),
544                    );
545                }
546                let result = self.constant_composite(ty, values.into_iter());
547                // TODO: Figure out how to do this. Compiling the below crashes both clspv *and* llvm-spirv:
548                /*
549                __constant struct A {
550                    float x;
551                    int y[];
552                } a = {1, {2, 3, 4}};
553
554                __kernel void foo(__global int* data, __constant int* c) {
555                __constant struct A* asdf = &a;
556                *data = *c + asdf->y[*c];
557                }
558                */
559                // NOTE(eddyb) the above description is a bit outdated, it's now
560                // clear `OpTypeRuntimeArray` does not belong in user code, and
561                // is only for dynamically-sized SSBOs and descriptor indexing,
562                // and a general solution looks similar to `union` handling, but
563                // for the length of a fixed-length array.
564                self.zombie_no_span(result.def_cx(self), "constant `OpTypeRuntimeArray` value");
565                result
566            }
567            SpirvType::Function { .. } => self
568                .tcx
569                .dcx()
570                .fatal("TODO: SpirvType::Function not supported yet in create_const_alloc"),
571            SpirvType::Image { .. } => self.tcx.dcx().fatal("cannot create a constant image value"),
572            SpirvType::Sampler => self
