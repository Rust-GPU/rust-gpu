98impl<'tcx> CodegenCx<'tcx> {
99    pub fn new(tcx: TyCtxt<'tcx>, codegen_unit: &'tcx CodegenUnit<'tcx>) -> Self {
100        // Validate the target spec, as the backend doesn't control `--target`.
101        let target_tuple = tcx.sess.opts.target_triple.tuple();
102        let target: SpirvTarget = target_tuple.parse().unwrap_or_else(|_| {
103            let qualifier = if !target_tuple.starts_with("spirv-") {
104                "non-SPIR-V "
105            } else {
106                ""
107            };
108            tcx.dcx().fatal(format!(
109                "{qualifier}target `{target_tuple}` not supported by `rustc_codegen_spirv`",
110            ))
111        });
112        let target_spec_mismatched_jsons = {
113            use rustc_target::json::ToJson;
114
115            // HACK(eddyb) this loads the same `serde_json` used by `rustc_target`.
116            extern crate serde_json;
117
118            let expected = &target.rustc_target();
119            let found = &tcx.sess.target;
120            match &tcx.sess.opts.target_triple {
121                // HACK(eddyb) if `--target=path/to/target/spec.json` was used,
122                // `tcx.sess.target.to_json()` could still differ from it, and
123                // ideally `spirv-builder` can be forced to pass an exact match.
124                //
125                // FIXME(eddyb) consider the `RUST_TARGET_PATH` env var alternative.
126                TargetTuple::TargetTuple(_) => {
127                    // FIXME(eddyb) this case should be impossible as upstream
128                    // `rustc` doesn't support `spirv-*` targets!
129                    (expected != found).then(|| [expected, found].map(|spec| spec.to_json()))
130                }
131                TargetTuple::TargetJson { contents, .. } => {
132                    let expected = expected.to_json();
133                    let found = serde_json::from_str(contents).unwrap();
134                    (expected != found).then_some([expected, found])
135                }
136            }
137        };
138        if let Some([expected, found]) = target_spec_mismatched_jsons {
139            let diff_keys = [&expected, &found]
140                .into_iter()
141                .flat_map(|json| json.as_object().into_iter().flat_map(|obj| obj.keys()))
142                .unique()
143                .filter(|k| expected.get(k) != found.get(k));
144
145            tcx.dcx()
146                .struct_fatal(format!("mismatched `{target_tuple}` target spec"))
147                .with_note(format!(
148                    "expected (built into `rustc_codegen_spirv`):\n{expected:#}"
149                ))
150                .with_note(match &tcx.sess.opts.target_triple {
151                    TargetTuple::TargetJson {
152                        path_for_rustdoc,
153                        contents,
154                        ..
155                    } if !path_for_rustdoc.as_os_str().is_empty() => {
156                        format!("found (`{}`):\n{contents}", path_for_rustdoc.display())
157                    }
158                    _ => format!("found:\n{found:#}"),
159                })
160                .with_help(format!(
161                    "mismatched properties: {}",
162                    diff_keys.map(|k| format!("{k:?}")).join(", ")
163                ))
164                .emit();
165        }
166
167        let sym = Symbols::get();
168
169        let mut feature_names = tcx
170            .sess
171            .target_features
172            .iter()
173            .map(|s| s.as_str())
174            .collect::<Vec<_>>();
175
176        // target_features is a HashSet, not a Vec, so we need to sort to have deterministic
177        // compilation - otherwise, the order of capabilities in binaries depends on the iteration
178        // order of the hashset. Sort by the string, since that's easy.
179        feature_names.sort_unstable();
180
181        let features = feature_names
182            .into_iter()
183            .map(|s| s.parse())
184            .collect::<Result<_, String>>()
185            .unwrap_or_else(|error| {
186                tcx.dcx().err(error);
187                Vec::new()
188            });
189
190        let codegen_args = CodegenArgs::from_session(tcx.sess);
191
192        Self {
193            tcx,
194            codegen_unit,
195            builder: BuilderSpirv::new(tcx, &sym, &target, &features),
196            fn_instances: Default::default(),
197            statics: Default::default(),
198            type_cache: Default::default(),
199            vtables: Default::default(),
200            ext_inst: Default::default(),
201            zombie_decorations: Default::default(),
202            sym,
203            instruction_table: InstructionTable::new(),
204            libm_intrinsics: Default::default(),
205            panic_entry_points: Default::default(),
206            fmt_args_new_fn_ids: Default::default(),
207            fmt_rt_arg_new_fn_ids_to_ty_and_spec: Default::default(),
208            buffer_load_intrinsics: Default::default(),
209            buffer_store_intrinsics: Default::default(),
210            from_trait_impls: Default::default(),
211            i8_i16_atomics_allowed: false,
212            codegen_args,
213        }
214    }

238    /// Zombie system:
239    ///
240    /// If something unrepresentable is encountered, we don't want to fail
241    /// the compilation. Instead, we emit something bogus (usually it's fairly
242    /// faithful, though, e.g. `u128` emits `OpTypeInt 128 0`), and then mark the
243    /// resulting ID as a "zombie". We continue compiling the rest of the crate,
244    /// then, at the very end, anything that transitively references a zombie value
245    /// is stripped from the binary.
246    ///
247    /// Errors will only be emitted (by `linker::zombies`) for reachable zombies.
248    pub fn zombie_with_span(&self, word: Word, span: Span, reason: &str) {
249        self.zombie_decorations.borrow_mut().insert(
250            word,
251            (
252                ZombieDecoration {
253                    // FIXME(eddyb) this could take advantage of `Cow` and use
254                    // either `&'static str` or `String`, on a case-by-case basis.
255                    reason: reason.to_string().into(),
256                },
257                SrcLocDecoration::from_rustc_span(span, &self.builder),
258            ),
259        );
260    }
261    pub fn zombie_no_span(&self, word: Word, reason: &str) {
262        self.zombie_with_span(word, DUMMY_SP, reason);
263    }
264
265    pub fn finalize_module(self) -> Module {
266        let mut result = self.builder.finalize();
267        result
268            .annotations
269            .extend(self.zombie_decorations.into_inner().into_iter().flat_map(
270                |(id, (zombie, src_loc))| {
271                    [zombie.encode_to_inst(id)]
272                        .into_iter()
273                        .chain(src_loc.map(|src_loc| src_loc.encode_to_inst(id)))
274                },
275            ));
276        result
277    }

279    pub fn set_linkage(&self, target: Word, name: String, linkage: LinkageType) {
280        self.emit_global().decorate(
281            target,
282            Decoration::LinkageAttributes,
283            once(Operand::LiteralString(name)).chain(once(Operand::LinkageType(linkage))),
284        );
285    }

672    pub fn do_disassemble(&self, module: &Module) {
673        fn compact_ids(module: &mut rspirv::dr::Function) -> u32 {
674            let mut remap = std::collections::HashMap::new();
675            let mut insert = |current_id: &mut u32| {
676                let len = remap.len();
677                *current_id = *remap.entry(*current_id).or_insert_with(|| len as u32 + 1);
678            };
679            module.all_inst_iter_mut().for_each(|inst| {
680                if let Some(result_id) = &mut inst.result_id {
681                    insert(result_id);
682                }
683                if let Some(result_type) = &mut inst.result_type {
684                    insert(result_type);
685                }
686                inst.operands.iter_mut().for_each(|op| {
687                    if let Some(w) = op.id_ref_any_mut() {
688                        insert(w);
689                    }
690                });
691            });
692            remap.len() as u32 + 1
693        }
694
695        use rspirv::binary::Disassemble;
696
697        if self.disassemble {
698            eprintln!("{}", module.disassemble());
699        }
700
701        if let Some(func) = &self.disassemble_fn {
702            let id = module
703                .debug_names
704                .iter()
705                .find(|inst| {
706                    inst.class.opcode == rspirv::spirv::Op::Name
707                        && inst.operands[1].unwrap_literal_string() == func
708                })
709                .unwrap_or_else(|| {
710                    panic!(
711                        "no function with the name `{}` found in:\n{}\n",
712                        func,
713                        module.disassemble()
714                    )
715                })
716                .operands[0]
717                .unwrap_id_ref();
718            let mut func = module
719                .functions
720                .iter()
721                .find(|f| f.def_id().unwrap() == id)
722                .unwrap()
723                .clone();
724            // Compact to make IDs more stable
725            compact_ids(&mut func);
726            eprintln!("{}", func.disassemble());
727        }
728
729        if let Some(entry) = &self.disassemble_entry {
730            let id = module
731                .entry_points
732                .iter()
733                .filter(|inst| inst.class.opcode == rspirv::spirv::Op::EntryPoint)
734                .find(|inst| inst.operands[2].unwrap_literal_string() == entry)
735                .unwrap_or_else(|| {
736                    panic!(
737                        "no entry point with the name `{}` found in:\n{}\n",
738                        entry,
739                        module.disassemble()
740                    )
741                })
742                .operands[1]
743                .unwrap_id_ref();
744            let mut func = module
745                .functions
746                .iter()
747                .find(|f| f.def_id().unwrap() == id)
748                .unwrap()
749                .clone();
750            // Compact to make IDs more stable
751            compact_ids(&mut func);
752            eprintln!("{}", func.disassemble());
753        }
754
755        if self.disassemble_globals {
756            for inst in module.global_inst_iter() {
757                eprintln!("{}", inst.disassemble());
758            }
759        }
760    }
761}
