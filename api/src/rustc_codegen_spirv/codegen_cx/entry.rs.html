25/// Various information about an entry-point parameter, which can only be deduced
26/// (and/or checked) in all cases by using the original reference/value Rust type
27/// (e.g. `&mut T` vs `&T` vs `T`).
28///
29/// This is in contrast to other information about "shader interface variables",
30/// that can rely on merely the SPIR-V type and/or `#[spirv(...)]` attributes.
31///
32/// See also `entry_param_deduce_from_rust_ref_or_value` (which computes this).
33struct EntryParamDeducedFromRustRefOrValue<'tcx> {
34    /// The type/layout for the data to pass onto the entry-point parameter,
35    /// either by-value (only for `Input`) or behind some kind of reference.
36    ///
37    /// That is, the original parameter type is (given `T = value_layout.ty`):
38    /// * `T` (iff `storage_class` is `Input`)
39    /// * `&T` (all shader interface storage classes other than `Input`/`Output`)
40    /// * `&mut T` (only writable storage classes)
41    value_layout: TyAndLayout<'tcx>,
42
43    /// The SPIR-V storage class to declare the shader interface variable in,
44    /// either deduced from the type (e.g. opaque handles use `UniformConstant`),
45    /// provided via `#[spirv(...)]` attributes, or an `Input`/`Output` default.
46    //
47    // HACK(eddyb) this can be `Err(SpecConstant)` to indicate this is actually
48    // an `OpSpecConstant` being exposed as if it were an `Input`
49    storage_class: Result<StorageClass, SpecConstant>,
50
51    /// Whether this entry-point parameter doesn't allow writes to the underlying
52    /// shader interface variable (i.e. is by-value, or `&T` where `T: Freeze`).
53    ///
54    /// For some storage classes, this can be mapped to `NonWritable` decorations
55    /// (only `StorageBuffer` for now, with few others, if any, plausible at all).
56    read_only: bool,
57}
58
59impl<'tcx> CodegenCx<'tcx> {
60    // Entry points declare their "interface" (all uniforms, inputs, outputs, etc.) as parameters.
61    // spir-v uses globals to declare the interface. So, we need to generate a lil stub for the
62    // "real" main that collects all those global variables and calls the user-defined main
63    // function.
64    pub fn entry_stub(
65        &self,
66        entry_instance: Instance<'tcx>,
67        fn_abi: &FnAbi<'tcx, Ty<'tcx>>,
68        name: String,
69        entry: Entry,
70    ) {
71        let entry_def_id = entry_instance.def_id();
72        let span = self
73            .tcx
74            .def_ident_span(entry_def_id)
75            .unwrap_or_else(|| self.tcx.def_span(entry_def_id));
76        let hir_params = {
77            let fn_local_def_id = if let Some(id) = entry_def_id.as_local() {
78                id
79            } else {
80                self.tcx
81                    .dcx()
82                    .span_err(span, format!("cannot declare {name} as an entry point"));
83                return;
84            };
85            self.tcx.hir_body_owned_by(fn_local_def_id).params
86        };
87        for (arg_abi, hir_param) in fn_abi.args.iter().zip(hir_params) {
88            match arg_abi.mode {
89                PassMode::Direct(_) => {}
90                PassMode::Pair(..) => {
91                    // FIXME(eddyb) implement `ScalarPair` `Input`s, or change
92                    // the `FnAbi` readjustment to only use `PassMode::Pair` for
93                    // pointers to `!Sized` types, but not other `ScalarPair`s.
94                    if !matches!(arg_abi.layout.ty.kind(), ty::Ref(..)) {
95                        self.tcx.dcx().span_err(
96                            hir_param.ty_span,
97                            format!(
98                                "entry point parameter type not yet supported \
99                                 (`{}` has `ScalarPair` ABI but is not a `&T`)",
100                                arg_abi.layout.ty
101                            ),
102                        );
103                    }
104                }
105                // FIXME(eddyb) support these (by just ignoring them) - if there
106                // is any validation concern, it should be done on the types.
107                PassMode::Ignore => self.tcx.dcx().span_fatal(
108                    hir_param.ty_span,
109                    format!(
110                        "entry point parameter type not yet supported \
111                        (`{}` has size `0`)",
112                        arg_abi.layout.ty
113                    ),
114                ),
115                _ => span_bug!(
116                    hir_param.ty_span,
117                    "query hooks should've made this `PassMode` impossible: {:#?}",
118                    arg_abi
119                ),
120            }
121        }
122        if fn_abi.ret.layout.ty.is_unit() {
123            assert_matches!(fn_abi.ret.mode, PassMode::Ignore);
124        } else {
125            self.tcx.dcx().span_err(
126                span,
127                format!(
128                    "entry point should return `()`, not `{}`",
129                    fn_abi.ret.layout.ty
130                ),
131            );
132        }
133
134        // let execution_model = entry.execution_model;
135        let stub = self.shader_entry_stub(
136            span,
137            entry_instance,
138            fn_abi,
139            hir_params,
140            name,
141            entry.execution_model,
142        );
143        let mut emit = self.emit_global();
144        entry
145            .execution_modes
146            .iter()
147            .for_each(|(execution_mode, execution_mode_extra)| {
148                emit.execution_mode(stub.id, *execution_mode, execution_mode_extra);
149            });
150    }

152    fn shader_entry_stub(
153        &self,
154        span: Span,
155        entry_instance: Instance<'tcx>,
156        entry_fn_abi: &FnAbi<'tcx, Ty<'tcx>>,
157        hir_params: &[hir::Param<'tcx>],
158        name: String,
159        execution_model: ExecutionModel,
160    ) -> SpirvFunctionCursor {
161        let stub_fn = {
162            let void = SpirvType::Void.def(span, self);
163            let fn_void_void = SpirvType::Function {
164                return_type: void,
165                arguments: &[],
166            }
167            .def(span, self);
168            let mut emit = self.emit_global();
169            let id = emit
170                .begin_function(void, None, FunctionControl::NONE, fn_void_void)
171                .unwrap();
172            let index_in_builder = emit.selected_function().unwrap();
173            emit.end_function().unwrap();
174            SpirvFunctionCursor {
175                ty: fn_void_void,
176                id,
177                index_in_builder,
178            }
179        };
180
181        let mut op_entry_point_interface_operands = vec![];
182
183        let mut bx = Builder::build(self, Builder::append_block(self, stub_fn, ""));
184        let mut call_args = vec![];
185        let mut decoration_locations = FxHashMap::default();
186        for (entry_arg_abi, hir_param) in entry_fn_abi.args.iter().zip(hir_params) {
187            bx.set_span(hir_param.span);
188            self.declare_shader_interface_for_param(
189                execution_model,
190                entry_arg_abi,
191                hir_param,
192                &mut op_entry_point_interface_operands,
193                &mut bx,
194                &mut call_args,
195                &mut decoration_locations,
196            );
197        }
198        bx.set_span(span);
199        bx.call(
200            self.get_fn(entry_instance).ty,
201            None,
202            Some(entry_fn_abi),
203            self.get_fn_addr(entry_instance),
204            &call_args,
205            None,
206            None,
207        );
208        bx.ret_void();
209
210        self.emit_global().entry_point(
211            execution_model,
212            stub_fn.id,
213            name,
214            op_entry_point_interface_operands,
215        );
216        stub_fn
217    }

219    /// Attempt to compute `EntryParamDeducedFromRustRefOrValue` (see its docs)
220    /// from `ref_or_value_layout` (and potentially some of `attrs`).
221    ///
222    // FIXME(eddyb) document this by itself.
223    fn entry_param_deduce_from_rust_ref_or_value(
224        &self,
225        ref_or_value_layout: TyAndLayout<'tcx>,
226        hir_param: &hir::Param<'tcx>,
227        attrs: &AggregatedSpirvAttributes,
228    ) -> EntryParamDeducedFromRustRefOrValue<'tcx> {
229        // FIXME(eddyb) attribute validation should be done ahead of time.
230        // FIXME(eddyb) also check the type for compatibility with being
231        // part of the interface, including potentially `Sync`ness etc.
232        // FIXME(eddyb) really need to require `T: Sync` for references
233        // (especially relevant with interior mutability!).
234        let (value_layout, explicit_mutbl, is_ref) = match *ref_or_value_layout.ty.kind() {
235            ty::Ref(_, pointee_ty, mutbl) => (self.layout_of(pointee_ty), mutbl, true),
236            _ => (ref_or_value_layout, hir::Mutability::Not, false),
237        };
238        let effective_mutbl = match explicit_mutbl {
239            // NOTE(eddyb) `T: !Freeze` used to detect "`T` has interior mutability"
240            // (i.e. "`&T` is a shared+mutable reference"), more specifically `T`
241            // containing `UnsafeCell` (but not behind any indirection), which
242            // includes many safe abstractions (e.g. `Cell`, `RefCell`, `Atomic*`).
243            hir::Mutability::Not
244                if is_ref
245                    && !value_layout
246                        .ty
247                        .is_freeze(self.tcx, ty::TypingEnv::fully_monomorphized()) =>
248            {
249                hir::Mutability::Mut
250            }
251            _ => explicit_mutbl,
252        };
253        // FIXME(eddyb) this should maybe be an extension method on `StorageClass`?
254        let expected_mutbl_for = |storage_class| match storage_class {
255            StorageClass::UniformConstant
256            | StorageClass::Input
257            | StorageClass::Uniform
258            | StorageClass::PushConstant => hir::Mutability::Not,
259
260            // FIXME(eddyb) further categorize this by which want interior
261            // mutability (+`Sync`!), likely almost all of them, and which
262            // can be per-lane-owning `&mut T`.
263            _ => hir::Mutability::Mut,
264        };
265        let value_spirv_type = value_layout.spirv_type(hir_param.ty_span, self);
266        // Some types automatically specify a storage class. Compute that here.
267        let element_ty = match self.lookup_type(value_spirv_type) {
268            SpirvType::Array { element, .. } | SpirvType::RuntimeArray { element } => {
269                self.lookup_type(element)
270            }
271            ty => ty,
272        };
273        let deduced_storage_class_from_ty = match element_ty {
274            SpirvType::Image { .. }
275            | SpirvType::Sampler
276            | SpirvType::SampledImage { .. }
277            | SpirvType::AccelerationStructureKhr { .. } => {
278                if is_ref {
279                    Some(StorageClass::UniformConstant)
280                } else {
281                    self.tcx.dcx().span_err(
282                        hir_param.ty_span,
283                        format!(
284                            "entry parameter type must be by-reference: `&{}`",
285                            value_layout.ty,
286                        ),
287                    );
288                    None
289                }
290            }
291            _ => None,
292        };
293        // Storage classes can be specified via attribute. Compute that here, and emit diagnostics.
294        let attr_storage_class = attrs.storage_class.map(|storage_class_attr| {
295            let storage_class = storage_class_attr.value;
296
297            if !is_ref {
298                self.tcx.dcx().span_fatal(
299                    hir_param.ty_span,
300                    format!(
301                        "invalid entry param type `{}` for storage class `{storage_class:?}` \
302                         (expected `&{}T`)",
303                        value_layout.ty,
304                        expected_mutbl_for(storage_class).prefix_str()
305                    ),
306                )
307            }
308
309            match deduced_storage_class_from_ty {
310                Some(deduced) if storage_class == deduced => self.tcx.dcx().span_warn(
311                    storage_class_attr.span,
312                    "redundant storage class attribute, storage class is deduced from type",
313                ),
314                Some(deduced) => {
315                    self.tcx
316                        .dcx()
317                        .struct_span_err(hir_param.span, "storage class mismatch")
318                        .with_span_label(
319                            storage_class_attr.span,
320                            format!("`{storage_class:?}` specified in attribute"),
321                        )
322                        .with_span_label(
323                            hir_param.ty_span,
324                            format!("`{deduced:?}` deduced from type"),
325                        )
326                        .with_help(format!(
327                            "remove storage class attribute to use `{deduced:?}` as storage class"
328                        ))
329                        .emit();
330                }
331                None => (),
332            }
333
334            storage_class
335        });
336        // If storage class was not deduced nor specified, compute the default (i.e. input/output)
337        let storage_class = deduced_storage_class_from_ty
338            .or(attr_storage_class)
339            .unwrap_or_else(|| match (is_ref, explicit_mutbl) {
340                (false, _) => StorageClass::Input,
341                (true, hir::Mutability::Mut) => StorageClass::Output,
342                (true, hir::Mutability::Not) => self.tcx.dcx().span_fatal(
343                    hir_param.ty_span,
344                    format!(
345                        "invalid entry param type `{}` (expected `{}` or `&mut {1}`)",
346                        ref_or_value_layout.ty, value_layout.ty
347                    ),
348                ),
349            });
350
351        // Validate reference mutability against the *final* storage class.
352        let read_only = effective_mutbl == hir::Mutability::Not;
353        if is_ref {
354            // FIXME(eddyb) named booleans make uses a bit more readable.
355            let ref_is_read_only = read_only;
356            let storage_class_requires_read_only =
357                expected_mutbl_for(storage_class) == hir::Mutability::Not;
358            if !ref_is_read_only && storage_class_requires_read_only {
359                let mut err = self.tcx.dcx().struct_span_err(
360                    hir_param.ty_span,
361                    format!(
362                        "entry-point requires {}...",
363                        match explicit_mutbl {
364                            hir::Mutability::Not => "interior mutability",
365                            hir::Mutability::Mut => "a mutable reference",
366                        }
367                    ),
368                );
369                {
370                    let note_message =
371                        format!("...but storage class `{storage_class:?}` is read-only");
372                    let (note_label_span, note_label) =
373                        if let Some(storage_class_attr) = attrs.storage_class {
374                            (
375                                storage_class_attr.span,
376                                format!("`{storage_class:?}` specified in attribute"),
377                            )
378                        } else {
379                            (
380                                hir_param.ty_span,
381                                format!("`{storage_class:?}` deduced from type"),
382                            )
383                        };
384                    // HACK(eddyb) have to use `MultiSpan` directly for labels,
385                    // as there's no `span_label` equivalent for `span_note`s.
386                    let mut note_multi_span: MultiSpan = vec![note_label_span].into();
387                    note_multi_span.push_span_label(note_label_span, note_label);
388                    err.span_note(note_multi_span, note_message);
389                }
390                err.emit();
391            }
392        }
393
394        // HACK(eddyb) only handle `attrs.spec_constant` after everything above
395        // would've assumed it was actually an implicitly-`Input`.
396        let mut storage_class = Ok(storage_class);
397        if let Some(spec_constant) = attrs.spec_constant {
398            if ref_or_value_layout.ty != self.tcx.types.u32 {
399                self.tcx.dcx().span_err(
400                    hir_param.ty_span,
401                    format!(
402                        "unsupported `#[spirv(spec_constant)]` type `{}` (expected `{}`)",
403                        ref_or_value_layout.ty, self.tcx.types.u32
404                    ),
405                );
406            } else if let Some(storage_class) = attrs.storage_class {
407                self.tcx.dcx().span_err(
408                    storage_class.span,
409                    "`#[spirv(spec_constant)]` cannot have a storage class",
410                );
411            } else {
412                assert_eq!(storage_class, Ok(StorageClass::Input));
413                assert!(!is_ref);
414                storage_class = Err(spec_constant.value);
415            }
416        }
417
418        EntryParamDeducedFromRustRefOrValue {
419            value_layout,
420            storage_class,
421            read_only,
422        }
423    }

425    #[allow(clippy::too_many_arguments)]
426    fn declare_shader_interface_for_param(
427        &self,
428        execution_model: ExecutionModel,
429        entry_arg_abi: &ArgAbi<'tcx, Ty<'tcx>>,
430        hir_param: &hir::Param<'tcx>,
431        op_entry_point_interface_operands: &mut Vec<Word>,
432        bx: &mut Builder<'_, 'tcx>,
433        call_args: &mut Vec<SpirvValue>,
434        decoration_locations: &mut FxHashMap<StorageClass, u32>,
435    ) {
436        let attrs = AggregatedSpirvAttributes::parse(self, self.tcx.hir_attrs(hir_param.hir_id));
437
438        let EntryParamDeducedFromRustRefOrValue {
439            value_layout,
440            storage_class,
441            read_only,
442        } = self.entry_param_deduce_from_rust_ref_or_value(entry_arg_abi.layout, hir_param, &attrs);
443        let value_spirv_type = value_layout.spirv_type(hir_param.ty_span, self);
444
445        let (var_id, spec_const_id) = match storage_class {
446            // Pre-allocate the module-scoped `OpVariable` *Result* ID.
447            Ok(_) => (
448                Ok(self.emit_global().id()),
449                Err("entry-point interface variable is not a `#[spirv(spec_constant)]`"),
450            ),
451            Err(SpecConstant { id, default }) => {
452                let mut emit = self.emit_global();
453                let spec_const_id =
454                    emit.spec_constant_bit32(value_spirv_type, default.unwrap_or(0));
455                emit.decorate(
456                    spec_const_id,
457                    Decoration::SpecId,
458                    [Operand::LiteralBit32(id)],
459                );
460                (
461                    Err("`#[spirv(spec_constant)]` is not an entry-point interface variable"),
462                    Ok(spec_const_id),
463                )
464            }
465        };
466
467        // Emit decorations deduced from the reference/value Rust type.
468        if read_only {
469            // NOTE(eddyb) it appears only `StorageBuffer`s simultaneously:
470            // - allow `NonWritable` decorations on shader interface variables
471            // - default to writable (i.e. the decoration actually has an effect)
472            if storage_class == Ok(StorageClass::StorageBuffer) {
473                self.emit_global()
474                    .decorate(var_id.unwrap(), Decoration::NonWritable, []);
475            }
476        }
477
478        // Certain storage classes require an `OpTypeStruct` decorated with `Block`,
479        // which we represent with `SpirvType::InterfaceBlock` (see its doc comment).
480        // This "interface block" construct is also required for "runtime arrays".
481        let is_unsized = self.lookup_type(value_spirv_type).sizeof(self).is_none();
482        let is_pair = matches!(entry_arg_abi.mode, PassMode::Pair(..));
483        let is_unsized_with_len = is_pair && is_unsized;
484        // HACK(eddyb) sanity check because we get the same information in two
485        // very different ways, and going out of sync could cause subtle issues.
486        assert_eq!(
487            is_unsized_with_len,
488            value_layout.is_unsized(),
489            "`{}` param mismatch in call ABI (is_pair={is_pair}) + \
490             SPIR-V type (is_unsized={is_unsized}) \
491             vs layout:\n{value_layout:#?}",
492            entry_arg_abi.layout.ty
493        );
494        if is_pair && !is_unsized {
495            // If PassMode is Pair, then we need to fill in the second part of the pair with a
496            // value. We currently only do that with unsized types, so if a type is a pair for some
497            // other reason (e.g. a tuple), we bail.
498            self.tcx
499                .dcx()
500                .span_fatal(hir_param.ty_span, "pair type not supported yet")
501        }
502        // FIXME(eddyb) should this talk about "typed buffers" instead of "interface blocks"?
503        // FIXME(eddyb) should we talk about "descriptor indexing" or
504        // actually use more reasonable terms like "resource arrays"?
505        let needs_interface_block_and_supports_descriptor_indexing = matches!(
506            storage_class,
507            Ok(StorageClass::Uniform | StorageClass::StorageBuffer)
508        );
509        let needs_interface_block = needs_interface_block_and_supports_descriptor_indexing
510            || storage_class == Ok(StorageClass::PushConstant);
511        // NOTE(eddyb) `#[spirv(typed_buffer)]` adds `SpirvType::InterfaceBlock`s
512        // which must bypass the automated ones (i.e. the user is taking control).
513        let has_explicit_interface_block = needs_interface_block_and_supports_descriptor_indexing
514            && {
515                // Peel off arrays first (used for "descriptor indexing").
516                let outermost_or_array_element = match self.lookup_type(value_spirv_type) {
517                    SpirvType::Array { element, .. } | SpirvType::RuntimeArray { element } => {
518                        element
519                    }
520                    _ => value_spirv_type,
521                };
522                matches!(
523                    self.lookup_type(outermost_or_array_element),
524                    SpirvType::InterfaceBlock { .. }
525                )
526            };
527        let var_ptr_spirv_type;
528        let (value_ptr, value_len) = if needs_interface_block && !has_explicit_interface_block {
529            let var_spirv_type = SpirvType::InterfaceBlock {
530                inner_type: value_spirv_type,
531            }
532            .def(hir_param.span, self);
533            var_ptr_spirv_type = self.type_ptr_to(var_spirv_type);
534
535            let zero_u32 = self.constant_u32(hir_param.span, 0).def_cx(self);
536            let value_ptr_spirv_type = self.type_ptr_to(value_spirv_type);
537            let value_ptr = bx
538                .emit()
539                .in_bounds_access_chain(
540                    value_ptr_spirv_type,
541                    None,
542                    var_id.unwrap(),
543                    [zero_u32].iter().cloned(),
544                )
545                .unwrap()
546                .with_type(value_ptr_spirv_type);
547
548            let value_len = if is_unsized_with_len {
549                match self.lookup_type(value_spirv_type) {
550                    SpirvType::RuntimeArray { .. } => {}
551                    _ => {
552                        self.tcx.dcx().span_err(
553                            hir_param.ty_span,
554                            "only plain slices are supported as unsized types",
555                        );
556                    }
557                }
558
559                // FIXME(eddyb) shouldn't this be `usize`?
560                let len_spirv_type = self.type_isize();
561                let len = bx
562                    .emit()
563                    .array_length(len_spirv_type, None, var_id.unwrap(), 0)
564                    .unwrap();
565
566                Some(len.with_type(len_spirv_type))
567            } else {
568                if is_unsized {
569                    // It's OK to use a RuntimeArray<u32> and not have a length parameter, but
570                    // it's just nicer ergonomics to use a slice.
571                    self.tcx
572                        .dcx()
573                        .span_warn(hir_param.ty_span, "use &[T] instead of &RuntimeArray<T>");
574                }
575                None
576            };
577
578            (Ok(value_ptr), value_len)
579        } else {
580            var_ptr_spirv_type = self.type_ptr_to(value_spirv_type);
581
582            // FIXME(eddyb) should we talk about "descriptor indexing" or
583            // actually use more reasonable terms like "resource arrays"?
584            let unsized_is_descriptor_indexing =
585                needs_interface_block_and_supports_descriptor_indexing
586                    || storage_class == Ok(StorageClass::UniformConstant);
587            if unsized_is_descriptor_indexing {
588                match self.lookup_type(value_spirv_type) {
589                    SpirvType::RuntimeArray { .. } => {
590                        if is_unsized_with_len {
591                            self.tcx.dcx().span_err(
592                                hir_param.ty_span,
593                                "descriptor indexing must use &RuntimeArray<T>, not &[T]",
594                            );
595                        }
596                    }
597                    _ => {
598                        if is_unsized {
599                            self.tcx.dcx().span_err(
600                                hir_param.ty_span,
601                                "only RuntimeArray is supported, not other unsized types",
602                            );
603                        }
604                    }
605                }
606            } else {
607                // FIXME(eddyb) determine, based on the type, what kind of type
608                // this is, to narrow it further to e.g. "buffer in a non-buffer
609                // storage class" or "storage class expects fixed data sizes".
610                if is_unsized {
611                    self.tcx.dcx().span_fatal(
612                        hir_param.ty_span,
613                        format!(
614                            "unsized types are not supported for {}",
615                            match storage_class {
616                                Ok(storage_class) => format!("storage class {storage_class:?}"),
617                                Err(SpecConstant { .. }) => "`#[spirv(spec_constant)]`".into(),
618                            },
619                        ),
620                    );
621                }
622            }
623
624            let value_len = if is_pair {
625                // We've already emitted an error, fill in a placeholder value
626                Some(bx.undef(self.type_isize()))
627            } else {
628                None
629            };
630
631            (
632                var_id.map(|var_id| var_id.with_type(var_ptr_spirv_type)),
633                value_len,
634            )
635        };
636
637        // Compute call argument(s) to match what the Rust entry `fn` expects,
638        // starting from the `value_ptr` pointing to a `value_spirv_type`
639        // (e.g. `Input` doesn't use indirection, so we have to load from it).
640        if let ty::Ref(..) = entry_arg_abi.layout.ty.kind() {
641            call_args.push(value_ptr.unwrap());
642            match entry_arg_abi.mode {
643                PassMode::Direct(_) => assert_eq!(value_len, None),
644                PassMode::Pair(..) => call_args.push(value_len.unwrap()),
645                _ => unreachable!(),
646            }
647        } else {
648            assert_matches!(entry_arg_abi.mode, PassMode::Direct(_));
649
650            let value = match storage_class {
651                Ok(_) => {
652                    assert_eq!(storage_class, Ok(StorageClass::Input));
653                    bx.load(
654                        entry_arg_abi.layout.spirv_type(hir_param.ty_span, bx),
655                        value_ptr.unwrap(),
656                        entry_arg_abi.layout.align.abi,
657                    )
658                }
659                Err(SpecConstant { .. }) => spec_const_id.unwrap().with_type(value_spirv_type),
660            };
661            call_args.push(value);
662            assert_eq!(value_len, None);
663        }
664
665        // FIXME(eddyb) check whether the storage class is compatible with the
666        // specific shader stage of this entry-point, and any decorations
667        // (e.g. Vulkan has specific rules for builtin storage classes).
668
669        // Emit `OpName` in the simple case of a pattern that's just a variable
670        // name (e.g. "foo" for `foo: Vec3`). While `OpName` is *not* supposed
671        // to be semantic, OpenGL and some tooling rely on it for reflection.
672        if let hir::PatKind::Binding(_, _, ident, _) = &hir_param.pat.kind {
673            self.emit_global()
674                .name(var_id.or(spec_const_id).unwrap(), ident.to_string());
675        }
676
677        // Emit `OpDecorate`s based on attributes.
678        let mut decoration_supersedes_location = false;
679        if let Some(builtin) = attrs.builtin {
680            if let Err(SpecConstant { .. }) = storage_class {
681                self.tcx.dcx().span_fatal(
682                    builtin.span,
683                    format!(
684                        "`#[spirv(spec_constant)]` cannot be `{:?}` builtin",
685                        builtin.value
686                    ),
687                );
688            }
689            self.emit_global().decorate(
690                var_id.unwrap(),
691                Decoration::BuiltIn,
692                std::iter::once(Operand::BuiltIn(builtin.value)),
693            );
694            decoration_supersedes_location = true;
695        }
696        if let Some(descriptor_set) = attrs.descriptor_set {
697            if let Err(SpecConstant { .. }) = storage_class {
698                self.tcx.dcx().span_fatal(
699                    descriptor_set.span,
700                    "`#[spirv(descriptor_set = ...)]` cannot apply to `#[spirv(spec_constant)]`",
701                );
702            }
703            self.emit_global().decorate(
704                var_id.unwrap(),
705                Decoration::DescriptorSet,
706                std::iter::once(Operand::LiteralBit32(descriptor_set.value)),
707            );
708            decoration_supersedes_location = true;
709        }
710        if let Some(binding) = attrs.binding {
711            if let Err(SpecConstant { .. }) = storage_class {
712                self.tcx.dcx().span_fatal(
713                    binding.span,
714                    "`#[spirv(binding = ...)]` cannot apply to `#[spirv(spec_constant)]`",
715                );
716            }
717            self.emit_global().decorate(
718                var_id.unwrap(),
719                Decoration::Binding,
720                std::iter::once(Operand::LiteralBit32(binding.value)),
721            );
722            decoration_supersedes_location = true;
723        }
724        if let Some(flat) = attrs.flat {
725            if let Err(SpecConstant { .. }) = storage_class {
726                self.tcx.dcx().span_fatal(
727                    flat.span,
728                    "`#[spirv(flat)]` cannot apply to `#[spirv(spec_constant)]`",
729                );
730            }
731            self.emit_global()
732                .decorate(var_id.unwrap(), Decoration::Flat, std::iter::empty());
733        }
734        if let Some(invariant) = attrs.invariant {
735            if storage_class != Ok(StorageClass::Output) {
736                self.tcx.dcx().span_fatal(
737                    invariant.span,
738                    "`#[spirv(invariant)]` is only valid on Output variables",
739                );
740            }
741            self.emit_global()
742                .decorate(var_id.unwrap(), Decoration::Invariant, std::iter::empty());
743        }
744        if let Some(per_primitive_ext) = attrs.per_primitive_ext {
745            match execution_model {
746                ExecutionModel::Fragment => {
747                    if storage_class != Ok(StorageClass::Input) {
748                        self.tcx.dcx().span_fatal(
749                            per_primitive_ext.span,
750                            "`#[spirv(per_primitive_ext)]` in fragment shaders is only valid on Input variables",
751                        );
752                    }
753                }
754                ExecutionModel::MeshNV | ExecutionModel::MeshEXT => {
755                    if storage_class != Ok(StorageClass::Output) {
756                        self.tcx.dcx().span_fatal(
757                            per_primitive_ext.span,
758                            "`#[spirv(per_primitive_ext)]` in mesh shaders is only valid on Output variables",
759                        );
760                    }
761                }
762                _ => {
763                    self.tcx.dcx().span_fatal(
764                        per_primitive_ext.span,
765                        "`#[spirv(per_primitive_ext)]` is only valid in fragment or mesh shaders",
766                    );
767                }
768            }
769
770            self.emit_global().decorate(
771                var_id.unwrap(),
772                Decoration::PerPrimitiveEXT,
773                std::iter::empty(),
774            );
775        }
776
777        let is_subpass_input = match self.lookup_type(value_spirv_type) {
778            SpirvType::Image {
779                dim: Dim::DimSubpassData,
780                ..
781            } => true,
782            SpirvType::RuntimeArray { element: elt, .. }
783            | SpirvType::Array { element: elt, .. } => {
784                matches!(
785                    self.lookup_type(elt),
786                    SpirvType::Image {
787                        dim: Dim::DimSubpassData,
788                        ..
789                    }
790                )
791            }
792            _ => false,
793        };
794        if let Some(attachment_index) = attrs.input_attachment_index {
795            if is_subpass_input && self.builder.has_capability(Capability::InputAttachment) {
796                self.emit_global().decorate(
797                    var_id.unwrap(),
798                    Decoration::InputAttachmentIndex,
799                    std::iter::once(Operand::LiteralBit32(attachment_index.value)),
800                );
801            } else if is_subpass_input {
802                self.tcx
803                    .dcx()
804                    .span_err(hir_param.ty_span, "Missing capability InputAttachment");
805            } else {
806                self.tcx.dcx().span_err(
807                    attachment_index.span,
808                    "#[spirv(input_attachment_index)] is only valid on Image types with dim = SubpassData"
809                );
810            }
811            decoration_supersedes_location = true;
812        } else if is_subpass_input {
813            self.tcx.dcx().span_err(
814                hir_param.ty_span,
815                "Image types with dim = SubpassData require #[spirv(input_attachment_index)] decoration",
816            );
817        }
818
819        if let Ok(storage_class) = storage_class {
820            self.check_for_bad_types(
821                execution_model,
822                hir_param.ty_span,
823                var_ptr_spirv_type,
824                storage_class,
825                attrs.builtin.is_some(),
826                attrs.flat,
827            );
828        }
829
830        // Assign locations from left to right, incrementing each storage class
831        // individually.
832        // TODO: Is this right for UniformConstant? Do they share locations with
833        // input/outpus?
834        let has_location = !decoration_supersedes_location
835            && matches!(
836                storage_class,
837                Ok(StorageClass::Input | StorageClass::Output | StorageClass::UniformConstant)
838            );
839        if has_location {
840            let location = decoration_locations
841                .entry(storage_class.unwrap())
842                .or_insert_with(|| 0);
843            self.emit_global().decorate(
844                var_id.unwrap(),
845                Decoration::Location,
846                std::iter::once(Operand::LiteralBit32(*location)),
847            );
848            *location += 1;
849        }
850
851        match storage_class {
852            Ok(storage_class) => {
853                let var = var_id.unwrap();
854
855                // Emit the `OpVariable` with its *Result* ID set to `var_id`.
856                self.emit_global()
857                    .variable(var_ptr_spirv_type, Some(var), storage_class, None);
858
859                // Record this `OpVariable` as needing to be added (if applicable),
860                // to the *Interface* operands of the `OpEntryPoint` instruction.
861                if self.emit_global().version().unwrap() > (1, 3) {
862                    // SPIR-V >= v1.4 includes all OpVariables in the interface.
863                    op_entry_point_interface_operands.push(var);
864                } else {
865                    // SPIR-V <= v1.3 only includes Input and Output in the interface.
866                    if storage_class == StorageClass::Input || storage_class == StorageClass::Output
867                    {
868                        op_entry_point_interface_operands.push(var);
869                    }
870                }
871            }
872            Err(not_var) => {
873                // Emitted earlier.
874                let SpecConstant { .. } = not_var;
875            }
876        }
877    }
878
879    // Booleans are only allowed in some storage classes. Error if they're in others.
880    // Integers and `f64`s must be decorated with `#[spirv(flat)]`.
881    fn check_for_bad_types(
882        &self,
883        execution_model: ExecutionModel,
884        span: Span,
885        ty: Word,
886        storage_class: StorageClass,
887        is_builtin: bool,
888        flat_attr: Option<Spanned<()>>,
889    ) {
890        // private and function are allowed here, but they can't happen.
891        if matches!(
892            storage_class,
893            StorageClass::Workgroup | StorageClass::CrossWorkgroup
894        ) {
895            return;
896        }
897
898        let mut has_bool = false;
899        let mut type_must_be_flat = false;
900        recurse(self, ty, &mut has_bool, &mut type_must_be_flat);
901
902        // SPIR-V technically allows all input/output variables to be booleans, not just builtins,
903        // but has a note:
904        // > Khronos Issue #363: OpTypeBool can be used in the Input and Output storage classes,
905        //   but the client APIs still only allow built-in Boolean variables (e.g. FrontFacing),
906        //   not user variables.
907        // spirv-val disallows non-builtin inputs/outputs, so we do too, I guess.
908        if has_bool
909            && !(is_builtin && matches!(storage_class, StorageClass::Input | StorageClass::Output))
910        {
911            self.tcx
912                .dcx()
913                .span_err(span, "entry-point parameter cannot contain `bool`s");
914        }
915
916        // Enforce Vulkan validation rules around `Flat` as accurately as possible,
917        // i.e. "interpolation control" can only be used "within" the rasterization
918        // pipeline (roughly: `vertex (outputs) -> ... -> (inputs for) fragment`),
919        // but not at the "outer" interface (vertex inputs/fragment outputs).
920        // Also, fragment inputs *require* it for some ("uninterpolatable") types.
921        // FIXME(eddyb) maybe this kind of `enum` could be placed elsewhere?
922        enum Force {
923            Disallow,
924            Require,
925        }
926        #[allow(clippy::match_same_arms)]
927        let flat_forced = match (execution_model, storage_class) {
928            // VUID-StandaloneSpirv-Flat-06202
929            // > The `Flat`, `NoPerspective`, `Sample`, and `Centroid` decorations **must**
930            // > not be used on variables with the `Input` storage class in a vertex shader
931            (ExecutionModel::Vertex, StorageClass::Input) => Some(Force::Disallow),
932
933            // VUID-StandaloneSpirv-Flat-04744
934            // > Any variable with integer or double-precision floating-point type and
935            // > with `Input` storage class in a fragment shader, **must** be decorated `Flat`
936            (ExecutionModel::Fragment, StorageClass::Input) if type_must_be_flat => {
937                // FIXME(eddyb) shouldn't this be automatic then? (maybe with a warning?)
938                Some(Force::Require)
939            }
940
941            // VUID-StandaloneSpirv-Flat-06201
942            // > The `Flat`, `NoPerspective`, `Sample`, and `Centroid` decorations **must**
943            // > not be used on variables with the `Output` storage class in a fragment shader
944            (ExecutionModel::Fragment, StorageClass::Output) => Some(Force::Disallow),
945
946            // VUID-StandaloneSpirv-Flat-04670
947            // > The `Flat`, `NoPerspective`, `Sample`, and `Centroid` decorations **must**
948            // > only be used on variables with the `Output` or `Input` storage class
949            (_, StorageClass::Input | StorageClass::Output) => None,
950            _ => Some(Force::Disallow),
951        };
952
953        let flat_mismatch = match (flat_forced, flat_attr) {
954            (Some(Force::Disallow), Some(flat_attr)) => Some((flat_attr.span, "cannot")),
955            // FIXME(eddyb) it would be useful to show the type that required it.
956            (Some(Force::Require), None) => Some((span, "must")),
957            _ => None,
958        };
959        if let Some((span, must_or_cannot)) = flat_mismatch {
960            self.tcx.dcx().span_err(
961                span,
962                format!(
963                    "`{execution_model:?}` entry-point `{storage_class:?}` parameter \
964                     {must_or_cannot} be decorated with `#[spirv(flat)]`"
965                ),
966            );
967        }
968
969        fn recurse(cx: &CodegenCx<'_>, ty: Word, has_bool: &mut bool, must_be_flat: &mut bool) {
970            match cx.lookup_type(ty) {
971                SpirvType::Bool => *has_bool = true,
972                SpirvType::Integer(_, _) | SpirvType::Float(64) => *must_be_flat = true,
973                SpirvType::Adt { field_types, .. } => {
974                    for &f in field_types {
975                        recurse(cx, f, has_bool, must_be_flat);
976                    }
977                }
978                SpirvType::Vector { element, .. }
979                | SpirvType::Matrix { element, .. }
980                | SpirvType::Array { element, .. }
981                | SpirvType::RuntimeArray { element }
982                | SpirvType::Pointer { pointee: element }
983                | SpirvType::InterfaceBlock {
984                    inner_type: element,
985                } => recurse(cx, element, has_bool, must_be_flat),
986                SpirvType::Function {
987                    return_type,
988                    arguments,
989                } => {
990                    recurse(cx, return_type, has_bool, must_be_flat);
991                    for &a in arguments {
992                        recurse(cx, a, has_bool, must_be_flat);
993                    }
994                }
995                _ => (),
996            }
997        }
998    }
