1//! Simplify `OpCompositeExtract` pointing to `OpCompositeConstruct`s / `OpCompositeInsert`s.
2//! Such constructions arise after inlining, when using multi-argument closures
3//! (and other `Fn*` trait implementations). These composites can frequently be invalid,
4//! containing pointers, `OpFunctionArgument`s, etc. After simplification, components
5//! will become valid targets for `OpLoad`/`OpStore`.
6use super::apply_rewrite_rules;
7use rspirv::dr::{Function, Instruction};
8use rspirv::spirv::Op;
9use rustc_data_structures::fx::FxHashMap;
10
11pub fn destructure_composites(function: &mut Function) {
12    let mut rewrite_rules = FxHashMap::default();
13    let reference: FxHashMap<_, _> = function
14        .all_inst_iter()
15        .filter_map(|inst| match inst.class.opcode {
16            Op::CompositeConstruct => Some((inst.result_id.unwrap(), inst.clone())),
17            Op::CompositeInsert if inst.operands.len() == 3 => {
18                Some((inst.result_id.unwrap(), inst.clone()))
19            }
20            _ => None,
21        })
22        .collect();
23
24    for inst in function.all_inst_iter_mut() {
25        // multi-index extraction for nested tuples/structs
26        if inst.class.opcode == Op::CompositeExtract && inst.operands.len() >= 2 {
27            let mut current_id = inst.operands[0].unwrap_id_ref();
28            let mut final_origin = None;
29
30            // step through each index sequentially to resolve deeply nested extracts
31            for index_operand in &inst.operands[1..] {
32                let index = index_operand.unwrap_literal_bit32();
33
34                let mut search_id = current_id;
35                let mut resolved_id = None;
36
37                while let Some(inst) = reference.get(&search_id) {
38                    match inst.class.opcode {
39                        Op::CompositeInsert => {
40                            let insert_index = inst.operands[2].unwrap_literal_bit32();
41                            if insert_index == index {
42                                resolved_id = Some(inst.operands[0].unwrap_id_ref());
43                                break;
44                            }
45                            // If not our index, continue down the insert chain
46                            search_id = inst.operands[1].unwrap_id_ref();
47                        }
48                        Op::CompositeConstruct => {
49                            resolved_id =
50                                inst.operands.get(index as usize).map(|o| o.unwrap_id_ref());
51                            break;
52                        }
53                        _ => unreachable!(),
54                    }
55                }
56
57                if let Some(res) = resolved_id {
58                    current_id = res;
59                    final_origin = Some(res);
60                } else {
61                    final_origin = None;
62                    break;
63                }
64            }
65
66            if let Some(origin_id) = final_origin {
67                rewrite_rules.insert(
68                    inst.result_id.unwrap(),
69                    rewrite_rules.get(&origin_id).map_or(origin_id, |id| *id),
70                );
71                *inst = Instruction::new(Op::Nop, None, None, vec![]);
72            }
73        }
74    }
75
76    // Transitive closure computation
77    let mut closed_rewrite_rules = rewrite_rules.clone();
78    for value in closed_rewrite_rules.values_mut() {
79        while let Some(next) = rewrite_rules.get(value) {
80            *value = *next;
81        }
82    }
83
84    // Remove instructions replaced by NOPs, as well as unused composite values.
85    for block in function.blocks.iter_mut() {
86        block
87            .instructions
88            .retain(|inst| inst.class.opcode != Op::Nop);
89    }
90    apply_rewrite_rules(&closed_rewrite_rules, &mut function.blocks);
91}
