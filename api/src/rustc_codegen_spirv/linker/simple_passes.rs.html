10/// Error marker type, indicating an integer/float type SPIR-V lacks support for.

13/// Returns the capability required for an integer type of the given width, if any.
14fn capability_for_int_width(
15    width: u32,
16) -> Result<Option<rspirv::spirv::Capability>, UnsupportedType> {
17    Ok(match width {
18        8 => Some(rspirv::spirv::Capability::Int8),
19        16 => Some(rspirv::spirv::Capability::Int16),
20        32 => None,
21        64 => Some(rspirv::spirv::Capability::Int64),
22        _ => return Err(UnsupportedType),
23    })
24}

38pub fn shift_ids(module: &mut Module, add: u32) {
39    module.all_inst_iter_mut().for_each(|inst| {
40        if let Some(result_id) = &mut inst.result_id {
41            *result_id += add;
42        }
43
44        if let Some(result_type) = &mut inst.result_type {
45            *result_type += add;
46        }
47
48        inst.operands.iter_mut().for_each(|op| {
49            if let Some(w) = op.id_ref_any_mut() {
50                *w += add;
51            }
52        });
53    });
54}

56/// spir-v requires basic blocks to be ordered so that if A dominates B, A appears before B (except
57/// in the case of backedges). Reverse post-order is a good ordering that satisfies this condition
58/// (with an "already visited set" that blocks going deeper, which solves both the fact that it's a
59/// DAG, not a tree, as well as backedges).
60pub fn block_ordering_pass(func: &mut Function) {
61    if func.blocks.len() < 2 {
62        return;
63    }
64    fn visit_postorder(
65        func: &Function,
66        visited: &mut FxHashSet<Word>,
67        postorder: &mut Vec<Word>,
68        current: Word,
69    ) {
70        if !visited.insert(current) {
71            return;
72        }
73        let current_block = func
74            .blocks
75            .iter()
76            .find(|b| b.label_id().unwrap() == current)
77            .unwrap();
78        let mut edges = outgoing_edges(current_block).collect::<Vec<_>>();
79        // HACK(eddyb) treat `OpSelectionMerge` as an edge, in case it points
80        // to an otherwise-unreachable block.
81        if let Some(before_last_idx) = current_block.instructions.len().checked_sub(2)
82            && let Some(before_last) = current_block.instructions.get(before_last_idx)
83            && before_last.class.opcode == Op::SelectionMerge
84        {
85            edges.push(before_last.operands[0].unwrap_id_ref());
86        }
87        // Reverse the order, so reverse-postorder keeps things tidy
88        for &outgoing in edges.iter().rev() {
89            visit_postorder(func, visited, postorder, outgoing);
90        }
91        postorder.push(current);
92    }
93
94    let mut visited = FxHashSet::default();
95    let mut postorder = Vec::new();
96
97    let entry_label = func.blocks[0].label_id().unwrap();
98    visit_postorder(func, &mut visited, &mut postorder, entry_label);
99
100    let mut old_blocks = take(&mut func.blocks);
101    // Order blocks according to reverse postorder
102    for &block in postorder.iter().rev() {
103        let index = old_blocks
104            .iter()
105            .position(|b| b.label_id().unwrap() == block)
106            .unwrap();
107        func.blocks.push(old_blocks.remove(index));
108    }
109    // Note: if old_blocks isn't empty here, that means there were unreachable blocks that were deleted.
110    assert_eq!(func.blocks[0].label_id().unwrap(), entry_label);
111}

113pub fn outgoing_edges(block: &Block) -> impl Iterator<Item = Word> + '_ {
114    let terminator = block.instructions.last().unwrap();
115    // https://www.khronos.org/registry/spir-v/specs/unified1/SPIRV.html#Termination
116    let operand_indices = match terminator.class.opcode {
117        Op::Branch => (0..1).step_by(1),
118        Op::BranchConditional => (1..3).step_by(1),
119        Op::Switch => (1..terminator.operands.len()).step_by(2),
120        Op::Return
121        | Op::ReturnValue
122        | Op::Kill
123        | Op::Unreachable
124        | Op::IgnoreIntersectionKHR
125        | Op::TerminateRayKHR
126        | Op::EmitMeshTasksEXT => (0..0).step_by(1),
127        _ => panic!("Invalid block terminator: {terminator:?}"),
128    };
129    operand_indices.map(move |i| terminator.operands[i].unwrap_id_ref())
130}

132pub fn compact_ids(module: &mut Module) -> u32 {
133    let mut remap = FxHashMap::default();
134
135    let mut insert = |current_id: u32| -> u32 {
136        let len = remap.len();
137        *remap.entry(current_id).or_insert_with(|| len as u32 + 1)
138    };
139
140    module.all_inst_iter_mut().for_each(|inst| {
141        if let Some(result_id) = &mut inst.result_id {
142            *result_id = insert(*result_id);
143        }
144
145        if let Some(result_type) = &mut inst.result_type {
146            *result_type = insert(*result_type);
147        }
148
149        inst.operands.iter_mut().for_each(|op| {
150            if let Some(w) = op.id_ref_any_mut() {
151                *w = insert(*w);
152            }
153        });
154    });
155
156    remap.len() as u32 + 1
157}

159pub fn sort_globals(module: &mut Module) {
160    // Function declarations come before definitions. TODO: Figure out if it's even possible to
161    // have a function declaration without a body in a fully linked module?
162    module.functions.sort_by_key(|f| !f.blocks.is_empty());
163}

165pub fn name_variables_pass(module: &mut Module) {
166    let variables = module
167        .types_global_values
168        .iter()
169        .filter(|inst| inst.class.opcode == Op::Variable)
170        .map(|inst| inst.result_id.unwrap())
171        .collect::<FxHashSet<Word>>();
172    module
173        .debug_names
174        .retain(|inst| variables.contains(&inst.operands[0].unwrap_id_ref()));
175    // FIXME(eddyb) why does this remove `OpLine` instructions?
176    module
177        .types_global_values
178        .retain(|inst| inst.class.opcode != Op::Line);
179    for func in &mut module.functions {
180        for block in &mut func.blocks {
181            block
182                .instructions
183                .retain(|inst| inst.class.opcode != Op::Line);
184        }
185    }
186}

188// Some instructions are only valid in fragment shaders. Check them.
189pub fn check_fragment_insts(sess: &Session, module: &Module) -> super::Result<()> {
190    let mut visited = vec![false; module.functions.len()];
191    let mut stack = Vec::new();
192    let mut names = None;
193    let func_id_to_idx: FxHashMap<Word, usize> = module
194        .functions
195        .iter()
196        .enumerate()
197        .map(|(index, func)| (func.def_id().unwrap(), index))
198        .collect();
199    let entries = module
200        .entry_points
201        .iter()
202        .filter(|i| i.operands[0].unwrap_execution_model() != ExecutionModel::Fragment)
203        .map(|i| func_id_to_idx[&i.operands[1].unwrap_id_ref()]);
204    let mut any_err = None;
205    for entry in entries {
206        let entry_had_err = visit(
207            sess,
208            module,
209            &mut visited,
210            &mut stack,
211            &mut names,
212            entry,
213            &func_id_to_idx,
214        )
215        .err();
216        any_err = any_err.or(entry_had_err);
217    }
218    return match any_err {
219        Some(err) => Err(err),
220        None => Ok(()),
221    };
222
223    fn visit<'m>(
224        sess: &Session,
225        module: &'m Module,
226        visited: &mut Vec<bool>,
227        stack: &mut Vec<Word>,
228        names: &mut Option<FxHashMap<Word, &'m str>>,
229        index: usize,
230        func_id_to_idx: &FxHashMap<Word, usize>,
231    ) -> super::Result<()> {
232        if visited[index] {
233            return Ok(());
234        }
235        visited[index] = true;
236        stack.push(module.functions[index].def_id().unwrap());
237        let mut any_err = None;
238        for inst in module.functions[index].all_inst_iter() {
239            if inst.class.opcode == Op::FunctionCall {
240                let callee = func_id_to_idx[&inst.operands[0].unwrap_id_ref()];
241                let callee_had_err =
242                    visit(sess, module, visited, stack, names, callee, func_id_to_idx).err();
243                any_err = any_err.or(callee_had_err);
244            }
245            if matches!(
246                inst.class.opcode,
247                Op::ImageSampleImplicitLod
248                    | Op::ImageSampleDrefImplicitLod
249                    | Op::ImageSampleProjImplicitLod
250                    | Op::ImageSampleProjDrefImplicitLod
251                    | Op::ImageQueryLod
252                    | Op::ImageSparseSampleImplicitLod
253                    | Op::ImageSparseSampleDrefImplicitLod
254                    | Op::DPdx
255                    | Op::DPdy
256                    | Op::Fwidth
257                    | Op::DPdxFine
258                    | Op::DPdyFine
259                    | Op::FwidthFine
260                    | Op::DPdxCoarse
261                    | Op::DPdyCoarse
262                    | Op::FwidthCoarse
263                    | Op::Kill
264            ) {
265                // These instructions are (usually) in system functions - if we get an error, allow
266                // the system function to be visited again from elsewhere to emit another error
267                // from another callsite.
268                visited[index] = false;
269
270                let names = names.get_or_insert_with(|| get_names(module));
271                let stack = stack.iter().rev().map(|&s| get_name(names, s).into_owned());
272                let note = once("Stack:".to_string())
273                    .chain(stack)
274                    .collect::<Vec<_>>()
275                    .join("\n");
276                any_err = Some(
277                    sess.dcx()
278                        .struct_err(format!(
279                            "{} cannot be used outside a fragment shader",
280                            inst.class.opname
281                        ))
282                        .with_note(note)
283                        .emit(),
284                );
285            }
286        }
287        stack.pop();
288
289        match any_err {
290            Some(err) => Err(err),
291            None => Ok(()),
292        }
293    }
294}

296/// Remove type-related capabilities that are not required by any types in the module.
297///
298/// This function specifically targets Int8, Int16, Int64, Float16, and Float64 capabilities,
299/// removing them if no types in the module require them. All other capabilities are preserved.
300/// This is part of the fix for issue #300 where constant casts were creating unnecessary types.
301//
302// FIXME(eddyb) move this to a SPIR-T pass (potentially even using sets of used
303// exts/caps that validation itself can collect while traversing the module).
304pub fn remove_unused_type_capabilities(module: &mut Module) {
305    use rspirv::spirv::Capability;
306
307    // Collect type-related capabilities that are actually needed
308    let mut needed_type_capabilities = FxHashSet::default();
309
310    // Scan all types to determine which type-related capabilities are needed
311    for inst in &module.types_global_values {
312        match inst.class.opcode {
313            Op::TypeInt => {
314                let width = inst.operands[0].unwrap_literal_bit32();
315                if let Ok(Some(cap)) = capability_for_int_width(width) {
316                    needed_type_capabilities.insert(cap);
317                }
318            }
319            Op::TypeFloat => {
320                let width = inst.operands[0].unwrap_literal_bit32();
321                if let Ok(Some(cap)) = capability_for_float_width(width) {
322                    needed_type_capabilities.insert(cap);
323                }
324            }
325            _ => {}
326        }
327    }
328
329    // Remove only type-related capabilities that aren't needed
330    module.capabilities.retain(|inst| {
331        let cap = inst.operands[0].unwrap_capability();
332        match cap {
333            // Only remove these type-related capabilities if they're not used
334            Capability::Int8
335            | Capability::Int16
336            | Capability::Int64
337            | Capability::Float16
338            | Capability::Float64 => needed_type_capabilities.contains(&cap),
339            // Keep all other capabilities
340            _ => true,
341        }
342    });
343}

345/// Remove all [`Decoration::NonUniform`] if this module does *not* have [`Capability::ShaderNonUniform`].
346/// This allows image asm to always declare `NonUniform` and not worry about conditional compilation.
347pub fn remove_non_uniform_decorations(_sess: &Session, module: &mut Module) -> super::Result<()> {
348    let has_shader_non_uniform_capability = module.capabilities.iter().any(|inst| {
349        inst.class.opcode == Op::Capability
350            && inst.operands[0].unwrap_capability() == Capability::ShaderNonUniform
351    });
352    if !has_shader_non_uniform_capability {
353        module.annotations.retain(|inst| {
354            !(inst.class.opcode == Op::Decorate
355                && inst.operands[1].unwrap_decoration() == Decoration::NonUniform)
356        });
357    }
358    Ok(())
