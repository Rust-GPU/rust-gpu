1//! Dead code elimination
2//!
3//! This pass removes any instruction that doesn't affect the module. It does so by considering all
4//! `OpEntryPoint` instructions to be "rooted", and then everything a rooted instruction touches is
5//! also rooted (done transitively). Then, any instruction not rooted is removed. It gets a little
6//! weird with things like `OpDecorate`, where the reference is reversed - an `OpDecorate` that
7//! *references* a rooted thing is also rooted, not the other way around - but that's the basic
8//! concept.
9
10use rspirv::dr::{Block, Function, Instruction, Module, Operand};
11use rspirv::spirv::{Decoration, LinkageType, Op, StorageClass, Word};
12use rustc_data_structures::fx::{FxIndexMap, FxIndexSet};
13use std::hash::Hash;
14
15pub fn dce(module: &mut Module) {
16    let mut rooted = collect_roots(module);
17    while spread_roots(module, &mut rooted) {}
18    kill_unrooted(module, &rooted);
19}
20
21pub fn collect_roots(module: &Module) -> FxIndexSet<Word> {
22    let mut rooted = FxIndexSet::default();
23
24    for inst in &module.entry_points {
25        root(inst, &mut rooted);
26    }
27
28    // NOTE(eddyb) such "link exports" roots are only relevant when `Options`'s
29    // `keep_link_export`s field is used to request that `Export`s are left in
30    // (primarily for unit testing - see also its doc comment).
31    for inst in &module.annotations {
32        if inst.class.opcode == Op::Decorate
33            && inst.operands[1].unwrap_decoration() == Decoration::LinkageAttributes
34            && inst.operands[3].unwrap_linkage_type() == LinkageType::Export
35        {
36            root(inst, &mut rooted);
37        }
38    }
39
40    rooted
41}

43// Exactly the same as Function::all_inst_iter, except return type is `impl DoubleEndedIterator`
44// instead of `impl Iterator`
45fn all_inst_iter(func: &Function) -> impl DoubleEndedIterator<Item = &Instruction> {
46    func.def
47        .iter()
48        .chain(func.parameters.iter())
49        .chain(
50            func.blocks
51                .iter()
52                .flat_map(|b| b.label.iter().chain(b.instructions.iter())),
53        )
54        .chain(func.end.iter())
55}
56
57fn spread_roots(module: &Module, rooted: &mut FxIndexSet<Word>) -> bool {
58    let mut any = false;
59    for inst in module.global_inst_iter() {
60        if let Some(id) = inst.result_id
61            && rooted.contains(&id)
62        {
63            any |= root(inst, rooted);
64        }
65    }
66    for func in &module.functions {
67        if rooted.contains(&func.def_id().unwrap()) {
68            // NB (Mobius 2021) - since later insts are much more likely to reference
69            // earlier insts, by reversing the iteration order, we're more likely to root the
70            // entire relevant function at once.
71            // See https://github.com/EmbarkStudios/rust-gpu/pull/691#discussion_r681477091
72            for inst in all_inst_iter(func).rev() {
73                if !instruction_is_pure(inst) {
74                    any |= root(inst, rooted);
75                } else if let Some(id) = inst.result_id
76                    && rooted.contains(&id)
77                {
78                    any |= root(inst, rooted);
79                }
80            }
81        }
82    }
83    any
84}
85
86fn root(inst: &Instruction, rooted: &mut FxIndexSet<Word>) -> bool {
87    let mut any = false;
88    if let Some(id) = inst.result_type {
89        any |= rooted.insert(id);
90    }
91    for op in &inst.operands {
92        if let Some(id) = op.id_ref_any() {
93            any |= rooted.insert(id);
94        }
95    }
96    any
97}

99fn is_rooted(inst: &Instruction, rooted: &FxIndexSet<Word>) -> bool {
100    if let Some(result_id) = inst.result_id {
101        rooted.contains(&result_id)
102    } else {
103        // For things like OpDecorate which apply attributes to rooted things, but are not
104        // referenced by roots
105        inst.operands
106            .iter()
107            .any(|op| op.id_ref_any().is_some_and(|w| rooted.contains(&w)))
108    }
109}

111fn kill_unrooted(module: &mut Module, rooted: &FxIndexSet<Word>) {
112    module
113        .ext_inst_imports
114        .retain(|inst| is_rooted(inst, rooted));
115    module
116        .execution_modes
117        .retain(|inst| is_rooted(inst, rooted));
118    module
119        .debug_string_source
120        .retain(|inst| is_rooted(inst, rooted));
121    module.debug_names.retain(|inst| is_rooted(inst, rooted));
122    module
123        .debug_module_processed
124        .retain(|inst| is_rooted(inst, rooted));
125    module.annotations.retain(|inst| is_rooted(inst, rooted));
126    module
127        .types_global_values
128        .retain(|inst| is_rooted(inst, rooted));
129    module
130        .functions
131        .retain(|f| is_rooted(f.def.as_ref().unwrap(), rooted));
132    for fun in &mut module.functions {
133        for block in &mut fun.blocks {
134            block
135                .instructions
136                .retain(|inst| !instruction_is_pure(inst) || is_rooted(inst, rooted));
137        }
138    }
139}

141pub fn dce_phi(blocks: &mut FxIndexMap<impl Eq + Hash, &mut Block>) {
142    let mut used = FxIndexSet::default();
143    loop {
144        let mut changed = false;
145        for inst in blocks.values().flat_map(|block| &block.instructions) {
146            if inst.class.opcode != Op::Phi || used.contains(&inst.result_id.unwrap()) {
147                for op in &inst.operands {
148                    if let Some(id) = op.id_ref_any() {
149                        changed |= used.insert(id);
150                    }
151                }
152            }
153        }
154        if !changed {
155            break;
156        }
157    }
158    for block in blocks.values_mut() {
159        block
160            .instructions
161            .retain(|inst| inst.class.opcode != Op::Phi || used.contains(&inst.result_id.unwrap()));
162    }
163}
