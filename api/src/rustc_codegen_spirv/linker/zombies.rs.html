15#[derive(Copy, Clone)]
16struct Zombie<'a> {
17    id: Word,
18    kind: &'a ZombieKind<'a>,
19}
20
21enum ZombieKind<'a> {
22    /// Definition annotated with `ZombieDecoration`.
23    Leaf,
24
25    /// Transitively zombie'd by using other zombies, from an instruction.
26    Uses(Vec<ZombieUse<'a>>),
27}
28
29struct ZombieUse<'a> {
30    used_zombie_id: Word,
31
32    /// Active debug "source location" instruction at the time of the use, if any
33    /// (both `OpLine` and `CustomInst::SetDebugSrcLoc` are supported).
34    use_debug_src_loc_inst: Option<&'a Instruction>,
35
36    origin: UseOrigin,
37}
38
39enum UseOrigin {
40    GlobalOperandOrResultType,
41    IntraFuncOperandOrResultType { parent_func_id: Word },
42    CallCalleeOperand { caller_func_id: Word },
43}
44
45struct Zombies<'a> {
46    /// ID of `OpExtInstImport` for our custom "extended instruction set",
47    /// if present (see `crate::custom_insts` for more details).
48    custom_ext_inst_set_import: Option<Word>,
49
50    id_to_zombie_kind: FxIndexMap<Word, ZombieKind<'a>>,
51}
52
53impl<'a> Zombies<'a> {
54    // FIXME(eddyb) rename all the other methods to say `_inst` explicitly.
55    fn get_zombie_by_id(&self, id: Word) -> Option<Zombie<'_>> {
56        self.id_to_zombie_kind
57            .get(&id)
58            .map(|kind| Zombie { id, kind })
59    }
60
61    fn zombies_used_from_inst<'b>(
62        &'b self,
63        inst: &'b Instruction,
64    ) -> impl Iterator<Item = Zombie<'b>> + 'b {
65        inst.result_type
66            .into_iter()
67            .chain(inst.operands.iter().filter_map(|op| op.id_ref_any()))
68            .filter_map(move |id| self.get_zombie_by_id(id))
69    }
70
71    fn spread(&mut self, module: &'a Module) -> bool {
72        let mut any = false;
73        // globals are easy
74        {
75            let mut debug_src_loc_inst = None;
76            for inst in module.global_inst_iter() {
77                match inst.class.opcode {
78                    Op::Line => debug_src_loc_inst = Some(inst),
79                    Op::NoLine => debug_src_loc_inst = None,
80                    Op::ExtInst
81                        if Some(inst.operands[0].unwrap_id_ref())
82                            == self.custom_ext_inst_set_import =>
83                    {
84                        match CustomOp::decode_from_ext_inst(inst) {
85                            CustomOp::SetDebugSrcLoc => debug_src_loc_inst = Some(inst),
86                            CustomOp::ClearDebugSrcLoc => debug_src_loc_inst = None,
87                            _ => {}
88                        }
89                    }
90                    _ => {}
91                }
92
93                if let Some(result_id) = inst.result_id {
94                    if self.id_to_zombie_kind.contains_key(&result_id) {
95                        continue;
96                    }
97                    let zombie_uses: Vec<_> = self
98                        .zombies_used_from_inst(inst)
99                        .map(|zombie| ZombieUse {
100                            used_zombie_id: zombie.id,
101                            use_debug_src_loc_inst: debug_src_loc_inst,
102                            origin: UseOrigin::GlobalOperandOrResultType,
103                        })
104                        .collect();
105                    if !zombie_uses.is_empty() {
106                        self.id_to_zombie_kind
107                            .insert(result_id, ZombieKind::Uses(zombie_uses));
108                        any = true;
109                    }
110                }
111            }
112        }
113        // No need to zombie defs within a function: If any def within a function is zombied, then the
114        // whole function is zombied. But, we don't have to mark the defs within a function as zombie,
115        // because the defs can't escape the function.
116        // HACK(eddyb) one exception to this is function-local variables, or the
117        // `OpBitcast`s of pointer casts, either of which which may actually be
118        // unused and as such cannot be allowed to always zombie the function.
119        for func in &module.functions {
120            let func_id = func.def_id().unwrap();
121            if self.id_to_zombie_kind.contains_key(&func_id) {
122                // Func is already zombie, no need to scan it again.
123                continue;
124            }
125
126            let mut all_zombie_uses_in_func = vec![];
127            let mut debug_src_loc_inst = None;
128            for inst in func.all_inst_iter() {
129                match inst.class.opcode {
130                    Op::Line => debug_src_loc_inst = Some(inst),
131                    // NOTE(eddyb) each block starts out with cleared debuginfo.
132                    Op::Label | Op::NoLine => debug_src_loc_inst = None,
133                    Op::ExtInst
134                        if Some(inst.operands[0].unwrap_id_ref())
135                            == self.custom_ext_inst_set_import =>
136                    {
137                        match CustomOp::decode_from_ext_inst(inst) {
138                            CustomOp::SetDebugSrcLoc => debug_src_loc_inst = Some(inst),
139                            CustomOp::ClearDebugSrcLoc => debug_src_loc_inst = None,
140                            _ => {}
141                        }
142                    }
143                    _ => {}
144                }
145
146                if [Op::Variable, Op::Bitcast].contains(&inst.class.opcode) {
147                    let result_id = inst.result_id.unwrap();
148                    if self.id_to_zombie_kind.contains_key(&result_id) {
149                        continue;
150                    }
151                    let zombie_uses: Vec<_> = self
152                        .zombies_used_from_inst(inst)
153                        .map(|zombie| ZombieUse {
154                            used_zombie_id: zombie.id,
155                            use_debug_src_loc_inst: debug_src_loc_inst,
156                            origin: UseOrigin::IntraFuncOperandOrResultType {
157                                parent_func_id: func_id,
158                            },
159                        })
160                        .collect();
161                    if !zombie_uses.is_empty() {
162                        self.id_to_zombie_kind
163                            .insert(result_id, ZombieKind::Uses(zombie_uses));
164                        any = true;
165                    }
166                    continue;
167                }
168
169                all_zombie_uses_in_func.extend(
170                    inst.result_id
171                        .and_then(|result_id| self.get_zombie_by_id(result_id))
172                        .into_iter()
173                        .chain(self.zombies_used_from_inst(inst))
174                        .map(|zombie| {
175                            let origin = if inst.class.opcode == Op::FunctionCall
176                                && inst.operands[0] == Operand::IdRef(zombie.id)
177                            {
178                                UseOrigin::CallCalleeOperand {
179                                    caller_func_id: func_id,
180                                }
181                            } else {
182                                UseOrigin::IntraFuncOperandOrResultType {
183                                    parent_func_id: func_id,
184                                }
185                            };
186                            ZombieUse {
187                                used_zombie_id: zombie.id,
188                                use_debug_src_loc_inst: debug_src_loc_inst,
189                                origin,
190                            }
191                        }),
192                );
193            }
194            if !all_zombie_uses_in_func.is_empty() {
195                self.id_to_zombie_kind
196                    .insert(func_id, ZombieKind::Uses(all_zombie_uses_in_func));
197                any = true;
198            }
199        }
200        any
201    }
202}

204struct ZombieReporter<'a> {
205    sess: &'a Session,
206    module: &'a Module,
207    zombies: &'a Zombies<'a>,
208
209    id_to_name: Option<FxHashMap<Word, &'a str>>,
210    span_regen: SpanRegenerator<'a>,
211}
212impl<'a> ZombieReporter<'a> {
213    fn new(sess: &'a Session, module: &'a Module, zombies: &'a Zombies<'a>) -> Self {
214        Self {
215            sess,
216            module,
217            zombies,
218
219            id_to_name: None,
220            span_regen: SpanRegenerator::new(sess.source_map(), module),
221        }
222    }
223
224    // If an entry point references a zombie'd value, then the entry point would normally get removed.
225    // That's an absolutely horrible experience to debug, though, so instead, create a nice error
226    // message containing the stack trace of how the entry point got to the zombie value.
227    fn report_all(mut self) -> super::Result<()> {
228        let mut result = Ok(());
229        // FIXME(eddyb) this loop means that every entry-point can potentially
230        // list out all the leaves, but that shouldn't be a huge issue.
231        for root_id in super::dce::collect_roots(self.module) {
232            if let Some(zombie) = self.zombies.get_zombie_by_id(root_id) {
233                for (_, err) in self.build_errors_keyed_by_leaf_id(zombie) {
234                    result = Err(err.emit());
235                }
236            }
237        }
238        result
239    }
240
241    fn add_use_note_to_err(
242        &mut self,
243        err: &mut Diag<'a>,
244        span: Span,
245        zombie: Zombie<'_>,
246        zombie_use: &ZombieUse<'_>,
247    ) {
248        let mut id_to_name = |id, kind| {
249            self.id_to_name
250                .get_or_insert_with(|| get_names(self.module))
251                .get(&id)
252                .map_or_else(
253                    || format!("unnamed {kind} (%{id})"),
254                    |&name| format!("`{name}`"),
255                )
256        };
257        let note = match zombie_use.origin {
258            UseOrigin::GlobalOperandOrResultType => {
259                format!("used by {}", id_to_name(zombie.id, "global"))
260            }
261            UseOrigin::IntraFuncOperandOrResultType { parent_func_id } => {
262                format!(
263                    "used from within {}",
264                    id_to_name(parent_func_id, "function")
265                )
266            }
267            UseOrigin::CallCalleeOperand { caller_func_id } => {
268                format!("called by {}", id_to_name(caller_func_id, "function"))
269            }
270        };
271        let span = zombie_use
272            .use_debug_src_loc_inst
273            .and_then(|inst| self.span_regen.src_loc_from_debug_inst(inst))
274            .and_then(|src_loc| self.span_regen.src_loc_to_rustc(src_loc))
275            .unwrap_or(span);
276        err.span_note(span, note);
277    }
278
279    fn build_errors_keyed_by_leaf_id(&mut self, zombie: Zombie<'_>) -> FxIndexMap<Word, Diag<'a>> {
280        // FIXME(eddyb) this is a bit inefficient, compared to some kind of
281        // "small map", but this is the error path, and being correct is more
282        // important here - in particular, we don't want to ignore *any* leaves.
283        let mut errors_keyed_by_leaf_id = FxIndexMap::default();
284
285        let span = self
286            .span_regen
287            .src_loc_for_id(zombie.id)
288            .and_then(|src_loc| self.span_regen.src_loc_to_rustc(src_loc))
289            .unwrap_or(DUMMY_SP);
290        match zombie.kind {
291            ZombieKind::Leaf => {
292                let reason = self.span_regen.zombie_for_id(zombie.id).unwrap().reason;
293                errors_keyed_by_leaf_id.insert(
294                    zombie.id,
295                    self.sess.dcx().struct_span_err(span, reason.to_string()),
296                );
297            }
298            ZombieKind::Uses(zombie_uses) => {
299                for zombie_use in zombie_uses {
300                    let used_zombie = self
301                        .zombies
302                        .get_zombie_by_id(zombie_use.used_zombie_id)
303                        .unwrap();
304                    for (leaf_id, err) in self.build_errors_keyed_by_leaf_id(used_zombie) {
305                        use rustc_data_structures::fx::IndexEntry as Entry;
306                        match errors_keyed_by_leaf_id.entry(leaf_id) {
307                            Entry::Occupied(_) => err.cancel(),
308                            Entry::Vacant(entry) => {
309                                self.add_use_note_to_err(
310                                    entry.insert(err),
311                                    span,
312                                    zombie,
313                                    zombie_use,
314                                );
315                            }
316                        }
317                    }
318                }
319            }
320        }
321        errors_keyed_by_leaf_id
322    }
323}

325pub fn report_and_remove_zombies(sess: &Session, module: &mut Module) -> super::Result<()> {
326    let mut zombies = Zombies {
327        // FIXME(eddyb) avoid repeating this across different passes/helpers.
328        custom_ext_inst_set_import: module
329            .ext_inst_imports
330            .iter()
331            .find(|inst| {
332                assert_eq!(inst.class.opcode, Op::ExtInstImport);
333                inst.operands[0].unwrap_literal_string() == &custom_insts::CUSTOM_EXT_INST_SET[..]
334            })
335            .map(|inst| inst.result_id.unwrap()),
336
337        id_to_zombie_kind: ZombieDecoration::decode_all(module)
338            .map(|(id, _)| (id, ZombieKind::Leaf))
339            .collect(),
340    };
341    // Note: This is O(n^2).
342    while zombies.spread(module) {}
343
344    let result = ZombieReporter::new(sess, module, &zombies).report_all();
345    if tracing::enabled!(target: "print_all_zombie", Level::DEBUG) {
346        let mut span_regen = SpanRegenerator::new(sess.source_map(), module);
347        for &zombie_id in zombies.id_to_zombie_kind.keys() {
348            let mut zombie_leaf_id = zombie_id;
349            let mut infection_chain = SmallVec::<[_; 4]>::new();
350            loop {
351                zombie_leaf_id = match zombies.get_zombie_by_id(zombie_leaf_id).unwrap().kind {
352                    ZombieKind::Leaf => break,
353                    // FIXME(eddyb) this is all very lossy and should probably go away.
354                    ZombieKind::Uses(zombie_uses) => zombie_uses[0].used_zombie_id,
355                };
356                infection_chain.push(zombie_leaf_id);
357            }
358
359            let reason = span_regen.zombie_for_id(zombie_leaf_id).unwrap().reason;
360            debug!(
361                target: "print_all_zombie",
362                "zombie'd %{zombie_id} because {reason}"
363            );
364            if !infection_chain.is_empty() {
365                debug!(
366                    target: "print_all_zombie",
367                    " (infected via {:?})", infection_chain
368                );
369            }
370            debug!(target: "print_all_zombie", "");
371        }
372    }
373
374    if tracing::enabled!(target: "print_zombie", Level::DEBUG) {
375        let mut span_regen = SpanRegenerator::new(sess.source_map(), module);
376        let names = get_names(module);
377        for f in &module.functions {
378            if let Some(zombie) = zombies.get_zombie_by_id(f.def_id().unwrap()) {
379                let mut zombie_leaf_id = zombie.id;
380                loop {
381                    zombie_leaf_id = match zombies.get_zombie_by_id(zombie_leaf_id).unwrap().kind {
382                        ZombieKind::Leaf => break,
383                        // FIXME(eddyb) this is all very lossy and should probably go away.
384                        ZombieKind::Uses(zombie_uses) => zombie_uses[0].used_zombie_id,
385                    };
386                }
387
388                let name = get_name(&names, f.def_id().unwrap());
389                let reason = span_regen.zombie_for_id(zombie_leaf_id).unwrap().reason;
390                debug!(
391                    target: "print_zombie",
392                    "function removed {name:?} because {reason:?}"
393                );
394            }
395        }
396    }
397
398    // FIXME(eddyb) this should be unnecessary, either something is unused, and
399    // it will get DCE'd *anyway*, or it caused an error.
400    {
401        // HACK(eddyb) cannot use the original map because it borrows the `Module`.
402        let all_zombies: FxHashSet<_> = zombies.id_to_zombie_kind.into_keys().collect();
403        let keep = |inst: &Instruction| {
404            if let Some(result_id) = inst.result_id {
405                !all_zombies.contains(&result_id)
406            } else {
407                let mut inst_ids = inst
408                    .result_type
409                    .into_iter()
410                    .chain(inst.operands.iter().filter_map(|op| op.id_ref_any()));
411                !inst_ids.any(|id| all_zombies.contains(&id))
412            }
413        };
414        module.capabilities.retain(keep);
415        module.extensions.retain(keep);
416        module.ext_inst_imports.retain(keep);
417        module.memory_model = module.memory_model.take().filter(keep);
418        module.entry_points.retain(keep);
419        module.execution_modes.retain(keep);
420        module.debug_string_source.retain(keep);
421        module.debug_names.retain(keep);
422        module.debug_module_processed.retain(keep);
423        module.annotations.retain(keep);
424        module.types_global_values.retain(keep);
425        module.functions.retain(|f| keep(f.def.as_ref().unwrap()));
426    }
427
428    result
