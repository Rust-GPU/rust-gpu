13#[derive(Copy, Clone)]
14struct Zombie<'a> {
15    id: Word,
16    kind: &'a ZombieKind<'a>,
17}
18
19enum ZombieKind<'a> {
20    /// Definition annotated with `ZombieDecoration`.
21    Leaf,
22
23    /// Transitively zombie'd by using other zombies, from an instruction.
24    Uses(Vec<ZombieUse<'a>>),
25}
26
27struct ZombieUse<'a> {
28    used_zombie_id: Word,
29
30    /// Active debug "source location" instruction at the time of the use, if any
31    /// (both `OpLine` and `CustomInst::SetDebugSrcLoc` are supported).
32    use_debug_src_loc_inst: Option<&'a Instruction>,
33
34    origin: UseOrigin,
35}
36
37enum UseOrigin {
38    GlobalOperandOrResultType,
39    IntraFuncOperandOrResultType { parent_func_id: Word },
40    CallCalleeOperand { caller_func_id: Word },
41}
42
43struct Zombies<'a> {
44    /// ID of `OpExtInstImport` for our custom "extended instruction set",
45    /// if present (see `crate::custom_insts` for more details).
46    custom_ext_inst_set_import: Option<Word>,
47
48    id_to_zombie_kind: FxIndexMap<Word, ZombieKind<'a>>,
49}
50
51impl<'a> Zombies<'a> {
52    // FIXME(eddyb) rename all the other methods to say `_inst` explicitly.
53    fn get_zombie_by_id(&self, id: Word) -> Option<Zombie<'_>> {
54        self.id_to_zombie_kind
55            .get(&id)
56            .map(|kind| Zombie { id, kind })
57    }
58
59    fn zombies_used_from_inst<'b>(
60        &'b self,
61        inst: &'b Instruction,
62    ) -> impl Iterator<Item = Zombie<'b>> + 'b {
63        inst.result_type
64            .into_iter()
65            .chain(inst.operands.iter().filter_map(|op| op.id_ref_any()))
66            .filter_map(move |id| self.get_zombie_by_id(id))
67    }
68
69    fn spread(&mut self, module: &'a Module) -> bool {
70        let mut any = false;
71        // globals are easy
72        {
73            let mut debug_src_loc_inst = None;
74            for inst in module.global_inst_iter() {
75                match inst.class.opcode {
76                    Op::Line => debug_src_loc_inst = Some(inst),
77                    Op::NoLine => debug_src_loc_inst = None,
78                    Op::ExtInst
79                        if Some(inst.operands[0].unwrap_id_ref())
80                            == self.custom_ext_inst_set_import =>
81                    {
82                        match CustomOp::decode_from_ext_inst(inst) {
83                            CustomOp::SetDebugSrcLoc => debug_src_loc_inst = Some(inst),
84                            CustomOp::ClearDebugSrcLoc => debug_src_loc_inst = None,
85                            _ => {}
86                        }
87                    }
88                    _ => {}
89                }
90
91                if let Some(result_id) = inst.result_id {
92                    if self.id_to_zombie_kind.contains_key(&result_id) {
93                        continue;
94                    }
95                    let zombie_uses: Vec<_> = self
96                        .zombies_used_from_inst(inst)
97                        .map(|zombie| ZombieUse {
98                            used_zombie_id: zombie.id,
99                            use_debug_src_loc_inst: debug_src_loc_inst,
100                            origin: UseOrigin::GlobalOperandOrResultType,
101                        })
102                        .collect();
103                    if !zombie_uses.is_empty() {
104                        self.id_to_zombie_kind
105                            .insert(result_id, ZombieKind::Uses(zombie_uses));
106                        any = true;
107                    }
108                }
109            }
110        }
111        // No need to zombie defs within a function: If any def within a function is zombied, then the
112        // whole function is zombied. But, we don't have to mark the defs within a function as zombie,
113        // because the defs can't escape the function.
114        // HACK(eddyb) one exception to this is function-local variables, or the
115        // `OpBitcast`s of pointer casts, either of which which may actually be
116        // unused and as such cannot be allowed to always zombie the function.
117        for func in &module.functions {
118            let func_id = func.def_id().unwrap();
119            if self.id_to_zombie_kind.contains_key(&func_id) {
120                // Func is already zombie, no need to scan it again.
121                continue;
122            }
123
124            let mut all_zombie_uses_in_func = vec![];
125            let mut debug_src_loc_inst = None;
126            for inst in func.all_inst_iter() {
127                match inst.class.opcode {
128                    Op::Line => debug_src_loc_inst = Some(inst),
129                    // NOTE(eddyb) each block starts out with cleared debuginfo.
130                    Op::Label | Op::NoLine => debug_src_loc_inst = None,
131                    Op::ExtInst
132                        if Some(inst.operands[0].unwrap_id_ref())
133                            == self.custom_ext_inst_set_import =>
134                    {
135                        match CustomOp::decode_from_ext_inst(inst) {
136                            CustomOp::SetDebugSrcLoc => debug_src_loc_inst = Some(inst),
137                            CustomOp::ClearDebugSrcLoc => debug_src_loc_inst = None,
138                            _ => {}
139                        }
140                    }
141                    _ => {}
142                }
143
144                if [Op::Variable, Op::Bitcast].contains(&inst.class.opcode) {
145                    let result_id = inst.result_id.unwrap();
146                    if self.id_to_zombie_kind.contains_key(&result_id) {
147                        continue;
148                    }
149                    let zombie_uses: Vec<_> = self
150                        .zombies_used_from_inst(inst)
151                        .map(|zombie| ZombieUse {
152                            used_zombie_id: zombie.id,
153                            use_debug_src_loc_inst: debug_src_loc_inst,
154                            origin: UseOrigin::IntraFuncOperandOrResultType {
155                                parent_func_id: func_id,
156                            },
157                        })
158                        .collect();
159                    if !zombie_uses.is_empty() {
160                        self.id_to_zombie_kind
161                            .insert(result_id, ZombieKind::Uses(zombie_uses));
162                        any = true;
163                    }
164                    continue;
165                }
166
167                all_zombie_uses_in_func.extend(
168                    inst.result_id
169                        .and_then(|result_id| self.get_zombie_by_id(result_id))
170                        .into_iter()
171                        .chain(self.zombies_used_from_inst(inst))
172                        .map(|zombie| {
173                            let origin = if inst.class.opcode == Op::FunctionCall
174                                && inst.operands[0] == Operand::IdRef(zombie.id)
175                            {
176                                UseOrigin::CallCalleeOperand {
177                                    caller_func_id: func_id,
178                                }
179                            } else {
180                                UseOrigin::IntraFuncOperandOrResultType {
181                                    parent_func_id: func_id,
182                                }
183                            };
184                            ZombieUse {
185                                used_zombie_id: zombie.id,
186                                use_debug_src_loc_inst: debug_src_loc_inst,
187                                origin,
188                            }
189                        }),
190                );
191            }
192            if !all_zombie_uses_in_func.is_empty() {
193                self.id_to_zombie_kind
194                    .insert(func_id, ZombieKind::Uses(all_zombie_uses_in_func));
195                any = true;
196            }
197        }
198        any
199    }
200}

202struct ZombieReporter<'a> {
203    sess: &'a Session,
204    module: &'a Module,
205    zombies: &'a Zombies<'a>,
206
207    id_to_name: Option<FxHashMap<Word, &'a str>>,
208    span_regen: SpanRegenerator<'a>,
209}
210impl<'a> ZombieReporter<'a> {
211    fn new(sess: &'a Session, module: &'a Module, zombies: &'a Zombies<'a>) -> Self {
212        Self {
213            sess,
214            module,
215            zombies,
216
217            id_to_name: None,
218            span_regen: SpanRegenerator::new(sess.source_map(), module),
219        }
220    }
221
222    // If an entry point references a zombie'd value, then the entry point would normally get removed.
223    // That's an absolutely horrible experience to debug, though, so instead, create a nice error
224    // message containing the stack trace of how the entry point got to the zombie value.
225    fn report_all(mut self) -> super::Result<()> {
226        let mut result = Ok(());
227        // FIXME(eddyb) this loop means that every entry-point can potentially
228        // list out all the leaves, but that shouldn't be a huge issue.
229        for root_id in super::dce::collect_roots(self.module) {
230            if let Some(zombie) = self.zombies.get_zombie_by_id(root_id) {
231                for (_, err) in self.build_errors_keyed_by_leaf_id(zombie) {
232                    result = Err(err.emit());
233                }
234            }
235        }
236        result
237    }
238
239    fn add_use_note_to_err(
240        &mut self,
241        err: &mut Diag<'a>,
242        span: Span,
243        zombie: Zombie<'_>,
244        zombie_use: &ZombieUse<'_>,
245    ) {
246        let mut id_to_name = |id, kind| {
247            self.id_to_name
248                .get_or_insert_with(|| get_names(self.module))
249                .get(&id)
250                .map_or_else(
251                    || format!("unnamed {kind} (%{id})"),
252                    |&name| format!("`{name}`"),
253                )
254        };
255        let note = match zombie_use.origin {
256            UseOrigin::GlobalOperandOrResultType => {
257                format!("used by {}", id_to_name(zombie.id, "global"))
258            }
259            UseOrigin::IntraFuncOperandOrResultType { parent_func_id } => {
260                format!(
261                    "used from within {}",
262                    id_to_name(parent_func_id, "function")
263                )
264            }
265            UseOrigin::CallCalleeOperand { caller_func_id } => {
266                format!("called by {}", id_to_name(caller_func_id, "function"))
267            }
268        };
269        let span = zombie_use
270            .use_debug_src_loc_inst
271            .and_then(|inst| self.span_regen.src_loc_from_debug_inst(inst))
272            .and_then(|src_loc| self.span_regen.src_loc_to_rustc(src_loc))
273            .unwrap_or(span);
274        err.span_note(span, note);
275    }
276
277    fn build_errors_keyed_by_leaf_id(&mut self, zombie: Zombie<'_>) -> FxIndexMap<Word, Diag<'a>> {
278        // FIXME(eddyb) this is a bit inefficient, compared to some kind of
279        // "small map", but this is the error path, and being correct is more
280        // important here - in particular, we don't want to ignore *any* leaves.
281        let mut errors_keyed_by_leaf_id = FxIndexMap::default();
282
283        let span = self
284            .span_regen
285            .src_loc_for_id(zombie.id)
286            .and_then(|src_loc| self.span_regen.src_loc_to_rustc(src_loc))
287            .unwrap_or(DUMMY_SP);
288        match zombie.kind {
289            ZombieKind::Leaf => {
290                let reason = self.span_regen.zombie_for_id(zombie.id).unwrap().reason;
291                errors_keyed_by_leaf_id.insert(
292                    zombie.id,
293                    self.sess.dcx().struct_span_err(span, reason.to_string()),
294                );
295            }
296            ZombieKind::Uses(zombie_uses) => {
297                for zombie_use in zombie_uses {
298                    let used_zombie = self
299                        .zombies
300                        .get_zombie_by_id(zombie_use.used_zombie_id)
301                        .unwrap();
302                    for (leaf_id, err) in self.build_errors_keyed_by_leaf_id(used_zombie) {
303                        use rustc_data_structures::fx::IndexEntry as Entry;
304                        match errors_keyed_by_leaf_id.entry(leaf_id) {
305                            Entry::Occupied(_) => err.cancel(),
306                            Entry::Vacant(entry) => {
307                                self.add_use_note_to_err(
308                                    entry.insert(err),
309                                    span,
310                                    zombie,
311                                    zombie_use,
312                                );
313                            }
314                        }
315                    }
316                }
317            }
318        }
319        errors_keyed_by_leaf_id
320    }
321}

323pub fn report_zombies(sess: &Session, module: &Module) -> super::Result<()> {
324    let mut zombies = Zombies {
325        // FIXME(eddyb) avoid repeating this across different passes/helpers.
326        custom_ext_inst_set_import: module
327            .ext_inst_imports
328            .iter()
329            .find(|inst| {
330                assert_eq!(inst.class.opcode, Op::ExtInstImport);
331                inst.operands[0].unwrap_literal_string() == &custom_insts::CUSTOM_EXT_INST_SET[..]
332            })
333            .map(|inst| inst.result_id.unwrap()),
334
335        id_to_zombie_kind: ZombieDecoration::decode_all(module)
336            .map(|(id, _)| (id, ZombieKind::Leaf))
337            .collect(),
338    };
339    // Note: This is O(n^2).
340    while zombies.spread(module) {}
341
342    ZombieReporter::new(sess, module, &zombies).report_all()
343}
