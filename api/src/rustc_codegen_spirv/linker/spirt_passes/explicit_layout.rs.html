18/// Erase explicit layout decorations from struct/array types, when used with
19/// storage classes that do not support them (as per the Vulkan spec).
20//
21// NOTE(eddyb) this is a stop-gap until `spirt::{mem,qptr}` can replace it.
22pub fn erase_when_invalid(module: &mut Module) {
23    let spv_spec = super::SpvSpecWithExtras::get();
24    let wk = &spv_spec.well_known;
25
26    let mut eraser = SelectiveEraser {
27        cx: &module.cx(),
28        wk,
29
30        transformed_types: FxHashMap::default(),
31        transformed_consts: FxHashMap::default(),
32        transformed_data_inst_forms: FxHashMap::default(),
33        seen_global_vars: FxHashSet::default(),
34        global_var_queue: VecDeque::new(),
35        seen_funcs: FxHashSet::default(),
36        func_queue: VecDeque::new(),
37
38        cached_erased_explicit_layout_types: FxHashMap::default(),
39        cached_erased_explicit_layout_consts: FxHashMap::default(),
40
41        parent_block: None,
42    };
43
44    // Seed the queues starting from the module exports.
45    for exportee in module.exports.values_mut() {
46        exportee
47            .inner_transform_with(&mut eraser)
48            .apply_to(exportee);
49    }
50
51    // Process the queues until they're all empty.
52    while !eraser.global_var_queue.is_empty() || !eraser.func_queue.is_empty() {
53        while let Some(gv) = eraser.global_var_queue.pop_front() {
54            eraser.in_place_transform_global_var_decl(&mut module.global_vars[gv]);
55        }
56        while let Some(func) = eraser.func_queue.pop_front() {
57            eraser.in_place_transform_func_decl(&mut module.funcs[func]);
58        }
59    }
60}

62// FIXME(eddyb) name could be better (avoiding overly verbose is a bit tricky).
63struct SelectiveEraser<'a> {
64    cx: &'a Context,
65    wk: &'static super::SpvWellKnownWithExtras,
66
67    // FIXME(eddyb) build some automation to avoid ever repeating these.
68    transformed_types: FxHashMap<Type, Transformed<Type>>,
69    transformed_consts: FxHashMap<Const, Transformed<Const>>,
70    transformed_data_inst_forms: FxHashMap<DataInstForm, Transformed<DataInstForm>>,
71    seen_global_vars: FxHashSet<GlobalVar>,
72    global_var_queue: VecDeque<GlobalVar>,
73    seen_funcs: FxHashSet<Func>,
74    func_queue: VecDeque<Func>,
75
76    // FIXME(eddyb) these overlap with some `transformed_*` fields above,
77    // but they're contextually transformed additionally.
78    // HACK(eddyb) these are now used via the `EraseExplicitLayout` newtype.
79    cached_erased_explicit_layout_types: FxHashMap<Type, Transformed<Type>>,
80    cached_erased_explicit_layout_consts: FxHashMap<Const, Transformed<Const>>,
81
82    // HACK(eddyb) this is to allow `in_place_transform_data_inst_def` inject
83    // new instructions into its parent block.
84    parent_block: Option<ControlNode>,
85}

87impl Transformer for SelectiveEraser<'_> {
88    // FIXME(eddyb) build some automation to avoid ever repeating these.
89    fn transform_type_use(&mut self, ty: Type) -> Transformed<Type> {
90        if let Some(&cached) = self.transformed_types.get(&ty) {
91            return cached;
92        }
93        let transformed = self
94            .transform_type_def(&self.cx[ty])
95            .map(|ty_def| self.cx.intern(ty_def));
96        self.transformed_types.insert(ty, transformed);
97        transformed
98    }
99    fn transform_const_use(&mut self, ct: Const) -> Transformed<Const> {
100        if let Some(&cached) = self.transformed_consts.get(&ct) {
101            return cached;
102        }
103        let transformed = self
104            .transform_const_def(&self.cx[ct])
105            .map(|ct_def| self.cx.intern(ct_def));
106        self.transformed_consts.insert(ct, transformed);
107        transformed
108    }
109    fn transform_data_inst_form_use(
110        &mut self,
111        data_inst_form: DataInstForm,
112    ) -> Transformed<DataInstForm> {
113        if let Some(&cached) = self.transformed_data_inst_forms.get(&data_inst_form) {
114            return cached;
115        }
116        let transformed = self
117            .transform_data_inst_form_def(&self.cx[data_inst_form])
118            .map(|data_inst_form_def| self.cx.intern(data_inst_form_def));
119        self.transformed_data_inst_forms
120            .insert(data_inst_form, transformed);
121        transformed
122    }
123
124    fn transform_global_var_use(&mut self, gv: GlobalVar) -> Transformed<GlobalVar> {
125        if self.seen_global_vars.insert(gv) {
126            self.global_var_queue.push_back(gv);
127        }
128        Transformed::Unchanged
129    }
130    fn transform_func_use(&mut self, func: Func) -> Transformed<Func> {
131        if self.seen_funcs.insert(func) {
132            self.func_queue.push_back(func);
133        }
134        Transformed::Unchanged
135    }

139    fn transform_type_def(&mut self, ty_def: &TypeDef) -> Transformed<TypeDef> {
140        let wk = self.wk;
141
142        let needs_erasure_of_explicit_layout = match &ty_def.kind {
143            TypeKind::SpvInst {
144                spv_inst,
145                type_and_const_inputs: _,
146            } if spv_inst.opcode == wk.OpTypePointer => match spv_inst.imms[..] {
147                [spv::Imm::Short(sc_kind, sc)] => {
148                    assert_eq!(sc_kind, wk.StorageClass);
149                    !self.addr_space_allows_explicit_layout(AddrSpace::SpvStorageClass(sc))
150                }
151                _ => unreachable!(),
152            },
153
154            _ => false,
155        };
156        if needs_erasure_of_explicit_layout {
157            ty_def.inner_transform_with(&mut EraseExplicitLayout(self))
158        } else {
159            ty_def.inner_transform_with(self)
160        }
161    }
162
163    fn in_place_transform_global_var_decl(&mut self, gv_decl: &mut GlobalVarDecl) {
164        let needs_erasure_of_explicit_layout =
165            !self.addr_space_allows_explicit_layout(gv_decl.addr_space);
166        if needs_erasure_of_explicit_layout {
167            gv_decl.inner_in_place_transform_with(&mut EraseExplicitLayout(self));
168        } else {
169            gv_decl.inner_in_place_transform_with(self);
170        }
171    }
172
173    fn in_place_transform_func_decl(&mut self, func_decl: &mut FuncDecl) {
174        // HACK(eddyb) to catch any instructions having their input/output types
175        // changed from under them, a separate visit has to be used before *any*
176        // region input/node output declarations in the function body may change.
177        if let DeclDef::Present(func_def_body) = &mut func_decl.def {
178            let mut errors_to_attach = vec![];
179            func_def_body.inner_visit_with(&mut super::VisitAllControlRegionsAndNodes {
180                state: (),
181                visit_control_region: |_: &mut _, _| {},
182                visit_control_node: |_: &mut _, func_at_node: FuncAt<'_, ControlNode>| {
183                    if let ControlNodeKind::Block { insts } = func_at_node.def().kind {
184                        for func_at_inst in func_at_node.at(insts) {
185                            if let Err(e) = self.pre_check_data_inst(func_at_inst) {
186                                errors_to_attach.push((func_at_inst.position, e));
187                            }
188                        }
189                    }
190                },
191            });
192            for (inst, err) in errors_to_attach {
193                func_def_body
194                    .at_mut(inst)
195                    .def()
196                    .attrs
197                    .push_diag(self.cx, err);
198            }
199        }
200
201        func_decl.inner_in_place_transform_with(self);
202    }
203
204    fn in_place_transform_control_node_def(
205        &mut self,
206        mut func_at_control_node: FuncAtMut<'_, ControlNode>,
207    ) {
208        let old_parent_block = self.parent_block.take();
209        if let ControlNodeKind::Block { .. } = func_at_control_node.reborrow().def().kind {
210            self.parent_block = Some(func_at_control_node.position);
211        }
212        func_at_control_node.inner_in_place_transform_with(self);
213        self.parent_block = old_parent_block;
214    }

216    fn in_place_transform_data_inst_def(&mut self, mut func_at_data_inst: FuncAtMut<'_, DataInst>) {
217        let cx = self.cx;
218        let wk = self.wk;
219
220        func_at_data_inst
221            .reborrow()
222            .inner_in_place_transform_with(self);
223
224        let func_at_data_inst_frozen = func_at_data_inst.reborrow().freeze();
225        let data_inst = func_at_data_inst_frozen.position;
226        let data_inst_def = func_at_data_inst_frozen.def();
227        let data_inst_form_def = &cx[data_inst_def.form];
228        let func = func_at_data_inst_frozen.at(());
229        let type_of_val = |v: Value| func.at(v).type_of(cx);
230        let pointee_type_of_ptr_val = |p: Value| match &cx[type_of_val(p)].kind {
231            TypeKind::SpvInst {
232                spv_inst,
233                type_and_const_inputs,
234            } if spv_inst.opcode == wk.OpTypePointer => match type_and_const_inputs[..] {
235                [TypeOrConst::Type(elem_type)] => Some(elem_type),
236                _ => unreachable!(),
237            },
238            _ => None,
239        };
240
241        let DataInstKind::SpvInst(spv_inst) = &data_inst_form_def.kind else {
242            return;
243        };
244
245        // FIXME(eddyb) filter attributes into debuginfo and
246        // semantic, and understand the semantic ones.
247        let attrs = data_inst_def.attrs;
248
249        let mk_bitcast_def = |in_value, out_type| DataInstDef {
250            attrs,
251            form: cx.intern(DataInstFormDef {
252                kind: DataInstKind::SpvInst(wk.OpBitcast.into()),
253                output_type: Some(out_type),
254            }),
255            inputs: [in_value].into_iter().collect(),
256        };
257
258        if spv_inst.opcode == wk.OpLoad {
259            let pointee_type = pointee_type_of_ptr_val(data_inst_def.inputs[0]);
260            let value_type = data_inst_form_def.output_type.unwrap();
261            // FIXME(eddyb) leave a BUG diagnostic in the `None` case?
262            if pointee_type.is_some_and(|ty| {
263                ty != value_type && ty == self.erase_explicit_layout_in_type(value_type)
264            }) {
265                let func = func_at_data_inst.at(());
266                let ControlNodeKind::Block { insts } =
267                    &mut func.control_nodes[self.parent_block.unwrap()].kind
268                else {
269                    unreachable!()
270                };
271
272                let fixed_load_inst = func.data_insts.define(
273                    cx,
274                    DataInstDef {
275                        attrs,
276                        form: cx.intern(DataInstFormDef {
277                            kind: data_inst_form_def.kind.clone(),
278                            output_type: Some(pointee_type.unwrap()),
279                        }),
280                        inputs: func.data_insts[data_inst].inputs.clone(),
281                    }
282                    .into(),
283                );
284                insts.insert_before(fixed_load_inst, data_inst, func.data_insts);
285                *func.data_insts[data_inst] =
286                    mk_bitcast_def(Value::DataInstOutput(fixed_load_inst), value_type);
287
288                self.disaggregate_bitcast(func.at(data_inst));
289            }
290        } else if spv_inst.opcode == wk.OpStore {
291            let pointee_type = pointee_type_of_ptr_val(data_inst_def.inputs[0]);
292            let value_type = type_of_val(data_inst_def.inputs[1]);
293            // FIXME(eddyb) leave a BUG diagnostic in the `None` case?
294            if pointee_type.is_some_and(|ty| {
295                ty != value_type && ty == self.erase_explicit_layout_in_type(value_type)
296            }) {
297                let func = func_at_data_inst.at(());
298                let stored_value = &mut func.data_insts[data_inst].inputs[1];
299
300                if let Value::Const(ct) = stored_value {
301                    EraseExplicitLayout(self)
302                        .transform_const_use(*ct)
303                        .apply_to(ct);
304                } else {
305                    let original_stored_value = *stored_value;
306
307                    let ControlNodeKind::Block { insts } =
308                        &mut func.control_nodes[self.parent_block.unwrap()].kind
309                    else {
310                        unreachable!()
311                    };
312                    let stored_value_cast_inst = func.data_insts.define(
313                        cx,
314                        mk_bitcast_def(original_stored_value, pointee_type.unwrap()).into(),
315                    );
316                    insts.insert_before(stored_value_cast_inst, data_inst, func.data_insts);
317                    func.data_insts[data_inst].inputs[1] =
318                        Value::DataInstOutput(stored_value_cast_inst);
319
320                    self.disaggregate_bitcast(func.at(stored_value_cast_inst));
321                }
322            }
323        } else if spv_inst.opcode == wk.OpCopyMemory {
324            let dst_ptr = data_inst_def.inputs[0];
325            let src_ptr = data_inst_def.inputs[1];
326            let [dst_pointee_type, src_pointee_type] =
327                [dst_ptr, src_ptr].map(pointee_type_of_ptr_val);
328            // FIXME(eddyb) leave a BUG diagnostic in the `None` case?
329            let mismatched_dst_src_types = match [dst_pointee_type, src_pointee_type] {
330                [Some(a), Some(b)] => {
331                    // FIXME(eddyb) there has to be a nicer way to write this??
332                    fn equal<T: Eq>([a, b]: [T; 2]) -> bool {
333                        a == b
334                    }
335                    !equal([a, b]) && equal([a, b].map(|ty| self.erase_explicit_layout_in_type(ty)))
336                }
337                _ => false,
338            };
339            if mismatched_dst_src_types {
340                let is_memory_access_imm =
341                    |imm| matches!(imm, &spv::Imm::Short(k, _) if k == wk.MemoryAccess);
342
343                // HACK(eddyb) this relies on `MemoryAccess` being non-recursive
344                // (in fact, its parameters seem to only be simple literals/IDs).
345                let (dst_imms, src_imms) =
346                    match (spv_inst.imms.iter().positions(is_memory_access_imm))
347                        .collect::<SmallVec<[_; 2]>>()[..]
348                    {
349                        [] | [0] => (&spv_inst.imms[..], &spv_inst.imms[..]),
350                        [0, i] => spv_inst.imms.split_at(i),
351                        _ => unreachable!(),
352                    };
353
354                let func = func_at_data_inst.at(());
355                let ControlNodeKind::Block { insts } =
356                    &mut func.control_nodes[self.parent_block.unwrap()].kind
357                else {
358                    unreachable!()
359                };
360
361                let load_inst = func.data_insts.define(
362                    cx,
363                    DataInstDef {
364                        attrs,
365                        form: cx.intern(DataInstFormDef {
366                            kind: DataInstKind::SpvInst(spv::Inst {
367                                opcode: wk.OpLoad,
368                                imms: src_imms.iter().copied().collect(),
369                            }),
370                            output_type: Some(src_pointee_type.unwrap()),
371                        }),
372                        inputs: [src_ptr].into_iter().collect(),
373                    }
374                    .into(),
375                );
376                insts.insert_before(load_inst, data_inst, func.data_insts);
377                let cast_inst = func.data_insts.define(
378                    cx,
379                    mk_bitcast_def(Value::DataInstOutput(load_inst), dst_pointee_type.unwrap())
380                        .into(),
381                );
382                insts.insert_before(cast_inst, data_inst, func.data_insts);
383
384                *func.data_insts[data_inst] = DataInstDef {
385                    attrs,
386                    form: cx.intern(DataInstFormDef {
387                        kind: DataInstKind::SpvInst(spv::Inst {
388                            opcode: wk.OpStore,
389                            imms: dst_imms.iter().copied().collect(),
390                        }),
391                        output_type: None,
392                    }),
393                    inputs: [dst_ptr, Value::DataInstOutput(cast_inst)]
394                        .into_iter()
395                        .collect(),
396                };
397
398                self.disaggregate_bitcast(func.at(cast_inst));
399            }
400        }
401    }
402}

404impl<'a> SelectiveEraser<'a> {
405    fn addr_space_allows_explicit_layout(&self, addr_space: AddrSpace) -> bool {
406        let wk = self.wk;
407
408        // FIXME(eddyb) this might need to include `Workgroup`, specifically when
409        // `WorkgroupMemoryExplicitLayoutKHR` is being relied upon.
410        [
411            wk.PushConstant,
412            wk.Uniform,
413            wk.StorageBuffer,
414            wk.PhysicalStorageBuffer,
415        ]
416        .map(AddrSpace::SpvStorageClass)
417        .contains(&addr_space)
418    }
419
420    // FIXME(eddyb) properly distinguish between zero-extension and sign-extension.
421    fn const_as_u32(&self, ct: Const) -> Option<u32> {
422        if let ConstKind::SpvInst {
423            spv_inst_and_const_inputs,
424        } = &self.cx[ct].kind
425        {
426            let (spv_inst, _const_inputs) = &**spv_inst_and_const_inputs;
427            if spv_inst.opcode == self.wk.OpConstant && spv_inst.imms.len() == 1 {
428                match spv_inst.imms[..] {
429                    [spv::Imm::Short(_, x)] => return Some(x),
430                    _ => unreachable!(),
431                }
432            }
433        }
434        None
435    }
436
437    fn aggregate_component_types(
438        &self,
439        ty: Type,
440    ) -> Option<impl ExactSizeIterator<Item = Type> + Clone + 'a> {
441        let cx = self.cx;
442        let wk = self.wk;
443
444        match &cx[ty].kind {
445            TypeKind::SpvInst {
446                spv_inst,
447                type_and_const_inputs,
448            } if spv_inst.opcode == wk.OpTypeStruct => {
449                Some(Either::Left(type_and_const_inputs.iter().map(
450                    |&ty_or_ct| match ty_or_ct {
451                        TypeOrConst::Type(ty) => ty,
452                        TypeOrConst::Const(_) => unreachable!(),
453                    },
454                )))
455            }
456            TypeKind::SpvInst {
457                spv_inst,
458                type_and_const_inputs,
459            } if spv_inst.opcode == wk.OpTypeArray => {
460                let [TypeOrConst::Type(elem_type), TypeOrConst::Const(count)] =
461                    type_and_const_inputs[..]
462                else {
463                    unreachable!()
464                };
465                let count = self.const_as_u32(count)?;
466                Some(Either::Right((0..count).map(move |_| elem_type)))
467            }
468            _ => None,
469        }
470    }
471
472    fn erase_explicit_layout_in_type(&mut self, mut ty: Type) -> Type {
473        EraseExplicitLayout(self)
474            .transform_type_use(ty)
475            .apply_to(&mut ty);
476        ty
477    }
478
479    // HACK(eddyb) this expands an illegal `OpBitcast` of a struct/array, into
480    // leaf values from the source aggregate that are then recomposed into the
481    // target aggregate - this should go away when SPIR-T `disaggregate` lands.
482    fn disaggregate_bitcast(&mut self, mut func_at_cast_inst: FuncAtMut<'_, DataInst>) {
483        let cx = self.cx;
484        let wk = self.wk;
485
486        let cast_inst = func_at_cast_inst.position;
487        let cast_def = func_at_cast_inst.reborrow().freeze().def().clone();
488        let cast_form_def = &cx[cast_def.form];
489
490        // FIXME(eddyb) filter attributes into debuginfo and
491        // semantic, and understand the semantic ones.
492        let attrs = cast_def.attrs;
493
494        assert!(cast_form_def.kind == DataInstKind::SpvInst(wk.OpBitcast.into()));
495        let in_value = cast_def.inputs[0];
496        let out_type = cast_form_def.output_type.unwrap();
497
498        let mut func = func_at_cast_inst.reborrow();
499        let in_type = func.reborrow().freeze().at(in_value).type_of(cx);
500
501        // FIXME(eddyb) there has to be a nicer way to write this??
502        fn equal<T: Eq>([a, b]: [T; 2]) -> bool {
503            a == b
504        }
505
506        let [in_component_types, out_component_types] = Some([in_type, out_type])
507            .filter(|&types| {
508                !equal(types) && equal(types.map(|ty| self.erase_explicit_layout_in_type(ty)))
509            })
510            .map(|types| types.map(|ty| self.aggregate_component_types(ty)))
511            .unwrap_or_default();
512
513        // NOTE(eddyb) such sanity checks should always succeed, because of the
514        // "in/out types are equal after erasure" check, earlier above.
515        assert_eq!(
516            in_component_types.as_ref().map(|iter| iter.len()),
517            out_component_types.as_ref().map(|iter| iter.len()),
518        );
519
520        let [Some(in_component_types), Some(out_component_types)] =
521            [in_component_types, out_component_types]
522        else {
523            return;
524        };
525
526        let components = (in_component_types.zip_eq(out_component_types).enumerate())
527            .map(|(component_idx, (component_in_type, component_out_type))| {
528                let component_idx = u32::try_from(component_idx).unwrap();
529
530                let component_cast_types =
531                    Some([component_in_type, component_out_type]).filter(|&types| !equal(types));
532                if let Some(component_cast_types) = component_cast_types {
533                    assert!(equal(
534                        component_cast_types.map(|ty| self.erase_explicit_layout_in_type(ty))
535                    ));
536                }
537
538                let component_extract_inst = func.data_insts.define(
539                    cx,
540                    DataInstDef {
541                        attrs,
542                        form: cx.intern(DataInstFormDef {
543                            kind: DataInstKind::SpvInst(spv::Inst {
544                                opcode: wk.OpCompositeExtract,
545                                imms: [spv::Imm::Short(wk.LiteralInteger, component_idx)]
546                                    .into_iter()
547                                    .collect(),
548                            }),
549                            output_type: Some(component_in_type),
550                        }),
551                        inputs: [in_value].into_iter().collect(),
552                    }
553                    .into(),
554                );
555
556                let ControlNodeKind::Block { insts } =
557                    &mut func.control_nodes[self.parent_block.unwrap()].kind
558                else {
559                    unreachable!()
560                };
561                insts.insert_before(component_extract_inst, cast_inst, func.data_insts);
562
563                let component_cast_inst = component_cast_types.map(|[_, component_out_type]| {
564                    let inst = func.data_insts.define(
565                        cx,
566                        DataInstDef {
567                            attrs,
568                            form: cx.intern(DataInstFormDef {
569                                kind: DataInstKind::SpvInst(wk.OpBitcast.into()),
570                                output_type: Some(component_out_type),
571                            }),
572                            inputs: [Value::DataInstOutput(component_extract_inst)]
573                                .into_iter()
574                                .collect(),
575                        }
576                        .into(),
577                    );
578                    insts.insert_before(inst, cast_inst, func.data_insts);
579
580                    inst
581                });
582
583                if let Some(component_cast_inst) = component_cast_inst {
584                    self.disaggregate_bitcast(func.reborrow().at(component_cast_inst));
585                }
586
587                Value::DataInstOutput(component_cast_inst.unwrap_or(component_extract_inst))
588            })
589            .collect();
590
591        *func.at(cast_inst).def() = DataInstDef {
592            attrs,
593            form: cx.intern(DataInstFormDef {
594                kind: DataInstKind::SpvInst(wk.OpCompositeConstruct.into()),
595                output_type: Some(out_type),
596            }),
597            inputs: components,
598        };
599    }
600
601    // HACK(eddyb) this runs on every `DataInst` in a function body, before the
602    // declarations of any region input/node output, are ever changed, to catch
603    // the cases that would need special handling, but lack it.
604    fn pre_check_data_inst(&mut self, func_at_inst: FuncAt<'_, DataInst>) -> Result<(), Diag> {
605        let cx = self.cx;
606        let wk = self.wk;
607
608        let data_inst_def = func_at_inst.def();
609        let data_inst_form_def = &cx[data_inst_def.form];
610
611        // FIXME(eddyb) consider preserving the actual type change in the error.
612        let any_types_will_change = (data_inst_form_def.output_type.into_iter())
613            .chain(
614                data_inst_def
615                    .inputs
616                    .iter()
617                    .map(|&v| func_at_inst.at(v).type_of(cx)),
618            )
619            .any(|ty| {
620                let mut new_ty = ty;
621                self.transform_type_use(ty).apply_to(&mut new_ty);
622                new_ty != ty
623            });
624        if !any_types_will_change {
625            return Ok(());
626        }
627
628        let spv_inst = match &data_inst_form_def.kind {
629            DataInstKind::FuncCall(_) => return Ok(()),
630
631            DataInstKind::SpvInst(spv_inst)
632                if [wk.OpLoad, wk.OpStore, wk.OpCopyMemory].contains(&spv_inst.opcode) =>
633            {
634                return Ok(());
635            }
636
637            DataInstKind::QPtr(_) => {
638                return Err(Diag::bug([
639                    "unhandled pointer type change in unexpected `qptr` instruction".into(),
640                ]));
641            }
642            &DataInstKind::SpvExtInst { ext_set, inst } => {
643                let ext_set = &cx[ext_set];
644                return Err(Diag::bug([format!(
645                    "unhandled pointer type change in extended SPIR-V \
646                     (`{ext_set}` / #{inst}) instruction"
647                )
648                .into()]));
649            }
650
651            DataInstKind::SpvInst(spv_inst) => spv_inst,
652        };
653
654        let sigs = crate::spirv_type_constraints::instruction_signatures(
655            rspirv::spirv::Op::from_u32(spv_inst.opcode.as_u16().into()).unwrap(),
656        );
657        let pointer_pointee_correlated_sigs: SmallVec<[_; 1]> = sigs
658            .unwrap_or(&[])
659            .iter()
660            .filter(|sig| {
661                use crate::spirv_type_constraints::{TyListPat, TyPat};
662
663                #[derive(Copy, Clone, Default)]
664                struct ConstrainedVars {
665                    direct: bool,
666                    in_pointee: bool,
667                }
668                impl std::ops::BitOr for ConstrainedVars {
669                    type Output = Self;
670                    fn bitor(self, rhs: Self) -> Self {
671                        Self {
672                            direct: self.direct | rhs.direct,
673                            in_pointee: self.in_pointee | rhs.in_pointee,
674                        }
675                    }
676                }
677                impl ConstrainedVars {
678                    fn collect_from(pat: &TyPat<'_>) -> Self {
679                        match pat {
680                            TyPat::Pointer(_, inner) => {
681                                let Self { direct, in_pointee } = Self::collect_from(inner);
682                                Self {
683                                    direct: false,
684                                    in_pointee: direct | in_pointee,
685                                }
686                            }
687
688                            TyPat::Any | TyPat::Void => Self::default(),
689                            TyPat::Var(_) => Self {
690                                direct: true,
691                                in_pointee: false,
692                            },
693                            TyPat::Either(a, b) => Self::collect_from(a) | Self::collect_from(b),
694                            TyPat::Array(inner)
695                            | TyPat::Vector(inner)
696                            | TyPat::Vector4(inner)
697                            | TyPat::Matrix(inner)
698                            | TyPat::Image(inner)
699                            | TyPat::Pipe(inner)
700                            | TyPat::SampledImage(inner)
701                            | TyPat::IndexComposite(inner) => Self::collect_from(inner),
702                            TyPat::Struct(fields) => Self::collect_from_list_leaves(fields),
703                            TyPat::Function(output, inputs) => {
704                                Self::collect_from(output) | Self::collect_from_list_leaves(inputs)
705                            }
706                        }
707                    }
708                    fn collect_from_list_leaves(pat: &TyListPat<'_>) -> Self {
709                        match pat {
710                            TyListPat::Any | TyListPat::Nil | TyListPat::Var(_) => Self::default(),
711                            TyListPat::Repeat(inner) => Self::collect_from_list_leaves(inner),
712                            TyListPat::Cons { first, suffix } => {
713                                Self::collect_from(first) | Self::collect_from_list_leaves(suffix)
714                            }
715                        }
716                    }
717                }
718
719                let mut min_expected_inputs = 0;
720                let mut constrained_vars = sig
721                    .output_type
722                    .map(ConstrainedVars::collect_from)
723                    .unwrap_or_default();
724
725                let mut inputs = sig.input_types;
726                while let TyListPat::Cons { first, suffix } = inputs {
727                    min_expected_inputs += 1;
728                    constrained_vars = constrained_vars | ConstrainedVars::collect_from(first);
729
730                    inputs = suffix;
731                }
732
733                if let (Ordering::Less, _) | (Ordering::Greater, TyListPat::Nil) =
734                    (data_inst_def.inputs.len().cmp(&min_expected_inputs), inputs)
735                {
736                    return false;
737                }
738
739                constrained_vars.direct && constrained_vars.in_pointee
740            })
741            .collect();
742        if !pointer_pointee_correlated_sigs.is_empty() {
743            return Err(Diag::bug([format!(
744                "unhandled pointer type change in `{}` SPIR-V instruction: \
745                 {pointer_pointee_correlated_sigs:#?}",
746                spv_inst.opcode.name()
747            )
748            .into()]));
749        }
750        Ok(())
751    }
752}

754// HACK(eddyb) wrapper modifying `Transformer` behavior of `SelectiveEraser`.
755struct EraseExplicitLayout<'a, 'b>(&'a mut SelectiveEraser<'b>);
756
757impl Transformer for EraseExplicitLayout<'_, '_> {
758    // FIXME(eddyb) build some automation to avoid ever repeating these.
759    fn transform_type_use(&mut self, ty: Type) -> Transformed<Type> {
760        if let Some(&cached) = self.0.cached_erased_explicit_layout_types.get(&ty) {
761            return cached;
762        }
763        let transformed = self
764            .transform_type_def(&self.0.cx[ty])
765            .map(|ty_def| self.0.cx.intern(ty_def));
766        self.0
767            .cached_erased_explicit_layout_types
768            .insert(ty, transformed);
769        transformed
770    }
771    fn transform_const_use(&mut self, ct: Const) -> Transformed<Const> {
772        if let Some(&cached) = self.0.cached_erased_explicit_layout_consts.get(&ct) {
773            return cached;
774        }
775        let transformed = self
776            .transform_const_def(&self.0.cx[ct])
777            .map(|ct_def| self.0.cx.intern(ct_def));
778        self.0
779            .cached_erased_explicit_layout_consts
780            .insert(ct, transformed);
781        transformed
782    }
783    fn transform_data_inst_form_use(&mut self, _: DataInstForm) -> Transformed<DataInstForm> {
784        unreachable!()
785    }
786
787    fn transform_global_var_use(&mut self, gv: GlobalVar) -> Transformed<GlobalVar> {
788        self.0.transform_global_var_use(gv)
789    }
790    fn transform_func_use(&mut self, func: Func) -> Transformed<Func> {
791        self.0.transform_func_use(func)
792    }
793
794    // NOTE(eddyb) above methods are plumbing, erasure methods are below.
795
796    fn transform_type_def(&mut self, ty_def: &TypeDef) -> Transformed<TypeDef> {
797        let wk = self.0.wk;
798
799        // HACK(eddyb) reconsider pointer types, based on *their* storage class
800        // (e.g. implicit-layout pointers to explicit-layout pointers, even if
801        // for Vulkan that's only possible by involving `PhysicalStorageBuffer`).
802        match &ty_def.kind {
803            TypeKind::SpvInst {
804                spv_inst,
805                type_and_const_inputs: _,
806            } if spv_inst.opcode == wk.OpTypePointer => {
807                return self.0.transform_type_def(ty_def);
808            }
809            _ => {}
810        }
811
812        let transformed = ty_def.inner_transform_with(self);
813
814        let old_attrs = match &transformed {
815            Transformed::Unchanged => ty_def.attrs,
816            Transformed::Changed(new_ty_def) => new_ty_def.attrs,
817        };
818
819        let new_attrs = self.0.cx.intern(AttrSetDef {
820            attrs: self.0.cx[old_attrs]
821                .attrs
822                .iter()
823                .filter(|attr| {
824                    // FIXME(eddyb) `rustfmt` breaks down for `matches!`.
825                    #[allow(clippy::match_like_matches_macro)]
826                    let is_explicit_layout_decoration = match attr {
827                        Attr::SpvAnnotation(attr_spv_inst)
828                            if (attr_spv_inst.opcode == wk.OpDecorate
829                                && [wk.ArrayStride, wk.MatrixStride]
830                                    .map(|d| spv::Imm::Short(wk.Decoration, d))
831                                    .contains(&attr_spv_inst.imms[0]))
832                                || (attr_spv_inst.opcode == wk.OpMemberDecorate
833                                    && attr_spv_inst.imms[1]
834                                        == spv::Imm::Short(wk.Decoration, wk.Offset)) =>
835                        {
836                            true
837                        }
838
839                        _ => false,
840                    };
841                    !is_explicit_layout_decoration
842                })
843                .cloned()
844                .collect(),
845        });
846
847        if old_attrs == new_attrs {
848            return transformed;
849        }
850
851        let mut ty_def = TypeDef {
852            attrs: ty_def.attrs,
853            kind: ty_def.kind.clone(),
854        };
855        transformed.apply_to(&mut ty_def);
856
857        ty_def.attrs = new_attrs;
858        Transformed::Changed(ty_def)
859    }
860}
