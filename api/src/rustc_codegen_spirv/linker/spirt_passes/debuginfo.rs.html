13/// Replace our custom extended instruction debuginfo with standard SPIR-V ones.
14//
15// FIXME(eddyb) also handle `SrcLocDecoration`s (when `rspirv` isn't used on the
16// SPIR-V output of `spirt::spv::lift`, as it's lossy wrt `OpLine`).
17pub fn convert_custom_debuginfo_to_spv(module: &mut Module) {
18    let cx = &module.cx();
19
20    // FIXME(eddyb) reuse this collection work in some kind of "pass manager".
21    let all_funcs = {
22        let mut collector = super::ReachableUseCollector {
23            cx,
24            module,
25
26            seen_types: FxIndexSet::default(),
27            seen_consts: FxIndexSet::default(),
28            seen_data_inst_forms: FxIndexSet::default(),
29            seen_global_vars: FxIndexSet::default(),
30            seen_funcs: FxIndexSet::default(),
31        };
32        for (export_key, &exportee) in &module.exports {
33            export_key.inner_visit_with(&mut collector);
34            exportee.inner_visit_with(&mut collector);
35        }
36        collector.seen_funcs
37    };
38
39    let mut transformer = CustomDebuginfoToSpv {
40        cx,
41        wk: &super::SpvSpecWithExtras::get().well_known,
42        custom_ext_inst_set: cx.intern(&custom_insts::CUSTOM_EXT_INST_SET[..]),
43    };
44    for func in all_funcs {
45        transformer.in_place_transform_func_decl(&mut module.funcs[func]);
46    }
47}
48
49struct CustomDebuginfoToSpv<'a> {
50    cx: &'a Context,
51    wk: &'static super::SpvWellKnownWithExtras,
52
53    /// Interned name for our custom "extended instruction set"
54    /// (see `crate::custom_insts` for more details).
55    custom_ext_inst_set: InternedStr,
56}
57
58impl Transformer for CustomDebuginfoToSpv<'_> {
59    fn in_place_transform_control_node_def(
60        &mut self,
61        mut func_at_control_node: spirt::func_at::FuncAtMut<'_, ControlNode>,
62    ) {
63        // HACK(eddyb) this relies on the fact that `ControlNodeKind::Block` maps
64        // to one original SPIR-V block, which may not necessarily be true, and
65        // steps should be taken elsewhere to explicitly unset debuginfo, instead
66        // of relying on the end of a SPIR-V block implicitly unsetting it all.
67        // NOTE(eddyb) allowing debuginfo to apply *outside* of a `Block` could
68        // be useful in allowing *some* structured control-flow to have debuginfo,
69        // but that would likely require more work on the SPIR-T side.
70        if let ControlNodeKind::Block { mut insts } = func_at_control_node.reborrow().def().kind {
71            let mut current_file_line_col = None;
72
73            // HACK(eddyb) buffering the `DataInst`s to remove from this block,
74            // as iterating and modifying a list at the same time isn't supported.
75            let mut insts_to_remove = SmallVec::<[_; 8]>::new();
76
77            let mut func_at_inst_iter = func_at_control_node.reborrow().at(insts).into_iter();
78            while let Some(func_at_inst) = func_at_inst_iter.next() {
79                let inst = func_at_inst.position;
80                let data_inst_def = func_at_inst.def();
81
82                // FIXME(eddyb) deduplicate with `spirt_passes::diagnostics`.
83                if let DataInstKind::SpvExtInst {
84                    ext_set,
85                    inst: ext_inst,
86                } = self.cx[data_inst_def.form].kind
87                    && ext_set == self.custom_ext_inst_set
88                {
89                    let custom_op = CustomOp::decode(ext_inst);
90                    match custom_op.with_operands(&data_inst_def.inputs) {
91                        CustomInst::SetDebugSrcLoc {
92                            file,
93                            line_start: line,
94                            line_end: _,
95                            col_start: col,
96                            col_end: _,
97                        } => {
98                            let const_kind = |v: Value| match v {
99                                Value::Const(ct) => &self.cx[ct].kind,
100                                _ => unreachable!(),
101                            };
102                            let const_str = |v: Value| match const_kind(v) {
103                                &ConstKind::SpvStringLiteralForExtInst(s) => s,
104                                _ => unreachable!(),
105                            };
106                            let const_u32 = |v: Value| match const_kind(v) {
107                                ConstKind::SpvInst {
108                                    spv_inst_and_const_inputs,
109                                } => {
110                                    let (spv_inst, _const_inputs) = &**spv_inst_and_const_inputs;
111                                    assert!(spv_inst.opcode == self.wk.OpConstant);
112                                    match spv_inst.imms[..] {
113                                        [spv::Imm::Short(_, x)] => x,
114                                        _ => unreachable!(),
115                                    }
116                                }
117                                _ => unreachable!(),
118                            };
119                            current_file_line_col =
120                                Some((const_str(file), const_u32(line), const_u32(col)));
121                            insts_to_remove.push(inst);
122                            continue;
123                        }
124                        CustomInst::ClearDebugSrcLoc => {
125                            current_file_line_col = None;
126                            insts_to_remove.push(inst);
127                            continue;
128                        }
129                        CustomInst::PushInlinedCallFrame { .. }
130                        | CustomInst::PopInlinedCallFrame => {
131                            insts_to_remove.push(inst);
132                            continue;
133                        }
134                        CustomInst::Abort { .. } => {
135                            assert!(
136                                !custom_op.is_debuginfo(),
137                                "`CustomOp::{custom_op:?}` debuginfo not lowered"
138                            );
139                        }
140                    }
141                }
142
143                // Add/remove the equivalent `Attr::SpvDebugLine` attribute.
144                // FIXME(eddyb) this could use more caching.
145                data_inst_def.attrs = self.cx.intern(AttrSetDef {
146                    attrs: self.cx[data_inst_def.attrs]
147                        .attrs
148                        .iter()
149                        .filter(|attr| !matches!(attr, Attr::SpvDebugLine { .. }))
150                        .cloned()
151                        .chain(
152                            current_file_line_col.map(|(file, line, col)| Attr::SpvDebugLine {
153                                file_path: OrdAssertEq(file),
154                                line,
155                                col,
156                            }),
157                        )
158                        .collect(),
159                });
160            }
161
162            // Finally remove the `DataInst`s buffered for removal earlier.
163            for inst in insts_to_remove {
164                insts.remove(inst, func_at_control_node.data_insts);
165            }
166            func_at_control_node.reborrow().def().kind = ControlNodeKind::Block { insts };
167        }
168
169        func_at_control_node.inner_in_place_transform_with(self);
170    }
171}
