18/// Apply "reduction rules" to `func_def_body`, replacing (pure) computations
19/// with one of their inputs or a constant (e.g. `x + 0 => x` or `1 + 2 => 3`),
20/// and at most only adding more `Select` outputs/`Loop` state (where necessary)
21/// but never any new instructions (unlike e.g. LLVM's instcombine).
22pub(crate) fn reduce_in_func(cx: &Context, func_def_body: &mut FuncDefBody) {
23    let wk = &super::SpvSpecWithExtras::get().well_known;
24
25    let parent_map = ParentMap::new(func_def_body);
26
27    // FIXME(eddyb) entity-keyed dense maps might be better for performance,
28    // but would require separate maps for separate `Value` cases.
29    let mut value_replacements = FxHashMap::default();
30
31    let mut reduction_cache = FxHashMap::default();
32
33    // HACK(eddyb) this is an annoying workaround for iterator invalidation
34    // (SPIR-T iterators don't cope well with the underlying data changing).
35    //
36    // FIXME(eddyb) replace SPIR-T `FuncAtMut<EntityListIter<T>>` with some
37    // kind of "list cursor", maybe even allowing removal during traversal.
38    let mut reduction_queue = vec![];
39
40    #[derive(Copy, Clone)]
41    enum ReductionTarget {
42        /// Replace uses of a `DataInst` with a reduced `Value`.
43        DataInst(DataInst),
44
45        /// Replace an `OpSwitch` `ControlNode` with an `if`-`else` one.
46        //
47        // HACK(eddyb) see comment in `handle_control_node` for more details.
48        SwitchToIfElse(ControlNode),
49    }
50
51    loop {
52        let old_value_replacements_len = value_replacements.len();
53
54        // HACK(eddyb) we want to transform `DataInstDef`s, while having the ability
55        // to (mutably) traverse the function, but `in_place_transform_data_inst_def`
56        // only gives us a `&mut DataInstDef` (without the `FuncAtMut` around it).
57        //
58        // HACK(eddyb) ignore the above, for now it's pretty bad due to iterator
59        // invalidation (see comment on `let reduction_queue` too).
60        let mut handle_control_node =
61            |func_at_control_node: FuncAt<'_, ControlNode>| match func_at_control_node.def() {
62                &ControlNodeDef {
63                    kind: ControlNodeKind::Block { insts },
64                    ..
65                } => {
66                    for func_at_inst in func_at_control_node.at(insts) {
67                        if let Ok(redu) = Reducible::try_from((cx, func_at_inst.def())) {
68                            let redu_target = ReductionTarget::DataInst(func_at_inst.position);
69                            reduction_queue.push((redu_target, redu));
70                        }
71                    }
72                }
73
74                ControlNodeDef {
75                    kind:
76                        ControlNodeKind::Select {
77                            kind,
78                            scrutinee,
79                            cases,
80                        },
81                    outputs,
82                } => {
83                    // FIXME(eddyb) this should probably be ran in the queue loop
84                    // below, to more quickly benefit from previous reductions.
85                    for i in 0..u32::try_from(outputs.len()).unwrap() {
86                        let output = Value::ControlNodeOutput {
87                            control_node: func_at_control_node.position,
88                            output_idx: i,
89                        };
90                        if let Entry::Vacant(entry) = value_replacements.entry(output) {
91                            let per_case_value = cases.iter().map(|&case| {
92                                func_at_control_node.at(case).def().outputs[i as usize]
93                            });
94                            if let Some(reduced) = try_reduce_select(
95                                cx,
96                                &parent_map,
97                                func_at_control_node.position,
98                                kind,
99                                *scrutinee,
100                                per_case_value,
101                            ) {
102                                entry.insert(reduced);
103                            }
104                        }
105                    }
106
107                    // HACK(eddyb) turn `switch x { case 0: A; case 1: B; default: ... }`
108                    // into `if ... {B} else {A}`, when `x` ends up limited in `0..=1`,
109                    // (such `switch`es come from e.g. `match`-ing enums w/ 2 variants)
110                    // allowing us to bypass SPIR-T current (and temporary) lossiness
111                    // wrt `default: OpUnreachable` (i.e. we prove the `default:` can't
112                    // be entered based on `x` not having values other than `0` or `1`)
113                    if let SelectionKind::SpvInst(spv_inst) = kind {
114                        if spv_inst.opcode == wk.OpSwitch && cases.len() == 3 {
115                            // FIXME(eddyb) this kind of `OpSwitch` decoding logic should
116                            // be done by SPIR-T ahead of time, not here.
117                            let num_logical_imms = cases.len() - 1;
118                            assert_eq!(spv_inst.imms.len() % num_logical_imms, 0);
119                            let logical_imm_size = spv_inst.imms.len() / num_logical_imms;
120                            // FIXME(eddyb) collect to array instead.
121                            let logical_imms_as_u32s: SmallVec<[_; 2]> = spv_inst
122                                .imms
123                                .chunks(logical_imm_size)
124                                .map(spv_imm_checked_trunc32)
125                                .collect();
126
127                            // FIMXE(eddyb) support more values than just `0..=1`.
128                            if logical_imms_as_u32s[..] == [Some(0), Some(1)] {
129                                let redu = Reducible {
130                                    op: PureOp::IntToBool,
131                                    output_type: cx.intern(TypeDef {
132                                        attrs: Default::default(),
133                                        kind: TypeKind::SpvInst {
134                                            spv_inst: wk.OpTypeBool.into(),
135                                            type_and_const_inputs: iter::empty().collect(),
136                                        },
137                                    }),
138                                    input: *scrutinee,
139                                };
140                                let redu_target =
141                                    ReductionTarget::SwitchToIfElse(func_at_control_node.position);
142                                reduction_queue.push((redu_target, redu));
143                            }
144                        }
145                    }
146                }
147
148                ControlNodeDef {
149                    kind:
150                        ControlNodeKind::Loop {
151                            body,
152                            initial_inputs,
153                            ..
154                        },
155                    ..
156                } => {
157                    // FIXME(eddyb) this should probably be ran in the queue loop
158                    // below, to more quickly benefit from previous reductions.
159                    let body_outputs = &func_at_control_node.at(*body).def().outputs;
160                    for (i, (&initial_input, &body_output)) in
161                        initial_inputs.iter().zip(body_outputs).enumerate()
162                    {
163                        let body_input = Value::ControlRegionInput {
164                            region: *body,
165                            input_idx: i as u32,
166                        };
167                        if body_output == body_input {
168                            value_replacements
169                                .entry(body_input)
170                                .or_insert(initial_input);
171                        }
172                    }
173                }
174
175                &ControlNodeDef {
176                    kind: ControlNodeKind::ExitInvocation { .. },
177                    ..
178                } => {}
179            };
180        func_def_body.inner_visit_with(&mut VisitAllControlRegionsAndNodes {
181            state: (),
182            visit_control_region: |_: &mut (), _| {},
183            visit_control_node: |_: &mut (), func_at_control_node| {
184                handle_control_node(func_at_control_node);
185            },
186        });
187
188        // FIXME(eddyb) should this loop become the only loop, by having loop
189        // reductions push the new instruction to `reduction_queue`? the problem
190        // then is that it's not trivial to figure out what else might benefit
191        // from another full scan, so perhaps the only solution is "demand-driven"
192        // (recursing into use->def, instead of processing defs).
193        let mut any_changes = false;
194        for (redu_target, redu) in reduction_queue.drain(..) {
195            if let Some(v) = redu.try_reduce(
196                cx,
197                func_def_body.at_mut(()),
198                &value_replacements,
199                &parent_map,
200                &mut reduction_cache,
201            ) {
202                any_changes = true;
203                match redu_target {
204                    ReductionTarget::DataInst(inst) => {
205                        value_replacements.insert(Value::DataInstOutput(inst), v);
206
207                        // Replace the reduced `DataInstDef` itself with `OpNop`,
208                        // removing the ability to use its "name" as a value.
209                        //
210                        // FIXME(eddyb) cache the interned `OpNop`.
211                        *func_def_body.at_mut(inst).def() = DataInstDef {
212                            attrs: Default::default(),
213                            form: cx.intern(DataInstFormDef {
214                                kind: DataInstKind::SpvInst(wk.OpNop.into()),
215                                output_type: None,
216                            }),
217                            inputs: iter::empty().collect(),
218                        };
219                    }
220
221                    // HACK(eddyb) see comment in `handle_control_node` for more details.
222                    ReductionTarget::SwitchToIfElse(control_node) => {
223                        let control_node_def = func_def_body.at_mut(control_node).def();
224                        match &control_node_def.kind {
225                            ControlNodeKind::Select { cases, .. } => match cases[..] {
226                                [_default, case_0, case_1] => {
227                                    control_node_def.kind = ControlNodeKind::Select {
228                                        kind: SelectionKind::BoolCond,
229                                        scrutinee: v,
230                                        cases: [case_1, case_0].iter().copied().collect(),
231                                    };
232                                }
233                                _ => unreachable!(),
234                            },
235                            _ => unreachable!(),
236                        }
237                    }
238                }
239            }
240        }
241
242        if !any_changes && old_value_replacements_len == value_replacements.len() {
243            break;
244        }
245
246        func_def_body.inner_in_place_transform_with(&mut ReplaceValueWith(|mut v| {
247            let old = v;
248            loop {
249                match v {
250                    Value::Const(_) => break,
251                    _ => match value_replacements.get(&v) {
252                        Some(&new) => v = new,
253                        None => break,
254                    },
255                }
256            }
257            if v != old {
258                any_changes = true;
259                Some(v)
260            } else {
261                None
262            }
263        }));
264    }
265}

267// FIXME(eddyb) maybe this kind of "parent map" should be provided by SPIR-T?
268#[derive(Default)]
269struct ParentMap {
270    data_inst_parent: EntityOrientedDenseMap<DataInst, ControlNode>,
271    control_node_parent: EntityOrientedDenseMap<ControlNode, ControlRegion>,
272    control_region_parent: EntityOrientedDenseMap<ControlRegion, ControlNode>,
273}
274
275impl ParentMap {
276    fn new(func_def_body: &FuncDefBody) -> Self {
277        let mut visitor = VisitAllControlRegionsAndNodes {
278            state: Self::default(),
279            visit_control_region:
280                |this: &mut Self, func_at_control_region: FuncAt<'_, ControlRegion>| {
281                    for func_at_child_control_node in func_at_control_region.at_children() {
282                        this.control_node_parent.insert(
283                            func_at_child_control_node.position,
284                            func_at_control_region.position,
285                        );
286                    }
287                },
288            visit_control_node: |this: &mut Self, func_at_control_node: FuncAt<'_, ControlNode>| {
289                let child_regions = match &func_at_control_node.def().kind {
290                    &ControlNodeKind::Block { insts } => {
291                        for func_at_inst in func_at_control_node.at(insts) {
292                            this.data_inst_parent
293                                .insert(func_at_inst.position, func_at_control_node.position);
294                        }
295                        &[][..]
296                    }
297
298                    ControlNodeKind::Select { cases, .. } => cases,
299                    ControlNodeKind::Loop { body, .. } => slice::from_ref(body),
300                    ControlNodeKind::ExitInvocation { .. } => &[][..],
301                };
302                for &child_region in child_regions {
303                    this.control_region_parent
304                        .insert(child_region, func_at_control_node.position);
305                }
306            },
307        };
308        func_def_body.inner_visit_with(&mut visitor);
309        visitor.state
310    }
311}

313/// If possible, find a single `Value` from `cases` (or even `scrutinee`),
314/// which would always be a valid result for `Select(kind, scrutinee, cases)`,
315/// regardless of which case gets (dynamically) taken.
316fn try_reduce_select(
317    cx: &Context,
318    parent_map: &ParentMap,
319    select_control_node: ControlNode,
320    // FIXME(eddyb) are these redundant with the `ControlNode` above?
321    kind: &SelectionKind,
322    scrutinee: Value,
323    cases: impl Iterator<Item = Value>,
324) -> Option<Value> {
325    let wk = &super::SpvSpecWithExtras::get().well_known;
326
327    let as_spv_const = |v: Value| match v {
328        Value::Const(ct) => match &cx[ct].kind {
329            ConstKind::SpvInst {
330                spv_inst_and_const_inputs,
331            } => {
332                let (spv_inst, _const_inputs) = &**spv_inst_and_const_inputs;
333                Some(spv_inst.opcode)
334            }
335            _ => None,
336        },
337        _ => None,
338    };
339
340    // Ignore `OpUndef`s, as they can be legally substituted with any other value.
341    let mut first_undef = None;
342    let mut non_undef_cases = cases.filter(|&case| {
343        let is_undef = as_spv_const(case) == Some(wk.OpUndef);
344        if is_undef && first_undef.is_none() {
345            first_undef = Some(case);
346        }
347        !is_undef
348    });
349    match (non_undef_cases.next(), non_undef_cases.next()) {
350        (None, _) => first_undef,
351
352        // `Select(c: bool, true, false)` can be replaced with just `c`.
353        (Some(x), Some(y))
354            if matches!(kind, SelectionKind::BoolCond)
355                && as_spv_const(x) == Some(wk.OpConstantTrue)
356                && as_spv_const(y) == Some(wk.OpConstantFalse) =>
357        {
358            assert!(non_undef_cases.next().is_none() && first_undef.is_none());
359
360            Some(scrutinee)
361        }
362
363        (Some(x), y) => {
364            if y.into_iter().chain(non_undef_cases).all(|z| z == x) {
365                // HACK(eddyb) closure here serves as `try` block.
366                let is_x_valid_outside_select = || {
367                    // Constants are always valid.
368                    if let Value::Const(_) = x {
369                        return Some(());
370                    }
371
372                    // HACK(eddyb) if the same value appears in two different
373                    // cases, it's definitely dominating the whole `Select`.
374                    if y.is_some() {
375                        return Some(());
376                    }
377
378                    // In general, `x` dominating the `Select` is what would
379                    // allow lifting an use of it outside the `Select`.
380                    let region_defining_x = match x {
381                        Value::Const(_) => unreachable!(),
382                        Value::ControlRegionInput { region, .. } => region,
383                        Value::ControlNodeOutput { control_node, .. } => {
384                            *parent_map.control_node_parent.get(control_node)?
385                        }
386                        Value::DataInstOutput(inst) => *parent_map
387                            .control_node_parent
388                            .get(*parent_map.data_inst_parent.get(inst)?)?,
389                    };
390
391                    // Fast-reject: if `x` is defined immediately inside one of
392                    // `select_control_node`'s cases, it's not a dominator.
393                    if parent_map.control_region_parent.get(region_defining_x)
394                        == Some(&select_control_node)
395                    {
396                        return None;
397                    }
398
399                    // Since we know `x` is used inside the `Select`, this only
400                    // needs to check that `x` is defined in a region that the
401                    // `Select` is nested in, as the only other possibility is
402                    // that the `x` is defined inside the `Select` - that is,
403                    // one of `x` and `Select` always dominates the other.
404                    //
405                    // FIXME(eddyb) this could be more efficient with some kind
406                    // of "region depth" precomputation but a potentially-slower
407                    // check doubles as a sanity check, for now.
408                    let mut region_containing_select =
409                        *parent_map.control_node_parent.get(select_control_node)?;
410                    loop {
411                        if region_containing_select == region_defining_x {
412                            return Some(());
413                        }
414                        region_containing_select = *parent_map.control_node_parent.get(
415                            *parent_map
416                                .control_region_parent
417                                .get(region_containing_select)?,
418                        )?;
419                    }
420                };
421                if is_x_valid_outside_select().is_some() {
422                    return Some(x);
423                }
424            }
425
426            None
427        }
428    }
429}

431/// Pure operation that transforms one `Value` into another `Value`.
432//
433// FIXME(eddyb) move this elsewhere? also, how should binops etc. be supported?
434// (one approach could be having a "focus input" that can be dynamic, with the
435// other inputs being `Const`s, i.e. partially applying all but one input)
436#[derive(Copy, Clone, PartialEq, Eq, Hash)]
437enum PureOp {
438    BitCast,
439    CompositeExtract {
440        elem_idx: spv::Imm,
441    },
442
443    /// Maps `0` to `false`, and `1` to `true`, but any other input values won't
444    /// allow reduction, which is used to signal `0..=1` isn't being guaranteed.
445    //
446    // HACK(eddyb) not a real operation, but a helper used to extract a `bool`
447    // equivalent for an `OpSwitch`'s scrutinee.
448    // FIXME(eddyb) proper SPIR-T range analysis should be implemented and such
449    // a reduction not attempted at all if the range is larger than `0..=1`
450    // (also, the actual operation can be replaced with `x == 1` or `x != 0`)
451    IntToBool,
452}
453
454impl TryFrom<&spv::Inst> for PureOp {
455    type Error = ();
456    fn try_from(spv_inst: &spv::Inst) -> Result<Self, ()> {
457        let wk = &super::SpvSpecWithExtras::get().well_known;
458
459        let op = spv_inst.opcode;
460        Ok(match spv_inst.imms[..] {
461            [] if op == wk.OpBitcast => Self::BitCast,
462
463            // FIXME(eddyb) support more than one index at a time, somehow.
464            [elem_idx] if op == wk.OpCompositeExtract => Self::CompositeExtract { elem_idx },
465
466            _ => return Err(()),
467        })
468    }
469}
470
471impl TryFrom<PureOp> for spv::Inst {
472    type Error = ();
473    fn try_from(op: PureOp) -> Result<Self, ()> {
474        let wk = &super::SpvSpecWithExtras::get().well_known;
475
476        let (opcode, imms) = match op {
477            PureOp::BitCast => (wk.OpBitcast, iter::empty().collect()),
478            PureOp::CompositeExtract { elem_idx } => {
479                (wk.OpCompositeExtract, iter::once(elem_idx).collect())
480            }
481
482            // HACK(eddyb) this is the only reason this is `TryFrom` not `From`.
483            PureOp::IntToBool => return Err(()),
484        };
485        Ok(Self { opcode, imms })
486    }
487}

489/// Potentially-reducible application of a `PureOp` (`op`) to `input`.
490#[derive(Copy, Clone, PartialEq, Eq, Hash)]
491struct Reducible<V = Value> {
492    op: PureOp,
493    output_type: Type,
494    input: V,
495}
496
497impl<V> Reducible<V> {
498    fn with_input<V2>(self, new_input: V2) -> Reducible<V2> {
499        Reducible {
500            op: self.op,
501            output_type: self.output_type,
502            input: new_input,
503        }
504    }
505}

507// FIXME(eddyb) instead of taking a `&Context`, could `Reducible` hold a `DataInstForm`?
508impl TryFrom<(&Context, &DataInstDef)> for Reducible {
509    type Error = ();
510    fn try_from((cx, inst_def): (&Context, &DataInstDef)) -> Result<Self, ()> {
511        let inst_form_def = &cx[inst_def.form];
512        if let DataInstKind::SpvInst(spv_inst) = &inst_form_def.kind {
513            let op = PureOp::try_from(spv_inst)?;
514            let output_type = inst_form_def.output_type.unwrap();
515            if let [input] = inst_def.inputs[..] {
516                return Ok(Self {
517                    op,
518                    output_type,
519                    input,
520                });
521            }
522        }
523        Err(())
524    }
525}

527impl Reducible {
528    // HACK(eddyb) `IntToBool` is the only reason this can return `None`.
529    fn try_into_inst(self, cx: &Context) -> Option<DataInstDef> {
530        let Self {
531            op,
532            output_type,
533            input,
534        } = self;
535        Some(DataInstDef {
536            attrs: Default::default(),
537            form: cx.intern(DataInstFormDef {
538                kind: DataInstKind::SpvInst(op.try_into().ok()?),
539                output_type: Some(output_type),
540            }),
541            inputs: iter::once(input).collect(),
542        })
543    }
544}

546/// Returns `Some(lowest32)` iff `imms` contains one *logical* SPIR-V immediate
547/// representing a (little-endian) integer which truncates (if wider than 32 bits)
548/// to `lowest32`, losslessly (i.e. the rest of the bits are all zeros).
549//
550// FIXME(eddyb) move this into some kind of utility/common helpers place.
551fn spv_imm_checked_trunc32(imms: &[spv::Imm]) -> Option<u32> {
552    match imms {
553        &[spv::Imm::Short(_, lowest32)] | &[spv::Imm::LongStart(_, lowest32), ..]
554            if imms[1..]
555                .iter()
556                .all(|imm| matches!(imm, spv::Imm::LongCont(_, 0))) =>
557        {
558            Some(lowest32)
559        }
560        _ => None,
561    }
562}

564impl Reducible<Const> {
565    // FIXME(eddyb) in theory this should always return `Some`.
566    fn try_reduce_const(&self, cx: &Context) -> Option<Const> {
567        let wk = &super::SpvSpecWithExtras::get().well_known;
568
569        let ct_def = &cx[self.input];
570        match (self.op, &ct_def.kind) {
571            (
572                _,
573                ConstKind::SpvInst {
574                    spv_inst_and_const_inputs,
575                },
576            ) if spv_inst_and_const_inputs.0.opcode == wk.OpUndef => Some(cx.intern(ConstDef {
577                attrs: ct_def.attrs,
578                ty: self.output_type,
579                kind: ct_def.kind.clone(),
580            })),
581
582            (
583                PureOp::BitCast,
584                ConstKind::SpvInst {
585                    spv_inst_and_const_inputs,
586                },
587            ) if spv_inst_and_const_inputs.0.opcode == wk.OpConstant => {
588                // `OpTypeInt`/`OpTypeFloat` bit width.
589                let scalar_width = |ty: Type| match &cx[ty].kind {
590                    TypeKind::SpvInst { spv_inst, .. }
591                        if [wk.OpTypeInt, wk.OpTypeFloat].contains(&spv_inst.opcode) =>
592                    {
593                        Some(spv_inst.imms[0])
594                    }
595                    _ => None,
596                };
597
598                match (scalar_width(ct_def.ty), scalar_width(self.output_type)) {
599                    (Some(from), Some(to)) if from == to => Some(cx.intern(ConstDef {
600                        attrs: ct_def.attrs,
601                        ty: self.output_type,
602                        kind: ct_def.kind.clone(),
603                    })),
604                    _ => None,
605                }
606            }
607
608            (
609                PureOp::CompositeExtract {
610                    elem_idx: spv::Imm::Short(_, elem_idx),
611                },
612                ConstKind::SpvInst {
613                    spv_inst_and_const_inputs,
614                },
615            ) if spv_inst_and_const_inputs.0.opcode == wk.OpConstantComposite => {
616                let (_spv_inst, const_inputs) = &**spv_inst_and_const_inputs;
617                Some(const_inputs[elem_idx as usize])
618            }
619
620            (
621                PureOp::IntToBool,
622                ConstKind::SpvInst {
623                    spv_inst_and_const_inputs,
624                },
625            ) if spv_inst_and_const_inputs.0.opcode == wk.OpConstant => {
626                let (spv_inst, _const_inputs) = &**spv_inst_and_const_inputs;
627                let bool_const_op = match spv_imm_checked_trunc32(&spv_inst.imms[..]) {
628                    Some(0) => wk.OpConstantFalse,
629                    Some(1) => wk.OpConstantTrue,
630                    _ => return None,
631                };
632                Some(cx.intern(ConstDef {
633                    attrs: Default::default(),
634                    ty: self.output_type,
635                    kind: ConstKind::SpvInst {
636                        spv_inst_and_const_inputs: Rc::new((
637                            bool_const_op.into(),
638                            iter::empty().collect(),
639                        )),
640                    },
641                }))
642            }
643
644            _ => None,
645        }
646    }
647}

649/// Outcome of a single step of a reduction (which may require more steps).
650enum ReductionStep {
651    Complete(Value),
652    Partial(Reducible),
653}
654
655impl Reducible<&DataInstDef> {
656    // FIXME(eddyb) force the input to actually be itself some kind of pure op.
657    fn try_reduce_output_of_data_inst(&self, cx: &Context) -> Option<ReductionStep> {
658        let wk = &super::SpvSpecWithExtras::get().well_known;
659
660        let input_inst_def = self.input;
661        if let DataInstKind::SpvInst(input_spv_inst) = &cx[input_inst_def.form].kind {
662            // NOTE(eddyb) do not destroy information left in e.g. comments.
663            #[allow(clippy::match_same_arms)]
664            match self.op {
665                PureOp::BitCast => {
666                    // FIXME(eddyb) reduce chains of bitcasts.
667                }
668
669                PureOp::CompositeExtract { elem_idx } => {
670                    if input_spv_inst.opcode == wk.OpCompositeInsert
671                        && input_spv_inst.imms.len() == 1
672                    {
673                        let new_elem = input_inst_def.inputs[0];
674                        let prev_composite = input_inst_def.inputs[1];
675                        return Some(if input_spv_inst.imms[0] == elem_idx {
676                            ReductionStep::Complete(new_elem)
677                        } else {
678                            ReductionStep::Partial(self.with_input(prev_composite))
679                        });
680                    }
681                }
682
683                PureOp::IntToBool => {
684                    // FIXME(eddyb) look into what instructions might end up
685                    // being used to transform booleans into integers.
686                }
687            }
688        }
689
690        None
691    }
692}

694impl Reducible {
695    // FIXME(eddyb) make this into some kind of local `ReduceCx` method.
696    fn try_reduce(
697        mut self,
698        cx: &Context,
699        // FIXME(eddyb) come up with a better convention for this!
700        func: FuncAtMut<'_, ()>,
701
702        value_replacements: &FxHashMap<Value, Value>,
703
704        parent_map: &ParentMap,
705
706        cache: &mut FxHashMap<Self, Option<Value>>,
707    ) -> Option<Value> {
708        // FIXME(eddyb) should we care about the cache *before* this loop below?
709
710        // HACK(eddyb) eagerly apply `value_replacements`.
711        // FIXME(eddyb) this could do the union-find trick of shortening chains
712        // the first time they're encountered, but also, if this process was more
713        // "demand-driven" (recursing into use->def, instead of processing defs),
714        // it might not require any of this complication.
715        while let Some(&replacement) = value_replacements.get(&self.input) {
716            self.input = replacement;
717        }
718
719        if let Some(&cached) = cache.get(&self) {
720            return cached;
721        }
722
723        let result = self.try_reduce_uncached(cx, func, value_replacements, parent_map, cache);
724
725        cache.insert(self, result);
726
727        result
728    }
729
730    // FIXME(eddyb) make this into some kind of local `ReduceCx` method.
731    fn try_reduce_uncached(
732        self,
733        cx: &Context,
734        // FIXME(eddyb) come up with a better convention for this!
735        mut func: FuncAtMut<'_, ()>,
736
737        value_replacements: &FxHashMap<Value, Value>,
738
739        parent_map: &ParentMap,
740
741        cache: &mut FxHashMap<Self, Option<Value>>,
742    ) -> Option<Value> {
743        match self.input {
744            Value::Const(ct) => self.with_input(ct).try_reduce_const(cx).map(Value::Const),
745            Value::ControlRegionInput {
746                region,
747                input_idx: state_idx,
748            } => {
749                let loop_node = *parent_map.control_region_parent.get(region)?;
750                // HACK(eddyb) this can't be a closure due to lifetime elision.
751                fn loop_initial_states(
752                    func_at_loop_node: FuncAtMut<'_, ControlNode>,
753                ) -> &mut SmallVec<[Value; 2]> {
754                    match &mut func_at_loop_node.def().kind {
755                        ControlNodeKind::Loop { initial_inputs, .. } => initial_inputs,
756                        _ => unreachable!(),
757                    }
758                }
759
760                let input_from_initial_state =
761                    loop_initial_states(func.reborrow().at(loop_node))[state_idx as usize];
762                let input_from_updated_state =
763                    func.reborrow().at(region).def().outputs[state_idx as usize];
764
765                let output_from_initial_state = self
766                    .with_input(input_from_initial_state)
767                    .try_reduce(cx, func.reborrow(), value_replacements, parent_map, cache)?;
768                // HACK(eddyb) this is here because it can fail, see the comment
769                // on `output_from_updated_state` for what's actually going on.
770                let output_from_updated_state_inst = self
771                    .with_input(input_from_updated_state)
772                    .try_into_inst(cx)?;
773
774                // Now that the reduction succeeded for the initial state,
775                // we can proceed with augmenting the loop with the extra state.
776                loop_initial_states(func.reborrow().at(loop_node)).push(output_from_initial_state);
777
778                let loop_state_decls = &mut func.reborrow().at(region).def().inputs;
779                let new_loop_state_idx = u32::try_from(loop_state_decls.len()).unwrap();
780                loop_state_decls.push(ControlRegionInputDecl {
781                    attrs: Default::default(),
782                    ty: self.output_type,
783                });
784
785                // HACK(eddyb) generating the instruction wholesale again is not
786                // the most efficient way to go about this, but avoiding getting
787                // stuck in a loop while processing a loop is also important.
788                //
789                // FIXME(eddyb) attempt to replace this with early-inserting in
790                // `cache` *then* returning.
791                let output_from_updated_state = func
792                    .data_insts
793                    .define(cx, output_from_updated_state_inst.into());
794                func.reborrow()
795                    .at(region)
796                    .def()
797                    .outputs
798                    .push(Value::DataInstOutput(output_from_updated_state));
799
800                // FIXME(eddyb) move this into some kind of utility/common helpers.
801                let loop_body_last_block = func
802                    .reborrow()
803                    .at(region)
804                    .def()
805                    .children
806                    .iter()
807                    .last
808                    .filter(|&node| {
809                        matches!(
810                            func.reborrow().at(node).def().kind,
811                            ControlNodeKind::Block { .. }
812                        )
813                    })
814                    .unwrap_or_else(|| {
815                        let new_block = func.control_nodes.define(
816                            cx,
817                            ControlNodeDef {
818                                kind: ControlNodeKind::Block {
819                                    insts: Default::default(),
820                                },
821                                outputs: Default::default(),
822                            }
823                            .into(),
824                        );
825                        func.control_regions[region]
826                            .children
827                            .insert_last(new_block, func.control_nodes);
828                        new_block
829                    });
830                match &mut func.control_nodes[loop_body_last_block].kind {
831                    ControlNodeKind::Block { insts } => {
832                        insts.insert_last(output_from_updated_state, func.data_insts);
833                    }
834                    _ => unreachable!(),
835                }
836
837                Some(Value::ControlRegionInput {
838                    region,
839                    input_idx: new_loop_state_idx,
840                })
841            }
842            Value::ControlNodeOutput {
843                control_node,
844                output_idx,
845            } => {
846                let cases = match &func.reborrow().at(control_node).def().kind {
847                    ControlNodeKind::Select { cases, .. } => cases,
848                    // NOTE(eddyb) only `Select`s can have outputs right now.
849                    _ => unreachable!(),
850                };
851
852                // FIXME(eddyb) remove all the cloning and undo additions of new
853                // outputs "upstream", if they end up unused (or let DCE do it?).
854                let cases = cases.clone();
855                let per_case_new_output: SmallVec<[_; 2]> = cases
856                    .iter()
857                    .map(|&case| {
858                        let per_case_input =
859                            func.reborrow().at(case).def().outputs[output_idx as usize];
860                        self.with_input(per_case_input).try_reduce(
861                            cx,
862                            func.reborrow(),
863                            value_replacements,
864                            parent_map,
865                            cache,
866                        )
867                    })
868                    .collect::<Option<_>>()?;
869
870                // Try to avoid introducing a new output, by reducing the merge
871                // of the per-case output values to a single value, if possible.
872                let (kind, scrutinee) = match &func.reborrow().at(control_node).def().kind {
873                    ControlNodeKind::Select {
874                        kind, scrutinee, ..
875                    } => (kind, *scrutinee),
876                    _ => unreachable!(),
877                };
878                if let Some(v) = try_reduce_select(
879                    cx,
880                    parent_map,
881                    control_node,
882                    kind,
883                    scrutinee,
884                    per_case_new_output.iter().copied(),
885                ) {
886                    return Some(v);
887                }
888
889                // Merge the per-case output values into a new output.
890                let control_node_output_decls = &mut func.reborrow().at(control_node).def().outputs;
891                let new_output_idx = u32::try_from(control_node_output_decls.len()).unwrap();
892                control_node_output_decls.push(ControlNodeOutputDecl {
893                    attrs: Default::default(),
894                    ty: self.output_type,
895                });
896                for (&case, new_output) in cases.iter().zip(per_case_new_output) {
897                    let per_case_outputs = &mut func.reborrow().at(case).def().outputs;
898                    assert_eq!(per_case_outputs.len(), new_output_idx as usize);
899                    per_case_outputs.push(new_output);
900                }
901                Some(Value::ControlNodeOutput {
902                    control_node,
903                    output_idx: new_output_idx,
904                })
905            }
906            Value::DataInstOutput(inst) => {
907                let inst_def = &*func.reborrow().at(inst).def();
908                match self
909                    .with_input(inst_def)
910                    .try_reduce_output_of_data_inst(cx)?
911                {
912                    ReductionStep::Complete(v) => Some(v),
913                    // FIXME(eddyb) actually use a loop instead of recursing here.
914                    ReductionStep::Partial(redu) => {
915                        redu.try_reduce(cx, func, value_replacements, parent_map, cache)
916                    }
917                }
918            }
919        }
920    }
921}
