10pub fn validate(module: &mut Module) {
11    let spv_spec = super::SpvSpecWithExtras::get();
12    let wk = &spv_spec.well_known;
13
14    let mut validator = Validator {
15        cx: &module.cx(),
16        wk,
17        spv_spec_caps: match wk.Capability.def() {
18            spv::spec::OperandKindDef::ValueEnum { variants } => variants,
19            _ => unreachable!(),
20        },
21
22        module_spv_dialect: match &module.dialect {
23            ModuleDialect::Spv(dialect) => dialect,
24        },
25
26        transformed_types: FxHashMap::default(),
27        transformed_consts: FxHashMap::default(),
28        transformed_data_inst_forms: FxHashMap::default(),
29        seen_global_vars: FxHashSet::default(),
30        global_var_queue: VecDeque::new(),
31        seen_funcs: FxHashSet::default(),
32        func_queue: VecDeque::new(),
33    };
34
35    // Seed the queues starting from the module exports.
36    for exportee in module.exports.values_mut() {
37        exportee
38            .inner_transform_with(&mut validator)
39            .apply_to(exportee);
40    }
41
42    // Process the queues until they're all empty.
43    while !validator.global_var_queue.is_empty() || !validator.func_queue.is_empty() {
44        while let Some(gv) = validator.global_var_queue.pop_front() {
45            validator.in_place_transform_global_var_decl(&mut module.global_vars[gv]);
46        }
47        while let Some(func) = validator.func_queue.pop_front() {
48            validator.in_place_transform_func_decl(&mut module.funcs[func]);
49        }
50    }
51}

53struct Validator<'a> {
54    cx: &'a Context,
55    wk: &'static super::SpvWellKnownWithExtras,
56    spv_spec_caps: &'static spv::spec::indexed::NamedIdxMap<
57        u16,
58        spv::spec::Enumerant,
59        spv::spec::indexed::KhrSegmented,
60    >,
61
62    module_spv_dialect: &'a spv::Dialect,
63
64    // FIXME(eddyb) build some automation to avoid ever repeating these.
65    transformed_types: FxHashMap<Type, Transformed<Type>>,
66    transformed_consts: FxHashMap<Const, Transformed<Const>>,
67    transformed_data_inst_forms: FxHashMap<DataInstForm, Transformed<DataInstForm>>,
68    seen_global_vars: FxHashSet<GlobalVar>,
69    global_var_queue: VecDeque<GlobalVar>,
70    seen_funcs: FxHashSet<Func>,
71    func_queue: VecDeque<Func>,
72}
73
74impl Transformer for Validator<'_> {
75    // FIXME(eddyb) build some automation to avoid ever repeating these.
76    fn transform_type_use(&mut self, ty: Type) -> Transformed<Type> {
77        if let Some(&cached) = self.transformed_types.get(&ty) {
78            return cached;
79        }
80        let transformed = self
81            .transform_type_def(&self.cx[ty])
82            .map(|ty_def| self.cx.intern(ty_def));
83        self.transformed_types.insert(ty, transformed);
84        transformed
85    }
86    fn transform_const_use(&mut self, ct: Const) -> Transformed<Const> {
87        if let Some(&cached) = self.transformed_consts.get(&ct) {
88            return cached;
89        }
90        let transformed = self
91            .transform_const_def(&self.cx[ct])
92            .map(|ct_def| self.cx.intern(ct_def));
93        self.transformed_consts.insert(ct, transformed);
94        transformed
95    }
96    fn transform_data_inst_form_use(
97        &mut self,
98        data_inst_form: DataInstForm,
99    ) -> Transformed<DataInstForm> {
100        if let Some(&cached) = self.transformed_data_inst_forms.get(&data_inst_form) {
101            return cached;
102        }
103        let transformed = self
104            .transform_data_inst_form_def(&self.cx[data_inst_form])
105            .map(|data_inst_form_def| self.cx.intern(data_inst_form_def));
106        self.transformed_data_inst_forms
107            .insert(data_inst_form, transformed);
108        transformed
109    }
110
111    fn transform_global_var_use(&mut self, gv: GlobalVar) -> Transformed<GlobalVar> {
112        if self.seen_global_vars.insert(gv) {
113            self.global_var_queue.push_back(gv);
114        }
115        Transformed::Unchanged
116    }
117    fn transform_func_use(&mut self, func: Func) -> Transformed<Func> {
118        if self.seen_funcs.insert(func) {
119            self.func_queue.push_back(func);
120        }
121        Transformed::Unchanged
122    }
123
124    // NOTE(eddyb) above methods are plumbing, validation methods are below.
125
126    fn transform_type_def(&mut self, ty_def: &TypeDef) -> Transformed<TypeDef> {
127        let valid = match &ty_def.kind {
128            TypeKind::SpvInst {
129                spv_inst,
130                type_and_const_inputs: _,
131            } => self.validate_spv_inst(spv_inst),
132
133            TypeKind::QPtr | TypeKind::SpvStringLiteralForExtInst => Ok(()),
134        };
135        let transformed = ty_def.inner_transform_with(self);
136        match valid {
137            Ok(()) => transformed,
138            Err(diag) => {
139                let mut ty_def = TypeDef {
140                    attrs: ty_def.attrs,
141                    kind: ty_def.kind.clone(),
142                };
143                transformed.apply_to(&mut ty_def);
144                ty_def.attrs.push_diag(self.cx, diag);
145                Transformed::Changed(ty_def)
146            }
147        }
148    }
149
150    fn transform_const_def(&mut self, ct_def: &ConstDef) -> Transformed<ConstDef> {
151        let valid = match &ct_def.kind {
152            ConstKind::SpvInst {
153                spv_inst_and_const_inputs,
154            } => {
155                let (spv_inst, _const_inputs) = &**spv_inst_and_const_inputs;
156                self.validate_spv_inst(spv_inst)
157            }
158
159            ConstKind::PtrToGlobalVar(_) | ConstKind::SpvStringLiteralForExtInst(_) => Ok(()),
160        };
161        let transformed = ct_def.inner_transform_with(self);
162        match valid {
163            Ok(()) => transformed,
164            Err(diag) => {
165                let mut ct_def = ConstDef {
166                    attrs: ct_def.attrs,
167                    ty: ct_def.ty,
168                    kind: ct_def.kind.clone(),
169                };
170                transformed.apply_to(&mut ct_def);
171                ct_def.attrs.push_diag(self.cx, diag);
172                Transformed::Changed(ct_def)
173            }
174        }
175    }
176
177    fn in_place_transform_data_inst_def(&mut self, mut func_at_data_inst: FuncAtMut<'_, DataInst>) {
178        func_at_data_inst
179            .reborrow()
180            .inner_in_place_transform_with(self);
181
182        let inst_def = func_at_data_inst.def();
183        let valid = match &self.cx[inst_def.form].kind {
184            DataInstKind::SpvInst(spv_inst) => self.validate_spv_inst(spv_inst),
185
186            DataInstKind::FuncCall(_) | DataInstKind::QPtr(_) | DataInstKind::SpvExtInst { .. } => {
187                Ok(())
188            }
189        };
190        if let Err(diag) = valid {
191            inst_def.attrs.push_diag(self.cx, diag);
192        }
193    }
194}

196impl Validator<'_> {
197    fn require_spv_exts_caps<'a>(
198        &self,
199        describe: impl FnOnce() -> String,
200        exts_providing: impl ExactSizeIterator<Item = &'a str> + Clone,
201        caps_enabling: impl ExactSizeIterator<Item = u32> + Clone,
202    ) -> Result<(), Diag> {
203        // FIXME(eddyb) find a consistent style between all the error messages.
204        let provided_by_core_spv_or_ext = exts_providing.len() == 0
205            || exts_providing.clone().any(|ext| {
206                self.module_spv_dialect.extensions.contains(ext)
207                    || min_spv_version_implying_ext(ext).is_some_and(|min_version| {
208                        let module_version = {
209                            let d = &self.module_spv_dialect;
210                            (d.version_major, d.version_minor)
211                        };
212                        module_version >= min_version
213                    })
214            });
215        if !provided_by_core_spv_or_ext {
216            let exts = exts_providing
217                .map(|ext| format!("`{ext}`"))
218                .collect::<Vec<_>>()
219                .join(", ");
220            return Err(Diag::err([
221                describe().into(),
222                format!(" requires one of these extensions: {exts}",).into(),
223            ]));
224        }
225
226        let enabled_by_default_or_cap = caps_enabling.len() == 0
227            || caps_enabling
228                .clone()
229                .any(|cap| self.module_spv_dialect.capabilities.contains(&cap))
230            || {
231                // HACK(eddyb) this is an expensive fallback, and should be
232                // precomputed ahead of time, but `spirt::spv::spec` exposing
233                // the necessary information is the bigger issue and should be
234                // solved first, before optimizing any of this.
235                let mut elabored_module_caps: FxIndexSet<_> = self
236                    .module_spv_dialect
237                    .capabilities
238                    .iter()
239                    .copied()
240                    .collect();
241                let mut i = 0;
242                while i < elabored_module_caps.len() {
243                    let cap = elabored_module_caps[i];
244                    if let Some(cap) = rspirv::spirv::Capability::from_u32(cap) {
245                        elabored_module_caps.extend(
246                            rspirv::dr::Operand::from(cap)
247                                .required_capabilities()
248                                .into_iter()
249                                .map(|cap| cap as u32),
250                        );
251                    }
252                    i += 1;
253                }
254                caps_enabling
255                    .clone()
256                    .any(|cap| elabored_module_caps.contains(&cap))
257            };
258        if !enabled_by_default_or_cap {
259            let caps = caps_enabling
260                .map(|cap| {
261                    let cap_name = u16::try_from(cap)
262                        .ok()
263                        .and_then(|cap| Some(self.spv_spec_caps.get_named(cap)?.0));
264                    match cap_name {
265                        Some(name) => format!("`{name}`"),
266                        None => format!("<unknown Capability 0x{cap:04x}>"),
267                    }
268                })
269                .collect::<Vec<_>>()
270                .join(", ");
271            return Err(Diag::err([
272                describe().into(),
273                format!(" requires one of these capabilities: {caps}",).into(),
274            ]));
275        }
276        Ok(())
277    }

279    fn validate_spv_inst(&self, spv_inst: &spv::Inst) -> Result<(), Diag> {
280        // FIXME(eddyb) make this information available through `spirt::spv::spec`.
281        let (exts_providing_inst, caps_enabling_insts) = {
282            let inst_def =
283                rspirv::grammar::CoreInstructionTable::lookup_opcode(spv_inst.opcode.as_u16())
284                    .unwrap();
285            (
286                inst_def.extensions.iter().copied(),
287                inst_def.capabilities.iter().map(|&cap| cap as u32),
288            )
289        };
290        self.require_spv_exts_caps(
291            || format!("SPIR-V `{}` instruction", spv_inst.opcode.name()),
292            exts_providing_inst,
293            caps_enabling_insts,
294        )?;
295
296        // HACK(eddyb) even if this seems wasteful in its allocation of
297        // strings, they should only happen once each per module, and
298        // also it wouldn't be hard to switch to some "small str" crate.
299        let int_or_float = |type_name: &str, cap_name: &str| {
300            // FIXME(eddyb) find a consistent style between all the error messages
301            // (mentioning `OpCapability` seems unfortunate, for example).
302            match self.spv_spec_caps.lookup(cap_name).map(u32::from) {
303                None => Err(format!("`{type_name}` type unsupported in SPIR-V")),
304                Some(cap) if !self.module_spv_dialect.capabilities.contains(&cap) => Err(format!(
305                    "`{type_name}` type used without `OpCapability {cap_name}`"
306                )),
307                Some(_) => Ok(()),
308            }
309            .map_err(|msg| Diag::err([msg.into()]))
310        };
311        match spv_inst.imms[..] {
312            [spv::Imm::Short(_, width), spv::Imm::Short(_, signedness)]
313                if spv_inst.opcode == self.wk.OpTypeInt && width != 32 =>
314            {
315                let signed = signedness != 0;
316                int_or_float(
317                    &format!("{}{width}", if signed { "i" } else { "u" }),
318                    &format!("Int{width}"),
319                )?;
320            }
321            [spv::Imm::Short(_, width)]
322                if spv_inst.opcode == self.wk.OpTypeFloat && width != 32 =>
323            {
324                int_or_float(&format!("f{width}"), &format!("Float{width}"))?;
325            }
326            _ => {}
327        }
328
329        // FIXME(eddyb) implement this after exposing enough of the information
330        // via `spirt::spv::spec` (will likely need some way to efficiently store
331        // `(extensions, capabilities)`, e.g. ad-hoc interning into one integer,
332        // because that will have to be added to every single e.g. `Enumerant`).
333        for &imm in &spv_inst.imms {
334            // HACK(eddyb) two simple cases being handled via `rspirv` as a demo.
335            let check_enum_via_rspirv = |enum_kind: spv::spec::OperandKind,
336                                         // HACK(eddyb) bypassing rustfmt failure mode.
337                                         #[allow(unused_parens)] as_rspirv_operand: (
338                                             fn(u32) -> Option<rspirv::dr::Operand>
339                                         )| {
340                if let spv::Imm::Short(kind, imm) = imm
341                    && kind == enum_kind
342                    && let Some(operand) = as_rspirv_operand(imm)
343                {
344                    self.require_spv_exts_caps(
345                        || {
346                            let (enum_name, enum_def) = kind.name_and_def();
347                            let enumerant_name = match enum_def {
348                                spv::spec::OperandKindDef::ValueEnum { variants } => {
349                                    u16::try_from(imm)
350                                        .ok()
351                                        .and_then(|imm| Some(variants.get_named(imm)?.0))
352                                }
353                                _ => None,
354                            };
355                            match enumerant_name {
356                                Some(name) => format!("SPIR-V `{enum_name}.{name}` operand"),
357                                None => format!("<unknown {enum_name} 0x{imm:04x}>"),
358                            }
359                        },
360                        operand.required_extensions().iter().copied(),
361                        operand
362                            .required_capabilities()
363                            .iter()
364                            .map(|&cap| cap as u32),
365                    )?;
366                }
367
368                Ok(())
369            };
370            check_enum_via_rspirv(self.wk.ImageFormat, |imm| {
371                Some(rspirv::spirv::ImageFormat::from_u32(imm)?.into())
372            })?;
373            check_enum_via_rspirv(self.wk.StorageClass, |imm| {
374                Some(rspirv::spirv::StorageClass::from_u32(imm)?.into())
375            })?;
376        }
377
378        Ok(())
379    }
380}

382// HACK(eddyb) due to neither SPIR-T, nor `rspirv`, exposing the SPIR-V version
383// which started provided an instruction/enumerator/etc. (i.e. supplanting the
384// need for an extension), this "first SPIR-V version incorporating an extension"
385// approximation was obtained by running (inside `spirt 0.4.0`'s source tree):
386//
387// jq -r '[(.instructions[], (.operand_kinds[] | (.enumerants//[])[])) \
388//   | {e:.extensions|arrays|select(length>0)|unique,v:.version}] as $all \
389//   | [$all[].e[]] | unique | map(\
390//     . as $e | $all | map(select(.e|contains([$e])) | (.v//"None")) \
391//       | max | select(. != "None") | {e:$e,v:(split(".") | map(tonumber))}\
392//   ) \
393//   | group_by(.v) | map({e:map(@json"\(.e)")|join("\n| "), v:.[0].v})[] \
394//   | "\(.e) => (\(.v[0]), \(.v[1])),"' \
395//   khronos-spec/SPIRV-Headers/include/spirv/unified1/spirv.core.grammar.json
396//
397fn min_spv_version_implying_ext(ext: &str) -> Option<(u8, u8)> {
398    Some(match ext {
399        "SPV_KHR_16bit_storage"
400        | "SPV_KHR_device_group"
401        | "SPV_KHR_multiview"
402        | "SPV_KHR_shader_draw_parameters"
403        | "SPV_KHR_storage_buffer_storage_class"
404        | "SPV_KHR_variable_pointers" => (1, 3),
405        "SPV_GOOGLE_decorate_string" | "SPV_KHR_no_integer_wrap_decoration" => (1, 4),
406        "SPV_EXT_descriptor_indexing"
407        | "SPV_EXT_physical_storage_buffer"
408        | "SPV_KHR_8bit_storage"
409        | "SPV_KHR_physical_storage_buffer"
410        | "SPV_KHR_vulkan_memory_model" => (1, 5),
411        "SPV_KHR_integer_dot_product" | "SPV_KHR_terminate_invocation" => (1, 6),
412        _ => return None,
413    })
