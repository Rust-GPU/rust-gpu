81fn id(header: &mut ModuleHeader) -> Word {
82    let result = header.bound;
83    header.bound += 1;
84    result

87fn apply_rewrite_rules<'a>(
88    rewrite_rules: &FxHashMap<Word, Word>,
89    blocks: impl IntoIterator<Item = &'a mut Block>,
90) {
91    let all_ids_mut = blocks
92        .into_iter()
93        .flat_map(|b| b.label.iter_mut().chain(b.instructions.iter_mut()))
94        .flat_map(|inst| {
95            inst.result_id
96                .iter_mut()
97                .chain(inst.result_type.iter_mut())
98                .chain(
99                    inst.operands
100                        .iter_mut()
101                        .filter_map(|op| op.id_ref_any_mut()),
102                )
103        });
104    for id in all_ids_mut {
105        if let Some(&rewrite) = rewrite_rules.get(id) {
106            *id = rewrite;
107        }
108    }
109}

111fn get_names(module: &Module) -> FxHashMap<Word, &str> {
112    let entry_names = module
113        .entry_points
114        .iter()
115        .filter(|i| i.class.opcode == Op::EntryPoint)
116        .map(|i| {
117            (
118                i.operands[1].unwrap_id_ref(),
119                i.operands[2].unwrap_literal_string(),
120            )
121        });
122    let debug_names = module
123        .debug_names
124        .iter()
125        .filter(|i| i.class.opcode == Op::Name)
126        .map(|i| {
127            (
128                i.operands[0].unwrap_id_ref(),
129                i.operands[1].unwrap_literal_string(),
130            )
131        });
132    // items later on take priority
133    entry_names.chain(debug_names).collect()
134}

136fn get_name<'a>(names: &FxHashMap<Word, &'a str>, id: Word) -> Cow<'a, str> {
137    names.get(&id).map_or_else(
138        || Cow::Owned(format!("Unnamed function ID %{id}")),
139        |&s| Cow::Borrowed(s),
140    )
141}

168pub fn link(
169    sess: &Session,
170    mut inputs: Vec<Module>,
171    opts: &Options,
172    outputs: &OutputFilenames,
173    disambiguated_crate_name_for_dumps: &OsStr,
174) -> Result<LinkResult> {
175    // HACK(eddyb) this is defined here to allow SPIR-T pretty-printing to apply
176    // to SPIR-V being dumped, outside of e.g. `--dump-spirt-passes`.
177    // FIXME(eddyb) this isn't used everywhere, sadly - to find those, search
178    // elsewhere for `.assemble()` and/or `spirv_tools::binary::from_binary`.
179    let spv_module_to_spv_words_and_spirt_module = |spv_module: &Module| {
180        let spv_words;
181        let spv_bytes = {
182            let _timer = sess.timer("assemble-to-spv_bytes-for-spirt");
183            spv_words = spv_module.assemble();
184            // FIXME(eddyb) this is wastefully cloning all the bytes, but also
185            // `spirt::Module` should have a method that takes `Vec<u32>`.
186            spirv_tools::binary::from_binary(&spv_words).to_vec()
187        };
188
189        // FIXME(eddyb) should've really been "spirt::Module::lower_from_spv_bytes".
190        let lower_from_spv_timer = sess.timer("spirt::Module::lower_from_spv_file");
191        let cx = std::rc::Rc::new(spirt::Context::new());
192        crate::custom_insts::register_to_spirt_context(&cx);
193        (
194            spv_words,
195            spirt::Module::lower_from_spv_bytes(cx, spv_bytes),
196            // HACK(eddyb) this is only returned for `SpirtDumpGuard`.
197            lower_from_spv_timer,
198        )
199    };
200
201    // FIXME(eddyb) deduplicate with `SpirtDumpGuard`.
202    let dump_spv_and_spirt = |spv_module: &Module, dump_file_path_stem: PathBuf| {
203        let (spv_words, spirt_module_or_err, _) =
204            spv_module_to_spv_words_and_spirt_module(spv_module);
205        std::fs::write(
206            dump_file_path_stem.with_extension("spv"),
207            spirv_tools::binary::from_binary(&spv_words),
208        )
209        .unwrap();
210
211        // FIXME(eddyb) reify SPIR-V -> SPIR-T errors so they're easier to debug.
212        if let Ok(mut module) = spirt_module_or_err {
213            // HACK(eddyb) avoid pretty-printing massive amounts of unused SPIR-T.
214            spirt::passes::link::minimize_exports(&mut module, |export_key| {
215                matches!(export_key, spirt::ExportKey::SpvEntryPoint { .. })
216            });
217
218            opts.spirt_cleanup_for_dumping(&mut module);
219
220            let pretty = spirt::print::Plan::for_module(&module).pretty_print();
221
222            // FIXME(eddyb) don't allocate whole `String`s here.
223            std::fs::write(
224                dump_file_path_stem.with_extension("spirt"),
225                pretty.to_string(),
226            )
227            .unwrap();
228            std::fs::write(
229                dump_file_path_stem.with_extension("spirt.html"),
230                pretty
231                    .render_to_html()
232                    .with_dark_mode_support()
233                    .to_html_doc(),
234            )
235            .unwrap();
236        }
237    };
238
239    let mut output = {
240        let _timer = sess.timer("link_merge");
241        // shift all the ids
242        let mut bound = inputs[0].header.as_ref().unwrap().bound - 1;
243        let version = inputs[0].header.as_ref().unwrap().version();
244
245        for module in inputs.iter_mut().skip(1) {
246            simple_passes::shift_ids(module, bound);
247            bound += module.header.as_ref().unwrap().bound - 1;
248            let this_version = module.header.as_ref().unwrap().version();
249            if version != this_version {
250                return Err(sess.dcx().err(format!(
251                    "cannot link two modules with different SPIR-V versions: v{}.{} and v{}.{}",
252                    version.0, version.1, this_version.0, this_version.1
253                )));
254            }
255        }
256
257        // merge the binaries
258        let mut output = crate::link::with_rspirv_loader(|loader| {
259            for module in inputs {
260                for inst in module.all_inst_iter() {
261                    use rspirv::binary::ParseAction;
262                    match loader.consume_instruction(inst.clone()) {
263                        ParseAction::Continue => {}
264                        ParseAction::Stop => unreachable!(),
265                        ParseAction::Error(err) => return Err(err),
266                    }
267                }
268            }
269            Ok(())
270        })
271        .unwrap();
272
273        let mut header = ModuleHeader::new(bound + 1);
274        header.set_version(version.0, version.1);
275        header.generator = 0x001B_0000;
276        output.header = Some(header);
277        output
278    };
279
280    if let Some(dir) = &opts.dump_post_merge {
281        dump_spv_and_spirt(&output, dir.join(disambiguated_crate_name_for_dumps));
282    }
283
284    // remove duplicates (https://github.com/KhronosGroup/SPIRV-Tools/blob/e7866de4b1dc2a7e8672867caeb0bdca49f458d3/source/opt/remove_duplicates_pass.cpp)
285    {
286        let _timer = sess.timer("link_remove_duplicates");
287        duplicates::remove_duplicate_extensions(&mut output);
288        duplicates::remove_duplicate_capabilities(&mut output);
289        duplicates::remove_duplicate_ext_inst_imports(&mut output);
290        duplicates::remove_duplicate_types(&mut output);
291        // jb-todo: strip identical OpDecoration / OpDecorationGroups
292    }
293
294    // find import / export pairs
295    {
296        let _timer = sess.timer("link_find_pairs");
297        import_export_link::run(opts, sess, &mut output)?;
298    }
299
300    {
301        let _timer = sess.timer("link_dce-post-link");
302        dce::dce(&mut output);
303    }
304
305    {
306        let _timer = sess.timer("link_fragment_inst_check");
307        simple_passes::check_fragment_insts(sess, &output)?;
308    }
309
310    // HACK(eddyb) this has to run before the `report_zombies` pass, so that
311    // any zombies that are passed as call arguments, but eventually unused,
312    // won't be (incorrectly) considered used.
313    {
314        let _timer = sess.timer("link_remove_unused_params");
315        output = param_weakening::remove_unused_params(output);
316    }
317
318    if opts.early_report_zombies {
319        let _timer = sess.timer("link_report_zombies");
320        zombies::report_zombies(sess, &output)?;
321    }
322
323    if opts.infer_storage_classes {
324        let _timer = sess.timer("specialize_generic_storage_class");
325        // HACK(eddyb) `specializer` requires functions' blocks to be in RPO order
326        // (i.e. `block_ordering_pass`) - this could be relaxed by using RPO visit
327        // inside `specializer`, but this is easier.
328        for func in &mut output.functions {
329            simple_passes::block_ordering_pass(func);
330        }
331        output = specializer::specialize(
332            opts,
333            output,
334            specializer::SimpleSpecialization {
335                specialize_operand: |operand| {
336                    matches!(operand, Operand::StorageClass(StorageClass::Generic))
337                },
338
339                // NOTE(eddyb) this can be anything that is guaranteed to pass
340                // validation - there are no constraints so this is either some
341                // unused pointer, or perhaps one created using `OpConstantNull`
342                // and simply never mixed with pointers that have a storage class.
343                // It would be nice to use `Generic` itself here so that we leave
344                // some kind of indication of it being unconstrained, but `Generic`
345                // requires additional capabilities, so we use `Function` instead.
346                // TODO(eddyb) investigate whether this can end up in a pointer
347                // type that's the value of a module-scoped variable, and whether
348                // `Function` is actually invalid! (may need `Private`)
349                concrete_fallback: Operand::StorageClass(StorageClass::Function),
350            },
351        );
352    }
353
354    // NOTE(eddyb) with SPIR-T, we can do `mem2reg` before inlining, too!
355    {
356        {
357            let _timer = sess.timer("link_dce-before-inlining");
358            dce::dce(&mut output);
359        }
360
361        let _timer = sess.timer("link_block_ordering_pass_and_mem2reg-before-inlining");
362        let mut pointer_to_pointee = FxHashMap::default();
363        let mut constants = FxHashMap::default();
364        let mut u32 = None;
365        for inst in &output.types_global_values {
366            match inst.class.opcode {
367                Op::TypePointer => {
368                    pointer_to_pointee
369                        .insert(inst.result_id.unwrap(), inst.operands[1].unwrap_id_ref());
370                }
371                Op::TypeInt
372                    if inst.operands[0].unwrap_literal_bit32() == 32
373                        && inst.operands[1].unwrap_literal_bit32() == 0 =>
374                {
375                    assert!(u32.is_none());
376                    u32 = Some(inst.result_id.unwrap());
377                }
378                Op::Constant if u32.is_some() && inst.result_type == u32 => {
379                    let value = inst.operands[0].unwrap_literal_bit32();
380                    constants.insert(inst.result_id.unwrap(), value);
381                }
382                _ => {}
383            }
384        }
385        for func in &mut output.functions {
386            simple_passes::block_ordering_pass(func);
387            // Note: mem2reg requires functions to be in RPO order (i.e. block_ordering_pass)
388            mem2reg::mem2reg(
389                output.header.as_mut().unwrap(),
390                &mut output.types_global_values,
391                &pointer_to_pointee,
392                &constants,
393                func,
394            );
395            destructure_composites::destructure_composites(func);
396        }
397    }
398
399    {
400        let _timer =
401            sess.timer("link_dce-and-remove_duplicate_debuginfo-after-mem2reg-before-inlining");
402        dce::dce(&mut output);
403        duplicates::remove_duplicate_debuginfo(&mut output);
404    }
405
406    // HACK(eddyb) this has to be after DCE, to not break SPIR-T w/ dead decorations.
407    if let Some(dir) = &opts.dump_pre_inline {
408        dump_spv_and_spirt(&output, dir.join(disambiguated_crate_name_for_dumps));
409    }
410
411    {
412        let _timer = sess.timer("link_inline");
413        inline::inline(sess, &mut output)?;
414    }
415
416    {
417        let _timer = sess.timer("link_dce-after-inlining");
418        dce::dce(&mut output);
419    }
420
421    // HACK(eddyb) this has to be after DCE, to not break SPIR-T w/ dead decorations.
422    if let Some(dir) = &opts.dump_post_inline {
423        dump_spv_and_spirt(&output, dir.join(disambiguated_crate_name_for_dumps));
424    }
425
426    {
427        let _timer = sess.timer("link_block_ordering_pass_and_mem2reg-after-inlining");
428        let mut pointer_to_pointee = FxHashMap::default();
429        let mut constants = FxHashMap::default();
430        let mut u32 = None;
431        for inst in &output.types_global_values {
432            match inst.class.opcode {
433                Op::TypePointer => {
434                    pointer_to_pointee
435                        .insert(inst.result_id.unwrap(), inst.operands[1].unwrap_id_ref());
436                }
437                Op::TypeInt
438                    if inst.operands[0].unwrap_literal_bit32() == 32
439                        && inst.operands[1].unwrap_literal_bit32() == 0 =>
440                {
441                    assert!(u32.is_none());
442                    u32 = Some(inst.result_id.unwrap());
443                }
444                Op::Constant if u32.is_some() && inst.result_type == u32 => {
445                    let value = inst.operands[0].unwrap_literal_bit32();
446                    constants.insert(inst.result_id.unwrap(), value);
447                }
448                _ => {}
449            }
450        }
451        for func in &mut output.functions {
452            simple_passes::block_ordering_pass(func);
453            // Note: mem2reg requires functions to be in RPO order (i.e. block_ordering_pass)
454            mem2reg::mem2reg(
455                output.header.as_mut().unwrap(),
456                &mut output.types_global_values,
457                &pointer_to_pointee,
458                &constants,
459                func,
460            );
461            destructure_composites::destructure_composites(func);
462        }
463    }
464
465    {
466        let _timer =
467            sess.timer("link_dce-and-remove_duplicate_debuginfo-after-mem2reg-after-inlining");
468        dce::dce(&mut output);
469        duplicates::remove_duplicate_debuginfo(&mut output);
470    }
471
472    {
473        let _timer = sess.timer("link_remove_non_uniform");
474        simple_passes::remove_non_uniform_decorations(sess, &mut output)?;
475    }
476
477    // NOTE(eddyb) SPIR-T pipeline is entirely limited to this block.
478    {
479        let (spv_words, module_or_err, lower_from_spv_timer) =
480            spv_module_to_spv_words_and_spirt_module(&output);
481        let module = &mut module_or_err.map_err(|e| {
482            let spv_path = outputs.temp_path_for_diagnostic("spirt-lower-from-spv-input.spv");
483
484            let was_saved_msg =
485                match std::fs::write(&spv_path, spirv_tools::binary::from_binary(&spv_words)) {
486                    Ok(()) => format!("was saved to {}", spv_path.display()),
487                    Err(e) => format!("could not be saved: {e}"),
488                };
489
490            sess.dcx()
491                .struct_err(format!("{e}"))
492                .with_note("while lowering SPIR-V module to SPIR-T (spirt::spv::lower)")
493                .with_note(format!("input SPIR-V module {was_saved_msg}"))
494                .emit()
495        })?;
496
497        let mut dump_guard = SpirtDumpGuard {
498            sess,
499            linker_options: opts,
500            outputs,
501            disambiguated_crate_name_for_dumps,
502
503            module,
504            per_pass_module_for_dumping: vec![],
505            any_spirt_bugs: false,
506        };
507        let module = &mut *dump_guard.module;
508        // FIXME(eddyb) set the name into `dump_guard` to be able to access it on panic.
509        let before_pass = |pass| sess.timer(pass);
510        let mut after_pass = |pass, module: &spirt::Module, timer| {
511            drop(timer);
512            if opts.dump_spirt_passes.is_some() {
513                dump_guard
514                    .per_pass_module_for_dumping
515                    .push((pass, module.clone()));
516            }
517        };
518        // HACK(eddyb) don't dump the unstructured state if not requested, as
519        // after SPIR-T 0.4.0 it's extremely verbose (due to def-use hermeticity).
520        if opts.spirt_keep_unstructured_cfg_in_dumps || !opts.structurize {
521            after_pass("lower_from_spv", module, lower_from_spv_timer);
522        } else {
523            drop(lower_from_spv_timer);
524        }
525
526        // NOTE(eddyb) this *must* run on unstructured CFGs, to do its job.
527        // FIXME(eddyb) no longer relying on structurization, try porting this
528        // to replace custom aborts in `Block`s and inject `ExitInvocation`s
529        // after them (truncating the `Block` and/or parent region if necessary).
530        {
531            let _timer = before_pass(
532                "spirt_passes::controlflow::convert_custom_aborts_to_unstructured_returns_in_entry_points",
533            );
534            spirt_passes::controlflow::convert_custom_aborts_to_unstructured_returns_in_entry_points(opts, module);
535        }
536
537        if opts.structurize {
538            let timer = before_pass("spirt::legalize::structurize_func_cfgs");
539            spirt::passes::legalize::structurize_func_cfgs(module);
540            after_pass("structurize_func_cfgs", module, timer);
541        }
542
543        if !opts.spirt_passes.is_empty() {
544            // FIXME(eddyb) why does this focus on functions, it could just be module passes??
545            spirt_passes::run_func_passes(
546                module,
547                &opts.spirt_passes,
548                |name, _module| before_pass(name),
549                &mut after_pass,
550            );
551        }
552
553        {
554            let timer = before_pass("spirt_passes::validate");
555            spirt_passes::validate::validate(module);
556            after_pass("validate", module, timer);
557        }
558
559        {
560            let _timer = before_pass("spirt_passes::diagnostics::report_diagnostics");
561            spirt_passes::diagnostics::report_diagnostics(sess, opts, module).map_err(
562                |spirt_passes::diagnostics::ReportedDiagnostics {
563                     rustc_errors_guarantee,
564                     any_errors_were_spirt_bugs,
565                 }| {
566                    dump_guard.any_spirt_bugs |= any_errors_were_spirt_bugs;
567                    rustc_errors_guarantee
568                },
569            )?;
570        }
571
572        // Replace our custom debuginfo instructions just before lifting to SPIR-V.
573        {
574            let _timer = before_pass("spirt_passes::debuginfo::convert_custom_debuginfo_to_spv");
575            spirt_passes::debuginfo::convert_custom_debuginfo_to_spv(module);
576        }
577
578        let spv_words = {
579            let _timer = before_pass("spirt::Module::lift_to_spv_module_emitter");
580            module.lift_to_spv_module_emitter().unwrap().words
581        };
582        // FIXME(eddyb) dump both SPIR-T and `spv_words` if there's an error here.
583        output = {
584            let _timer = sess.timer("parse-spv_words-from-spirt");
585            crate::link::with_rspirv_loader(|loader| {
586                rspirv::binary::parse_words(&spv_words, loader)
587            })
588            .unwrap()
589        };
590    }
591
592    // Ensure that no references remain, to our custom "extended instruction set".
593    for inst in &output.ext_inst_imports {
594        assert_eq!(inst.class.opcode, Op::ExtInstImport);
595        let ext_inst_set = inst.operands[0].unwrap_literal_string();
596        if ext_inst_set.starts_with(custom_insts::CUSTOM_EXT_INST_SET_PREFIX) {
597            let expected = &custom_insts::CUSTOM_EXT_INST_SET[..];
598            if ext_inst_set == expected {
599                return Err(sess.dcx().err(format!(
600                    "`OpExtInstImport {ext_inst_set:?}` should not have been \
601                         left around after SPIR-T passes"
602                )));
603            } else {
604                return Err(sess.dcx().err(format!(
605                    "unsupported `OpExtInstImport {ext_inst_set:?}`
606                     (expected {expected:?} name - version mismatch?)"
607                )));
608            }
609        }
610    }
611
612    // FIXME(eddyb) rewrite these passes to SPIR-T ones, so we don't have to
613    // parse the output of `spirt::spv::lift` back into `rspirv` - also, for
614    // multi-module, it's much simpler with SPIR-T, just replace `module.exports`
615    // with a single-entry map, run `spirt::spv::lift` (or even `spirt::print`)
616    // on `module`, then put back the full original `module.exports` map.
617    {
618        let _timer = sess.timer("peephole_opts");
619        let types = peephole_opts::collect_types(&output);
620        for func in &mut output.functions {
621            peephole_opts::composite_construct(&types, func);
622            peephole_opts::vector_ops(output.header.as_mut().unwrap(), &types, func);
623            peephole_opts::bool_fusion(output.header.as_mut().unwrap(), &types, func);
624        }
625    }
626
627    {
628        let _timer = sess.timer("link_remove_unused_type_capabilities");
629        simple_passes::remove_unused_type_capabilities(&mut output);
630    }
631
632    {
633        let _timer = sess.timer("link_gather_all_interface_vars_from_uses");
634        entry_interface::gather_all_interface_vars_from_uses(&mut output);
635    }
636
637    if opts.spirv_metadata == SpirvMetadata::NameVariables {
638        let _timer = sess.timer("link_name_variables");
639        simple_passes::name_variables_pass(&mut output);
640    }
641
642    {
643        let _timer = sess.timer("link_sort_globals");
644        simple_passes::sort_globals(&mut output);
645    }
646
647    let mut output = if opts.module_output_type == ModuleOutputType::Multiple {
648        let mut file_stem_to_entry_name_and_module = BTreeMap::new();
649        for (i, entry) in output.entry_points.iter().enumerate() {
650            let mut module = output.clone();
651            module.entry_points.clear();
652            module.entry_points.push(entry.clone());
653            let entry_name = entry.operands[2].unwrap_literal_string().to_string();
654            let mut file_stem = OsString::from(
655                sanitize_filename::sanitize_with_options(
656                    &entry_name,
657                    sanitize_filename::Options {
658                        replacement: "-",
659                        ..Default::default()
660                    },
661                )
662                .replace("--", "-"),
663            );
664            // It's always possible to find an unambiguous `file_stem`, but it
665            // may take two tries (or more, in bizzare/adversarial cases).
666            let mut disambiguator = Some(i);
667            loop {
668                use std::collections::btree_map::Entry;
669                match file_stem_to_entry_name_and_module.entry(file_stem) {
670                    Entry::Vacant(entry) => {
671                        entry.insert((entry_name, module));
672                        break;
673                    }
674                    // FIXME(eddyb) false positive: `file_stem` was moved out of,
675                    // so assigning it is necessary, but clippy doesn't know that.
676                    #[allow(clippy::assigning_clones)]
677                    Entry::Occupied(entry) => {
678                        // FIXME(eddyb) there's no way to access the owned key
679                        // passed to `BTreeMap::entry` from `OccupiedEntry`.
680                        file_stem = entry.key().clone();
681                        file_stem.push(".");
682                        match disambiguator.take() {
683                            Some(d) => file_stem.push(d.to_string()),
684                            None => file_stem.push("next"),
685                        }
686                    }
687                }
688            }
689        }
690        LinkResult::MultipleModules {
691            file_stem_to_entry_name_and_module,
692        }
693    } else {
694        LinkResult::SingleModule(Box::new(output))
695    };
696
697    let output_module_iter = match &mut output {
698        LinkResult::SingleModule(m) => Either::Left(std::iter::once((None, &mut **m))),
699        LinkResult::MultipleModules {
700            file_stem_to_entry_name_and_module,
701        } => Either::Right(
702            file_stem_to_entry_name_and_module
703                .iter_mut()
704                .map(|(file_stem, (_, m))| (Some(file_stem), m)),
705        ),
706    };
707    for (file_stem, output) in output_module_iter {
708        // Run DCE again, even if module_output_type == ModuleOutputType::Multiple - the first DCE ran before
709        // structurization and mem2reg (for perf reasons), and mem2reg may remove references to
710        // invalid types, so we need to DCE again.
711        {
712            let _timer = sess.timer("link_dce-post-split");
713            dce::dce(output);
714        }
715
716        // HACK(eddyb) this has to be after DCE, to not break SPIR-T w/ dead decorations.
717        if let Some(dir) = &opts.dump_post_split {
718            let mut file_name = disambiguated_crate_name_for_dumps.to_os_string();
719            if let Some(file_stem) = file_stem {
720                file_name.push(".");
721                file_name.push(file_stem);
722            }
723
724            dump_spv_and_spirt(output, dir.join(file_name));
725        }
726
727        {
728            let _timer = sess.timer("link_remove_duplicate_debuginfo");
729            duplicates::remove_duplicate_debuginfo(output);
730        }
731
732        if opts.compact_ids {
733            let _timer = sess.timer("link_compact_ids");
734            // compact the ids https://github.com/KhronosGroup/SPIRV-Tools/blob/e02f178a716b0c3c803ce31b9df4088596537872/source/opt/compact_ids_pass.cpp#L43
735            output.header.as_mut().unwrap().bound = simple_passes::compact_ids(output);
736        };
737
738        // FIXME(eddyb) convert these into actual `OpLine`s with a SPIR-T pass,
739        // but that'd require keeping the modules in SPIR-T form (once lowered),
740        // and never loading them back into `rspirv` once lifted back to SPIR-V.
741        SrcLocDecoration::remove_all(output);
742
743        // FIXME(eddyb) might make more sense to rewrite these away on SPIR-T.
744        ZombieDecoration::remove_all(output);
745    }
746
747    Ok(output)
748}

750/// Helper for dumping SPIR-T on drop, which allows panics to also dump,
751/// not just successful compilation (i.e. via `--dump-spirt-passes`).
752struct SpirtDumpGuard<'a> {
753    sess: &'a Session,
754    linker_options: &'a Options,
755    outputs: &'a OutputFilenames,
756    disambiguated_crate_name_for_dumps: &'a OsStr,
757
758    module: &'a mut spirt::Module,
759    per_pass_module_for_dumping: Vec<(&'static str, spirt::Module)>,
760    any_spirt_bugs: bool,
761}
762
763impl Drop for SpirtDumpGuard<'_> {
764    fn drop(&mut self) {
765        self.any_spirt_bugs |= std::thread::panicking();
766
767        let mut dump_spirt_file_path =
768            self.linker_options
769                .dump_spirt_passes
770                .as_ref()
771                .map(|dump_dir| {
772                    dump_dir
773                        .join(self.disambiguated_crate_name_for_dumps)
774                        .with_extension("spirt")
775                });
776
777        // FIXME(eddyb) this won't allow seeing the individual passes, but it's
778        // better than nothing (theoretically the whole "SPIR-T pipeline" could
779        // be put in a loop so that everything is redone with per-pass tracking,
780        // but that requires keeping around e.g. the initial SPIR-V for longer,
781        // and probably invoking the "SPIR-T pipeline" here, as looping is hard).
782        if self.any_spirt_bugs && dump_spirt_file_path.is_none() {
783            if self.per_pass_module_for_dumping.is_empty() {
784                self.per_pass_module_for_dumping
785                    .push(("", self.module.clone()));
786            }
787            dump_spirt_file_path = Some(self.outputs.temp_path_for_diagnostic("spirt"));
788        }
789
790        if let Some(dump_spirt_file_path) = &dump_spirt_file_path {
791            for (_, module) in &mut self.per_pass_module_for_dumping {
792                self.linker_options.spirt_cleanup_for_dumping(module);
793            }
794
795            // FIXME(eddyb) catch panics during pretty-printing itself, and
796            // tell the user to use `--dump-spirt-passes` (and resolve the
797            // second FIXME below so it does anything) - also, that may need
798            // quieting the panic handler, likely controlled by a `thread_local!`
799            // (while the panic handler is global), but that would be useful
800            // for collecting a panic message (assuming any of this is worth it).
801            // FIXME(eddyb) when per-pass versions are available, try building
802            // plans for individual versions, or maybe repeat `Plan::for_versions`
803            // without the last version if it initially panicked?
804            let plan = spirt::print::Plan::for_versions(
805                self.module.cx_ref(),
806                self.per_pass_module_for_dumping
807                    .iter()
808                    .map(|(pass, module)| (format!("after {pass}"), module)),
809            );
810            let pretty = plan.pretty_print();
811
812            // FIXME(eddyb) don't allocate whole `String`s here.
813            std::fs::write(dump_spirt_file_path, pretty.to_string()).unwrap();
814            std::fs::write(
815                dump_spirt_file_path.with_extension("spirt.html"),
816                pretty
817                    .render_to_html()
818                    .with_dark_mode_support()
819                    .to_html_doc(),
820            )
821            .unwrap();
822            if self.any_spirt_bugs {
823                let mut note = self.sess.dcx().struct_note("SPIR-T bugs were encountered");
824                note.help(format!(
825                    "pretty-printed SPIR-T was saved to {}.html",
826                    dump_spirt_file_path.display()
827                ));
828                if self.linker_options.dump_spirt_passes.is_none() {
829                    note.help("re-run with `RUSTGPU_CODEGEN_ARGS=\"--dump-spirt-passes=$PWD\"` for more details");
830                }
831                note.note("pretty-printed SPIR-T is preferred when reporting Rust-GPU issues");
832                note.emit();
833            }
834        }
835    }
