34#[derive(Copy, Clone, Debug, Ord, PartialOrd, Eq, PartialEq, Hash)]
35pub enum SpirvValueKind {
36    Def(Word),
37
38    /// The ID of a global instruction matching a `SpirvConst`, but which cannot
39    /// pass validation. Used to error (or attach zombie spans), at the usesites
40    /// of such constants, instead of where they're generated (and cached).
41    IllegalConst(Word),
42
43    /// This can only happen in one specific case - which is as a result of
44    /// `codegen_buffer_store_intrinsic`, that function is supposed to return
45    /// `OpTypeVoid`, however because it gets inline by the compiler it can't.
46    /// Instead we return this, and trigger an error if we ever end up using the
47    /// result of this function call (which we can't).
48    IllegalTypeUsed(Word),
49
50    // FIXME(eddyb) this shouldn't be needed, but `rustc_codegen_ssa` still relies
51    // on converting `Function`s to `Value`s even for direct calls, the `Builder`
52    // should just have direct and indirect `call` variants (or a `Callee` enum).
53    FnAddr {
54        function: Word,
55    },
56
57    /// Deferred pointer cast, for the `Logical` addressing model (which doesn't
58    /// really support raw pointers in the way Rust expects to be able to use).
59    ///
60    /// The cast's target pointer type is the `ty` of the `SpirvValue` that has
61    /// `LogicalPtrCast` as its `kind`, as it would be redundant to have it here.
62    LogicalPtrCast {
63        /// Pointer value being cast.
64        original_ptr: Word,
65
66        /// Pointer type of `original_ptr`.
67        original_ptr_ty: Word,
68
69        /// Result ID for the `OpBitcast` instruction representing the cast,
70        /// to attach zombies to.
71        //
72        // HACK(eddyb) having an `OpBitcast` only works by being DCE'd away,
73        // or by being replaced with a noop in `qptr::lower`.
74        bitcast_result_id: Word,
75    },
76}

84impl SpirvValue {
85    pub fn strip_ptrcasts(self) -> Self {
86        match self.kind {
87            SpirvValueKind::LogicalPtrCast {
88                original_ptr,
89                original_ptr_ty,
90                bitcast_result_id: _,
91            } => original_ptr.with_type(original_ptr_ty),
92
93            _ => self,
94        }
95    }
96
97    pub fn const_fold_load(self, cx: &CodegenCx<'_>) -> Option<Self> {
98        match self.kind {
99            SpirvValueKind::Def(id) | SpirvValueKind::IllegalConst(id) => {
100                let &entry = cx.builder.id_to_const.borrow().get(&id)?;
101                match entry.val {
102                    SpirvConst::PtrTo { pointee } => {
103                        let ty = match cx.lookup_type(self.ty) {
104                            SpirvType::Pointer { pointee } => pointee,
105                            ty => bug!("load called on value that wasn't a pointer: {:?}", ty),
106                        };
107                        // FIXME(eddyb) deduplicate this `if`-`else` and its other copies.
108                        let kind = if entry.legal.is_ok() {
109                            SpirvValueKind::Def(pointee)
110                        } else {
111                            SpirvValueKind::IllegalConst(pointee)
112                        };
113                        Some(SpirvValue { kind, ty })
114                    }
115                    _ => None,
116                }
117            }
118
119            _ => None,
120        }
121    }
122
123    // Important: we *cannot* use bx.emit() here, because this is called in
124    // contexts where the emitter is already locked. Doing so may cause subtle
125    // rare bugs.
126    pub fn def(self, bx: &builder::Builder<'_, '_>) -> Word {
127        self.def_with_span(bx, bx.span())
128    }
129
130    // def and def_cx are separated, because Builder has a span associated with
131    // what it's currently emitting.
132    pub fn def_cx(self, cx: &CodegenCx<'_>) -> Word {
133        self.def_with_span(cx, DUMMY_SP)
134    }
135
136    pub fn def_with_span(self, cx: &CodegenCx<'_>, span: Span) -> Word {
137        match self.kind {
138            SpirvValueKind::Def(id) => id,
139
140            SpirvValueKind::IllegalConst(id) => {
141                let entry = &cx.builder.id_to_const.borrow()[&id];
142                let msg = match entry.legal.unwrap_err() {
143                    IllegalConst::Shallow(cause) => {
144                        if let (
145                            LeafIllegalConst::CompositeContainsPtrTo,
146                            SpirvConst::Composite(_fields),
147                        ) = (cause, &entry.val)
148                        {
149                            // FIXME(eddyb) materialize this at runtime, using
150                            // `OpCompositeConstruct` (transitively, i.e. after
151                            // putting every field through `SpirvValue::def`),
152                            // if we have a `Builder` to do that in.
153                            // FIXME(eddyb) this isn't possible right now, as
154                            // the builder would be dynamically "locked" anyway
155                            // (i.e. attempting to do `bx.emit()` would panic).
156                        }
157
158                        cause.message()
159                    }
160
161                    IllegalConst::Indirect(cause) => cause.message(),
162                };
163
164                cx.zombie_with_span(id, span, msg);
165
166                id
167            }
168
169            SpirvValueKind::IllegalTypeUsed(id) => {
170                cx.tcx
171                    .dcx()
172                    .struct_span_err(span, "Can't use type as a value")
173                    .with_note(format!("Type: *{}", cx.debug_type(id)))
174                    .emit();
175
176                id
177            }
178
179            SpirvValueKind::FnAddr { .. } => {
180                cx.builder
181                    .const_to_id
182                    .borrow()
183                    .get(&WithType {
184                        ty: self.ty,
185                        val: SpirvConst::ZombieUndefForFnAddr,
186                    })
187                    .expect("FnAddr didn't go through proper undef registration")
188                    .val
189            }
190
191            SpirvValueKind::LogicalPtrCast {
192                original_ptr: _,
193                original_ptr_ty,
194                bitcast_result_id,
195            } => {
196                cx.zombie_with_span(
197                    bitcast_result_id,
198                    span,
199                    &format!(
200                        "cannot cast between pointer types\
201                         \nfrom `{}`\
202                         \n  to `{}`",
203                        cx.debug_type(original_ptr_ty),
204                        cx.debug_type(self.ty)
205                    ),
206                );
207
208                bitcast_result_id
209            }
210        }
211    }
212}

228pub enum SpirvConst<'a, 'tcx> {
229    /// Constants of boolean, integer or floating-point type (up to 128-bit).
230    Scalar(u128),
231
232    Null,
233    Undef,
234
235    /// Like `Undef`, but cached separately to avoid `FnAddr` zombies accidentally
236    /// applying to non-zombie `Undef`s of the same types.
237    // FIXME(eddyb) include the function ID so that multiple `fn` pointers to
238    // different functions, but of the same type, don't overlap their zombies.
239    ZombieUndefForFnAddr,
240
241    Composite(&'a [Word]),
242
243    /// Pointer to constant data, i.e. `&pointee`, represented as an `OpVariable`
244    /// in the `Private` storage class, and with `pointee` as its initializer.
245    PtrTo {
246        pointee: Word,
247    },
248
249    /// Symbolic result for the `const_data_from_alloc` method, to allow deferring
250    /// the actual value generation until after a pointer to this value is cast
251    /// to its final type (e.g. that will be loaded as).
252    //
253    // FIXME(eddyb) replace this with `qptr` handling of constant data.
254    ConstDataFromAlloc(ConstAllocation<'tcx>),
255}

257impl<'tcx> SpirvConst<'_, 'tcx> {
258    /// Replace `&[T]` fields with `&'tcx [T]` ones produced by calling
259    /// `tcx.arena.dropless.alloc_slice(...)` - this is done late for two reasons:
260    /// 1. it avoids allocating in the arena when the cache would be hit anyway,
261    ///    which would create "garbage" (as in, unreachable allocations)
262    ///    (ideally these would also be interned, but that's even more refactors)
263    /// 2. an empty slice is disallowed (as it's usually handled as a special
264    ///    case elsewhere, e.g. `rustc`'s `ty::List` - sadly we can't use that)
265    fn tcx_arena_alloc_slices(self, cx: &CodegenCx<'tcx>) -> SpirvConst<'tcx, 'tcx> {
266        fn arena_alloc_slice<'tcx, T: Copy>(cx: &CodegenCx<'tcx>, xs: &[T]) -> &'tcx [T] {
267            if xs.is_empty() {
268                &[]
269            } else {
270                cx.tcx.arena.dropless.alloc_slice(xs)
271            }
272        }
273
274        match self {
275            // FIXME(eddyb) these are all noop cases, could they be automated?
276            SpirvConst::Scalar(v) => SpirvConst::Scalar(v),
277            SpirvConst::Null => SpirvConst::Null,
278            SpirvConst::Undef => SpirvConst::Undef,
279            SpirvConst::ZombieUndefForFnAddr => SpirvConst::ZombieUndefForFnAddr,
280            SpirvConst::PtrTo { pointee } => SpirvConst::PtrTo { pointee },
281
282            SpirvConst::Composite(fields) => SpirvConst::Composite(arena_alloc_slice(cx, fields)),
283
284            SpirvConst::ConstDataFromAlloc(alloc) => SpirvConst::ConstDataFromAlloc(alloc),
285        }
286    }
287}

295/// Primary causes for a `SpirvConst` to be deemed illegal.
296#[derive(Copy, Clone, Debug)]
297enum LeafIllegalConst {
298    /// `SpirvConst::Composite` containing a `SpirvConst::PtrTo` as a field.
299    /// This is illegal because `OpConstantComposite` must have other constants
300    /// as its operands, and `OpVariable`s are never considered constant.
301    // FIXME(eddyb) figure out if this is an accidental omission in SPIR-V.
302    CompositeContainsPtrTo,
303
304    /// `ConstDataFromAlloc` constant, which cannot currently be materialized
305    /// to SPIR-V (and requires to be wrapped in `PtrTo` and bitcast, first).
306    //
307    // FIXME(eddyb) replace this with `qptr` handling of constant data.
308    UntypedConstDataFromAlloc,
309}
310
311impl LeafIllegalConst {
312    fn message(&self) -> &'static str {
313        match *self {
314            Self::CompositeContainsPtrTo => {
315                "constant arrays/structs cannot contain pointers to other constants"
316            }
317            Self::UntypedConstDataFromAlloc => {
318                "`const_data_from_alloc` result wasn't passed through `static_addr_of`, \
319                 then `const_bitcast` (which would've given it a type)"
320            }
321        }
322    }
323}

325#[derive(Copy, Clone, Debug)]
326enum IllegalConst {
327    /// This `SpirvConst` is (or contains) a "leaf" illegal constant. As there
328    /// is no indirection, some of these could still be materialized at runtime,
329    /// using e.g. `OpCompositeConstruct` instead of `OpConstantComposite`.
330    Shallow(LeafIllegalConst),
331
332    /// This `SpirvConst` is (or contains/points to) a `PtrTo` which points to
333    /// a "leaf" illegal constant. As the data would have to live for `'static`,
334    /// there is no way to materialize it as a pointer in SPIR-V. However, it
335    /// could still be legalized during codegen by e.g. folding loads from it.
336    Indirect(LeafIllegalConst),
337}

345/// `HashMap` key type (for `debug_file_cache` in `BuilderSpirv`), which is
346/// equivalent to a whole `rustc` `SourceFile`, but has O(1) `Eq` and `Hash`
347/// implementations (i.e. not involving the path or the contents of the file).
348///
349/// This is possible because we can compare `Arc<SourceFile>`s by equality, as
350/// `rustc`'s `SourceMap` already ensures that only one `SourceFile` will be
351/// allocated for some given file. For hashing, we could hash the address, or
352///
353struct DebugFileKey(Arc<SourceFile>);
354
355impl PartialEq for DebugFileKey {
356    fn eq(&self, other: &Self) -> bool {
357        let (Self(self_sf), Self(other_sf)) = (self, other);
358        Arc::ptr_eq(self_sf, other_sf)
359    }
360}
361impl Eq for DebugFileKey {}
362
363impl Hash for DebugFileKey {
364    fn hash<H: Hasher>(&self, state: &mut H) {
365        let Self(sf) = self;
366        sf.stable_id.hash(state);
367        sf.src_hash.hash(state);
368    }
369}

399/// Cursor system:
400///
401/// The LLVM module builder model (and therefore `codegen_ssa`) assumes that there is a central
402/// module object, then, builder objects are created pointing at that central module object (e.g.
403/// for adding instructions to a basic block).  Several of these builder objects can be live at the
404/// same time, mutating the central module object all at once.  Unfortunately, rspirv doesn't work
405/// like that. Instead, there is a single builder object, which owns a module and a "cursor". This
406/// cursor indicates to the builder where to append instructions when an instruction is added -
407/// e.g. if `add()` is called, then `OpAdd` is appended to the basic block pointed to by the cursor.
408///
409/// So! We emulate the LLVM system by treating the rspirv Builder as the "central module object",
410/// then, when a "builder object" is created, we store a reference to a `RefCell<rspirv builder>`,
411/// *as well as* a copy of the cursor for that particular builder. Whenever the `RefCell` is
412/// borrowed, then we stomp over the rspirv cursor with our copy, causing the duration of that
413/// `RefCell` borrow to use that cursor.
414///
415/// So, if you're writing code inside `crate::builder::Builder`, then `self.emit()` will use
416/// `self.cursor` (the current basic block) as that "stomp-over" cursor and return a mutable
417/// reference to the rspirv builder. If you're writing code elsewhere (`codegen_cx::CodegenCx`),
418/// then `self.emit_global()` will use the generic "global cursor" and return a mutable reference
419/// to the rspirv builder with no basic block nor function selected, i.e. any instructions emitted
420/// will be in the global section.
421//
422// FIXME(eddyb) try updating documentation like the above.
423// FIXME(eddyb) figure out how to replace `BuilderCursor` with something like
424// `Option<SpirvBlockCursor>`, but that can't handle "in function outside BB".
425#[derive(Debug, Default, Copy, Clone)]
426#[must_use = "BuilderCursor should usually be assigned to the Builder.cursor field"]
427struct BuilderCursor {
428    fn_id_and_idx: Option<(Word, usize)>,
429    block_id_and_idx: Option<(Word, usize)>,
430}

450impl<'tcx> BuilderSpirv<'tcx> {
451    pub fn new(
452        tcx: TyCtxt<'tcx>,
453        sym: &Symbols,
454        target: &SpirvTarget,
455        features: &[TargetFeature],
456    ) -> Self {
457        let version = target.spirv_version();
458        let memory_model = target.memory_model();
459
460        let mut builder = Builder::new();
461        builder.set_version(version.0, version.1);
462        builder.module_mut().header.as_mut().unwrap().generator = 0x001B_0000;
463
464        let mut enabled_capabilities = FxHashSet::default();
465
466        fn add_cap(
467            builder: &mut Builder,
468            enabled_capabilities: &mut FxHashSet<Capability>,
469            cap: Capability,
470        ) {
471            // This should be the only callsite of Builder::capability (aside from tests), to make
472            // sure the hashset stays in sync.
473            builder.capability(cap);
474            enabled_capabilities.insert(cap);
475        }
476        fn add_ext(builder: &mut Builder, ext: Symbol) {
477            // This should be the only callsite of Builder::extension (aside from tests), to make
478            // sure the hashset stays in sync.
479            builder.extension(ext.as_str());
480        }
481
482        for feature in features {
483            match *feature {
484                TargetFeature::Capability(cap) => {
485                    add_cap(&mut builder, &mut enabled_capabilities, cap);
486                }
487                TargetFeature::Extension(ext) => {
488                    add_ext(&mut builder, ext);
489                }
490            }
491        }
492
493        add_cap(&mut builder, &mut enabled_capabilities, Capability::Shader);
494        if memory_model == MemoryModel::Vulkan {
495            if version < (1, 5) {
496                add_ext(&mut builder, sym.spv_khr_vulkan_memory_model);
497            }
498            add_cap(
499                &mut builder,
500                &mut enabled_capabilities,
501                Capability::VulkanMemoryModel,
502            );
503        }
504
505        // The linker will always be ran on this module
506        add_cap(&mut builder, &mut enabled_capabilities, Capability::Linkage);
507
508        builder.memory_model(AddressingModel::Logical, memory_model);
509
510        Self {
511            source_map: tcx.sess.source_map(),
512            dropless_arena: &tcx.arena.dropless,
513            builder: RefCell::new(builder),
514            const_to_id: Default::default(),
515            id_to_const: Default::default(),
516            debug_file_cache: Default::default(),
517            enabled_capabilities,
518        }
519    }

542    /// See comment on `BuilderCursor`
543    fn builder(&self, cursor: BuilderCursor) -> RefMut<'_, Builder> {
544        let mut builder = self.builder.borrow_mut();
545
546        let [maybe_fn_idx, maybe_block_idx] = [cursor.fn_id_and_idx, cursor.block_id_and_idx]
547            .map(|id_and_idx| id_and_idx.map(|(_, idx)| idx));
548
549        let fn_changed = builder.selected_function() != maybe_fn_idx;
550        if fn_changed {
551            builder.select_function(maybe_fn_idx).unwrap();
552        }
553
554        // Only check the function/block IDs if either of their indices changed.
555        if let Some((fn_id, fn_idx)) = cursor.fn_id_and_idx
556            && (fn_changed || builder.selected_block() != maybe_block_idx)
557        {
558            builder.select_block(maybe_block_idx).unwrap();
559
560            let function = &builder.module_ref().functions[fn_idx];
561            if fn_changed {
562                assert_eq!(function.def_id(), Some(fn_id));
563            }
564            if let Some((block_id, block_idx)) = cursor.block_id_and_idx {
565                assert_eq!(function.blocks[block_idx].label_id(), Some(block_id));
566            }
567        }
568
569        builder
570    }

593    pub(crate) fn def_constant_cx(
594        &self,
595        ty: Word,
596        val: SpirvConst<'_, 'tcx>,
597        cx: &CodegenCx<'tcx>,
598    ) -> SpirvValue {
599        let scalar_ty = match val {
600            SpirvConst::Scalar(_) => Some(cx.lookup_type(ty)),
601            _ => None,
602        };
603
604        // HACK(eddyb) this is done so late (just before interning `val`) to
605        // minimize any potential misuse from direct `def_constant` calls.
606        let val = match (val, scalar_ty) {
607            (SpirvConst::Scalar(val), Some(SpirvType::Integer(bits, signed))) => {
608                let size = Size::from_bits(bits);
609                SpirvConst::Scalar(if signed {
610                    size.sign_extend(val) as u128
611                } else {
612                    size.truncate(val)
613                })
614            }
615            _ => val,
616        };
617
618        let val_with_type = WithType { ty, val };
619        if let Some(entry) = self.const_to_id.borrow().get(&val_with_type) {
620            // FIXME(eddyb) deduplicate this `if`-`else` and its other copies.
621            let kind = if entry.legal.is_ok() {
622                SpirvValueKind::Def(entry.val)
623            } else {
624                SpirvValueKind::IllegalConst(entry.val)
625            };
626            return SpirvValue { kind, ty };
627        }
628        let val = val_with_type.val;
629
630        // FIXME(eddyb) make this an extension method on `rspirv::dr::Builder`?
631        let const_op = |builder: &mut Builder, op, lhs, maybe_rhs: Option<_>| {
632            // HACK(eddyb) remove after `OpSpecConstantOp` support gets added to SPIR-T.
633            let spirt_has_const_op = false;
634
635            if !spirt_has_const_op {
636                let zombie = builder.undef(ty, None);
637                cx.zombie_with_span(
638                    zombie,
639                    DUMMY_SP,
640                    &format!("unsupported constant of type `{}`", cx.debug_type(ty)),
641                );
642                return zombie;
643            }
644
645            let id = builder.id();
646            builder
647                .module_mut()
648                .types_global_values
649                .push(Instruction::new(
650                    Op::SpecConstantOp,
651                    Some(ty),
652                    Some(id),
653                    [
654                        Operand::LiteralSpecConstantOpInteger(op),
655                        Operand::IdRef(lhs),
656                    ]
657                    .into_iter()
658                    .chain(maybe_rhs.map(Operand::IdRef))
659                    .collect(),
660                ));
661            id
662        };
663
664        let mut builder = self.global_builder();
665        let id = match val {
666            SpirvConst::Scalar(v) => match scalar_ty.unwrap() {
667                SpirvType::Integer(..=32, _) | SpirvType::Float(..=32) => {
668                    builder.constant_bit32(ty, v as u32)
669                }
670                SpirvType::Integer(64, _) | SpirvType::Float(64) => {
671                    builder.constant_bit64(ty, v as u64)
672                }
673                SpirvType::Integer(128, false) => {
674                    // HACK(eddyb) avoid borrow conflicts.
675                    drop(builder);
676
677                    let const_64_u32_id = cx.const_u32(64).def_cx(cx);
678                    let [lo_id, hi_id] =
679                        [v as u64, (v >> 64) as u64].map(|half| cx.const_u64(half).def_cx(cx));
680
681                    builder = self.global_builder();
682                    let mut const_op =
683                        |op, lhs, maybe_rhs| const_op(&mut builder, op, lhs, maybe_rhs);
684                    let [lo_u128_id, hi_shifted_u128_id] =
685                        [(lo_id, None), (hi_id, Some(const_64_u32_id))].map(
686                            |(half_u64_id, shift)| {
687                                let mut half_u128_id = const_op(Op::UConvert, half_u64_id, None);
688                                if let Some(shift_amount_id) = shift {
689                                    half_u128_id = const_op(
690                                        Op::ShiftLeftLogical,
691                                        half_u128_id,
692                                        Some(shift_amount_id),
693                                    );
694                                }
695                                half_u128_id
696                            },
697                        );
698                    const_op(Op::BitwiseOr, lo_u128_id, Some(hi_shifted_u128_id))
699                }
700                SpirvType::Integer(128, true) | SpirvType::Float(128) => {
701                    // HACK(eddyb) avoid borrow conflicts.
702                    drop(builder);
703
704                    let v_u128_id = cx.const_u128(v).def_cx(cx);
705
706                    builder = self.global_builder();
707                    const_op(&mut builder, Op::Bitcast, v_u128_id, None)
708                }
709                SpirvType::Bool => match v {
710                    0 => builder.constant_false(ty),
711                    1 => builder.constant_true(ty),
712                    _ => cx
713                        .tcx
714                        .dcx()
715                        .fatal(format!("invalid constant value for bool: {v}")),
716                },
717                other => cx.tcx.dcx().fatal(format!(
718                    "SpirvConst::Scalar does not support type {}",
719                    other.debug(ty, cx)
720                )),
721            },
722
723            SpirvConst::Null => builder.constant_null(ty),
724            SpirvConst::Undef
725            | SpirvConst::ZombieUndefForFnAddr
726            | SpirvConst::ConstDataFromAlloc(_) => builder.undef(ty, None),
727
728            SpirvConst::Composite(v) => builder.constant_composite(ty, v.iter().copied()),
729
730            SpirvConst::PtrTo { pointee } => {
731                builder.variable(ty, None, StorageClass::Private, Some(pointee))
732            }
733        };
734        #[allow(clippy::match_same_arms)]
735        let legal = match val {
736            SpirvConst::Scalar(_) => Ok(()),
737
738            SpirvConst::Null => {
739                // FIXME(eddyb) check that the type supports `OpConstantNull`.
740                Ok(())
741            }
742            SpirvConst::Undef => {
743                // FIXME(eddyb) check that the type supports `OpUndef`.
744                Ok(())
745            }
746
747            SpirvConst::ZombieUndefForFnAddr => {
748                // This can be considered legal as it's already marked as zombie.
749                // FIXME(eddyb) is it possible for the original zombie to lack a
750                // span, and should we go through `IllegalConst` in order to be
751                // able to attach a proper usesite span?
752                Ok(())
753            }
754
755            SpirvConst::Composite(v) => v
756                .iter()
757                .map(|field| {
758                    let field_entry = &self.id_to_const.borrow()[field];
759                    field_entry.legal.and(
760                        // `field` is itself some legal `SpirvConst`, but can we have
761                        // it as part of an `OpConstantComposite`?
762                        match field_entry.val {
763                            SpirvConst::PtrTo { .. } => Err(IllegalConst::Shallow(
764                                LeafIllegalConst::CompositeContainsPtrTo,
765                            )),
766                            _ => Ok(()),
767                        },
768                    )
769                })
770                .reduce(|a, b| {
771                    match (a, b) {
772                        (Ok(()), Ok(())) => Ok(()),
773                        (Err(illegal), Ok(())) | (Ok(()), Err(illegal)) => Err(illegal),
774
775                        // Combining two causes of an illegal `SpirvConst` has to
776                        // take into account which is "worse", i.e. which imposes
777                        // more restrictions on how the resulting value can be used.
778                        // `Indirect` is worse than `Shallow` because it cannot be
779                        // materialized at runtime in the same way `Shallow` can be.
780                        (Err(illegal @ IllegalConst::Indirect(_)), Err(_))
781                        | (Err(_), Err(illegal @ IllegalConst::Indirect(_)))
782                        | (
783                            Err(illegal @ IllegalConst::Shallow(_)),
784                            Err(IllegalConst::Shallow(_)),
785                        ) => Err(illegal),
786                    }
787                })
788                .unwrap_or(Ok(())),
789
790            SpirvConst::PtrTo { pointee } => match self.id_to_const.borrow()[&pointee].legal {
791                Ok(()) => Ok(()),
792
793                // `Shallow` becomes `Indirect` when placed behind a pointer.
794                Err(IllegalConst::Shallow(cause) | IllegalConst::Indirect(cause)) => {
795                    Err(IllegalConst::Indirect(cause))
796                }
797            },
798
799            SpirvConst::ConstDataFromAlloc(_) => Err(IllegalConst::Shallow(
800                LeafIllegalConst::UntypedConstDataFromAlloc,
801            )),
802        };
803        let val = val.tcx_arena_alloc_slices(cx);
804        assert_matches!(
805            self.const_to_id
806                .borrow_mut()
807                .insert(WithType { ty, val }, WithConstLegality { val: id, legal }),
808            None
809        );
810        assert_matches!(
811            self.id_to_const
812                .borrow_mut()
813                .insert(id, WithConstLegality { val, legal }),
814            None
815        );
816        // FIXME(eddyb) deduplicate this `if`-`else` and its other copies.
817        let kind = if legal.is_ok() {
818            SpirvValueKind::Def(id)
819        } else {
820            SpirvValueKind::IllegalConst(id)
821        };
822        SpirvValue { kind, ty }
823    }

845    pub fn file_line_col_range_for_debuginfo(
846        &self,
847        span: Span,
848    ) -> (DebugFileSpirv<'tcx>, Range<(u32, u32)>) {
849        // HACK(eddyb) this is similar to what `#[track_caller]` does, and it
850        // allows us to point to the use site of a macro, instead of inside the
851        // macro (but ideally we would record the entire macro backtrace).
852        let span = span.ctxt().outer_expn().expansion_cause().unwrap_or(span);
853
854        let (lo, hi) = (span.lo(), span.hi());
855
856        let lo_loc = self.source_map.lookup_char_pos(lo);
857        let lo_line_col = (lo_loc.line as u32, lo_loc.col_display as u32);
858
859        // Only use `hi` if the span is actually a range within a file.
860        let hi_line_col = if lo <= hi {
861            let hi_loc = self.source_map.lookup_char_pos(hi);
862            if lo_loc.file.start_pos == hi_loc.file.start_pos {
863                (hi_loc.line as u32, hi_loc.col_display as u32)
864            } else {
865                lo_line_col
866            }
867        } else {
868            lo_line_col
869        };
870
871        (self.def_debug_file(lo_loc.file), lo_line_col..hi_line_col)
872    }

874    fn def_debug_file(&self, sf: Arc<SourceFile>) -> DebugFileSpirv<'tcx> {
875        *self
876            .debug_file_cache
877            .borrow_mut()
878            .entry(DebugFileKey(sf))
879            .or_insert_with_key(|DebugFileKey(sf)| {
880                let mut builder = self.global_builder();
881
882                // FIXME(eddyb) it would be nicer if we could just rely on
883                // `RealFileName::to_string_lossy` returning `Cow<'_, str>`,
884                // but sadly that `'_` is the lifetime of the temporary `Arc`,
885                // not `'tcx`, so we have to arena-allocate to get `&'tcx str`.
886                let file_name = match &sf.name {
887                    FileName::Real(name) => {
888                        name.to_string_lossy(FileNameDisplayPreference::Remapped)
889                    }
890                    _ => sf.name.prefer_remapped_unconditionaly().to_string().into(),
891                };
892                let file_name = {
893                    // FIXME(eddyb) it should be possible to arena-allocate a
894                    // `&str` directly, but it would require upstream changes,
895                    // and strings are handled by string interning in `rustc`.
896                    fn arena_alloc_slice<'tcx, T: Copy>(
897                        dropless_arena: &'tcx DroplessArena,
898                        xs: &[T],
899                    ) -> &'tcx [T] {
900                        if xs.is_empty() {
901                            &[]
902                        } else {
903                            dropless_arena.alloc_slice(xs)
904                        }
905                    }
906                    str::from_utf8(arena_alloc_slice(self.dropless_arena, file_name.as_bytes()))
907                        .unwrap()
908                };
909                let file_name_op_string_id = builder.string(file_name.to_owned());
910
911                let file_contents = self
912                    .source_map
913                    .span_to_snippet(Span::with_root_ctxt(sf.start_pos, sf.end_position()))
914                    .ok();
915
916                // HACK(eddyb) this logic is duplicated from `spirt::spv::lift`.
917                let op_source_and_continued_chunks = file_contents.as_ref().map(|contents| {
918                    // The maximum word count is `2**16 - 1`, the first word is
919                    // taken up by the opcode & word count, and one extra byte is
920                    // taken up by the nil byte at the end of the LiteralString.
921                    const MAX_OP_SOURCE_CONT_CONTENTS_LEN: usize = (0xffff - 1) * 4 - 1;
922
923                    // `OpSource` has 3 more operands than `OpSourceContinued`,
924                    // and each of them take up exactly one word.
925                    const MAX_OP_SOURCE_CONTENTS_LEN: usize =
926                        MAX_OP_SOURCE_CONT_CONTENTS_LEN - 3 * 4;
927
928                    let (op_source_str, mut all_op_source_continued_str) =
929                        contents.split_at(contents.len().min(MAX_OP_SOURCE_CONTENTS_LEN));
930
931                    // FIXME(eddyb) `spirt::spv::lift` should use this.
932                    let all_op_source_continued_str_chunks = iter::from_fn(move || {
933                        let contents_rest = &mut all_op_source_continued_str;
934                        if contents_rest.is_empty() {
935                            return None;
936                        }
937
938                        // FIXME(eddyb) test with UTF-8! this `split_at` should
939                        // actually take *less* than the full possible size, to
940                        // avoid cutting a UTF-8 sequence.
941                        let (cont_chunk, rest) = contents_rest
942                            .split_at(contents_rest.len().min(MAX_OP_SOURCE_CONT_CONTENTS_LEN));
943                        *contents_rest = rest;
944                        Some(cont_chunk)
945                    });
946                    (op_source_str, all_op_source_continued_str_chunks)
947                });
948
949                if let Some((op_source_str, all_op_source_continued_str_chunks)) =
950                    op_source_and_continued_chunks
951                {
952                    builder.source(
953                        SourceLanguage::Unknown,
954                        0,
955                        Some(file_name_op_string_id),
956                        Some(op_source_str),
957                    );
958                    for cont_chunk in all_op_source_continued_str_chunks {
959                        builder.source_continued(cont_chunk);
960                    }
961                }
962
963                DebugFileSpirv {
964                    file_name,
965                    file_name_op_string_id,
966                }
967            })
968    }

970    pub fn set_global_initializer(&self, global: Word, initializer: Word) {
971        let mut builder = self.builder.borrow_mut();
972        let module = builder.module_mut();
973        let index = module
974            .types_global_values
975            .iter()
976            .enumerate()
977            .find_map(|(index, inst)| {
978                if inst.result_id == Some(global) {
979                    Some(index)
980                } else {
981                    None
982                }
983            })
984            .expect("set_global_initializer global not found");
985        // Remove and push it to the end, to keep spir-v definition order.
986        let mut inst = module.types_global_values.remove(index);
987        assert_eq!(inst.class.opcode, Op::Variable);
988        assert_eq!(
989            inst.operands.len(),
990            1,
991            "global already has initializer defined: {global}"
992        );
993        inst.operands.push(Operand::IdRef(initializer));
994        module.types_global_values.push(inst);
995    }
