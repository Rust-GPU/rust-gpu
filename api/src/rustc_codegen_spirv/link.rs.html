35pub fn link(
36    sess: &Session,
37    codegen_results: &CodegenResults,
38    metadata: &EncodedMetadata,
39    outputs: &OutputFilenames,
40    crate_name: &str,
41) {
42    let output_metadata = sess.opts.output_types.contains_key(&OutputType::Metadata);
43    for &crate_type in sess.opts.crate_types.iter() {
44        if (sess.opts.unstable_opts.no_codegen || !sess.opts.output_types.should_codegen())
45            && !output_metadata
46            && crate_type == CrateType::Executable
47        {
48            continue;
49        }
50
51        if invalid_output_for_target(sess, crate_type) {
52            bug!(
53                "invalid output type `{:?}` for target os `{}`",
54                crate_type,
55                sess.opts.target_triple
56            );
57        }
58
59        for obj in codegen_results
60            .modules
61            .iter()
62            .filter_map(|m| m.object.as_ref())
63        {
64            check_file_is_writeable(obj, sess);
65        }
66
67        if outputs.outputs.should_codegen() {
68            let out_filename = out_filename(sess, crate_type, outputs, Symbol::intern(crate_name));
69            let out_filename_file_for_writing = out_filename.file_for_writing(
70                outputs,
71                OutputType::Exe,
72                crate_name,
73                sess.invocation_temp.as_deref(),
74            );
75            match crate_type {
76                CrateType::Rlib => {
77                    link_rlib(
78                        sess,
79                        codegen_results,
80                        metadata,
81                        &out_filename_file_for_writing,
82                    );
83                }
84                CrateType::Executable | CrateType::Cdylib | CrateType::Dylib => {
85                    // HACK(eddyb) there's no way way to access `outputs.filestem`,
86                    // so we pay the cost of building a whole `PathBuf` instead.
87                    let disambiguated_crate_name_for_dumps = outputs
88                        .with_extension("")
89                        .file_name()
90                        .unwrap()
91                        .to_os_string();
92
93                    link_exe(
94                        sess,
95                        crate_type,
96                        &out_filename_file_for_writing,
97                        codegen_results,
98                        outputs,
99                        &disambiguated_crate_name_for_dumps,
100                    );
101                }
102                other => {
103                    sess.dcx()
104                        .err(format!("CrateType {other:?} not supported yet"));
105                }
106            }
107            match out_filename {
108                OutFileName::Real(_) => {
109                    // Already written to, above.
110                }
111                OutFileName::Stdout => {
112                    // HACK(eddyb) wrote a file above, time to read it back out.
113                    std::io::copy(
114                        &mut std::io::BufReader::new(
115                            std::fs::File::open(out_filename_file_for_writing).unwrap(),
116                        ),
117                        &mut std::io::stdout(),
118                    )
119                    .unwrap();
120                }
121            }
122        }
123    }
124}

126fn link_rlib(
127    sess: &Session,
128    codegen_results: &CodegenResults,
129    metadata: &EncodedMetadata,
130    out_filename: &Path,
131) {
132    let mut file_list = Vec::<&Path>::new();
133    for obj in codegen_results
134        .modules
135        .iter()
136        .filter_map(|m| m.object.as_ref())
137    {
138        file_list.push(obj);
139    }
140    for lib in codegen_results.crate_info.used_libraries.iter() {
141        if let NativeLibKind::Static {
142            bundle: None | Some(true),
143            ..
144        } = lib.kind
145        {
146            sess.dcx().err(format!(
147                "adding native library to rlib not supported yet: {}",
148                lib.name
149            ));
150        }
151    }
152
153    create_archive(&file_list, metadata.stub_or_full(), out_filename);
154}

156fn link_exe(
157    sess: &Session,
158    crate_type: CrateType,
159    out_filename: &Path,
160    codegen_results: &CodegenResults,
161    outputs: &OutputFilenames,
162    disambiguated_crate_name_for_dumps: &OsStr,
163) {
164    let mut objects = Vec::new();
165    let mut rlibs = Vec::new();
166    for obj in codegen_results
167        .modules
168        .iter()
169        .filter_map(|m| m.object.as_ref())
170    {
171        objects.push(obj.clone());
172    }
173
174    link_local_crate_native_libs_and_dependent_crate_libs(
175        &mut rlibs,
176        sess,
177        crate_type,
178        codegen_results,
179    );
180
181    let cg_args = CodegenArgs::from_session(sess);
182
183    // HACK(eddyb) this removes the `.json` in `.spv.json`, from `out_filename`.
184    let out_path_spv = out_filename.with_extension("");
185
186    let link_result = do_link(
187        sess,
188        &cg_args,
189        &objects,
190        &rlibs,
191        outputs,
192        disambiguated_crate_name_for_dumps,
193    );
194    let compile_result = match link_result {
195        linker::LinkResult::SingleModule(module) => {
196            let entry_points = entry_points(&module);
197            post_link_single_module(sess, &cg_args, *module, &out_path_spv, None);
198            CompileResult {
199                entry_points,
200                module: ModuleResult::SingleModule(out_path_spv),
201            }
202        }
203        linker::LinkResult::MultipleModules {
204            file_stem_to_entry_name_and_module,
205        } => {
206            let out_dir = out_path_spv.with_extension("spvs");
207            if !out_dir.is_dir() {
208                std::fs::create_dir_all(&out_dir).unwrap();
209            }
210
211            let entry_name_to_file_path: BTreeMap<_, _> = file_stem_to_entry_name_and_module
212                .into_iter()
213                .map(|(file_stem, (entry_name, module))| {
214                    let mut out_file_name = file_stem;
215                    out_file_name.push(".spv");
216                    let out_file_path = out_dir.join(out_file_name);
217                    post_link_single_module(
218                        sess,
219                        &cg_args,
220                        module,
221                        &out_file_path,
222                        Some(disambiguated_crate_name_for_dumps),
223                    );
224                    (entry_name, out_file_path)
225                })
226                .collect();
227            CompileResult {
228                entry_points: entry_name_to_file_path.keys().cloned().collect(),
229                module: ModuleResult::MultiModule(entry_name_to_file_path),
230            }
231        }
232    };
233
234    let file = File::create(out_filename).unwrap();
235    // FIXME(eddyb) move this functionality into `rustc_codegen_spirv_types`.
236    rustc_codegen_spirv_types::serde_json::to_writer(BufWriter::new(file), &compile_result)
237        .unwrap();
238}

240fn entry_points(module: &rspirv::dr::Module) -> Vec<String> {
241    module
242        .entry_points
243        .iter()
244        .filter(|inst| inst.class.opcode == rspirv::spirv::Op::EntryPoint)
245        .map(|inst| inst.operands[2].unwrap_literal_string().to_string())
246        .collect()
247}

249fn post_link_single_module(
250    sess: &Session,
251    cg_args: &CodegenArgs,
252    module: Module,
253    out_filename: &Path,
254    dump_prefix: Option<&OsStr>,
255) {
256    cg_args.do_disassemble(&module);
257    let spv_binary = module.assemble();
258
259    if let Some(dir) = &cg_args.dump_post_link {
260        // FIXME(eddyb) rename `filename` with `file_path` to make this less confusing.
261        let out_filename_file_name = out_filename.file_name().unwrap();
262        let dump_path = match dump_prefix {
263            Some(prefix) => dir.join(prefix).with_extension(out_filename_file_name),
264            None => dir.join(out_filename_file_name),
265        };
266        std::fs::write(dump_path, spirv_tools::binary::from_binary(&spv_binary)).unwrap();
267    }
268
269    let val_options = spirv_tools::val::ValidatorOptions {
270        relax_struct_store: cg_args.relax_struct_store,
271        relax_logical_pointer: cg_args.relax_logical_pointer,
272        before_legalization: false,
273        relax_block_layout: cg_args.relax_block_layout,
274        uniform_buffer_standard_layout: cg_args.uniform_buffer_standard_layout,
275        scalar_block_layout: cg_args.scalar_block_layout,
276        skip_block_layout: cg_args.skip_block_layout,
277        max_limits: vec![],
278    };
279    let opt_options = spirv_tools::opt::Options {
280        validator_options: Some(val_options.clone()),
281        max_id_bound: None,
282        preserve_bindings: cg_args.preserve_bindings,
283        preserve_spec_constants: false,
284    };
285
286    let spv_binary = if sess.opts.optimize != OptLevel::No
287        || (sess.opts.debuginfo == DebugInfo::None && cg_args.spirv_metadata == SpirvMetadata::None)
288    {
289        if cg_args.run_spirv_opt {
290            let _timer = sess.timer("link_spirv_opt");
291            do_spirv_opt(sess, cg_args, spv_binary, out_filename, opt_options)
292        } else {
293            let reason = match (sess.opts.optimize, sess.opts.debuginfo == DebugInfo::None) {
294                (OptLevel::No, true) => "debuginfo=None".to_string(),
295                (optlevel, false) => format!("optlevel={optlevel:?}"),
296                (optlevel, true) => format!("optlevel={optlevel:?}, debuginfo=None"),
297            };
298            sess.dcx().warn(format!(
299                "`spirv-opt` should have ran ({reason}) but was disabled by `--no-spirv-opt`"
300            ));
301            spv_binary
302        }
303    } else {
304        spv_binary
305    };
306
307    if cg_args.run_spirv_val {
308        do_spirv_val(sess, &spv_binary, out_filename, val_options);
309    }
310
311    {
312        let save_modules_timer = sess.timer("link_save_modules");
313        if let Err(e) = std::fs::write(out_filename, spirv_tools::binary::from_binary(&spv_binary))
314        {
315            let mut err = sess
316                .dcx()
317                .struct_err("failed to serialize spirv-binary to disk");
318            err.note(format!("module `{}`", out_filename.display()));
319            err.note(format!("I/O error: {e:#}"));
320            err.emit();
321        }
322
323        drop(save_modules_timer);
324    }
325}

327fn do_spirv_opt(
328    sess: &Session,
329    cg_args: &CodegenArgs,
330    spv_binary: Vec<u32>,
331    filename: &Path,
332    options: spirv_tools::opt::Options,
333) -> Vec<u32> {
334    use spirv_tools::{
335        error,
336        opt::{self, Optimizer},
337    };
338
339    let mut optimizer = opt::create(sess.target.options.env.parse().ok());
340
341    match sess.opts.optimize {
342        OptLevel::No => {}
343        OptLevel::Less | OptLevel::More | OptLevel::Aggressive => {
344            optimizer.register_performance_passes();
345        }
346        OptLevel::Size | OptLevel::SizeMin => {
347            optimizer.register_size_passes();
348        }
349    }
350
351    if sess.opts.debuginfo == DebugInfo::None && cg_args.spirv_metadata == SpirvMetadata::None {
352        optimizer
353            .register_pass(opt::Passes::EliminateDeadConstant)
354            .register_pass(opt::Passes::StripDebugInfo);
355    }
356
357    let result = optimizer.optimize(
358        &spv_binary,
359        &mut |msg: error::Message| {
360            use error::MessageLevel as Level;
361
362            // TODO: Adds spans here? Not sure how useful with binary, but maybe?
363
364            let mut err = match msg.level {
365                // We have to manually construct this after `forget_guarantee` was removed in
366                // <https://github.com/rust-lang/rust/commit/2cd14bc9394ca6675e08d02c02c5f9abfa813616>
367                Level::Error | Level::Fatal | Level::InternalError => {
368                    Diag::<()>::new(sess.dcx(), rustc_errors::Level::Error, msg.message)
369                }
370                Level::Warning => sess.dcx().struct_warn(msg.message),
371                Level::Info | Level::Debug => sess.dcx().struct_note(msg.message),
372            };
373
374            err.note(format!("module `{}`", filename.display()));
375            err.emit();
376        },
377        Some(options),
378    );
379
380    match result {
381        Ok(spirv_tools::binary::Binary::OwnedU32(words)) => words,
382        Ok(binary) => binary.as_words().to_vec(),
383        Err(e) => {
384            let mut err = sess.dcx().struct_warn(e.to_string());
385            err.note("spirv-opt failed, leaving as unoptimized");
386            err.note(format!("module `{}`", filename.display()));
387            err.emit();
388            spv_binary
389        }
390    }
391}

393fn do_spirv_val(
394    sess: &Session,
395    spv_binary: &[u32],
396    filename: &Path,
397    options: spirv_tools::val::ValidatorOptions,
398) {
399    use spirv_tools::val::{self, Validator};
400
401    let validator = val::create(sess.target.options.env.parse().ok());
402
403    if let Err(e) = validator.validate(spv_binary, Some(options)) {
404        let mut err = sess.dcx().struct_err(e.to_string());
405        err.note("spirv-val failed");
406        err.note(format!("module `{}`", filename.display()));
407        err.emit();
408    }
409}

411fn link_local_crate_native_libs_and_dependent_crate_libs(
412    rlibs: &mut Vec<PathBuf>,
413    sess: &Session,
414    crate_type: CrateType,
415    codegen_results: &CodegenResults,
416) {
417    if sess.opts.unstable_opts.link_native_libraries {
418        add_local_native_libraries(sess, codegen_results);
419    }
420    add_upstream_rust_crates(sess, rlibs, codegen_results, crate_type);
421    if sess.opts.unstable_opts.link_native_libraries {
422        add_upstream_native_libraries(sess, codegen_results, crate_type);
423    }
424}

426fn add_local_native_libraries(sess: &Session, codegen_results: &CodegenResults) {
427    let relevant_libs = codegen_results
428        .crate_info
429        .used_libraries
430        .iter()
431        .filter(|l| relevant_lib(sess, l));
432    assert_eq!(relevant_libs.count(), 0);
433}

435fn add_upstream_rust_crates(
436    sess: &Session,
437    rlibs: &mut Vec<PathBuf>,
438    codegen_results: &CodegenResults,
439    crate_type: CrateType,
440) {
441    let data = codegen_results
442        .crate_info
443        .dependency_formats
444        .get(&crate_type)
445        .expect("failed to find crate type in dependency format list");
446    for &cnum in &codegen_results.crate_info.used_crates {
447        let src = &codegen_results.crate_info.used_crate_source[&cnum];
448        match data[cnum] {
449            Linkage::NotLinked | Linkage::IncludedFromDylib => {}
450            Linkage::Static => rlibs.push(src.rlib.as_ref().unwrap().0.clone()),
451            //Linkage::Dynamic => rlibs.push(src.dylib.as_ref().unwrap().0.clone()),
452            Linkage::Dynamic => {
453                sess.dcx().err("TODO: Linkage::Dynamic not supported yet");
454            }
455        }
456    }
457}

459fn add_upstream_native_libraries(
460    sess: &Session,
461    codegen_results: &CodegenResults,
462    crate_type: CrateType,
463) {
464    let data = codegen_results
465        .crate_info
466        .dependency_formats
467        .get(&crate_type)
468        .expect("failed to find crate type in dependency format list");
469
470    for &cnum in &codegen_results.crate_info.used_crates {
471        for lib in codegen_results.crate_info.native_libraries[&cnum].iter() {
472            if !relevant_lib(sess, lib) {
473                continue;
474            }
475            match lib.kind {
476                NativeLibKind::Static {
477                    bundle: Some(false),
478                    ..
479                } if data[cnum] != Linkage::Static => {}
480
481                NativeLibKind::Static {
482                    bundle: None | Some(true),
483                    ..
484                } => {}
485
486                _ => sess.dcx().fatal(format!(
487                    "`NativeLibKind::{:?}` (name={:?}) not supported yet",
488                    lib.kind, lib.name
489                )),
490            }
491        }
492    }
493}

504fn create_archive(files: &[&Path], metadata: &[u8], out_filename: &Path) {
505    let files_with_names = files.iter().map(|file| {
506        (
507            file,
508            file.file_name()
509                .unwrap()
510                .to_str()
511                .expect("archive file names should be valid ASCII/UTF-8"),
512        )
513    });
514    let out_file = File::create(out_filename).unwrap();
515    let mut builder = GnuBuilder::new(
516        out_file,
517        iter::once(METADATA_FILENAME)
518            .chain(files_with_names.clone().map(|(_, name)| name))
519            .map(|name| name.as_bytes().to_vec())
520            .collect(),
521    );
522    builder
523        .append(
524            &Header::new(METADATA_FILENAME.as_bytes().to_vec(), metadata.len() as u64),
525            metadata,
526        )
527        .unwrap();
528
529    let mut filenames = FxHashSet::default();
530    filenames.insert(METADATA_FILENAME);
531    for (file, name) in files_with_names {
532        assert!(
533            filenames.insert(name),
534            "Duplicate filename in archive: {:?}",
535            file.file_name().unwrap()
536        );
537
538        // NOTE(eddyb) we can't use `append_path` or `append_file`, as they
539        // record too much metadata by default (mtime/UID/GID, at least),
540        // which is determintal to reproducible build artifacts, but also
541        // can misbehave in environments with high UIDs/GIDs (see #889).
542        let file = File::open(file).unwrap();
543        let header = Header::new(name.as_bytes().to_vec(), file.metadata().unwrap().len());
544        // NOTE(eddyb) either `fs::File`, or the result of `fs::read`, could fit
545        // here, but `fs::File` has specialized file->file copying on some OSes.
546        builder.append(&header, file).unwrap();
547    }
548    builder.into_inner().unwrap();
549}

551// HACK(eddyb) hiding the actual implementation to avoid `rspirv::dr::Loader`
552// being hardcoded (as future work may need to customize it for various reasons).
553pub fn with_rspirv_loader<E>(
554    f: impl FnOnce(&mut dyn rspirv::binary::Consumer) -> Result<(), E>,
555) -> Result<rspirv::dr::Module, E> {
556    let mut loader = rspirv::dr::Loader::new();
557    f(&mut loader)?;
558    Ok(loader.module())
559}

561/// This is the actual guts of linking: the rest of the link-related functions are just digging through rustc's
562/// shenanigans to collect all the object files we need to link.
563fn do_link(
564    sess: &Session,
565    cg_args: &CodegenArgs,
566    objects: &[PathBuf],
567    rlibs: &[PathBuf],
568    outputs: &OutputFilenames,
569    disambiguated_crate_name_for_dumps: &OsStr,
570) -> linker::LinkResult {
571    let load_modules_timer = sess.timer("link_load_modules");
572
573    let mut modules = Vec::new();
574    let mut add_module = |file_name: &OsStr, bytes: &[u8]| {
575        let module =
576            with_rspirv_loader(|loader| rspirv::binary::parse_bytes(bytes, loader)).unwrap();
577        if let Some(dir) = &cg_args.dump_pre_link {
578            // FIXME(eddyb) is it a good idea to re-`assemble` the `rspirv::dr`
579            // module, or should this just save the original bytes?
580            std::fs::write(
581                dir.join(file_name).with_extension("spv"),
582                spirv_tools::binary::from_binary(&module.assemble()),
583            )
584            .unwrap();
585        }
586        modules.push(module);
587    };
588
589    // `objects` are the plain obj files we need to link - usually produced by the final crate.
590    for obj in objects {
591        add_module(obj.file_name().unwrap(), &std::fs::read(obj).unwrap());
592    }
593
594    // `rlibs` are archive files we've created in `create_archive`, usually produced by crates that are being
595    // referenced. We need to unpack them and add the modules inside.
596    for rlib in rlibs {
597        let mut archive = Archive::new(File::open(rlib).unwrap());
598        while let Some(entry) = archive.next_entry() {
599            let mut entry = entry.unwrap();
600            if entry.header().identifier() != METADATA_FILENAME.as_bytes() {
601                // std::fs::read adds 1 to the size, so do the same here - see comment:
602                // https://github.com/rust-lang/rust/blob/72868e017bdade60603a25889e253f556305f996/library/std/src/fs.rs#L200-L202
603                let mut bytes = Vec::with_capacity(entry.header().size() as usize + 1);
604                entry.read_to_end(&mut bytes).unwrap();
605
606                let file_name = std::str::from_utf8(entry.header().identifier()).unwrap();
607                add_module(OsStr::new(file_name), &bytes);
608            }
609        }
610    }
611
612    drop(load_modules_timer);
613
614    // Do the link...
615    let link_result = linker::link(
616        sess,
617        modules,
618        &cg_args.linker_opts,
619        outputs,
620        disambiguated_crate_name_for_dumps,
621    );
622
623    if let Ok(v) = link_result {
624        v
625    } else {
626        sess.dcx().abort_if_errors();
627        bug!("Linker errored, but no error reported");
628    }
629}

633pub(crate) fn run_thin(
634    cgcx: &CodegenContext<SpirvCodegenBackend>,
635    modules: Vec<(String, SpirvModuleBuffer)>,
636    cached_modules: Vec<(SerializedModule<SpirvModuleBuffer>, WorkProduct)>,
637) -> Result<(Vec<LtoModuleCodegen<SpirvCodegenBackend>>, Vec<WorkProduct>), FatalError> {
638    if cgcx.opts.cg.linker_plugin_lto.enabled() {
639        unreachable!("We should never reach this case if the LTO step is deferred to the linker");
640    }
641    assert!(
642        cgcx.lto == Lto::ThinLocal,
643        "no actual LTO implemented in Rust-GPU"
644    );
645    let mut thin_buffers = Vec::with_capacity(modules.len());
646    let mut module_names = Vec::with_capacity(modules.len() + cached_modules.len());
647
648    for (name, buffer) in modules {
649        let cname = CString::new(name.clone()).unwrap();
650        thin_buffers.push(buffer);
651        module_names.push(cname);
652    }
653
654    let mut serialized_modules = Vec::with_capacity(cached_modules.len());
655
656    for (sm, wp) in cached_modules {
657        let _slice_u8 = sm.data();
658        serialized_modules.push(sm);
659        module_names.push(CString::new(wp.cgu_name).unwrap());
660    }
661
662    let shared = Arc::new(ThinShared {
663        data: (),
664        thin_buffers,
665        serialized_modules,
666        module_names,
667    });
668
669    let mut opt_jobs = vec![];
670    for (module_index, _) in shared.module_names.iter().enumerate() {
671        opt_jobs.push(LtoModuleCodegen::Thin(ThinModule {
672            shared: shared.clone(),
673            idx: module_index,
674        }));
675    }
676
677    Ok((opt_jobs, vec![]))
678}
