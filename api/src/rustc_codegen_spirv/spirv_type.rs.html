17/// Spir-v types are represented as simple Words, which are the `result_id` of instructions like
18/// `OpTypeInteger`. Sometimes, however, we want to inspect one of these Words and ask questions
19/// like "Is this an `OpTypeInteger`? How many bits does it have?". This struct holds all of that
20/// information. All types that are emitted are registered in `CodegenCx`, so you can always look
21/// up the definition of a `Word` via `cx.lookup_type`. Note that this type doesn't actually store
22/// the `result_id` of the type declaration instruction, merely the contents.
23//
24// FIXME(eddyb) should `SpirvType`s be behind `&'tcx` from `tcx.arena.dropless`?
25#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash)]
26pub enum SpirvType<'tcx> {
27    Void,
28    Bool,
29    Integer(u32, bool),
30    Float(u32),
31    /// This uses the rustc definition of "adt", i.e. a struct, enum, or union
32    Adt {
33        /// Not emitted into SPIR-V, but used to avoid too much deduplication,
34        /// which could result in one SPIR-V `OpType*` having many names
35        /// (not in itself an issue, but it makes error reporting harder).
36        def_id: Option<DefId>,
37
38        align: Align,
39        size: Option<Size>,
40        field_types: &'tcx [Word],
41        field_offsets: &'tcx [Size],
42        field_names: Option<&'tcx [Symbol]>,
43    },
44    Vector {
45        element: Word,
46        /// Note: vector count is literal.
47        count: u32,
48    },
49    Matrix {
50        element: Word,
51        /// Note: matrix count is literal.
52        count: u32,
53    },
54    Array {
55        element: Word,
56        /// Note: array count is ref to constant.
57        count: SpirvValue,
58    },
59    RuntimeArray {
60        element: Word,
61    },
62    Pointer {
63        pointee: Word,
64    },
65    Function {
66        return_type: Word,
67        arguments: &'tcx [Word],
68    },
69    Image {
70        sampled_type: Word,
71        dim: Dim,
72        depth: u32,
73        arrayed: u32,
74        multisampled: u32,
75        sampled: u32,
76        image_format: ImageFormat,
77    },
78    Sampler,
79    SampledImage {
80        image_type: Word,
81    },
82
83    /// `OpTypeStruct` decorated with `Block`, required by Vulkan (and OpenGL)
84    /// for `PushConstant`, `Uniform` and `StorageBuffer` interface variables.
85    InterfaceBlock {
86        inner_type: Word,
87    },
88
89    AccelerationStructureKhr,
90    RayQueryKhr,
91}

94    /// Note: `Builder::type_*` should be called *nowhere else* but here, to ensure
95    /// `CodegenCx::type_defs` stays up-to-date
96    pub fn def(self, def_span: Span, cx: &CodegenCx<'_>) -> Word {
97        if let Some(cached) = cx.type_cache.get(&self) {
98            return cached;
99        }
100        let id = Some(cx.emit_global().id());
101        let result = match self {
102            Self::Void => cx.emit_global().type_void_id(id),
103            Self::Bool => cx.emit_global().type_bool_id(id),
104            Self::Integer(width, signed) => cx.emit_global().type_int_id(id, width, signed as u32),
105            Self::Float(width) => cx.emit_global().type_float_id(id, width),
106            Self::Adt {
107                def_id: _,
108                align: _,
109                size: _,
110                field_types,
111                field_offsets,
112                field_names,
113            } => {
114                let mut emit = cx.emit_global();
115                let result = emit.type_struct_id(id, field_types.iter().cloned());
116                // The struct size is only used in our own sizeof_in_bits() (used in e.g. ArrayStride decoration)
117                for (index, offset) in field_offsets.iter().copied().enumerate() {
118                    emit.member_decorate(
119                        result,
120                        index as u32,
121                        Decoration::Offset,
122                        [Operand::LiteralBit32(offset.bytes() as u32)]
123                            .iter()
124                            .cloned(),
125                    );
126                }
127                if let Some(field_names) = field_names {
128                    for (index, field_name) in field_names.iter().enumerate() {
129                        emit.member_name(result, index as u32, field_name.as_str());
130                    }
131                }
132                result
133            }
134            Self::Vector { element, count } => cx.emit_global().type_vector_id(id, element, count),
135            Self::Matrix { element, count } => cx.emit_global().type_matrix_id(id, element, count),
136            Self::Array { element, count } => {
137                let result = cx
138                    .emit_global()
139                    .type_array_id(id, element, count.def_cx(cx));
140                Self::decorate_array_stride(result, element, cx);
141                result
142            }
143            Self::RuntimeArray { element } => {
144                let result = cx.emit_global().type_runtime_array_id(id, element);
145                Self::decorate_array_stride(result, element, cx);
146                result
147            }
148            Self::Pointer { pointee } => {
149                // NOTE(eddyb) we emit `StorageClass::Generic` here, but later
150                // the linker will specialize the entire SPIR-V module to use
151                // storage classes inferred from `OpVariable`s.
152                let result = cx
153                    .emit_global()
154                    .type_pointer(id, StorageClass::Generic, pointee);
155                // no pointers to functions
156                if let SpirvType::Function { .. } = cx.lookup_type(pointee) {
157                    // FIXME(eddyb) use the `SPV_INTEL_function_pointers` extension.
158                    cx.zombie_with_span(result, def_span, "function pointer types are not allowed");
159                }
160                result
161            }
162            Self::Function {
163                return_type,
164                arguments,
165            } => cx
166                .emit_global()
167                .type_function_id(id, return_type, arguments.iter().cloned()),
168            Self::Image {
169                sampled_type,
170                dim,
171                depth,
172                arrayed,
173                multisampled,
174                sampled,
175                image_format,
176            } => cx.emit_global().type_image_id(
177                id,
178                sampled_type,
179                dim,
180                depth,
181                arrayed,
182                multisampled,
183                sampled,
184                image_format,
185                None,
186            ),
187            Self::Sampler => cx.emit_global().type_sampler_id(id),
188            Self::AccelerationStructureKhr => {
189                cx.emit_global().type_acceleration_structure_khr_id(id)
190            }
191            Self::RayQueryKhr => cx.emit_global().type_ray_query_khr_id(id),
192            Self::SampledImage { image_type } => {
193                cx.emit_global().type_sampled_image_id(id, image_type)
194            }
195
196            Self::InterfaceBlock { inner_type } => {
197                let mut emit = cx.emit_global();
198                let result = emit.type_struct_id(id, iter::once(inner_type));
199                emit.decorate(result, Decoration::Block, iter::empty());
200                emit.member_decorate(
201                    result,
202                    0,
203                    Decoration::Offset,
204                    [Operand::LiteralBit32(0)].iter().cloned(),
205                );
206                result
207            }
208        };
209        cx.type_cache_def(result, self.tcx_arena_alloc_slices(cx), def_span);
210        result
211    }

213    fn decorate_array_stride(result: u32, element: u32, cx: &CodegenCx<'_>) {
214        let mut emit = cx.emit_global();
215        let ty = cx.lookup_type(element);
216        if let Some(element_size) = ty.physical_size(cx) {
217            // ArrayStride decoration wants in *bytes*
218            emit.decorate(
219                result,
220                Decoration::ArrayStride,
221                iter::once(Operand::LiteralBit32(element_size.bytes() as u32)),
222            );
223        }
224    }

226    /// `def_with_id` is used by the `RecursivePointeeCache` to handle `OpTypeForwardPointer`: when
227    /// emitting the subsequent `OpTypePointer`, the ID is already known and must be re-used.
228    pub fn def_with_id(self, cx: &CodegenCx<'_>, def_span: Span, id: Word) -> Word {
229        if let Some(cached) = cx.type_cache.get(&self) {
230            assert_eq!(cached, id);
231            return cached;
232        }
233        let result = match self {
234            Self::Pointer { pointee } => {
235                // NOTE(eddyb) we emit `StorageClass::Generic` here, but later
236                // the linker will specialize the entire SPIR-V module to use
237                // storage classes inferred from `OpVariable`s.
238                let result =
239                    cx.emit_global()
240                        .type_pointer(Some(id), StorageClass::Generic, pointee);
241                // no pointers to functions
242                if let SpirvType::Function { .. } = cx.lookup_type(pointee) {
243                    // FIXME(eddyb) use the `SPV_INTEL_function_pointers` extension.
244                    cx.zombie_with_span(result, def_span, "function pointer types are not allowed");
245                }
246                result
247            }
248            ref other => cx
249                .tcx
250                .dcx()
251                .fatal(format!("def_with_id invalid for type {other:?}")),
252        };
253        cx.type_cache_def(result, self.tcx_arena_alloc_slices(cx), def_span);
254        result
255    }

257    /// In addition to `SpirvType::def`, also name the resulting type (with `OpName`).
258    pub fn def_with_name<'tcx>(
259        self,
260        cx: &CodegenCx<'tcx>,
261        def_span: Span,
262        name_key: TyLayoutNameKey<'tcx>,
263    ) -> Word {
264        let id = self.def(def_span, cx);
265
266        // Only emit `OpName` if this is the first time we see this name.
267        let mut type_names = cx.type_cache.type_names.borrow_mut();
268        if type_names.entry(id).or_default().insert(name_key) {
269            cx.emit_global().name(id, name_key.to_string());
270        }
271
272        id
273    }

275    pub fn sizeof(&self, cx: &CodegenCx<'_>) -> Option<Size> {
276        let result = match *self {
277            // Types that have a dynamic size, or no concept of size at all.
278            Self::Void | Self::RuntimeArray { .. } | Self::Function { .. } => return None,
279
280            Self::Bool => Size::from_bytes(1),
281            Self::Integer(width, _) | Self::Float(width) => Size::from_bits(width),
282            Self::Adt { size, .. } => size?,
283            Self::Vector { element, count } => {
284                cx.lookup_type(element).sizeof(cx)? * count.next_power_of_two() as u64
285            }
286            Self::Matrix { element, count } => cx.lookup_type(element).sizeof(cx)? * count as u64,
287            Self::Array { element, count } => {
288                cx.lookup_type(element).sizeof(cx)?
289                    * cx.builder
290                        .lookup_const_scalar(count)
291                        .unwrap()
292                        .try_into()
293                        .unwrap()
294            }
295            Self::Pointer { .. } => cx.tcx.data_layout.pointer_size,
296            Self::Image { .. }
297            | Self::AccelerationStructureKhr
298            | Self::RayQueryKhr
299            | Self::Sampler
300            | Self::SampledImage { .. }
301            | Self::InterfaceBlock { .. } => Size::from_bytes(4),
302        };
303        Some(result)
304    }

306    pub fn alignof(&self, cx: &CodegenCx<'_>) -> Align {
307        match *self {
308            // Types that have no concept of size or alignment.
309            Self::Void | Self::Function { .. } => Align::from_bytes(0).unwrap(),
310
311            Self::Bool => Align::from_bytes(1).unwrap(),
312            Self::Integer(width, _) | Self::Float(width) => Align::from_bits(width as u64).unwrap(),
313            Self::Adt { align, .. } => align,
314            // Vectors have size==align
315            Self::Vector { .. } => Align::from_bytes(
316                self.sizeof(cx)
317                    .expect("alignof: Vectors must be sized")
318                    .bytes(),
319            )
320            .expect("alignof: Vectors must have power-of-2 size"),
321            Self::Array { element, .. }
322            | Self::RuntimeArray { element }
323            | Self::Matrix { element, .. } => cx.lookup_type(element).alignof(cx),
324            Self::Pointer { .. } => cx.tcx.data_layout.pointer_align.abi,
325            Self::Image { .. }
326            | Self::AccelerationStructureKhr
327            | Self::RayQueryKhr
328            | Self::Sampler
329            | Self::SampledImage { .. }
330            | Self::InterfaceBlock { .. } => Align::from_bytes(4).unwrap(),
331        }
332    }

334    /// Get the physical size of the type needed for explicit layout decorations.
335    #[allow(clippy::match_same_arms)]
336    pub fn physical_size(&self, cx: &CodegenCx<'_>) -> Option<Size> {
337        match *self {
338            // TODO(jwollen) Handle physical pointers (PhysicalStorageBuffer)
339            Self::Pointer { .. } => None,
340
341            Self::Adt { size, .. } => size,
342
343            Self::Array { element, count } => Some(
344                cx.lookup_type(element).physical_size(cx)?
345                    * cx.builder
346                        .lookup_const_scalar(count)
347                        .unwrap()
348                        .try_into()
349                        .unwrap(),
350            ),
351
352            // Always unsized types
353            Self::InterfaceBlock { .. } | Self::RayQueryKhr | Self::SampledImage { .. } => None,
354
355            // Descriptor types
356            Self::Image { .. } | Self::AccelerationStructureKhr | Self::Sampler => None,
357
358            // Primitive types
359            ty => ty.sizeof(cx),
360        }
361    }

363    /// Replace `&[T]` fields with `&'tcx [T]` ones produced by calling
364    /// `tcx.arena.dropless.alloc_slice(...)` - this is done late for two reasons:
365    /// 1. it avoids allocating in the arena when the cache would be hit anyway,
366    ///    which would create "garbage" (as in, unreachable allocations)
367    ///    (ideally these would also be interned, but that's even more refactors)
368    /// 2. an empty slice is disallowed (as it's usually handled as a special
369    ///    case elsewhere, e.g. `rustc`'s `ty::List` - sadly we can't use that)
370    fn tcx_arena_alloc_slices<'tcx>(self, cx: &CodegenCx<'tcx>) -> SpirvType<'tcx> {
371        fn arena_alloc_slice<'tcx, T: Copy>(cx: &CodegenCx<'tcx>, xs: &[T]) -> &'tcx [T] {
372            if xs.is_empty() {
373                &[]
374            } else {
375                cx.tcx.arena.dropless.alloc_slice(xs)
376            }
377        }
378
379        match self {
380            // FIXME(eddyb) these are all noop cases, could they be automated?
381            SpirvType::Void => SpirvType::Void,
382            SpirvType::Bool => SpirvType::Bool,
383            SpirvType::Integer(width, signedness) => SpirvType::Integer(width, signedness),
384            SpirvType::Float(width) => SpirvType::Float(width),
385            SpirvType::Vector { element, count } => SpirvType::Vector { element, count },
386            SpirvType::Matrix { element, count } => SpirvType::Matrix { element, count },
387            SpirvType::Array { element, count } => SpirvType::Array { element, count },
388            SpirvType::RuntimeArray { element } => SpirvType::RuntimeArray { element },
389            SpirvType::Pointer { pointee } => SpirvType::Pointer { pointee },
390            SpirvType::Image {
391                sampled_type,
392                dim,
393                depth,
394                arrayed,
395                multisampled,
396                sampled,
397                image_format,
398            } => SpirvType::Image {
399                sampled_type,
400                dim,
401                depth,
402                arrayed,
403                multisampled,
404                sampled,
405                image_format,
406            },
407            SpirvType::Sampler => SpirvType::Sampler,
408            SpirvType::SampledImage { image_type } => SpirvType::SampledImage { image_type },
409            SpirvType::InterfaceBlock { inner_type } => SpirvType::InterfaceBlock { inner_type },
410            SpirvType::AccelerationStructureKhr => SpirvType::AccelerationStructureKhr,
411            SpirvType::RayQueryKhr => SpirvType::RayQueryKhr,
412
413            // Only these variants have any slices to arena-allocate.
414            SpirvType::Adt {
415                def_id,
416                align,
417                size,
418                field_types,
419                field_offsets,
420                field_names,
421            } => SpirvType::Adt {
422                def_id,
423                align,
424                size,
425                field_types: arena_alloc_slice(cx, field_types),
426                field_offsets: arena_alloc_slice(cx, field_offsets),
427                field_names: field_names.map(|field_names| arena_alloc_slice(cx, field_names)),
428            },
429            SpirvType::Function {
430                return_type,
431                arguments,
432            } => SpirvType::Function {
433                return_type,
434                arguments: arena_alloc_slice(cx, arguments),
435            },
436        }
437    }

453/// Types can be recursive, e.g. a struct can contain a pointer to itself. So, we need to keep
454/// track of a stack of what types are currently being printed, to not infinitely loop.
455/// Unfortunately, unlike `fmt::Display`, we can't easily pass down the "stack" of
456/// currently-being-printed types, so we use a global static.
457static DEBUG_STACK: LazyLock<Mutex<Vec<Word>>> = LazyLock::new(|| Mutex::new(Vec::new()));
458
459impl fmt::Debug for SpirvTypePrinter<'_, '_> {
460    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
461        {
462            let mut debug_stack = DEBUG_STACK.lock().unwrap();
463            if debug_stack.contains(&self.id) {
464                return write!(f, "<recursive type id={}>", self.id);
465            }
466            debug_stack.push(self.id);
467        }
468        let res = match self.ty {
469            SpirvType::Void => f.debug_struct("Void").field("id", &self.id).finish(),
470            SpirvType::Bool => f.debug_struct("Bool").field("id", &self.id).finish(),
471            SpirvType::Integer(width, signedness) => f
472                .debug_struct("Integer")
473                .field("id", &self.id)
474                .field("width", &width)
475                .field("signedness", &signedness)
476                .finish(),
477            SpirvType::Float(width) => f
478                .debug_struct("Float")
479                .field("id", &self.id)
480                .field("width", &width)
481                .finish(),
482            SpirvType::Adt {
483                def_id,
484                align,
485                size,
486                field_types,
487                field_offsets,
488                field_names,
489            } => {
490                let fields = field_types
491                    .iter()
492                    .map(|&f| self.cx.debug_type(f))
493                    .collect::<Vec<_>>();
494                f.debug_struct("Adt")
495                    .field("id", &self.id)
496                    .field("def_id", &def_id)
497                    .field("align", &align)
498                    .field("size", &size)
499                    .field("field_types", &fields)
500                    .field("field_offsets", &field_offsets)
501                    .field("field_names", &field_names)
502                    .finish()
503            }
504            SpirvType::Vector { element, count } => f
505                .debug_struct("Vector")
506                .field("id", &self.id)
507                .field("element", &self.cx.debug_type(element))
508                .field("count", &count)
509                .finish(),
510            SpirvType::Matrix { element, count } => f
511                .debug_struct("Matrix")
512                .field("id", &self.id)
513                .field("element", &self.cx.debug_type(element))
514                .field("count", &count)
515                .finish(),
516            SpirvType::Array { element, count } => f
517                .debug_struct("Array")
518                .field("id", &self.id)
519                .field("element", &self.cx.debug_type(element))
520                .field(
521                    "count",
522                    &self
523                        .cx
524                        .builder
525                        .lookup_const_scalar(count)
526                        .expect("Array type has invalid count value"),
527                )
528                .finish(),
529            SpirvType::RuntimeArray { element } => f
530                .debug_struct("RuntimeArray")
531                .field("id", &self.id)
532                .field("element", &self.cx.debug_type(element))
533                .finish(),
534            SpirvType::Pointer { pointee } => f
535                .debug_struct("Pointer")
536                .field("id", &self.id)
537                .field("pointee", &self.cx.debug_type(pointee))
538                .finish(),
539            SpirvType::Function {
540                return_type,
541                arguments,
542            } => {
543                let args = arguments
544                    .iter()
545                    .map(|&a| self.cx.debug_type(a))
546                    .collect::<Vec<_>>();
547                f.debug_struct("Function")
548                    .field("id", &self.id)
549                    .field("return_type", &self.cx.lookup_type(return_type))
550                    .field("arguments", &args)
551                    .finish()
552            }
553            SpirvType::Image {
554                sampled_type,
555                dim,
556                depth,
557                arrayed,
558                multisampled,
559                sampled,
560                image_format,
561            } => f
562                .debug_struct("Image")
563                .field("id", &self.id)
564                .field("sampled_type", &self.cx.debug_type(sampled_type))
565                .field("dim", &dim)
566                .field("depth", &depth)
567                .field("arrayed", &arrayed)
568                .field("multisampled", &multisampled)
569                .field("sampled", &sampled)
570                .field("image_format", &image_format)
571                .finish(),
572            SpirvType::Sampler => f.debug_struct("Sampler").field("id", &self.id).finish(),
573            SpirvType::SampledImage { image_type } => f
574                .debug_struct("SampledImage")
575                .field("id", &self.id)
576                .field("image_type", &self.cx.debug_type(image_type))
577                .finish(),
578            SpirvType::InterfaceBlock { inner_type } => f
579                .debug_struct("InterfaceBlock")
580                .field("id", &self.id)
581                .field("inner_type", &self.cx.debug_type(inner_type))
582                .finish(),
583            SpirvType::AccelerationStructureKhr => f.debug_struct("AccelerationStructure").finish(),
584            SpirvType::RayQueryKhr => f.debug_struct("RayQuery").finish(),
585        };
586        {
587            let mut debug_stack = DEBUG_STACK.lock().unwrap();
588            debug_stack.pop();
589        }
590        res
591    }
592}

594/// Types can be recursive, e.g. a struct can contain a pointer to itself. So, we need to keep
595/// track of a stack of what types are currently being printed, to not infinitely loop. So, we only
596/// use `fmt::Display::fmt` as an "entry point", and then call through to our own (recursive)
597/// custom function that has a parameter for the current stack. Make sure to not call Display on a
598/// type inside the custom function!
599impl fmt::Display for SpirvTypePrinter<'_, '_> {
600    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
601        self.display(&mut Vec::new(), f)
602    }
603}

605impl SpirvTypePrinter<'_, '_> {
606    fn display(&self, stack: &mut Vec<Word>, f: &mut fmt::Formatter<'_>) -> fmt::Result {
607        fn ty(
608            cx: &CodegenCx<'_>,
609            stack: &mut Vec<Word>,
610            f: &mut fmt::Formatter<'_>,
611            ty: Word,
612        ) -> fmt::Result {
613            if stack.contains(&ty) {
614                f.write_str("<recursive type>")
615            } else {
616                stack.push(ty);
617                let result = cx.debug_type(ty).display(stack, f);
618                assert_eq!(ty, stack.pop().unwrap());
619                result
620            }
621        }
622        match self.ty {
623            SpirvType::Void => f.write_str("void"),
624            SpirvType::Bool => f.write_str("bool"),
625            SpirvType::Integer(width, signedness) => {
626                let prefix = if signedness { "i" } else { "u" };
627                write!(f, "{prefix}{width}")
628            }
629            SpirvType::Float(width) => write!(f, "f{width}"),
630            SpirvType::Adt {
631                def_id: _,
632                align: _,
633                size: _,
634                field_types,
635                field_offsets: _,
636                ref field_names,
637            } => {
638                write!(f, "struct")?;
639
640                // HACK(eddyb) use the first name (in insertion order, i.e.
641                // from the first invocation of `def_with_name` for this type)
642                // even when this may not be correct - a better solution could
643                // be to pick the shortest name (which could work well when
644                // newtypes are involved).
645                let first_name = {
646                    let type_names = self.cx.type_cache.type_names.borrow();
647                    type_names
648                        .get(&self.id)
649                        .and_then(|names| names.iter().next().copied())
650                };
651
652                if let Some(name) = first_name {
653                    write!(f, " {name}")?;
654                }
655
656                f.write_str(" { ")?;
657                for (index, &field) in field_types.iter().enumerate() {
658                    let suffix = if index + 1 == field_types.len() {
659                        ""
660                    } else {
661                        ", "
662                    };
663                    if let Some(field_names) = field_names {
664                        write!(f, "{}: ", field_names[index])?;
665                    }
666                    ty(self.cx, stack, f, field)?;
667                    write!(f, "{suffix}")?;
668                }
669                f.write_str(" }")
670            }
671            SpirvType::Vector { element, count } | SpirvType::Matrix { element, count } => {
672                ty(self.cx, stack, f, element)?;
673                write!(f, "x{count}")
674            }
675            SpirvType::Array { element, count } => {
676                let len = self.cx.builder.lookup_const_scalar(count);
677                let len = len.expect("Array type has invalid count value");
678                f.write_str("[")?;
679                ty(self.cx, stack, f, element)?;
680                write!(f, "; {len}]")
681            }
682            SpirvType::RuntimeArray { element } => {
683                f.write_str("[")?;
684                ty(self.cx, stack, f, element)?;
685                f.write_str("]")
686            }
687            SpirvType::Pointer { pointee } => {
688                f.write_str("*")?;
689                ty(self.cx, stack, f, pointee)
690            }
691            SpirvType::Function {
692                return_type,
693                arguments,
694            } => {
695                f.write_str("fn(")?;
696                for (index, &arg) in arguments.iter().enumerate() {
697                    let suffix = if index + 1 == arguments.len() {
698                        ""
699                    } else {
700                        ", "
701                    };
702                    ty(self.cx, stack, f, arg)?;
703                    write!(f, "{suffix}")?;
704                }
705                f.write_str(") -> ")?;
706                ty(self.cx, stack, f, return_type)
707            }
708            SpirvType::Image {
709                sampled_type,
710                dim,
711                depth,
712                arrayed,
713                multisampled,
714                sampled,
715                image_format,
716            } => f
717                .debug_struct("Image")
718                .field("sampled_type", &self.cx.debug_type(sampled_type))
719                .field("dim", &dim)
720                .field("depth", &depth)
721                .field("arrayed", &arrayed)
722                .field("multisampled", &multisampled)
723                .field("sampled", &sampled)
724                .field("image_format", &image_format)
725                .finish(),
726            SpirvType::Sampler => f.write_str("Sampler"),
727            SpirvType::SampledImage { image_type } => f
728                .debug_struct("SampledImage")
729                .field("image_type", &self.cx.debug_type(image_type))
730                .finish(),
731            SpirvType::InterfaceBlock { inner_type } => {
732                f.write_str("interface block { ")?;
733                ty(self.cx, stack, f, inner_type)?;
734                f.write_str(" }")
735            }
736            SpirvType::AccelerationStructureKhr => f.write_str("AccelerationStructureKhr"),
737            SpirvType::RayQueryKhr => f.write_str("RayQuery"),
738        }
739    }
740}

742#[derive(Default)]
743pub struct TypeCache<'tcx> {
744    pub id_to_spirv_type: RefCell<FxHashMap<Word, SpirvType<'tcx>>>,
745    pub spirv_type_to_id: RefCell<FxHashMap<SpirvType<'tcx>, Word>>,
746
747    /// Recursive pointer breaking
748    pub recursive_pointee_cache: RecursivePointeeCache<'tcx>,
749    /// Set of names for a type (only `SpirvType::Adt` currently).
750    /// The same `OpType*` may have multiple names if it's e.g. a generic
751    /// `struct` where the generic parameters result in the same field types.
752    type_names: RefCell<FxHashMap<Word, IndexSet<TyLayoutNameKey<'tcx>>>>,
753}
754
755impl<'tcx> TypeCache<'tcx> {
756    fn get(&self, ty: &SpirvType<'_>) -> Option<Word> {
757        self.spirv_type_to_id.borrow().get(ty).copied()
758    }
759
760    #[track_caller]
761    pub fn lookup(&self, id: Word) -> SpirvType<'tcx> {
762        *self
763            .id_to_spirv_type
764            .borrow()
765            .get(&id)
766            .expect("tried to lookup ID that wasn't a type, or has no definition")
767    }
768}
769
770impl<'tcx> CodegenCx<'tcx> {
771    fn type_cache_def(&self, id: Word, ty: SpirvType<'tcx>, def_span: Span) {
772        if let Some(old_ty) = self.type_cache.id_to_spirv_type.borrow_mut().insert(id, ty) {
773            span_bug!(
774                def_span,
775                "SPIR-V type with ID %{id} is being redefined\n\
776                old type: {old_ty}\n\
777                new type: {ty}",
778                old_ty = old_ty.debug(id, self),
779                ty = ty.debug(id, self)
780            );
781        }
782
783        if let Some(old_id) = self.type_cache.spirv_type_to_id.borrow_mut().insert(ty, id) {
784            span_bug!(
785                def_span,
786                "SPIR-V type is changing IDs (%{old_id} -> %{id}):\n\
787                {ty}",
788                ty = ty.debug(id, self)
789            );
790        }
791    }
792}
