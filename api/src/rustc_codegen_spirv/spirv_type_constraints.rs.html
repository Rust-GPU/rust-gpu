1//! SPIR-V type constraints. Can be used to perform a subset of validation,
2//! or for inference purposes.
3//!
4//! Only type/storage-class equality is currently handled here, no concrete
5//! type/storage-class constraints, nor anything involving non-type/storage-class
6//! operands. While more constraints could be supported, encoding all the possible
7//! rules for them may be challenging.
8//!
9//! Type constraints could be provided in two representations:
10//! * static/generic: the constraints are built up from generic types
11//!   (i.e. like iterator combinators), and passed to a generic trait method
12//!   to "apply" them, allowing efficient code to be generated, similar to
13//!   e.g. handcrafted imperative inference logic, but more flexible
14//! * dynamic/value: the constraints are a tree of `enum` values (like an AST)
15//!   that has to be visited (using recursion and `match`) to apply, which is
16//!   less efficient than the "static representation", but even more flexible
17//!   (as full introspection of the constraints is possible)
18//!
19//! Only the "dynamic representation" is currently implemented, as it's simpler.
20//! If the "static representation" ends up being required (for performance reasons),
21//! the "dynamic representation" could be generated from it using associated `const`s.
22
23use rspirv::spirv::Op;

25/// Helper trait to allow macros to work uniformly across different pattern types.
26trait Pat {
27    /// Unconstrained pattern, i.e. that matches everything.
28    const ANY: Self;
29}
30
31/// Pattern for a SPIR-V storage class, dynamic representation (see module-level docs).
32#[derive(Debug, PartialEq, Eq)]
33pub enum StorageClassPat {
34    /// Unconstrained storage class.
35    Any,
36
37    /// Storage class variable: all occurrences of `Var(i)` with the same `i` must be
38    /// identical storage classes. For convenience, these associated consts are provided:
39    /// * `StorageClassPat::S` for `StorageClassPat::Var(0)`
40    Var(usize),
41}
42
43impl Pat for StorageClassPat {
44    const ANY: Self = Self::Any;
45}
46
47impl StorageClassPat {
48    pub const S: Self = Self::Var(0);
49}

51/// Pattern for a SPIR-V type, dynamic representation (see module-level docs).
52#[derive(Debug, PartialEq, Eq)]
53pub enum TyPat<'a> {
54    /// Unconstrained type.
55    Any,
56
57    /// Type variable: all occurrences of `Var(i)` with the same `i` must be
58    /// identical types. For convenience, these associated consts are provided:
59    /// * `TyPat::T` for `TyPat::Var(0)`
60    Var(usize),
61
62    /// One of the two patterns must match.
63    Either(&'a TyPat<'a>, &'a TyPat<'a>),
64
65    /// `OpTypeVoid`: this is the only exception to the limitations of
66    /// "no concrete type constraints", and is used solely to disambiguate
67    /// image sampling instructions, which are specified using phrasing like:
68    /// > Its components must be the same as *Sampled Type* of the underlying
69    /// > `OpTypeImage` (unless that underlying *Sampled Type* is `OpTypeVoid`).
70    Void,
71
72    /// `OpTypePointer`, with an inner pattern for its *Storage Class* operand,
73    /// and another for its *Type* operand.
74    Pointer(&'a StorageClassPat, &'a TyPat<'a>),
75
76    // FIXME(eddyb) try to DRY the same-shape patterns below.
77    //
78    /// `OpTypeArray`, with an inner pattern for its *Element Type* operand.
79    Array(&'a TyPat<'a>),
80
81    /// `OpTypeVector`, with an inner pattern for its *Component Type* operand.
82    Vector(&'a TyPat<'a>),
83
84    /// `OpTypeVector`, with an inner pattern for its *Component Type* operand,
85    /// and a *Component Count* of `4`, used solely for image sampling instructions.
86    Vector4(&'a TyPat<'a>),
87
88    /// `OpTypeMatrix`, with an inner pattern for its *Column Type* operand.
89    Matrix(&'a TyPat<'a>),
90
91    /// `OpTypeImage`, with an inner pattern for its *Sampled Type* operand.
92    Image(&'a TyPat<'a>),
93
94    /// `OpTypePipe`, with an inner pattern for what is referred to as its
95    /// "data type", despite `OpTypePipe` itself not taking any operands.
96    // FIXME(eddyb) figure out why the spec talks about `OpTypePipe`s' "data type"s
97    Pipe(&'a TyPat<'a>),
98
99    /// `OpTypeSampledImage`, with an inner pattern for its *Image Type* operand.
100    SampledImage(&'a TyPat<'a>),
101
102    /// `OpTypeStruct`, with an inner pattern for its *Member N Type* operands.
103    Struct(&'a TyListPat<'a>),
104
105    /// `OpTypeFunction`, with an inner pattern for its *Return Type* operand,
106    /// and another for its *Parameter N Type* operands.
107    Function(&'a TyPat<'a>, &'a TyListPat<'a>),
108
109    /// Index the composite type matched by the inner pattern (like a `Var`)
110    /// by all of the value operands of the instruction which don't have their
111    /// types consumed explicitly, i.e. the indices operands correspond to
112    /// the `TyListPat::Any` part of the `inputs` of the `InstSig`,
113    /// e.g. in `sig! { (_, T, _, ..) -> IndexComposite(T) }`, the output type
114    /// has to be identical to the result of indexing `T` by all the value
115    /// operands *other than* the first 3.
116    // FIXME(eddyb) reconsider this design, to be less magical
117    IndexComposite(&'a TyPat<'a>),
118}
119
120impl Pat for TyPat<'_> {
121    const ANY: Self = Self::Any;
122}
123
124impl TyPat<'_> {
125    pub const T: Self = Self::Var(0);
126}

128/// Pattern for a list of SPIR-V types, dynamic representation (see module-level docs).
129///
130/// "Type lists" are used for `OpTypeStruct` fields, `OpTypeFunction` parameters,
131/// and operand types of instructions signatures.
132#[derive(Debug, PartialEq, Eq)]
133pub enum TyListPat<'a> {
134    /// Unconstrained type list (any length, and the types can freely vary).
135    Any,
136
137    /// Type list variable: all occurrences of `Var(i)` with the same `i` must be
138    /// identical type lists. For convenience, these associated consts are provided:
139    /// * `TyListPat::TS` for `TyListPat::Var(0)`
140    Var(usize),
141
142    /// Uniform repeat type list: equivalent to repeating the inner type list
143    /// pattern (which must be finite), enough times to cover the whole list.
144    Repeat(&'a TyListPat<'a>),
145
146    /// Empty type list.
147    Nil,
148
149    /// One type followed by another type list.
150    Cons {
151        first: &'a TyPat<'a>,
152        suffix: &'a TyListPat<'a>,
153    },
154}
155
156impl Pat for TyListPat<'_> {
157    const ANY: Self = Self::Any;
158}
159
160impl TyListPat<'_> {
161    pub const TS: Self = Self::Var(0);
162}

164/// Instruction "signature", dynamic representation (see module-level docs).
165#[derive(Copy, Clone, Debug)]
166pub struct InstSig<'a> {
167    /// Pattern for an instruction's sole storage class operand, if applicable.
168    // FIXME(eddyb) integrate this with `input_types` - it's non-trivial because
169    // that matches *the types of* ID operands, not the operands themselves.
170    pub storage_class: Option<&'a StorageClassPat>,
171
172    /// Patterns for the complete list of types of the instruction's ID operands,
173    /// where non-value operands (i.e. IDs of instructions without a *Result Type*)
174    /// can only match `TyPat::Any`.
175    pub input_types: &'a TyListPat<'a>,
176
177    /// Pattern for the instruction's *Result Type* operand, if applicable.
178    pub output_type: Option<&'a TyPat<'a>>,
179}
180
181/// Returns an array of valid signatures for an instruction with opcode `op`,
182/// or `None` if there aren't any known type constraints for that instruction.
183pub fn instruction_signatures(op: Op) -> Option<&'static [InstSig<'static>]> {
184    // Restrict the names the `pat!` macro can take as pattern constructors.
185    mod pat_ctors {
186        pub const S: super::StorageClassPat = super::StorageClassPat::S;
187        // NOTE(eddyb) it would be really nice if we could import `TyPat::{* - Any, Var}`,
188        // i.e. all but those two variants.
189        pub use super::TyPat::{
190            Array, Either, Function, Image, IndexComposite, Matrix, Pipe, Pointer, SampledImage,
191            Struct, Vector, Vector4, Void,
192        };
193        pub const T: super::TyPat<'_> = super::TyPat::T;
194        pub use super::TyListPat::Repeat;
195        pub const TS: super::TyListPat<'_> = super::TyListPat::TS;
196    }
197
198    macro_rules! pat {
199        (_) => { &Pat::ANY };
200        ($ctor:ident $(($($inner:tt $(($($inner_args:tt)+))?),+))?) => {
201            &pat_ctors::$ctor $(($(pat!($inner $(($($inner_args)+))?)),+))?
202        };
203        ([]) => { &TyListPat::Nil };
204        ([..]) => { &TyListPat::Any };
205        ([...$($rest:tt)+]) => { pat!($($rest)+) };
206        ([$first:tt $(($($first_args:tt)+))? $(, $($rest:tt)*)?]) => {
207            &TyListPat::Cons {
208                first: pat!($first $(($($first_args)+))?),
209                suffix: pat!([$($($rest)*)?]),
210            }
211        }
212    }
213    macro_rules! optionify {
214        () => {
215            None
216        };
217        ($x:expr) => {
218            Some($x)
219        };
220    }
221    macro_rules! sig {
222        ($(
223            $({$($storage_class:tt)*})?
224            ($($in_tys:tt)*)
225            $(-> $out_ty:tt $(($($out_ty_args:tt)+))?)?
226        )|+) => {
227            return Some(&[$(InstSig {
228                storage_class: optionify!($(pat!($($storage_class)*))?),
229                input_types: pat!([$($in_tys)*]),
230                output_type: optionify!($(pat!($out_ty $(($($out_ty_args)+))?))?),
231            }),*])
232        };
233    }
234
235    macro_rules! reserved {
236        ($ext:ident) => {
237            unreachable!(concat!("Op{:?} is reserved for ", stringify!($ext)), op)
238        };
239    }

246        Op::SizeOf => {}

249        Op::SourceContinued
250        | Op::Source
251        | Op::SourceExtension
252        | Op::Name
253        | Op::MemberName
254        | Op::String
255        | Op::Line => {}
256        //
257        Op::NoLine => {}
258        // SPIR-V 1.1
259        Op::ModuleProcessed => {}
260
261        // 3.37.3. Annotation Instructions
262        Op::Decorate
263        | Op::MemberDecorate
264        | Op::DecorationGroup
265        | Op::GroupDecorate
266        | Op::GroupMemberDecorate => {}
267        // SPIR-V 1.2
268        Op::DecorateId => {}
269        // SPIR-V 1.4
270        Op::DecorateString | Op::MemberDecorateString => {}
271
272        // 3.37.4. Extension Instructions
273        Op::Extension | Op::ExtInstImport | Op::ExtInst => {}
274
275        // 3.37.5. Mode-Setting Instructions
276        Op::MemoryModel | Op::EntryPoint | Op::ExecutionMode | Op::Capability => {}
277        // SPIR-V 1.2
278        Op::ExecutionModeId => {}
279
280        // 3.37.6. Type-Declaration Instructions
281        Op::TypeVoid
282        | Op::TypeBool
283        | Op::TypeInt
284        | Op::TypeFloat
285        | Op::TypeVector
286        | Op::TypeMatrix
287        | Op::TypeImage
288        | Op::TypeSampler
289        | Op::TypeSampledImage
290        | Op::TypeArray
291        | Op::TypeRuntimeArray
292        | Op::TypeStruct
293        | Op::TypeOpaque
294        | Op::TypePointer
295        | Op::TypeFunction
296        | Op::TypeEvent
297        | Op::TypeDeviceEvent
298        | Op::TypeReserveId
299        | Op::TypeQueue
300        | Op::TypePipe
301        | Op::TypeForwardPointer => {}
302        // SPIR-V 1.1, Capability: PipeStorage
303        Op::TypePipeStorage => {}
304        // SPIR-V 1.1, Capability: NamedBarrier
