42// HACK(eddyb) `InlineAsmOperandRef` lacks `#[derive(Clone)]`
43fn inline_asm_operand_ref_clone<'tcx, B: BackendTypes + ?Sized>(
44    operand: &InlineAsmOperandRef<'tcx, B>,
45) -> InlineAsmOperandRef<'tcx, B> {
46    use InlineAsmOperandRef::*;
47
48    match operand {
49        &In { reg, value } => In { reg, value },
50        &Out { reg, late, place } => Out { reg, late, place },
51        &InOut {
52            reg,
53            late,
54            in_value,
55            out_place,
56        } => InOut {
57            reg,
58            late,
59            in_value,
60            out_place,
61        },
62        Const { string } => Const {
63            string: string.clone(),
64        },
65        &SymFn { instance } => SymFn { instance },
66        &SymStatic { def_id } => SymStatic { def_id },
67        &Label { label } => Label { label },
68    }
69}

71impl<'a, 'tcx> AsmBuilderMethods<'tcx> for Builder<'a, 'tcx> {
72    /* Example asm and the template it compiles to:
73    asm!(
74        "mov {0}, {1}",
75        "add {0}, {number}",
76        out(reg) o,
77        in(reg) i,
78        number = const 5,
79    );
80    [
81    String("mov "),
82    Placeholder { operand_idx: 0, modifier: None, span: src/lib.rs:19:18: 19:21 (#0) },
83    String(", "),
84    Placeholder { operand_idx: 1, modifier: None, span: src/lib.rs:19:23: 19:26 (#0) },
85    String("\n"),
86    String("add "),
87    Placeholder { operand_idx: 0, modifier: None, span: src/lib.rs:20:18: 20:21 (#0) },
88    String(", "),
89    Placeholder { operand_idx: 2, modifier: None, span: src/lib.rs:20:23: 20:31 (#0) }
90    ]
91     */
92
93    fn codegen_inline_asm(
94        &mut self,
95        template: &[InlineAsmTemplatePiece],
96        operands: &[InlineAsmOperandRef<'tcx, Self>],
97        options: InlineAsmOptions,
98        _line_spans: &[Span],
99        _instance: Instance<'_>,
100        _dest: Option<Self::BasicBlock>,
101        _catch_funclet: Option<(Self::BasicBlock, Option<&Self::Funclet>)>,
102    ) {
103        const SUPPORTED_OPTIONS: InlineAsmOptions = InlineAsmOptions::NORETURN;
104        let unsupported_options = options & !SUPPORTED_OPTIONS;
105        if !unsupported_options.is_empty() {
106            self.err(format!("asm flags not supported: {unsupported_options:?}"));
107        }
108
109        // HACK(eddyb) get more accurate pointers types, for pointer operands,
110        // from the Rust types available in their respective `OperandRef`s.
111        let mut operands: SmallVec<[_; 8]> =
112            operands.iter().map(inline_asm_operand_ref_clone).collect();
113        for operand in &mut operands {
114            let (in_value, out_place) = match operand {
115                InlineAsmOperandRef::In { value, .. } => (Some(value), None),
116                InlineAsmOperandRef::InOut {
117                    in_value,
118                    out_place,
119                    ..
120                } => (Some(in_value), out_place.as_mut()),
121                InlineAsmOperandRef::Out { place, .. } => (None, place.as_mut()),
122
123                InlineAsmOperandRef::Const { .. }
124                | InlineAsmOperandRef::SymFn { .. }
125                | InlineAsmOperandRef::SymStatic { .. }
126                | InlineAsmOperandRef::Label { .. } => (None, None),
127            };
128
129            if let Some(in_value) = in_value
130                && let (BackendRepr::Scalar(scalar), OperandValue::Immediate(in_value_spv)) =
131                    (in_value.layout.backend_repr, &mut in_value.val)
132                && let Primitive::Pointer(_) = scalar.primitive()
133            {
134                let in_value_precise_type = in_value.layout.spirv_type(self.span(), self);
135                *in_value_spv = self.pointercast(*in_value_spv, in_value_precise_type);
136            }
137            if let Some(out_place) = out_place {
138                let out_place_precise_type = out_place.layout.spirv_type(self.span(), self);
139                let out_place_precise_ptr_type = self.type_ptr_to(out_place_precise_type);
140                out_place.val.llval =
141                    self.pointercast(out_place.val.llval, out_place_precise_ptr_type);
142            }
143        }
144
145        // vec of lines, and each line is vec of tokens
146        let mut tokens = vec![vec![]];
147        for piece in template {
148            match piece {
149                InlineAsmTemplatePiece::String(asm) => {
150                    // We cannot use str::lines() here because we don't want the behavior of "the
151                    // last newline is optional", we want an empty string for the last line if
152                    // there is no newline terminator.
153                    // Lambda copied from std LinesAnyMap
154                    let lines = asm.split('\n').map(|line| {
155                        let l = line.len();
156                        if l > 0 && line.as_bytes()[l - 1] == b'\r' {
157                            &line[0..l - 1]
158                        } else {
159                            line
160                        }
161                    });
162                    for (index, line) in lines.enumerate() {
163                        if index != 0 {
164                            // There was a newline, add a new line.
165                            tokens.push(vec![]);
166                        }
167                        let mut chars = line.chars();
168                        while let Some(token) = self.lex_word(&mut chars) {
169                            tokens.last_mut().unwrap().push(token);
170                        }
171                    }
172                }
173                &InlineAsmTemplatePiece::Placeholder {
174                    operand_idx,
175                    modifier,
176                    span,
177                } => {
178                    if let Some(modifier) = modifier {
179                        self.tcx
180                            .dcx()
181                            .span_err(span, format!("asm modifiers are not supported: {modifier}"));
182                    }
183                    let line = tokens.last_mut().unwrap();
184                    let typeof_kind = line.last().and_then(|prev| match prev {
185                        Token::Word("typeof") => Some(TypeofKind::Plain),
186                        Token::Word("typeof*") => Some(TypeofKind::Dereference),
187                        _ => None,
188                    });
189                    match typeof_kind {
190                        Some(kind) => {
191                            *line.last_mut().unwrap() =
192                                Token::Typeof(&operands[operand_idx], span, kind);
193                        }
194                        None => match &operands[operand_idx] {
195                            InlineAsmOperandRef::Const { string } => line.push(Token::Word(string)),
196                            item => line.push(Token::Placeholder(item, span)),
197                        },
198                    }
199                }
200            }
201        }
202
203        let mut id_map = FxHashMap::default();
204        let mut defined_ids = FxHashSet::default();
205        let mut id_to_type_map = FxHashMap::default();
206        for operand in &operands {
207            if let InlineAsmOperandRef::In { reg: _, value } = operand {
208                let value = value.immediate();
209                id_to_type_map.insert(value.def(self), value.ty);
210            }
211        }
212
213        let mut asm_block = AsmBlock::Open;
214        for line in tokens {
215            self.codegen_asm(
216                &mut id_map,
217                &mut defined_ids,
218                &mut id_to_type_map,
219                &mut asm_block,
220                line.into_iter(),
221            );
222        }
223
224        match (options.contains(InlineAsmOptions::NORETURN), asm_block) {
225            (true, AsmBlock::Open) => {
226                self.err("`noreturn` requires a terminator at the end");
227            }
228            (true, AsmBlock::End(_)) => {
229                // `noreturn` appends an `OpUnreachable` after the asm block.
230                // This requires starting a new block for this.
231                let label = self.emit().id();
232                self.emit()
233                    .insert_into_block(
234                        dr::InsertPoint::End,
235                        dr::Instruction::new(Op::Label, None, Some(label), vec![]),
236                    )
237                    .unwrap();
238            }
239            (false, AsmBlock::Open) => (),
240            (false, AsmBlock::End(terminator)) => {
241                self.err(format!(
242                    "trailing terminator `Op{terminator:?}` requires `options(noreturn)`"
243                ));
244            }
245        }
246        for (id, num) in id_map {
247            if !defined_ids.contains(&num) {
248                self.err(format!("%{id} is used but not defined"));
249            }
250        }
251    }
252}

281    fn lex_word<'a>(&self, line: &mut std::str::Chars<'a>) -> Option<Token<'a, 'cx, 'tcx>> {
282        loop {
283            let start = line.as_str();
284            match line.next()? {
285                // skip over leading whitespace
286                ch if ch.is_whitespace() => {}
287                // lex a string
288                '"' => {
289                    let mut cooked = String::new();
290                    loop {
291                        match line.next() {
292                            None => {
293                                self.err("Unterminated string in instruction");
294                                return None;
295                            }
296                            Some('"') => break,
297                            Some('\\') => {
298                                let escape = match line.next() {
299                                    None => {
300                                        self.err("Unterminated string in instruction");
301                                        return None;
302                                    }
303                                    Some('n') => '\n',
304                                    Some('r') => '\r',
305                                    Some('t') => '\t',
306                                    Some('0') => '\0',
307                                    Some('\\') => '\\',
308                                    Some('\'') => '\'',
309                                    Some('"') => '"',
310                                    Some(escape) => {
311                                        self.err(format!("invalid escape '\\{escape}'"));
312                                        return None;
313                                    }
314                                };
315                                cooked.push(escape);
316                            }
317                            Some(ch) => {
318                                cooked.push(ch);
319                            }
320                        }
321                    }
322                    break Some(Token::String(cooked));
323                }
324                // lex a word
325                _ => {
326                    let end = loop {
327                        let end = line.as_str();
328                        match line.next() {
329                            Some(ch) if !ch.is_whitespace() => {}
330                            _ => break end,
331                        }
332                    };
333                    let word = &start[..(start.len() - end.len())];
334                    break Some(Token::Word(word));
335                }
336            }
337        }
338    }

340    fn insert_inst(
341        &mut self,
342        id_map: &mut FxHashMap<&str, Word>,
343        defined_ids: &mut FxHashSet<Word>,
344        asm_block: &mut AsmBlock,
345        inst: dr::Instruction,
346    ) {
347        // Types declared must be registered in our type system.
348        let new_result_id = match inst.class.opcode {
349            Op::TypeVoid => SpirvType::Void.def(self.span(), self),
350            Op::TypeBool => SpirvType::Bool.def(self.span(), self),
351            Op::TypeInt => SpirvType::Integer(
352                inst.operands[0].unwrap_literal_bit32(),
353                inst.operands[1].unwrap_literal_bit32() != 0,
354            )
355            .def(self.span(), self),
356            Op::TypeFloat => {
357                SpirvType::Float(inst.operands[0].unwrap_literal_bit32()).def(self.span(), self)
358            }
359            Op::TypeStruct => {
360                self.err("OpTypeStruct in asm! is not supported yet");
361                return;
362            }
363            Op::TypeVector => SpirvType::Vector {
364                element: inst.operands[0].unwrap_id_ref(),
365                count: inst.operands[1].unwrap_literal_bit32(),
366            }
367            .def(self.span(), self),
368            Op::TypeMatrix => SpirvType::Matrix {
369                element: inst.operands[0].unwrap_id_ref(),
370                count: inst.operands[1].unwrap_literal_bit32(),
371            }
372            .def(self.span(), self),
373            Op::TypeArray => {
374                self.err("OpTypeArray in asm! is not supported yet");
375                return;
376            }
377            Op::TypeRuntimeArray => SpirvType::RuntimeArray {
378                element: inst.operands[0].unwrap_id_ref(),
379            }
380            .def(self.span(), self),
381            Op::TypePointer => {
382                let storage_class = inst.operands[0].unwrap_storage_class();
383                if storage_class != StorageClass::Generic {
384                    self.struct_err("TypePointer in asm! requires `Generic` storage class")
385                        .with_note(format!(
386                            "`{storage_class:?}` storage class was specified"
387                        ))
388                        .with_help(format!(
389                            "the storage class will be inferred automatically (e.g. to `{storage_class:?}`)"
390                        ))
391                        .emit();
392                }
393                SpirvType::Pointer {
394                    pointee: inst.operands[1].unwrap_id_ref(),
395                }
396                .def(self.span(), self)
397            }
398            Op::TypeImage => SpirvType::Image {
399                sampled_type: inst.operands[0].unwrap_id_ref(),
400                dim: inst.operands[1].unwrap_dim(),
401                depth: inst.operands[2].unwrap_literal_bit32(),
402                arrayed: inst.operands[3].unwrap_literal_bit32(),
403                multisampled: inst.operands[4].unwrap_literal_bit32(),
404                sampled: inst.operands[5].unwrap_literal_bit32(),
405                image_format: inst.operands[6].unwrap_image_format(),
406            }
407            .def(self.span(), self),
408            Op::TypeSampledImage => SpirvType::SampledImage {
409                image_type: inst.operands[0].unwrap_id_ref(),
410            }
411            .def(self.span(), self),
412            Op::TypeSampler => SpirvType::Sampler.def(self.span(), self),
413            Op::TypeAccelerationStructureKHR => {
414                SpirvType::AccelerationStructureKhr.def(self.span(), self)
415            }
416            Op::TypeRayQueryKHR => SpirvType::RayQueryKhr.def(self.span(), self),
417            Op::Variable => {
418                // OpVariable with Function storage class should be emitted inside the function,
419                // however, all other OpVariables should appear in the global scope instead.
420                if inst.operands[0].unwrap_storage_class() == StorageClass::Function {
421                    let mut builder = self.emit();
422                    builder.select_block(Some(0)).unwrap();
423                    builder
424                        .insert_into_block(dr::InsertPoint::Begin, inst)
425                        .unwrap();
426                } else {
427                    self.emit_global()
428                        .insert_types_global_values(dr::InsertPoint::End, inst);
429                }
430                return;
431            }
432
433            op => {
434                // NOTE(eddyb) allowing the instruction to be added below avoids
435                // spurious "`noreturn` requires a terminator at the end" errors.
436                if let Op::Return | Op::ReturnValue = op {
437                    self.struct_err(format!(
438                        "using `Op{op:?}` to return from within `asm!` is disallowed"
439                    ))
440                    .with_note(
441                        "resuming execution, without falling through the end \
442                        of the `asm!` block, is always undefined behavior",
443                    )
444                    .emit();
445                }
446
447                self.emit()
448                    .insert_into_block(dr::InsertPoint::End, inst)
449                    .unwrap();
450
451                *asm_block = match *asm_block {
452                    AsmBlock::Open => {
453                        if reflect::is_block_terminator(op) {
454                            AsmBlock::End(op)
455                        } else {
456                            AsmBlock::Open
457                        }
458                    }
459                    AsmBlock::End(terminator) => {
460                        if op != Op::Label {
461                            self.err(format!(
462                                "expected `OpLabel` after terminator `Op{terminator:?}`"
463                            ));
464                        }
465
466                        AsmBlock::Open
467                    }
468                };
469
470                return;
471            }
472        };
473        for value in id_map.values_mut() {
474            if *value == inst.result_id.unwrap() {
475                *value = new_result_id;
476            }
477        }
478        if defined_ids.remove(&inst.result_id.unwrap()) {
479            // Note this may be a duplicate insert, if the type was deduplicated.
480            defined_ids.insert(new_result_id);
481        }
482    }

484    fn codegen_asm<'a>(
485        &mut self,
486        id_map: &mut FxHashMap<&'a str, Word>,
487        defined_ids: &mut FxHashSet<Word>,
488        id_to_type_map: &mut FxHashMap<Word, Word>,
489        asm_block: &mut AsmBlock,
490        mut tokens: impl Iterator<Item = Token<'a, 'cx, 'tcx>>,
491    ) where
492        'cx: 'a,
493        'tcx: 'a,
494    {
495        let mut first_token = match tokens.next() {
496            Some(tok) => tok,
497            None => return,
498        };
499        // Parse result_id in front of instruction:
500        // %z = OpAdd %ty %x %y
501        let out_register = if match first_token {
502            Token::Placeholder(_, _) => true,
503            Token::Word(id_str) if id_str.starts_with('%') => true,
504            Token::Word(_) | Token::String(_) | Token::Typeof(_, _, _) => false,
505        } {
506            let result_id = match self.parse_id_out(id_map, defined_ids, first_token) {
507                Some(result_id) => result_id,
508                None => return,
509            };
510            if let Some(Token::Word("=")) = tokens.next() {
511            } else {
512                self.err("expected equals after result id specifier");
513                return;
514            }
515            first_token = if let Some(tok) = tokens.next() {
516                tok
517            } else {
518                self.err("expected instruction after equals");
519                return;
520            };
521            Some(result_id)
522        } else {
523            None
524        };
525        let inst_name = match first_token {
526            Token::Word(inst_name) => inst_name,
527            Token::String(_) => {
528                self.err("cannot use a string as an instruction");
529                return;
530            }
531            Token::Placeholder(_, span) | Token::Typeof(_, span, _) => {
532                self.tcx
533                    .dcx()
534                    .span_err(span, "cannot use a dynamic value as an instruction type");
535                return;
536            }
537        };
538        let inst_class = inst_name
539            .strip_prefix("Op")
540            .and_then(|n| self.cx.instruction_table.table.get(n));
541        let inst_class = if let Some(inst) = inst_class {
542            inst
543        } else {
544            self.err(format!("unknown spirv instruction {inst_name}"));
545            return;
546        };
547        let result_id = match out_register {
548            Some(OutRegister::Regular(reg)) => Some(reg),
549            Some(OutRegister::Place(_)) => Some(self.emit().id()),
550            None => None,
551        };
552        let mut instruction = dr::Instruction {
553            class: inst_class,
554            result_type: None,
555            result_id,
556            operands: vec![],
557        };
558        self.parse_operands(id_map, id_to_type_map, tokens, &mut instruction);
559        if let Some(result_type) = instruction.result_type {
560            id_to_type_map.insert(instruction.result_id.unwrap(), result_type);
561        }
562        self.insert_inst(id_map, defined_ids, asm_block, instruction);
563        if let Some(OutRegister::Place(place)) = out_register {
564            self.emit()
565                .store(
566                    place.val.llval.def(self),
567                    result_id.unwrap(),
568                    None,
569                    std::iter::empty(),
570                )
571                .unwrap();
572        }
573    }

575    fn parse_operands<'a>(
576        &mut self,
577        id_map: &mut FxHashMap<&'a str, Word>,
578        id_to_type_map: &FxHashMap<Word, Word>,
579        mut tokens: impl Iterator<Item = Token<'a, 'cx, 'tcx>>,
580        instruction: &mut dr::Instruction,
581    ) where
582        'cx: 'a,
583        'tcx: 'a,
584    {
585        let mut saw_id_result = false;
586        let mut need_result_type_infer = false;
587
588        let mut logical_operand_stack = instruction
589            .class
590            .operands
591            .iter()
592            .cloned()
593            .collect::<std::collections::VecDeque<_>>();
594
595        while let Some(LogicalOperand { kind, quantifier }) = logical_operand_stack.pop_front() {
596            if kind == OperandKind::IdResult {
597                assert_eq!(quantifier, OperandQuantifier::One);
598                if instruction.result_id.is_none() {
599                    self.err(format!(
600                        "instruction {} expects a result id",
601                        instruction.class.opname
602                    ));
603                }
604                saw_id_result = true;
605                continue;
606            }
607
608            if kind == OperandKind::IdResultType {
609                assert_eq!(quantifier, OperandQuantifier::One);
610                if let Some(token) = tokens.next() {
611                    if let Token::Word("_") = token {
612                        need_result_type_infer = true;
613                    } else if let Some(id) = self.parse_id_in(id_map, token) {
614                        instruction.result_type = Some(id);
615                    }
616                } else {
617                    self.err(format!(
618                        "instruction {} expects a result type",
619                        instruction.class.opname
620                    ));
621                }
622                continue;
623            }
624
625            let operands_start = instruction.operands.len();
626
627            match quantifier {
628                OperandQuantifier::One => {
629                    if !self.parse_one_operand(id_map, instruction, kind, &mut tokens) {
630                        self.err(format!(
631                            "expected operand after instruction: {}",
632                            instruction.class.opname
633                        ));
634                        return;
635                    }
636                }
637                OperandQuantifier::ZeroOrOne => {
638                    let _ = self.parse_one_operand(id_map, instruction, kind, &mut tokens);
639                    // If this return false, well, it's optional, do nothing
640                }
641                OperandQuantifier::ZeroOrMore => {
642                    while self.parse_one_operand(id_map, instruction, kind, &mut tokens) {}
643                }
644            }
645
646            // Parsed operands can add more optional operands that need to be parsed
647            // to an instruction - so push then on the stack here, after parsing
648            for op in instruction.operands[operands_start..].iter() {
649                logical_operand_stack.extend(op.additional_operands());
650            }
651        }
652
653        if !saw_id_result && instruction.result_id.is_some() {
654            self.err(format!(
655                "instruction {} does not expect a result id",
656                instruction.class.opname
657            ));
658        }
659        if tokens.next().is_some() {
660            self.tcx.dcx().err(format!(
661                "too many operands to instruction: {}",
662                instruction.class.opname
663            ));
664        }
665
666        if need_result_type_infer {
667            assert!(instruction.result_type.is_none());
668
669            match self.infer_result_type(id_to_type_map, instruction) {
670                Some(result_type) => instruction.result_type = Some(result_type),
671                None => self.err(format!(
672                    "instruction {} cannot have its result type inferred",
673                    instruction.class.opname
674                )),
675            }
676        }
677    }

679    fn infer_result_type(
680        &self,
681        id_to_type_map: &FxHashMap<Word, Word>,
682        instruction: &dr::Instruction,
683    ) -> Option<Word> {
684        use crate::spirv_type_constraints::{InstSig, TyListPat, TyPat, instruction_signatures};
685
686        #[derive(Debug)]
687        struct Unapplicable;
688
689        /// Recursively match `ty` against `pat`, returning one of:
690        /// * `Ok([None])`: `pat` matched but contained no type variables
691        /// * `Ok([Some(var)])`: `pat` matched and `var` is the type variable
692        /// * `Err(Mismatch)`: `pat` didn't match or isn't supported right now
693        fn match_ty_pat(
694            cx: &CodegenCx<'_>,
695            pat: &TyPat<'_>,
696            ty: Word,
697        ) -> Result<[Option<Word>; 1], Unapplicable> {
698            match pat {
699                TyPat::Any => Ok([None]),
700                &TyPat::T => Ok([Some(ty)]),
701                TyPat::Either(a, b) => {
702                    match_ty_pat(cx, a, ty).or_else(|Unapplicable| match_ty_pat(cx, b, ty))
703                }
704                _ => match (pat, cx.lookup_type(ty)) {
705                    (TyPat::Any | &TyPat::T | TyPat::Either(..), _) => unreachable!(),
706
707                    (TyPat::Void, SpirvType::Void) => Ok([None]),
708                    (TyPat::Pointer(_, pat), SpirvType::Pointer { pointee: ty, .. })
709                    | (TyPat::Vector(pat), SpirvType::Vector { element: ty, .. })
710                    | (
711                        TyPat::Vector4(pat),
712                        SpirvType::Vector {
713                            element: ty,
714                            count: 4,
715                        },
716                    )
717                    | (
718                        TyPat::Image(pat),
719                        SpirvType::Image {
720                            sampled_type: ty, ..
721                        },
722                    )
723                    | (TyPat::SampledImage(pat), SpirvType::SampledImage { image_type: ty }) => {
724                        match_ty_pat(cx, pat, ty)
725                    }
726                    _ => Err(Unapplicable),
727                },
728            }
729        }
730
731        #[derive(Debug)]
732        struct Ambiguous;
733
734        /// Construct a type from `pat`, replacing `TyPat::Var(i)` with `ty_vars[i]`.
735        /// `leftover_operands` is used for `IndexComposite` patterns, if any exist.
736        /// If the pattern isn't constraining enough to determine an unique type,
737        /// `Err(Ambiguous)` is returned instead.
738        fn subst_ty_pat(
739            cx: &CodegenCx<'_>,
740            pat: &TyPat<'_>,
741            ty_vars: &[Option<Word>],
742            leftover_operands: &[dr::Operand],
743        ) -> Result<Word, Ambiguous> {
744            Ok(match pat {
745                &TyPat::Var(i) => match ty_vars.get(i) {
746                    Some(&Some(ty)) => ty,
747                    _ => return Err(Ambiguous),
748                },
749
750                TyPat::Pointer(_, pat) => SpirvType::Pointer {
751                    pointee: subst_ty_pat(cx, pat, ty_vars, leftover_operands)?,
752                }
753                .def(DUMMY_SP, cx),
754
755                TyPat::Vector4(pat) => SpirvType::Vector {
756                    element: subst_ty_pat(cx, pat, ty_vars, leftover_operands)?,
757                    count: 4,
758                }
759                .def(DUMMY_SP, cx),
760
761                TyPat::SampledImage(pat) => SpirvType::SampledImage {
762                    image_type: subst_ty_pat(cx, pat, ty_vars, leftover_operands)?,
763                }
764                .def(DUMMY_SP, cx),
765
766                TyPat::IndexComposite(pat) => {
767                    let mut ty = subst_ty_pat(cx, pat, ty_vars, leftover_operands)?;
768                    for index in leftover_operands {
769                        let index_to_usize = || match *index {
770                            // FIXME(eddyb) support more than just literals,
771                            // by looking up `IdRef`s as constant integers.
772                            dr::Operand::LiteralBit32(i) => usize::try_from(i).ok(),
773
774                            _ => None,
775                        };
776                        ty = match cx.lookup_type(ty) {
777                            SpirvType::Array { element, .. }
778                            | SpirvType::RuntimeArray { element }
779                            // HACK(eddyb) this is pretty bad because it's not
780                            // checking that the index is an `OpConstant 0`, but
781                            // there's no other valid choice anyway.
782                            | SpirvType::InterfaceBlock { inner_type: element } => element,
783
784                            SpirvType::Adt { field_types, .. } => *index_to_usize()
785                                .and_then(|i| field_types.get(i))
786                                .ok_or(Ambiguous)?,
787
788                            // FIXME(eddyb) support more than just arrays and structs.
789                            _ => return Err(Ambiguous),
790                        };
791                    }
792                    ty
793                }
794
795                _ => return Err(Ambiguous),
796            })
797        }
798
799        // FIXME(eddyb) try multiple signatures until one fits.
800        let mut sig = match instruction_signatures(instruction.class.opcode)? {
801            [
802                sig @ InstSig {
803                    output_type: Some(_),
804                    ..
805                },
806            ] => *sig,
807            _ => return None,
808        };
809
810        let mut combined_ty_vars = [None];
811
812        let mut operands = instruction.operands.iter();
813        let mut next_id_operand = || operands.find_map(|o| o.id_ref_any());
814        while let TyListPat::Cons { first: pat, suffix } = *sig.input_types {
815            sig.input_types = suffix;
816
817            let match_result = match id_to_type_map.get(&next_id_operand()?) {
818                Some(&ty) => match_ty_pat(self, pat, ty),
819
820                // Non-value ID operand (or value operand of unknown type),
821                // only `TyPat::Any` is valid.
822                None => match pat {
823                    TyPat::Any => Ok([None]),
824                    _ => Err(Unapplicable),
825                },
826            };
827
828            let ty_vars = match match_result {
829                Ok(ty_vars) => ty_vars,
830                Err(Unapplicable) => return None,
831            };
832
833            for (&var, combined_var) in ty_vars.iter().zip(&mut combined_ty_vars) {
834                if let Some(var) = var {
835                    match *combined_var {
836                        Some(combined_var) => {
837                            // FIXME(eddyb) this could use some error reporting
838                            // (it's a type mismatch), although we could also
839                            // just use the first type and let validation take
840                            // care of the mismatch
841                            if var != combined_var {
842                                return None;
843                            }
844                        }
845                        None => *combined_var = Some(var),
846                    }
847                }
848            }
849        }
850        match sig.input_types {
851            TyListPat::Cons { .. } => unreachable!(),
852
853            TyListPat::Any => {}
854            TyListPat::Nil => {
855                if next_id_operand().is_some() {
856                    return None;
857                }
858            }
859            _ => return None,
860        }
861
862        // HACK(eddyb) clippy false positive, `.ok()` loses information.
863        #[allow(clippy::manual_ok_err)]
864        match subst_ty_pat(
865            self,
866            sig.output_type.unwrap(),
867            &combined_ty_vars,
868            operands.as_slice(),
869        ) {
870            Ok(ty) => Some(ty),
871            Err(Ambiguous) => None,
872        }
873    }

888    fn parse_id_out<'a>(
889        &mut self,
890        id_map: &mut FxHashMap<&'a str, Word>,
891        defined_ids: &mut FxHashSet<Word>,
892        token: Token<'a, 'cx, 'tcx>,
893    ) -> Option<OutRegister<'tcx>> {
894        match token {
895            Token::Word(word) => {
896                if let Some(id) = word.strip_prefix('%') {
897                    Some(OutRegister::Regular({
898                        let num = *id_map.entry(id).or_insert_with(|| self.emit().id());
899                        if !defined_ids.insert(num) {
900                            self.err(format!("%{id} is defined more than once"));
901                        }
902                        num
903                    }))
904                } else {
905                    self.err("expected ID");
906                    None
907                }
908            }
909            Token::String(_) => {
910                self.err("expected ID, not string");
911                None
912            }
913            Token::Typeof(_, span, _) => {
914                self.tcx
915                    .dcx()
916                    .span_err(span, "cannot assign to a typeof expression");
917                None
918            }
919            Token::Placeholder(hole, span) => match hole {
920                InlineAsmOperandRef::In { reg, value: _ } => {
921                    self.check_reg(span, reg);
922                    self.tcx
923                        .dcx()
924                        .span_err(span, "in register cannot be assigned to");
925                    None
926                }
927                InlineAsmOperandRef::Out {
928                    reg,
929                    late: _,
930                    place,
931                } => {
932                    self.check_reg(span, reg);
933                    if let Some(place) = place {
934                        Some(OutRegister::Place(*place))
935                    } else {
936                        self.tcx.dcx().span_err(span, "missing place for register");
937                        None
938                    }
939                }
940                InlineAsmOperandRef::InOut {
941                    reg,
942                    late: _,
943                    in_value: _,
944                    out_place,
945                } => {
946                    self.check_reg(span, reg);
947                    if let Some(out_place) = out_place {
948                        Some(OutRegister::Place(*out_place))
949                    } else {
950                        self.tcx.dcx().span_err(span, "missing place for register");
951                        None
952                    }
953                }
954                InlineAsmOperandRef::Const { string: _ } => {
955                    self.tcx
956                        .dcx()
957                        .span_err(span, "cannot write to const asm argument");
958                    None
959                }
960                InlineAsmOperandRef::SymFn { instance: _ } => {
961                    self.tcx
962                        .dcx()
963                        .span_err(span, "cannot write to function asm argument");
964                    None
965                }
966                InlineAsmOperandRef::SymStatic { def_id: _ } => {
967                    self.tcx
968                        .dcx()
969                        .span_err(span, "cannot write to static variable asm argument");
970                    None
971                }
972                InlineAsmOperandRef::Label { label: _ } => {
973                    self.tcx
974                        .dcx()
975                        .span_err(span, "cannot write to label asm argument");
976                    None
977                }
978            },
979        }
980    }

982    fn parse_id_in<'a>(
983        &mut self,
984        id_map: &mut FxHashMap<&'a str, Word>,
985        token: Token<'a, 'cx, 'tcx>,
986    ) -> Option<Word> {
987        match token {
988            Token::Word(word) => {
989                if let Some(id) = word.strip_prefix('%') {
990                    Some(*id_map.entry(id).or_insert_with(|| self.emit().id()))
991                } else {
992                    self.err("expected ID");
993                    None
994                }
995            }
996            Token::String(_) => {
997                self.err("expected ID, not string");
998                None
999            }
1000            Token::Typeof(hole, span, kind) => match hole {
1001                InlineAsmOperandRef::In { reg, value } => {
1002                    self.check_reg(span, reg);
1003                    let ty = value.immediate().ty;
1004                    Some(match kind {
1005                        TypeofKind::Plain => ty,
1006                        TypeofKind::Dereference => match self.lookup_type(ty) {
1007                            SpirvType::Pointer { pointee } => pointee,
1008                            other => {
1009                                self.tcx.dcx().span_err(
1010                                    span,
1011                                    format!(
1012                                        "cannot use typeof* on non-pointer type: {}",
1013                                        other.debug(ty, self)
1014                                    ),
1015                                );
1016                                ty
1017                            }
1018                        },
1019                    })
1020                }
1021                InlineAsmOperandRef::Out {
1022                    reg,
1023                    late: _,
1024                    place,
1025                } => {
1026                    self.check_reg(span, reg);
1027                    if let Some(place) = place {
1028                        match self.lookup_type(place.val.llval.ty) {
1029                            SpirvType::Pointer { pointee } => Some(pointee),
1030                            other => {
1031                                self.tcx.dcx().span_err(
1032                                    span,
1033                                    format!(
1034                                        "out register type not pointer: {}",
1035                                        other.debug(place.val.llval.ty, self)
1036                                    ),
1037                                );
1038                                None
1039                            }
1040                        }
1041                    } else {
1042                        self.tcx
1043                            .dcx()
1044                            .span_err(span, "missing place for out register typeof");
1045                        None
1046                    }
1047                }
1048                InlineAsmOperandRef::InOut {
1049                    reg,
1050                    late: _,
1051                    in_value,
1052                    out_place: _,
1053                } => {
1054                    self.check_reg(span, reg);
1055                    Some(in_value.immediate().ty)
1056                }
1057                InlineAsmOperandRef::Const { string: _ } => {
1058                    self.tcx
1059                        .dcx()
1060                        .span_err(span, "cannot take the type of a const asm argument");
1061                    None
1062                }
1063                InlineAsmOperandRef::SymFn { instance: _ } => {
1064                    self.tcx
1065                        .dcx()
1066                        .span_err(span, "cannot take the type of a function asm argument");
1067                    None
1068                }
1069                InlineAsmOperandRef::SymStatic { def_id: _ } => {
1070                    self.tcx.dcx().span_err(
1071                        span,
1072                        "cannot take the type of a static variable asm argument",
1073                    );
1074                    None
1075                }
1076                InlineAsmOperandRef::Label { label: _ } => {
1077                    self.tcx
1078                        .dcx()
1079                        .span_err(span, "cannot take the type of a label asm argument");
1080                    None
1081                }
1082            },
1083            Token::Placeholder(hole, span) => match hole {
1084                InlineAsmOperandRef::In { reg, value } => {
1085                    self.check_reg(span, reg);
1086                    Some(value.immediate().def(self))
1087                }
1088                InlineAsmOperandRef::Out {
1089                    reg,
1090                    late: _,
1091                    place: _,
1092                } => {
1093                    self.check_reg(span, reg);
1094                    self.tcx
1095                        .dcx()
1096                        .span_err(span, "out register cannot be used as a value");
1097                    None
1098                }
1099                InlineAsmOperandRef::InOut {
1100                    reg,
1101                    late: _,
1102                    in_value,
1103                    out_place: _,
1104                } => {
1105                    self.check_reg(span, reg);
1106                    Some(in_value.immediate().def(self))
1107                }
1108                InlineAsmOperandRef::Const { string: _ } => {
1109                    self.tcx
1110                        .dcx()
1111                        .span_err(span, "const asm argument not supported yet");
1112                    None
1113                }
1114                InlineAsmOperandRef::SymFn { instance: _ } => {
1115                    self.tcx
1116                        .dcx()
1117                        .span_err(span, "function asm argument not supported yet");
1118                    None
1119                }
1120                InlineAsmOperandRef::SymStatic { def_id: _ } => {
1121                    self.tcx
1122                        .dcx()
1123                        .span_err(span, "static variable asm argument not supported yet");
1124                    None
1125                }
1126                InlineAsmOperandRef::Label { label: _ } => {
1127                    self.tcx
1128                        .dcx()
1129                        .span_err(span, "label asm argument not supported yet");
1130                    None
1131                }
1132            },
1133        }
1134    }

1136    fn parse_one_operand<'a>(
1137        &mut self,
1138        id_map: &mut FxHashMap<&'a str, Word>,
1139        inst: &mut dr::Instruction,
1140        kind: OperandKind,
1141        tokens: &mut impl Iterator<Item = Token<'a, 'cx, 'tcx>>,
1142    ) -> bool
1143    where
1144        'cx: 'a,
1145        'tcx: 'a,
1146    {
1147        let token = match tokens.next() {
1148            Some(tok) => tok,
1149            None => return false,
1150        };
1151        let word = match token {
1152            Token::Word(word) => Some(word),
1153            Token::String(_) | Token::Placeholder(_, _) | Token::Typeof(_, _, _) => None,
1154        };
1155        match (kind, word) {
1156            (OperandKind::IdResultType | OperandKind::IdResult, _) => {
1157                bug!("should be handled by parse_operands");
1158            }
1159            (OperandKind::IdMemorySemantics, _) => {
1160                if let Some(id) = self.parse_id_in(id_map, token) {
1161                    inst.operands.push(dr::Operand::IdMemorySemantics(id));
1162                }
1163            }
1164            (OperandKind::IdScope, _) => {
1165                if let Some(id) = self.parse_id_in(id_map, token) {
1166                    inst.operands.push(dr::Operand::IdScope(id));
1167                }
1168            }
1169            (OperandKind::IdRef, _) => {
1170                if let Some(id) = self.parse_id_in(id_map, token) {
1171                    inst.operands.push(dr::Operand::IdRef(id));
1172                }
1173            }
1174
1175            (OperandKind::LiteralInteger, Some(word)) => match word.parse() {
1176                Ok(v) => inst.operands.push(dr::Operand::LiteralBit32(v)),
1177                Err(e) => self.err(format!("invalid integer: {e}")),
1178            },
1179            (OperandKind::LiteralFloat, Some(word)) => match word.parse() {
1180                Ok(v) => inst
1181                    .operands
1182                    .push(dr::Operand::LiteralBit32(f32::to_bits(v))),
1183                Err(e) => self.err(format!("invalid float: {e}")),
1184            },
1185            (OperandKind::LiteralString, _) => {
1186                if let Token::String(value) = token {
1187                    inst.operands.push(dr::Operand::LiteralString(value));
1188                }
1189            }
1190            (OperandKind::LiteralContextDependentNumber, Some(word)) => {
1191                assert!(matches!(inst.class.opcode, Op::Constant | Op::SpecConstant));
1192                let ty = inst.result_type.unwrap();
1193                fn parse(ty: SpirvType<'_>, w: &str) -> Result<dr::Operand, String> {
1194                    fn fmt(x: impl ToString) -> String {
1195                        x.to_string()
1196                    }
1197                    Ok(match ty {
1198                        SpirvType::Integer(8, false) => {
1199                            dr::Operand::LiteralBit32(w.parse::<u8>().map_err(fmt)? as u32)
1200                        }
1201                        SpirvType::Integer(16, false) => {
1202                            dr::Operand::LiteralBit32(w.parse::<u16>().map_err(fmt)? as u32)
1203                        }
1204                        SpirvType::Integer(32, false) => {
1205                            dr::Operand::LiteralBit32(w.parse::<u32>().map_err(fmt)?)
1206                        }
1207                        SpirvType::Integer(64, false) => {
1208                            dr::Operand::LiteralBit64(w.parse::<u64>().map_err(fmt)?)
1209                        }
1210                        SpirvType::Integer(8, true) => {
1211                            dr::Operand::LiteralBit32(w.parse::<i8>().map_err(fmt)? as i32 as u32)
1212                        }
1213                        SpirvType::Integer(16, true) => {
1214                            dr::Operand::LiteralBit32(w.parse::<i16>().map_err(fmt)? as i32 as u32)
1215                        }
1216                        SpirvType::Integer(32, true) => {
1217                            dr::Operand::LiteralBit32(w.parse::<i32>().map_err(fmt)? as u32)
1218                        }
1219                        SpirvType::Integer(64, true) => {
1220                            dr::Operand::LiteralBit64(w.parse::<i64>().map_err(fmt)? as u64)
1221                        }
1222                        SpirvType::Float(32) => {
1223                            dr::Operand::LiteralBit32(w.parse::<f32>().map_err(fmt)?.to_bits())
1224                        }
1225                        SpirvType::Float(64) => {
1226                            dr::Operand::LiteralBit64(w.parse::<f64>().map_err(fmt)?.to_bits())
1227                        }
1228                        _ => return Err("expected number literal in OpConstant".to_string()),
1229                    })
1230                }
1231                match parse(self.lookup_type(ty), word) {
1232                    Ok(op) => inst.operands.push(op),
1233                    Err(err) => self.err(err),
1234                }
1235            }
1236            (OperandKind::LiteralExtInstInteger, Some(word)) => match word.parse() {
1237                Ok(v) => inst.operands.push(dr::Operand::LiteralExtInstInteger(v)),
1238                Err(e) => self.err(format!("invalid integer: {e}")),
1239            },
1240            (OperandKind::LiteralSpecConstantOpInteger, Some(word)) => {
1241                match self.instruction_table.table.get(word) {
1242                    Some(v) => {
1243                        inst.operands
1244                            .push(dr::Operand::LiteralSpecConstantOpInteger(v.opcode));
1245                    }
1246                    None => self.err("invalid instruction in OpSpecConstantOp"),
1247                }
1248            }
1249            (OperandKind::PairLiteralIntegerIdRef, _) => {
1250                self.err("PairLiteralIntegerIdRef not supported yet");
1251            }
1252            (OperandKind::PairIdRefLiteralInteger, _) => {
1253                if let Some(id) = self.parse_id_in(id_map, token) {
1254                    inst.operands.push(dr::Operand::IdRef(id));
1255                    match tokens.next() {
1256                        Some(Token::Word(word)) => match word.parse() {
1257                            Ok(v) => inst.operands.push(dr::Operand::LiteralBit32(v)),
1258                            Err(e) => {
1259                                self.err(format!("invalid integer: {e}"));
1260                            }
1261                        },
1262                        Some(Token::String(_)) => {
1263                            self.err(format!("expected a literal, not a string for a {kind:?}"));
1264                        }
1265                        Some(Token::Placeholder(_, span)) => {
1266                            self.tcx.dcx().span_err(
1267                                span,
1268                                format!("expected a literal, not a dynamic value for a {kind:?}"),
1269                            );
1270                        }
1271                        Some(Token::Typeof(_, span, _)) => {
1272                            self.tcx.dcx().span_err(
1273                                span,
1274                                format!("expected a literal, not a type for a {kind:?}"),
1275                            );
1276                        }
1277                        None => {
1278                            self.err("expected operand after instruction");
1279                        }
1280                    }
1281                }

1716fn parse_bitflags_operand<T: std::ops::BitOr<Output = T> + Copy>(
1717    values: &'static [(&'static str, T)],
1718    word: &str,
1719) -> Option<T> {
1720    let mut result = None;
1721    'outer: for item in word.split('|') {
1722        for &(key, value) in values {
1723            if item == key {
1724                result = Some(result.map_or(value, |x| x | value));
1725                continue 'outer;
1726            }
1727        }
1728        return None;
1729    }
1730    result
