43macro_rules! simple_op {
44    (
45        $func_name:ident, $inst_name:ident
46        $(, fold_const {
47            $(int($fold_int_lhs:ident, $fold_int_rhs:ident) => $fold_int:expr)?
48        })?
49    ) => {
50        fn $func_name(&mut self, lhs: Self::Value, rhs: Self::Value) -> Self::Value {
51            assert_ty_eq!(self, lhs.ty, rhs.ty);
52            let result_type = lhs.ty;
53
54            $(if let Some(const_lhs) = self.builder.lookup_const(lhs) {
55                if let Some(const_rhs) = self.builder.lookup_const(rhs) {
56                    match self.lookup_type(result_type) {
57                        $(SpirvType::Integer(bits, signed) => {
58                            let size = Size::from_bits(bits);
59                            let as_u128 = |const_val| {
60                                let x = match const_val {
61                                    SpirvConst::Scalar(x) => x,
62                                    _ => return None,
63                                };
64                                Some(if signed {
65                                    size.sign_extend(x) as u128
66                                } else {
67                                    size.truncate(x)
68                                })
69                            };
70                            if let Some($fold_int_lhs) = as_u128(const_lhs) {
71                                if let Some($fold_int_rhs) = as_u128(const_rhs) {
72                                    return self.const_uint_big(result_type, $fold_int);
73                                }
74                            }
75                        })?
76                        _ => {}
77                    }
78                }
79            })?
80
81            self.emit()
82                .$inst_name(result_type, None, lhs.def(self), rhs.def(self))
83                .unwrap()
84                .with_type(result_type)
85        }
86    };
87}

112fn memset_fill_u16(b: u8) -> u16 {
113    b as u16 | ((b as u16) << 8)
114}

116fn memset_fill_u32(b: u8) -> u32 {
117    b as u32 | ((b as u32) << 8) | ((b as u32) << 16) | ((b as u32) << 24)
118}

125        | ((b as u64) << 32)

131fn memset_dynamic_scalar(
132    builder: &mut Builder<'_, '_>,
133    fill_var: Word,
134    byte_width: usize,
135    is_float: bool,
136) -> Word {
137    let composite_type = SpirvType::Vector {
138        element: SpirvType::Integer(8, false).def(builder.span(), builder),
139        count: byte_width as u32,
140    }
141    .def(builder.span(), builder);
142    let composite = builder
143        .emit()
144        .composite_construct(composite_type, None, iter::repeat_n(fill_var, byte_width))
145        .unwrap();
146    let result_type = if is_float {
147        SpirvType::Float(byte_width as u32 * 8)
148    } else {
149        SpirvType::Integer(byte_width as u32 * 8, false)
150    };
151    builder
152        .emit()
153        .bitcast(result_type.def(builder.span(), builder), None, composite)
154        .unwrap()
155}

159    fn ordering_to_semantics_def(&mut self, ordering: AtomicOrdering) -> SpirvValue {
160        let mut invalid_seq_cst = false;
161        let semantics = match ordering {
162            AtomicOrdering::Relaxed => MemorySemantics::NONE,
163            AtomicOrdering::Acquire => MemorySemantics::MAKE_VISIBLE | MemorySemantics::ACQUIRE,
164            AtomicOrdering::Release => MemorySemantics::MAKE_AVAILABLE | MemorySemantics::RELEASE,
165            AtomicOrdering::AcqRel => {
166                MemorySemantics::MAKE_AVAILABLE
167                    | MemorySemantics::MAKE_VISIBLE
168                    | MemorySemantics::ACQUIRE_RELEASE
169            }
170            AtomicOrdering::SeqCst => {
171                let builder = self.emit();
172                let memory_model = builder.module_ref().memory_model.as_ref().unwrap();
173                if memory_model.operands[1].unwrap_memory_model() == MemoryModel::Vulkan {
174                    invalid_seq_cst = true;
175                }
176                MemorySemantics::MAKE_AVAILABLE
177                    | MemorySemantics::MAKE_VISIBLE
178                    | MemorySemantics::SEQUENTIALLY_CONSISTENT
179            }
180        };
181        let semantics = self.constant_u32(self.span(), semantics.bits());
182        if invalid_seq_cst {
183            self.zombie(
184                semantics.def(self),
185                "cannot use `AtomicOrdering::SeqCst` on Vulkan memory model \
186                 (check if `AcqRel` fits your needs)",
187            );
188        }
189        semantics
190    }

193    fn memset_const_pattern(&self, ty: &SpirvType<'tcx>, fill_byte: u8) -> Word {
194        match *ty {
195            SpirvType::Void => self.fatal("memset invalid on void pattern"),
196            SpirvType::Bool => self.fatal("memset invalid on bool pattern"),
197            SpirvType::Integer(width, false) => match width {
198                8 => self.constant_u8(self.span(), fill_byte).def(self),
199                16 => self
200                    .constant_u16(self.span(), memset_fill_u16(fill_byte))
201                    .def(self),
202                32 => self
203                    .constant_u32(self.span(), memset_fill_u32(fill_byte))
204                    .def(self),
205                64 => self
206                    .constant_u64(self.span(), memset_fill_u64(fill_byte))
207                    .def(self),
208                _ => self.fatal(format!(
209                    "memset on integer width {width} not implemented yet"
210                )),
211            },
212            SpirvType::Integer(width, true) => match width {
213                8 => self.constant_i8(self.span(), fill_byte as i8).def(self),
214                16 => self
215                    .constant_i16(self.span(), memset_fill_u16(fill_byte) as i16)
216                    .def(self),
217                32 => self
218                    .constant_i32(self.span(), memset_fill_u32(fill_byte) as i32)
219                    .def(self),
220                64 => self
221                    .constant_i64(self.span(), memset_fill_u64(fill_byte) as i64)
222                    .def(self),
223                _ => self.fatal(format!(
224                    "memset on integer width {width} not implemented yet"
225                )),
226            },
227            SpirvType::Float(width) => match width {
228                32 => self
229                    .constant_f32(self.span(), f32::from_bits(memset_fill_u32(fill_byte)))
230                    .def(self),
231                64 => self
232                    .constant_f64(self.span(), f64::from_bits(memset_fill_u64(fill_byte)))
233                    .def(self),
234                _ => self.fatal(format!("memset on float width {width} not implemented yet")),
235            },
236            SpirvType::Adt { .. } => self.fatal("memset on structs not implemented yet"),
237            SpirvType::Vector { element, count } | SpirvType::Matrix { element, count } => {
238                let elem_pat = self.memset_const_pattern(&self.lookup_type(element), fill_byte);
239                self.constant_composite(
240                    ty.def(self.span(), self),
241                    iter::repeat_n(elem_pat, count as usize),
242                )
243                .def(self)
244            }
245            SpirvType::Array { element, count } => {
246                let elem_pat = self.memset_const_pattern(&self.lookup_type(element), fill_byte);
247                let count = self.builder.lookup_const_scalar(count).unwrap() as usize;
248                self.constant_composite(ty.def(self.span(), self), iter::repeat_n(elem_pat, count))
249                    .def(self)
250            }
251            SpirvType::RuntimeArray { .. } => {
252                self.fatal("memset on runtime arrays not implemented yet")
253            }
254            SpirvType::Pointer { .. } => self.fatal("memset on pointers not implemented yet"),
255            SpirvType::Function { .. } => self.fatal("memset on functions not implemented yet"),
256            SpirvType::Image { .. } => self.fatal("cannot memset image"),
257            SpirvType::Sampler => self.fatal("cannot memset sampler"),
258            SpirvType::SampledImage { .. } => self.fatal("cannot memset sampled image"),
259            SpirvType::InterfaceBlock { .. } => self.fatal("cannot memset interface block"),
260            SpirvType::AccelerationStructureKhr => {
261                self.fatal("cannot memset acceleration structure")
262            }
263            SpirvType::RayQueryKhr => self.fatal("cannot memset ray query"),
264        }
265    }

325    fn memset_constant_size(&mut self, ptr: SpirvValue, pat: SpirvValue, size_bytes: u64) {
326        let size_elem = self
327            .lookup_type(pat.ty)
328            .sizeof(self)
329            .expect("Memset on unsized values not supported");
330        let count = size_bytes / size_elem.bytes();
331        if count == 1 {
332            self.store(pat, ptr, Align::from_bytes(0).unwrap());
333        } else {
334            for index in 0..count {
335                let const_index = self.constant_u32(self.span(), index as u32);
336                let gep_ptr = self.inbounds_gep(pat.ty, ptr, &[const_index]);
337                self.store(pat, gep_ptr, Align::from_bytes(0).unwrap());
338            }
339        }
340    }

344    fn memset_dynamic_size(&mut self, ptr: SpirvValue, pat: SpirvValue, size_bytes: SpirvValue) {
345        let size_elem = self
346            .lookup_type(pat.ty)
347            .sizeof(self)
348            .expect("Unable to memset a dynamic sized object");
349        let size_elem_const = self.constant_int(size_bytes.ty, size_elem.bytes().into());
350        let zero = self.constant_int(size_bytes.ty, 0);
351        let one = self.constant_int(size_bytes.ty, 1);
352        let zero_align = Align::from_bytes(0).unwrap();
353
354        let header_bb = self.append_sibling_block("memset_header");
355        let body_bb = self.append_sibling_block("memset_body");
356        let exit_bb = self.append_sibling_block("memset_exit");
357
358        let count = self.udiv(size_bytes, size_elem_const);
359        let index = self.alloca(self.lookup_type(count.ty).sizeof(self).unwrap(), zero_align);
360        self.store(zero, index, zero_align);
361        self.br(header_bb);
362
363        self.switch_to_block(header_bb);
364        let current_index = self.load(count.ty, index, zero_align);
365        let cond = self.icmp(IntPredicate::IntULT, current_index, count);
366        self.cond_br(cond, body_bb, exit_bb);
367
368        self.switch_to_block(body_bb);
369        let gep_ptr = self.gep(pat.ty, ptr, &[current_index]);
370        self.store(pat, gep_ptr, zero_align);
371        let current_index_plus_1 = self.add(current_index, one);
372        self.store(current_index_plus_1, index, zero_align);
373        self.br(header_bb);
374
375        self.switch_to_block(exit_bb);
376    }

415    /// If `ptr`'s pointee type contains any prefix field/element of size `size`,
416    /// i.e. some leaf which can be used for all accesses of size `size`, return
417    /// `ptr` adjusted to point to the innermost such leaf, and the leaf's type.
418    //
419    // FIXME(eddyb) technically this duplicates `pointercast`, but the main use
420    // of `pointercast` is being replaced by this, and this can be more efficient.
421    //
422    // HACK(eddyb) temporary workaround for untyped pointers upstream.
423    // FIXME(eddyb) replace with untyped memory SPIR-V + `qptr` or similar.
424    #[instrument(level = "trace", skip(self))]
425    fn adjust_pointer_for_sized_access(
426        &mut self,
427        ptr: SpirvValue,
428        size: Size,
429    ) -> Option<(SpirvValue, <Self as BackendTypes>::Type)> {
430        let ptr = ptr.strip_ptrcasts();
431        let mut leaf_ty = match self.lookup_type(ptr.ty) {
432            SpirvType::Pointer { pointee } => pointee,
433            other => self.fatal(format!("`ptr` is non-pointer type: {other:?}")),
434        };
435
436        trace!(
437            "before nested adjust_pointer_for_sized_access. `leaf_ty`: {}",
438            self.debug_type(leaf_ty)
439        );
440
441        let mut indices = SmallVec::<[_; 8]>::new();
442        while let Some((inner_indices, inner_ty)) = self.recover_access_chain_from_offset(
443            leaf_ty,
444            Size::ZERO,
445            Some(size)..=Some(size),
446            None,
447        ) {
448            indices.extend(inner_indices);
449            leaf_ty = inner_ty;
450        }
451
452        trace!(
453            "after nested adjust_pointer_for_sized_access. `leaf_ty`: {}",
454            self.debug_type(leaf_ty)
455        );
456
457        let leaf_ptr_ty = (self.lookup_type(leaf_ty).sizeof(self) == Some(size))
458            .then(|| self.type_ptr_to(leaf_ty))?;
459        let leaf_ptr = if indices.is_empty() {
460            assert_ty_eq!(self, ptr.ty, leaf_ptr_ty);
461            ptr
462        } else {
463            let indices = indices
464                .into_iter()
465                .map(|idx| self.constant_u32(self.span(), idx).def(self))
466                .collect::<Vec<_>>();
467            self.emit()
468                .in_bounds_access_chain(leaf_ptr_ty, None, ptr.def(self), indices)
469                .unwrap()
470                .with_type(leaf_ptr_ty)
471        };
472
473        trace!(
474            "adjust_pointer_for_sized_access returning {} {}",
475            self.debug_type(leaf_ptr.ty),
476            self.debug_type(leaf_ty)
477        );
478        Some((leaf_ptr, leaf_ty))
479    }

481    /// If possible, return the appropriate `OpAccessChain` indices for going
482    /// from a pointer to `ty`, to a pointer to some leaf field/element having
483    /// a size that fits `leaf_size_range` (and, optionally, the type `leaf_ty`),
484    /// while adding `offset` bytes.
485    ///
486    /// That is, try to turn `((_: *T) as *u8).add(offset) as *Leaf` into a series
487    /// of struct field and array/vector element accesses.
488    #[instrument(level = "trace", skip(self), fields(ty = ?self.debug_type(ty), leaf_size_or_unsized_range, leaf_ty = ?leaf_ty))]
489    fn recover_access_chain_from_offset(
490        &self,
491        mut ty: <Self as BackendTypes>::Type,
492        mut offset: Size,
493        // FIXME(eddyb) using `None` for "unsized" is a pretty bad design.
494        leaf_size_or_unsized_range: RangeInclusive<Option<Size>>,
495        leaf_ty: Option<<Self as BackendTypes>::Type>,
496    ) -> Option<(SmallVec<[u32; 8]>, <Self as BackendTypes>::Type)> {
497        assert_ne!(Some(ty), leaf_ty);
498        if let Some(leaf_ty) = leaf_ty {
499            trace!(
500                "recovering access chain: leaf_ty: {:?}",
501                self.debug_type(leaf_ty)
502            );
503        } else {
504            trace!("recovering access chain: leaf_ty: None");
505        }
506
507        // HACK(eddyb) this has the correct ordering (`Sized(_) < Unsized`).
508        #[derive(Copy, Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
509        enum MaybeSized {
510            Sized(Size),
511            Unsized,
512        }
513        let leaf_size_range = {
514            let r = leaf_size_or_unsized_range;
515            let [start, end] =
516                [r.start(), r.end()].map(|x| x.map_or(MaybeSized::Unsized, MaybeSized::Sized));
517            start..=end
518        };
519
520        trace!("leaf_size_range: {:?}", leaf_size_range);
521
522        // NOTE(eddyb) `ty` and `ty_kind`/`ty_size` should be kept in sync.
523        let mut ty_kind = self.lookup_type(ty);
524
525        let mut indices = SmallVec::new();
526        loop {
527            let ty_size;
528            match ty_kind {
529                SpirvType::Adt {
530                    field_types,
531                    field_offsets,
532                    ..
533                } => {
534                    trace!("recovering access chain from ADT");
535                    let (i, field_ty, field_ty_kind, field_ty_size, offset_in_field) = field_offsets
536                        .iter()
537                        .enumerate()
538                        .find_map(|(i, &field_offset)| {
539                            if field_offset > offset {
540                                return None;
541                            }
542
543                            // Grab the actual field type to be able to confirm that
544                            // the leaf is somewhere inside the field.
545                            let field_ty = field_types[i];
546                            let field_ty_kind = self.lookup_type(field_ty);
547                            let field_ty_size = field_ty_kind
548                                .sizeof(self).map_or(MaybeSized::Unsized, MaybeSized::Sized);
549
550                            let offset_in_field = offset - field_offset;
551                            if MaybeSized::Sized(offset_in_field) < field_ty_size
552                                // If the field is a zero sized type, check the
553                                // expected size and type to get the correct entry
554                                || offset_in_field == Size::ZERO
555                                    && leaf_size_range.contains(&MaybeSized::Sized(Size::ZERO)) && leaf_ty == Some(field_ty)
556                            {
557                                Some((i, field_ty, field_ty_kind, field_ty_size, offset_in_field))
558                            } else {
559                                None
560                            }
561                        })?;
562
563                    ty = field_ty;
564                    trace!("setting ty = field_ty: {:?}", self.debug_type(field_ty));
565                    ty_kind = field_ty_kind;
566                    trace!("setting ty_kind = field_ty_kind: {:?}", field_ty_kind);
567                    ty_size = field_ty_size;
568                    trace!("setting ty_size = field_ty_size: {:?}", field_ty_size);
569
570                    indices.push(i as u32);
571                    offset = offset_in_field;
572                    trace!("setting offset = offset_in_field: {:?}", offset_in_field);
573                }
574                SpirvType::Vector { element, .. }
575                | SpirvType::Array { element, .. }
576                | SpirvType::RuntimeArray { element }
577                | SpirvType::Matrix { element, .. } => {
578                    trace!("recovering access chain from Vector, Array, RuntimeArray, or Matrix");
579                    ty = element;
580                    trace!("setting ty = element: {:?}", self.debug_type(element));
581                    ty_kind = self.lookup_type(ty);
582                    trace!("looked up ty kind: {:?}", ty_kind);
583
584                    let stride = ty_kind.sizeof(self)?;
585                    ty_size = MaybeSized::Sized(stride);
586
587                    indices.push((offset.bytes() / stride.bytes()).try_into().ok()?);
588                    offset = Size::from_bytes(offset.bytes() % stride.bytes());
589                }
590                _ => {
591                    trace!("recovering access chain from SOMETHING ELSE, RETURNING NONE");
592                    return None;
593                }
594            }
595
596            // Avoid digging beyond the point the leaf could actually fit.
597            if ty_size < *leaf_size_range.start() {
598                trace!("avoiding digging beyond the point the leaf could actually fit");
599                return None;
600            }
601
602            if offset == Size::ZERO
603                && leaf_size_range.contains(&ty_size)
604                && leaf_ty.is_none_or(|leaf_ty| leaf_ty == ty)
605            {
606                trace!("successful recovery leaf type: {:?}", self.debug_type(ty));
607                trace!("successful recovery indices: {:?}", indices);
608                return Some((indices, ty));
609            }
610        }
611    }

614    #[instrument(level = "trace", skip(self), fields(ty = ?self.debug_type(ty), ptr, combined_indices = ?combined_indices.iter().map(|x| (self.debug_type(x.ty), x.kind)).collect::<Vec<_>>(), is_inbounds))]
615    fn maybe_inbounds_gep(
616        &mut self,
617        // Represents the type of the element that `ptr` is assumed to point to
618        // *before* applying the *first* index (`ptr_base_index`). This type is used
619        // primarily to calculate the size for the initial offset.
620        ty: Word,
621        //  The base pointer value for the GEP operation.
622        ptr: SpirvValue,
623        // A slice of indices used for the GEP calculation. The *first* index is treated
624        // as the main offset from the base pointer `ptr`, scaled by the size of `ty`.
625        // Subsequent indices navigate through nested aggregate types (structs, arrays,
626        // vectors, etc.) starting from `ty`.
627        combined_indices: &[SpirvValue],
628        // If true, generate an `OpInBoundsAccessChain`, which has stricter requirements
629        // but allows more optimization. If false, generate `OpAccessChain`.
630        is_inbounds: bool,
631    ) -> SpirvValue {
632        // Separate the first index (used for the base offset) from the rest of the
633        // indices (used for navigating within the aggregate type). `ptr_base_index` is
634        // the index applied directly to `ptr`, effectively an offset multiplier based
635        // on the size of `ty`. `indices` are the subsequent indices used to drill down
636        // into fields or elements of `ty`.
637        // https://llvm.org/docs/GetElementPtr.html
638        // "An OpAccessChain instruction is the equivalent of an LLVM getelementptr instruction where the first index element is zero."
639        // https://github.com/gpuweb/gpuweb/issues/33
640        let (&ptr_base_index, indices) = combined_indices.split_first().unwrap();
641
642        // HACK(eddyb) this effectively removes any real support for GEPs with
643        // any `indices` (beyond `ptr_base_index`), which should now be the case
644        // across `rustc_codegen_ssa` (see also comment inside `inbounds_gep`).
645        if !indices.is_empty() {
646            self.fatal(format!(
647                "[RUST-GPU BUG] `inbounds_gep` or `gep` called \
648                 with {} combined indices (expected only 1)",
649                combined_indices.len(),
650            ));
651        }
652
653        self.ptr_offset_strided(ptr, ty, ptr_base_index, is_inbounds)
654    }

656    /// Array-indexing-like pointer arithmetic, i.e. `(ptr: *T).offset(index)`
657    /// (or `wrapping_offset` instead of `offset`, for `is_inbounds = false`),
658    /// where `T` is given by `stride_elem_ty` (named so for extra clarity).
659    ///
660    /// This can produce legal SPIR-V by using 3 strategies:
661    /// 1. noop, i.e. returning `ptr` unmodified, comparable to a `pointercast`
662    ///    (but instead letting downstream users do any casts they might need,
663    ///     themselves - also, upstream untyped pointers mean that no pointer
664    ///     can be expected to have any specific pointee type)
665    /// 2. `recover_access_chain_from_offset` for constant offsets
666    ///    (e.g. from `ptradd`/`inbounds_ptradd` used to access `struct` fields)
667    /// 3. merging onto an array `OpAccessChain` with the same `stride_elem_ty`
668    ///    (possibly `&array[0]` from `pointercast` doing `*[T; N]` -> `*T`)
669    ///
670    /// Also, `pointercast` (used downstream, or as part of strategy 3.) helps
671    /// with producing legal SPIR-V, as it allows deferring whole casts chains,
672    /// and has a couple success modes of its own:
673    /// - itself can also use `recover_access_chain_from_offset`, supporting
674    ///   `struct`/array casts e.g. `*(T, U, ...)` -> `*T` / `*[T; N]` -> `*T`
675    /// - even if a specific cast is unsupported, legal SPIR-V can still be
676    ///   later obtained (thanks to `SpirvValueKind::LogicalPtrCast`), if all
677    ///   uses of that cast rely on `pointercast` and/or `strip_ptrcasts`, e.g.:
678    ///   - another `ptr_offset_strided` call (with a different offset)
679    ///   - `adjust_pointer_for_typed_access`/`adjust_pointer_for_sized_access`
680    ///     (themselves used by e.g. loads, stores, copies, etc.)
681    //
682    // FIXME(eddyb) maybe move the above `pointercast` section to its own docs?
683    #[instrument(level = "trace", skip(self), fields(ptr, stride_elem_ty = ?self.debug_type(stride_elem_ty), index, is_inbounds))]
684    fn ptr_offset_strided(
685        &mut self,
686        ptr: SpirvValue,
687        stride_elem_ty: Word,
688        index: SpirvValue,
689        is_inbounds: bool,
690    ) -> SpirvValue {
691        // Precompute a constant `index * stride` (i.e. effective pointer offset)
692        // if possible, as both strategies 1 and 2 rely on knowing this value.
693        let const_offset = self.builder.lookup_const_scalar(index).and_then(|idx| {
694            let idx_u64 = u64::try_from(idx).ok()?;
695            let stride = self.lookup_type(stride_elem_ty).sizeof(self)?;
696            Some(idx_u64 * stride)
697        });
698
699        // Strategy 1: do nothing for a `0` offset (and `stride_elem_ty` can be
700        // safely ignored, because any typed uses will `pointercast` if needed).
701        if const_offset == Some(Size::ZERO) {
702            trace!("ptr_offset_strided: strategy 1 picked: offset 0 => noop");
703
704            return ptr;
705        }
706
707        // Strategy 2: try recovering an `OpAccessChain` from a constant offset.
708        if let Some(const_offset) = const_offset {
709            // Remove any (redundant) pointer casts applied to the input `ptr`,
710            // to obtain the "most original" pointer (which ideally will be e.g.
711            // a whole `OpVariable`, or the result of a previous `OpAccessChain`).
712            let original_ptr = ptr.strip_ptrcasts();
713            let original_pointee_ty = match self.lookup_type(original_ptr.ty) {
714                SpirvType::Pointer { pointee } => pointee,
715                other => self.fatal(format!("pointer arithmetic on non-pointer type {other:?}")),
716            };
717
718            if let Some((const_indices, leaf_pointee_ty)) = self.recover_access_chain_from_offset(
719                original_pointee_ty,
720                const_offset,
721                Some(Size::ZERO)..=None,
722                None,
723            ) {
724                trace!(
725                    "ptr_offset_strided: strategy 2 picked: offset {const_offset:?} \
726                     => access chain w/ {const_indices:?}"
727                );
728
729                let leaf_ptr_ty = self.type_ptr_to(leaf_pointee_ty);
730                let original_ptr_id = original_ptr.def(self);
731                let const_indices_ids = const_indices
732                    .into_iter()
733                    .map(|idx| self.constant_u32(self.span(), idx).def(self))
734                    .collect();
735
736                return self.emit_access_chain(
737                    leaf_ptr_ty,
738                    original_ptr_id,
739                    None,
740                    const_indices_ids,
741                    is_inbounds,
742                );
743            }
744        }
745
746        // Strategy 3: try merging onto an `OpAccessChain` of matching type,
747        // and this starts by `pointercast`-ing to that type for two reasons:
748        // - this is the only type a merged `OpAccessChain` could produce
749        // - `pointercast` can itself produce a new `OpAccessChain` in the
750        //   right circumstances (e.g. `&array[0]` for `*[T; N]` -> `*T`)
751        let ptr = self.pointercast(ptr, self.type_ptr_to(stride_elem_ty));
752        let ptr_id = ptr.def(self);
753
754        let maybe_original_access_chain = {
755            let builder = self.emit();
756            let module = builder.module_ref();
757            let current_func_blocks = builder
758                .selected_function()
759                .and_then(|func_idx| Some(&module.functions.get(func_idx)?.blocks[..]))
760                .unwrap_or_default();
761
762            // NOTE(eddyb) reverse search (`rfind`) used in the hopes that the
763            // instruction producing the value with ID `ptr_id` is more likely to
764            // have been added more recently to the function, even though there's
765            // still a risk of this causing a whole-function traversal.
766            //
767            // FIXME(eddyb) consider tracking that via e.g. `SpirvValueKind`.
768            current_func_blocks
769                .iter()
770                .flat_map(|b| &b.instructions)
771                .rfind(|inst| inst.result_id == Some(ptr_id))
772                .filter(|inst| {
773                    matches!(inst.class.opcode, Op::AccessChain | Op::InBoundsAccessChain)
774                })
775                .map(|inst| {
776                    let base_ptr = inst.operands[0].unwrap_id_ref();
777                    let indices = inst.operands[1..]
778                        .iter()
779                        .map(|op| op.unwrap_id_ref())
780                        .collect::<Vec<_>>();
781                    (base_ptr, indices)
782                })
783        };
784        if let Some((original_ptr, original_indices)) = maybe_original_access_chain {
785            trace!("ptr_offset_strided: strategy 3 picked: merging access chains");
786
787            let mut merged_indices = original_indices;
788
789            let last_index_id = merged_indices.last_mut().unwrap();
790            *last_index_id = self.add(last_index_id.with_type(index.ty), index).def(self);
791
792            return self.emit_access_chain(ptr.ty, original_ptr, None, merged_indices, is_inbounds);
793        }
794
795        // None of the legalizing strategies above applied, so this operation
796        // isn't really supported (and will error if actually used from a shader).
797        //
798        // FIXME(eddyb) supersede via SPIR-T pointer legalization (e.g. `qptr`).
799        trace!("ptr_offset_strided: falling back to (illegal) `OpPtrAccessChain`");
800
801        let result_ptr = if is_inbounds {
802            self.emit()
803                .in_bounds_ptr_access_chain(ptr.ty, None, ptr_id, index.def(self), vec![])
804        } else {
805            self.emit()
806                .ptr_access_chain(ptr.ty, None, ptr_id, index.def(self), vec![])
807        }
808        .unwrap();
809        self.zombie(
810            result_ptr,
811            "cannot offset a pointer to an arbitrary element",
812        );
813        result_ptr.with_type(ptr.ty)
814    }

816    #[instrument(
817        level = "trace",
818        skip(self),
819        fields(
820            result_type = ?self.debug_type(result_type),
821             pointer,
822             ptr_base_index,
823             indices,
824             is_inbounds
825            )
826        )]
827    fn emit_access_chain(
828        &mut self,
829        result_type: <Self as BackendTypes>::Type,
830        pointer: Word,
831        ptr_base_index: Option<SpirvValue>,
832        indices: Vec<Word>,
833        is_inbounds: bool,
834    ) -> SpirvValue {
835        let mut builder = self.emit();
836
837        let non_zero_ptr_base_index =
838            ptr_base_index.filter(|&idx| self.builder.lookup_const_scalar(idx) != Some(0));
839        if let Some(ptr_base_index) = non_zero_ptr_base_index {
840            let result = if is_inbounds {
841                builder.in_bounds_ptr_access_chain(
842                    result_type,
843                    None,
844                    pointer,
845                    ptr_base_index.def(self),
846                    indices,
847                )
848            } else {
849                builder.ptr_access_chain(
850                    result_type,
851                    None,
852                    pointer,
853                    ptr_base_index.def(self),
854                    indices,
855                )
856            }
857            .unwrap();
858            self.zombie(result, "cannot offset a pointer to an arbitrary element");
859            result
860        } else {
861            if is_inbounds {
862                builder.in_bounds_access_chain(result_type, None, pointer, indices)
863            } else {
864                builder.access_chain(result_type, None, pointer, indices)
865            }
866            .unwrap()
867        }
868        .with_type(result_type)
869    }

871    #[instrument(level = "trace", skip(self))]
872    fn fptoint_sat(
873        &mut self,
874        signed: bool,
875        val: SpirvValue,
876        dest_ty: <Self as BackendTypes>::Type,
877    ) -> SpirvValue {
878        // This uses the old llvm emulation to implement saturation
879
880        let src_ty = self.cx.val_ty(val);
881        let (float_ty, int_ty) = if self.cx.type_kind(src_ty) == TypeKind::Vector {
882            assert_eq!(
883                self.cx.vector_length(src_ty),
884                self.cx.vector_length(dest_ty)
885            );
886            (self.cx.element_type(src_ty), self.cx.element_type(dest_ty))
887        } else {
888            (src_ty, dest_ty)
889        };
890        let int_width = self.cx().int_width(int_ty);
891        let float_width = self.cx().float_width(float_ty);
892        // LLVM's fpto[su]i returns undef when the input x is infinite, NaN, or does not fit into the
893        // destination integer type after rounding towards zero. This `undef` value can cause UB in
894        // safe code (see issue #10184), so we implement a saturating conversion on top of it:
895        // Semantically, the mathematical value of the input is rounded towards zero to the next
896        // mathematical integer, and then the result is clamped into the range of the destination
897        // integer type. Positive and negative infinity are mapped to the maximum and minimum value of
898        // the destination integer type. NaN is mapped to 0.
899        //
900        // Define f_min and f_max as the largest and smallest (finite) floats that are exactly equal to
901        // a value representable in int_ty.
902        // They are exactly equal to int_ty::{MIN,MAX} if float_ty has enough significand bits.
903        // Otherwise, int_ty::MAX must be rounded towards zero, as it is one less than a power of two.
904        // int_ty::MIN, however, is either zero or a negative power of two and is thus exactly
905        // representable. Note that this only works if float_ty's exponent range is sufficiently large.
906        // f16 or 256 bit integers would break this property. Right now the smallest float type is f32
907        // with exponents ranging up to 127, which is barely enough for i128::MIN = -2^127.
908        // On the other hand, f_max works even if int_ty::MAX is greater than float_ty::MAX. Because
909        // we're rounding towards zero, we just get float_ty::MAX (which is always an integer).
910        // This already happens today with u128::MAX = 2^128 - 1 > f32::MAX.
911        let int_max = |signed: bool, int_width: u64| -> u128 {
912            let shift_amount = 128 - int_width;
913            if signed {
914                i128::MAX as u128 >> shift_amount
915            } else {
916                u128::MAX >> shift_amount
917            }
918        };
919        let int_min = |signed: bool, int_width: u64| -> i128 {
920            if signed {
921                i128::MIN >> (128 - int_width)
922            } else {
923                0
924            }
925        };
926
927        let compute_clamp_bounds_single = |signed: bool, int_width: u64| -> (u128, u128) {
928            let rounded_min =
929                ieee::Single::from_i128_r(int_min(signed, int_width), Round::TowardZero);
930            assert_eq!(rounded_min.status, Status::OK);
931            let rounded_max =
932                ieee::Single::from_u128_r(int_max(signed, int_width), Round::TowardZero);
933            assert!(rounded_max.value.is_finite());
934            (rounded_min.value.to_bits(), rounded_max.value.to_bits())
935        };
936        let compute_clamp_bounds_double = |signed: bool, int_width: u64| -> (u128, u128) {
937            let rounded_min =
938                ieee::Double::from_i128_r(int_min(signed, int_width), Round::TowardZero);
939            assert_eq!(rounded_min.status, Status::OK);
940            let rounded_max =
941                ieee::Double::from_u128_r(int_max(signed, int_width), Round::TowardZero);
942            assert!(rounded_max.value.is_finite());
943            (rounded_min.value.to_bits(), rounded_max.value.to_bits())
944        };
945        // To implement saturation, we perform the following steps:
946        //
947        // 1. Cast x to an integer with fpto[su]i. This may result in undef.
948        // 2. Compare x to f_min and f_max, and use the comparison results to select:
949        //  a) int_ty::MIN if x < f_min or x is NaN
950        //  b) int_ty::MAX if x > f_max
951        //  c) the result of fpto[su]i otherwise
952        // 3. If x is NaN, return 0.0, otherwise return the result of step 2.
953        //
954        // This avoids resulting undef because values in range [f_min, f_max] by definition fit into the
955        // destination type. It creates an undef temporary, but *producing* undef is not UB. Our use of
956        // undef does not introduce any non-determinism either.
957        // More importantly, the above procedure correctly implements saturating conversion.
958        // Proof (sketch):
959        // If x is NaN, 0 is returned by definition.
960        // Otherwise, x is finite or infinite and thus can be compared with f_min and f_max.
961        // This yields three cases to consider:
962        // (1) if x in [f_min, f_max], the result of fpto[su]i is returned, which agrees with
963        //     saturating conversion for inputs in that range.
964        // (2) if x > f_max, then x is larger than int_ty::MAX. This holds even if f_max is rounded
965        //     (i.e., if f_max < int_ty::MAX) because in those cases, nextUp(f_max) is already larger
966        //     than int_ty::MAX. Because x is larger than int_ty::MAX, the return value of int_ty::MAX
967        //     is correct.
968        // (3) if x < f_min, then x is smaller than int_ty::MIN. As shown earlier, f_min exactly equals
969        //     int_ty::MIN and therefore the return value of int_ty::MIN is correct.
970        // QED.
971
972        let float_bits_to_llval = |bx: &mut Self, bits| {
973            let bits_llval = match float_width {
974                32 => bx.cx().const_u32(bits as u32),
975                64 => bx.cx().const_u64(bits as u64),
976                n => bug!("unsupported float width {}", n),
977            };
978            bx.bitcast(bits_llval, float_ty)
979        };
980        let (f_min, f_max) = match float_width {
981            32 => compute_clamp_bounds_single(signed, int_width),
982            64 => compute_clamp_bounds_double(signed, int_width),
983            n => bug!("unsupported float width {}", n),
984        };
985        let f_min = float_bits_to_llval(self, f_min);
986        let f_max = float_bits_to_llval(self, f_max);
987        let int_max = self.cx().const_uint_big(int_ty, int_max(signed, int_width));
988        let int_min = self
989            .cx()
990            .const_uint_big(int_ty, int_min(signed, int_width) as u128);
991        let zero = self.cx().const_uint(int_ty, 0);
992
993        // If we're working with vectors, constants must be "splatted": the constant is duplicated
994        // into each lane of the vector.  The algorithm stays the same, we are just using the
995        // same constant across all lanes.
996        let maybe_splat = |bx: &mut Self, val| {
997            if bx.cx().type_kind(dest_ty) == TypeKind::Vector {
998                bx.vector_splat(bx.vector_length(dest_ty), val)
999            } else {
1000                val
1001            }
1002        };
1003        let f_min = maybe_splat(self, f_min);
1004        let f_max = maybe_splat(self, f_max);
1005        let int_max = maybe_splat(self, int_max);
1006        let int_min = maybe_splat(self, int_min);
1007        let zero = maybe_splat(self, zero);
1008
1009        // Step 1 ...
1010        let fptosui_result = if signed {
1011            self.fptosi(val, dest_ty)
1012        } else {
1013            self.fptoui(val, dest_ty)
1014        };
1015        let less_or_nan = self.fcmp(RealPredicate::RealULT, val, f_min);
1016        let greater = self.fcmp(RealPredicate::RealOGT, val, f_max);
1017
1018        // Step 2: We use two comparisons and two selects, with %s1 being the
1019        // result:
1020        //     %less_or_nan = fcmp ult %x, %f_min
1021        //     %greater = fcmp olt %x, %f_max
1022        //     %s0 = select %less_or_nan, int_ty::MIN, %fptosi_result
1023        //     %s1 = select %greater, int_ty::MAX, %s0
1024        // Note that %less_or_nan uses an *unordered* comparison. This
1025        // comparison is true if the operands are not comparable (i.e., if x is
1026        // NaN). The unordered comparison ensures that s1 becomes int_ty::MIN if
1027        // x is NaN.
1028        //
1029        // Performance note: Unordered comparison can be lowered to a "flipped"
1030        // comparison and a negation, and the negation can be merged into the
1031        // select. Therefore, it not necessarily any more expensive than an
1032        // ordered ("normal") comparison. Whether these optimizations will be
1033        // performed is ultimately up to the backend, but at least x86 does
1034        // perform them.
1035        let s0 = self.select(less_or_nan, int_min, fptosui_result);
1036        let s1 = self.select(greater, int_max, s0);
1037
1038        // Step 3: NaN replacement.
1039        // For unsigned types, the above step already yielded int_ty::MIN == 0 if x is NaN.
1040        // Therefore we only need to execute this step for signed integer types.
1041        if signed {
1042            // LLVM has no isNaN predicate, so we use (x == x) instead
1043            let cmp = self.fcmp(RealPredicate::RealOEQ, val, val);
1044            self.select(cmp, s1, zero)
1045        } else {
1046            s1
1047        }
1048    }

1050    // HACK(eddyb) helper shared by `typed_alloca` and `alloca`.
1051    #[instrument(level = "trace", skip(self), fields(ty = ?self.debug_type(ty)))]
1052    fn declare_func_local_var(
1053        &mut self,
1054        ty: <Self as BackendTypes>::Type,
1055        _align: Align,
1056    ) -> SpirvValue {
1057        let ptr_ty = self.type_ptr_to(ty);
1058
1059        // "All OpVariable instructions in a function must be the first instructions in the first block."
1060        let mut builder = self.emit();
1061        builder.select_block(Some(0)).unwrap();
1062        let index = {
1063            let block = &builder.module_ref().functions[builder.selected_function().unwrap()]
1064                .blocks[builder.selected_block().unwrap()];
1065            block
1066                .instructions
1067                .iter()
1068                .enumerate()
1069                .find_map(|(index, inst)| {
1070                    if inst.class.opcode != Op::Variable {

1073                        None
1074                    }
1075                })
1076                .unwrap_or(InsertPoint::End)
1077        };
1078        // TODO: rspirv doesn't have insert_variable function
1079        let result_id = builder.id();

1081            Op::Variable,
1082            Some(ptr_ty),
1083            Some(result_id),
1084            vec![Operand::StorageClass(StorageClass::Function)],
1085        );
1086        builder.insert_into_block(index, inst).unwrap();
1087        result_id.with_type(ptr_ty)
1088    }

1263        fn construct_8(self_: &Builder<'_, '_>, signed: bool, v: u128) -> Operand {
1264            if v > u8::MAX as u128 {
1265                self_.fatal(format!(
1266                    "Switches to values above u8::MAX not supported: {v:?}"

1269                // this cast chain can probably be collapsed, but, whatever, be safe
1270                Operand::LiteralBit32(v as u8 as i8 as i32 as u32)
1271            } else {
1272                Operand::LiteralBit32(v as u8 as u32)

1274        }

1276            if v > u16::MAX as u128 {

1278                    "Switches to values above u16::MAX not supported: {v:?}"
1279                ))
1280            } else if signed {
1281                Operand::LiteralBit32(v as u16 as i16 as i32 as u32)
1282            } else {
1283                Operand::LiteralBit32(v as u16 as u32)

1287            if v > u32::MAX as u128 {
1288                self_.fatal(format!(
1289                    "Switches to values above u32::MAX not supported: {v:?}"
1290                ))
1291            } else {
1292                Operand::LiteralBit32(v as u32)
1293            }
1294        }

1296            if v > u64::MAX as u128 {
1297                self_.fatal(format!(
1298                    "Switches to values above u64::MAX not supported: {v:?}"
1299                ))
1300            } else {
1301                Operand::LiteralBit64(v as u64)
1302            }
1303        }

1305        let (signed, construct_case) = match self.lookup_type(v.ty) {
1306            SpirvType::Integer(width, signed) => {

1312                    other => self.fatal(format!(

1315                };

1317            }
1318            other => self.fatal(format!(
1319                "switch selector cannot have non-integer type {}",
1320                other.debug(v.ty, self)
1321            )),
1322        };

1324            .map(|(i, b)| (construct_case(self, signed, i), b.id))
1325            .collect::<Vec<_>>();

1327            .switch(v.def(self), else_llbb.id, cases)
1328            .unwrap();

1481    #[instrument(level = "trace", skip(self))]
1482    fn checked_binop(
1483        &mut self,
1484        oop: OverflowOp,
1485        ty: Ty<'_>,
1486        lhs: Self::Value,
1487        rhs: Self::Value,
1488    ) -> (Self::Value, Self::Value) {
1489        // adopted partially from https://github.com/ziglang/zig/blob/master/src/codegen/spirv.zig
1490        let is_add = match oop {
1491            OverflowOp::Add => true,
1492            OverflowOp::Sub => false,
1493            OverflowOp::Mul => {
1494                // NOTE(eddyb) this needs to be `undef`, not `false`/`true`, because
1495                // we don't want the user's boolean constants to keep the zombie alive.
1496                let bool = SpirvType::Bool.def(self.span(), self);
1497                let overflowed = self.undef(bool);
1498
1499                let result = (self.mul(lhs, rhs), overflowed);
1500                self.zombie(result.1.def(self), "checked mul is not supported yet");
1501                return result;
1502            }
1503        };
1504        let signed = match ty.kind() {
1505            ty::Int(_) => true,
1506            ty::Uint(_) => false,
1507            other => self.fatal(format!(
1508                "Unexpected {} type: {other:#?}",
1509                match oop {
1510                    OverflowOp::Add => "checked add",
1511                    OverflowOp::Sub => "checked sub",
1512                    OverflowOp::Mul => "checked mul",
1513                }
1514            )),
1515        };
1516
1517        let result = if is_add {
1518            self.add(lhs, rhs)
1519        } else {
1520            self.sub(lhs, rhs)
1521        };
1522
1523        let overflowed = if signed {
1524            // when adding, overflow could happen if
1525            // - rhs is positive and result < lhs; or
1526            // - rhs is negative and result > lhs
1527            // this is equivalent to (rhs < 0) == (result > lhs)
1528            //
1529            // when subtracting, overflow happens if
1530            // - rhs is positive and result > lhs; or
1531            // - rhs is negative and result < lhs
1532            // this is equivalent to (rhs < 0) == (result < lhs)
1533            let rhs_lt_zero = self.icmp(IntPredicate::IntSLT, rhs, self.constant_int(rhs.ty, 0));
1534            let result_gt_lhs = self.icmp(
1535                if is_add {
1536                    IntPredicate::IntSGT
1537                } else {
1538                    IntPredicate::IntSLT
1539                },
1540                result,
1541                lhs,
1542            );
1543            self.icmp(IntPredicate::IntEQ, rhs_lt_zero, result_gt_lhs)
1544        } else {
1545            // for unsigned addition, overflow occurred if the result is less than any of the operands.
1546            // for subtraction, overflow occurred if the result is greater.
1547            self.icmp(
1548                if is_add {
1549                    IntPredicate::IntULT
1550                } else {
1551                    IntPredicate::IntUGT
1552                },
1553                result,
1554                lhs,
1555            )
1556        };
1557
1558        (result, overflowed)
1559    }

1562    // immediates, but if they're behind a pointer, they're compiled to u8. The reason for this is
1563    // because LLVM is bad at bools behind pointers (something something u1 bitmasking on load).
1564    //
1565    // SPIR-V allows bools behind *some* pointers, and disallows others - specifically, it allows
1566    // bools behind the storage classes Workgroup, CrossWorkgroup, Private, Function, Input, and

1568    // see, no bools allowed". So, we always compile rust bools to SPIR-V bools instead of u8 as
1569    // rustc does, even if they're behind a pointer, and error if bools are in an interface (the
1570    // user should choose u8, u32, or something else instead). That means that immediate types and
1571    // memory types are the same, and no conversion needs to happen here.

1638    fn load_operand(
1639        &mut self,
1640        place: PlaceRef<'tcx, Self::Value>,
1641    ) -> OperandRef<'tcx, Self::Value> {
1642        if place.layout.is_zst() {
1643            return OperandRef::zero_sized(place.layout);
1644        }
1645
1646        let val = if place.val.llextra.is_some() {
1647            OperandValue::Ref(place.val)
1648        } else if self.cx.is_backend_immediate(place.layout) {
1649            let llval = self.load(
1650                place.layout.spirv_type(self.span(), self),
1651                place.val.llval,
1652                place.val.align,
1653            );
1654            OperandValue::Immediate(llval)
1655        } else if let BackendRepr::ScalarPair(a, b) = place.layout.backend_repr {
1656            let b_offset = a
1657                .primitive()
1658                .size(self)
1659                .align_to(b.primitive().align(self).abi);
1660
1661            let mut load = |i, scalar: Scalar, align| {
1662                let llptr = if i == 0 {
1663                    place.val.llval
1664                } else {
1665                    self.inbounds_ptradd(place.val.llval, self.const_usize(b_offset.bytes()))
1666                };
1667                let load = self.load(
1668                    self.scalar_pair_element_backend_type(place.layout, i, false),
1669                    llptr,
1670                    align,
1671                );
1672                self.to_immediate_scalar(load, scalar)
1673            };
1674
1675            OperandValue::Pair(
1676                load(0, a, place.val.align),
1677                load(1, b, place.val.align.restrict_for_offset(b_offset)),
1678            )
1679        } else {
1680            OperandValue::Ref(place.val)
1681        };
1682        OperandRef {
1683            val,
1684            layout: place.layout,
1685        }
1686    }

1942    #[instrument(level = "trace", skip(self), fields(val_type = ?self.debug_type(val.ty), dest_ty = ?self.debug_type(dest_ty)))]
1943    fn bitcast(&mut self, val: Self::Value, dest_ty: Self::Type) -> Self::Value {
1944        if val.ty == dest_ty {
1945            val
1946        } else {
1947            let val_ty_kind = self.lookup_type(val.ty);
1948            let dest_ty_kind = self.lookup_type(dest_ty);
1949
1950            // HACK(eddyb) account for bitcasts from/to aggregates not being legal
1951            // in SPIR-V, but still being used to paper over untyped pointers,
1952            // by unpacking/repacking newtype-shaped aggregates as-needed.
1953            let unpack_newtype = |ty, kind| {
1954                let span = span!(Level::DEBUG, "unpack_newtype");
1955                let _guard = span.enter();
1956
1957                if !matches!(kind, SpirvType::Adt { .. } | SpirvType::Array { .. }) {
1958                    return None;
1959                }
1960                let size = kind.sizeof(self)?;
1961                let mut leaf_ty = ty;
1962
1963                // FIXME(eddyb) this isn't efficient, `recover_access_chain_from_offset`
1964                // could instead be doing all the extra digging itself.
1965                let mut indices = SmallVec::<[_; 8]>::new();
1966                while let Some((inner_indices, inner_ty)) = self.recover_access_chain_from_offset(
1967                    leaf_ty,
1968                    Size::ZERO,
1969                    Some(size)..=Some(size),
1970                    None,
1971                ) {
1972                    indices.extend(inner_indices);
1973                    leaf_ty = inner_ty;
1974                }
1975
1976                (!indices.is_empty()).then_some((indices, leaf_ty))
1977            };
1978            // Unpack input newtypes, and bitcast the leaf inside, instead.
1979            if let Some((indices, in_leaf_ty)) = unpack_newtype(val.ty, val_ty_kind) {
1980                let in_leaf = self
1981                    .emit()
1982                    .composite_extract(in_leaf_ty, None, val.def(self), indices)
1983                    .unwrap()
1984                    .with_type(in_leaf_ty);
1985
1986                trace!(
1987                    "unpacked newtype. val: {} -> in_leaf_ty: {}",
1988                    self.debug_type(val.ty),
1989                    self.debug_type(in_leaf_ty),
1990                );
1991                return self.bitcast(in_leaf, dest_ty);
1992            }
1993
1994            // Repack output newtypes, after bitcasting the leaf inside, instead.
1995            if let Some((indices, out_leaf_ty)) = unpack_newtype(dest_ty, dest_ty_kind) {
1996                trace!(
1997                    "unpacked newtype: dest: {} -> out_leaf_ty: {}",
1998                    self.debug_type(dest_ty),
1999                    self.debug_type(out_leaf_ty),
2000                );
2001                let out_leaf = self.bitcast(val, out_leaf_ty);
2002                let out_agg_undef = self.undef(dest_ty);
2003                trace!("returning composite insert");
2004                return self
2005                    .emit()
2006                    .composite_insert(
2007                        dest_ty,
2008                        None,
2009                        out_leaf.def(self),
2010                        out_agg_undef.def(self),
2011                        indices,
2012                    )
2013                    .unwrap()
2014                    .with_type(dest_ty);
2015            }
2016
2017            let val_is_ptr = matches!(val_ty_kind, SpirvType::Pointer { .. });
2018            let dest_is_ptr = matches!(dest_ty_kind, SpirvType::Pointer { .. });
2019
2020            // Reuse the pointer-specific logic in `pointercast` for `*T -> *U`.
2021            if val_is_ptr && dest_is_ptr {
2022                trace!("val and dest are both pointers");
2023                return self.pointercast(val, dest_ty);
2024            }
2025
2026            trace!(
2027                "before emitting: val ty: {} -> dest ty: {}",
2028                self.debug_type(val.ty),
2029                self.debug_type(dest_ty)
2030            );
2031
2032            let result = self
2033                .emit()
2034                .bitcast(dest_ty, None, val.def(self))
2035                .unwrap()
2036                .with_type(dest_ty);
2037
2038            if val_is_ptr || dest_is_ptr {
2039                self.zombie(
2040                    result.def(self),
2041                    &format!(
2042                        "cannot cast between pointer and non-pointer types\
2043                         \nfrom `{}`\
2044                         \n  to `{}`",
2045                        self.debug_type(val.ty),
2046                        self.debug_type(dest_ty)
2047                    ),
2048                );
2049            }
2050
2051            result
2052        }
2053    }

2055    fn intcast(&mut self, val: Self::Value, dest_ty: Self::Type, is_signed: bool) -> Self::Value {
2056        if val.ty == dest_ty {
2057            // I guess?
2058            return val;
2059        }
2060
2061        // If casting a constant, directly create a constant of the target type. This
2062        // avoids creating intermediate types that might require additional
2063        // capabilities. For example, casting a u8 constant to u32 will directly create
2064        // a u32 constant, avoiding the need for Int8 capability if it is not used
2065        // elsewhere.
2066        if let Some(const_val) = self.builder.lookup_const_scalar(val) {
2067            let src_ty = self.lookup_type(val.ty);
2068            let dst_ty_spv = self.lookup_type(dest_ty);
2069
2070            // Try to optimize the constant cast
2071            let optimized_result = match (src_ty, dst_ty_spv) {
2072                // Integer to integer cast
2073                (SpirvType::Integer(src_width, _), SpirvType::Integer(dst_width, _)) => {
2074                    // Only optimize if we're widening. This avoids creating the source
2075                    // type when it's safe to do so. For narrowing casts (e.g., u32 as
2076                    // u8), we need the proper truncation behavior that the regular cast
2077                    // provides.
2078                    if src_width < dst_width {
2079                        Some(self.constant_int(dest_ty, const_val))
2080                    } else {
2081                        None
2082                    }
2083                }
2084                // Bool to integer cast - const_val will be 0 or 1
2085                (SpirvType::Bool, SpirvType::Integer(_, _)) => {
2086                    Some(self.constant_int(dest_ty, const_val))
2087                }
2088                // Integer to bool cast - compare with zero
2089                (SpirvType::Integer(_, _), SpirvType::Bool) => {
2090                    Some(self.constant_bool(self.span(), const_val != 0))
2091                }
2092                _ => None,
2093            };
2094
2095            if let Some(result) = optimized_result {
2096                return result;
2097            }
2098        }
2099
2100        match (self.lookup_type(val.ty), self.lookup_type(dest_ty)) {
2101            // sign change
2102            (
2103                SpirvType::Integer(val_width, val_signedness),
2104                SpirvType::Integer(dest_width, dest_signedness),
2105            ) if val_width == dest_width && val_signedness != dest_signedness => self
2106                .emit()
2107                .bitcast(dest_ty, None, val.def(self))
2108                .unwrap()
2109                .with_type(dest_ty),
2110            // width change, and optional sign change
2111            (SpirvType::Integer(_, _), SpirvType::Integer(_, dest_signedness)) => {
2112                // spir-v spec doesn't seem to say that signedness needs to match the operands, only that the signedness
2113                // of the destination type must match the instruction's signedness.
2114                if dest_signedness {
2115                    self.emit().s_convert(dest_ty, None, val.def(self))
2116                } else {
2117                    self.emit().u_convert(dest_ty, None, val.def(self))
2118                }
2119                .unwrap()
2120                .with_type(dest_ty)
2121            }
2122            // bools are ints in llvm, so we have to implement this here
2123            (SpirvType::Bool, SpirvType::Integer(_, _)) => {
2124                // spir-v doesn't have a direct conversion instruction
2125                let if_true = self.constant_int(dest_ty, 1);
2126                let if_false = self.constant_int(dest_ty, 0);
2127                self.emit()
2128                    .select(
2129                        dest_ty,
2130                        None,
2131                        val.def(self),
2132                        if_true.def(self),
2133                        if_false.def(self),
2134                    )
2135                    .unwrap()
2136                    .with_type(dest_ty)
2137            }
2138            (SpirvType::Integer(_, _), SpirvType::Bool) => {
2139                // spir-v doesn't have a direct conversion instruction, glslang emits OpINotEqual
2140                let zero = self.constant_int(val.ty, 0);
2141                self.emit()
2142                    .i_not_equal(dest_ty, None, val.def(self), zero.def(self))
2143                    .unwrap()
2144                    .with_type(dest_ty)
2145            }
2146            (val_ty, dest_ty_spv) => self.fatal(format!(
2147                "TODO: intcast not implemented yet: val={val:?} val.ty={val_ty:?} dest_ty={dest_ty_spv:?} is_signed={is_signed}"
2148            )),
2149        }
2150    }

2152    #[instrument(level = "trace", skip(self), fields(ptr, ptr_ty = ?self.debug_type(ptr.ty), dest_ty = ?self.debug_type(dest_ty)))]
2153    fn pointercast(&mut self, ptr: Self::Value, dest_ty: Self::Type) -> Self::Value {
2154        // HACK(eddyb) reuse the special-casing in `const_bitcast`, which relies
2155        // on adding a pointer type to an untyped pointer (to some const data).
2156        if let SpirvValueKind::IllegalConst(_) = ptr.kind {
2157            trace!("illegal const");
2158            return self.const_bitcast(ptr, dest_ty);
2159        }
2160
2161        if ptr.ty == dest_ty {
2162            trace!("ptr.ty == dest_ty");
2163            return ptr;
2164        }
2165
2166        // Strip a previous `pointercast`, to reveal the original pointer type.
2167        let ptr = ptr.strip_ptrcasts();
2168
2169        trace!(
2170            "ptr type after strippint pointer cases: {}",
2171            self.debug_type(ptr.ty),
2172        );
2173
2174        if ptr.ty == dest_ty {
2175            return ptr;
2176        }
2177
2178        let ptr_pointee = match self.lookup_type(ptr.ty) {
2179            SpirvType::Pointer { pointee } => pointee,
2180            other => self.fatal(format!(
2181                "pointercast called on non-pointer source type: {other:?}"
2182            )),
2183        };
2184        let dest_pointee = match self.lookup_type(dest_ty) {
2185            SpirvType::Pointer { pointee } => pointee,
2186            other => self.fatal(format!(
2187                "pointercast called on non-pointer dest type: {other:?}"
2188            )),
2189        };
2190        let dest_pointee_size = self.lookup_type(dest_pointee).sizeof(self);
2191
2192        if let Some((indices, _)) = self.recover_access_chain_from_offset(
2193            ptr_pointee,
2194            Size::ZERO,
2195            dest_pointee_size..=dest_pointee_size,
2196            Some(dest_pointee),
2197        ) {
2198            trace!("`recover_access_chain_from_offset` returned something");
2199            trace!(
2200                "ptr_pointee: {}, dest_pointee {}",
2201                self.debug_type(ptr_pointee),
2202                self.debug_type(dest_pointee),
2203            );
2204            let indices = indices
2205                .into_iter()
2206                .map(|idx| self.constant_u32(self.span(), idx).def(self))
2207                .collect::<Vec<_>>();
2208            self.emit()
2209                .in_bounds_access_chain(dest_ty, None, ptr.def(self), indices)
2210                .unwrap()
2211                .with_type(dest_ty)
2212        } else {
2213            trace!("`recover_access_chain_from_offset` returned `None`");
2214            trace!(
2215                "ptr_pointee: {}, dest_pointee {}",
2216                self.debug_type(ptr_pointee),
2217                self.debug_type(dest_pointee),
