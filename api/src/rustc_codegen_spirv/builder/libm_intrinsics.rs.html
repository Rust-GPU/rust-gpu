201impl Builder<'_, '_> {
202    pub fn call_libm_intrinsic(
203        &mut self,
204        intrinsic: LibmIntrinsic,
205        result_type: Word,
206        args: &[SpirvValue],
207    ) -> SpirvValue {
208        match intrinsic {
209            LibmIntrinsic::GLOp(op) => self.gl_op(op, result_type, args),
210            LibmIntrinsic::Custom(LibmCustomIntrinsic::SinCos) => {
211                assert_eq!(args.len(), 1);
212                let x = args[0];
213                let sin = self.gl_op(GLOp::Sin, x.ty, [x]).def(self);
214                let cos = self.gl_op(GLOp::Cos, x.ty, [x]).def(self);
215                self.emit()
216                    .composite_construct(result_type, None, [sin, cos].iter().copied())
217                    .unwrap()
218                    .with_type(result_type)
219            }
220            LibmIntrinsic::Custom(LibmCustomIntrinsic::Fmod) => {
221                assert_eq!(args.len(), 2);
222                self.emit()
223                    .f_mod(result_type, None, args[0].def(self), args[1].def(self))
224                    .unwrap()
225                    .with_type(result_type)
226            }
227            LibmIntrinsic::Custom(LibmCustomIntrinsic::CopySign) => {
228                assert_eq!(args.len(), 2);
229                self.copysign(args[0], args[1])
230            }
231            LibmIntrinsic::Custom(LibmCustomIntrinsic::Cbrt) => {
232                assert_eq!(args.len(), 1);
233                self.gl_op(
234                    GLOp::Pow,
235                    result_type,
236                    [args[0], self.constant_float(args[0].ty, 1.0 / 3.0)],
237                )
238            }
239            LibmIntrinsic::Custom(LibmCustomIntrinsic::Log10) => {
240                assert_eq!(args.len(), 1);
241                // log10(x) == (1 / ln(10)) * ln(x)
242                let mul = self.constant_float(args[0].ty, 1.0 / 10.0f64.ln());
243                let ln = self.gl_op(GLOp::Log, result_type, [args[0]]);
244                self.fmul(mul, ln)
245            }
246            LibmIntrinsic::Custom(LibmCustomIntrinsic::Log1p) => {
247                assert_eq!(args.len(), 1);
248                let one = self.constant_float(args[0].ty, 1.0);
249                let add = self.fadd(args[0], one);
250                self.gl_op(GLOp::Log, result_type, [add])
251            }
252            LibmIntrinsic::Custom(LibmCustomIntrinsic::Exp10) => {
253                assert_eq!(args.len(), 1);
254                // exp10(x) == exp(x * log(10));
255                let log10 = self.constant_float(args[0].ty, 10.0f64.ln());
256                let mul = self.fmul(args[0], log10);
257                self.gl_op(GLOp::Exp, result_type, [mul])
258            }
259            LibmIntrinsic::Custom(LibmCustomIntrinsic::Expm1) => {
260                let exp = self.gl_op(GLOp::Exp, args[0].ty, [args[0]]);
261                let one = self.constant_float(exp.ty, 1.0);
262                self.fsub(exp, one)
263            }
264            LibmIntrinsic::Custom(LibmCustomIntrinsic::Erf) => {
265                self.undef_zombie(result_type, "Erf not supported yet")
266            }
267            LibmIntrinsic::Custom(LibmCustomIntrinsic::Erfc) => {
268                self.undef_zombie(result_type, "Erfc not supported yet")
269            }
270            LibmIntrinsic::Custom(LibmCustomIntrinsic::Fdim) => {
271                self.undef_zombie(result_type, "Fdim not supported yet")
272            }
273            LibmIntrinsic::Custom(LibmCustomIntrinsic::Hypot) => {
274                self.undef_zombie(result_type, "Hypot not supported yet")
275            }
276            LibmIntrinsic::Custom(LibmCustomIntrinsic::Ilogb) => {
277                self.undef_zombie(result_type, "Ilogb not supported yet")
278            }
279            LibmIntrinsic::Custom(LibmCustomIntrinsic::J0) => {
280                self.undef_zombie(result_type, "J0 not supported yet")
281            }
282            LibmIntrinsic::Custom(LibmCustomIntrinsic::Y0) => {
283                self.undef_zombie(result_type, "Y0 not supported yet")
284            }
285            LibmIntrinsic::Custom(LibmCustomIntrinsic::J1) => {
286                self.undef_zombie(result_type, "J1 not supported yet")
287            }
288            LibmIntrinsic::Custom(LibmCustomIntrinsic::Y1) => {
289                self.undef_zombie(result_type, "Y1 not supported yet")
290            }
291            LibmIntrinsic::Custom(LibmCustomIntrinsic::Jn) => {
292                self.undef_zombie(result_type, "Jn not supported yet")
293            }
294            LibmIntrinsic::Custom(LibmCustomIntrinsic::Yn) => {
295                self.undef_zombie(result_type, "Yn not supported yet")
296            }
297            LibmIntrinsic::Custom(LibmCustomIntrinsic::Lgamma) => {
298                self.undef_zombie(result_type, "Lgamma not supported yet")
299            }
300            LibmIntrinsic::Custom(LibmCustomIntrinsic::LgammaR) => {
301                self.undef_zombie(result_type, "LgammaR not supported yet")
302            }
303            LibmIntrinsic::Custom(LibmCustomIntrinsic::Tgamma) => {
304                self.undef_zombie(result_type, "Tgamma not supported yet")
305            }
306            LibmIntrinsic::Custom(LibmCustomIntrinsic::NextAfter) => {
307                self.undef_zombie(result_type, "NextAfter not supported yet")
308            }
309            LibmIntrinsic::Custom(LibmCustomIntrinsic::Remainder) => {
310                self.undef_zombie(result_type, "Remainder not supported yet")
311            }
312            LibmIntrinsic::Custom(LibmCustomIntrinsic::RemQuo) => {
313                self.undef_zombie(result_type, "RemQuo not supported yet")
314            }
315            LibmIntrinsic::Custom(LibmCustomIntrinsic::Scalbn) => {
316                self.undef_zombie(result_type, "Scalbn not supported yet")
317            }
318        }
319    }
