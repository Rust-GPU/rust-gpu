16    fn load_err(&mut self, original_type: Word, invalid_type: Word) -> SpirvValue {
17        let mut err = self.struct_err(format!(
18            "cannot load type {} in an untyped buffer load",
19            self.debug_type(original_type)
20        ));
21        if original_type != invalid_type {
22            err.note(format!(
23                "due to containing type {}",
24                self.debug_type(invalid_type)
25            ));
26        }
27        err.emit();
28        self.undef(invalid_type)
29    }

31    fn load_u32(
32        &mut self,
33        array: SpirvValue,
34        dynamic_index: SpirvValue,
35        constant_offset: u32,
36    ) -> SpirvValue {
37        let actual_index = if constant_offset != 0 {
38            let const_offset_val = self.constant_u32(DUMMY_SP, constant_offset);
39            self.add(dynamic_index, const_offset_val)
40        } else {
41            dynamic_index
42        };
43        let u32_ty = SpirvType::Integer(32, false).def(DUMMY_SP, self);
44        let u32_ptr = self.type_ptr_to(u32_ty);
45        let array = array.def(self);
46        let actual_index = actual_index.def(self);
47        let ptr = self
48            .emit()
49            .in_bounds_access_chain(u32_ptr, None, array, [actual_index])
50            .unwrap()
51            .with_type(u32_ptr);
52        if self.builder.has_capability(Capability::ShaderNonUniform) {
53            // apply NonUniform to the operation and the index
54            self.emit()
55                .decorate(ptr.def(self), Decoration::NonUniform, []);
56            self.emit()
57                .decorate(actual_index, Decoration::NonUniform, []);
58        }
59        self.load(u32_ty, ptr, Align::ONE)
60    }

62    #[allow(clippy::too_many_arguments)]
63    fn load_vec_mat_arr(
64        &mut self,
65        original_type: Word,
66        result_type: Word,
67        array: SpirvValue,
68        dynamic_word_index: SpirvValue,
69        constant_word_offset: u32,
70        element: Word,
71        count: u32,
72    ) -> SpirvValue {
73        let element_size_bytes = match self.lookup_type(element).sizeof(self) {
74            Some(size) => size,
75            None => return self.load_err(original_type, result_type),
76        };
77        if element_size_bytes.bytes() % 4 != 0 {
78            return self.load_err(original_type, result_type);
79        }
80        let element_size_words = (element_size_bytes.bytes() / 4) as u32;
81        let args = (0..count)
82            .map(|index| {
83                self.recurse_load_type(
84                    original_type,
85                    element,
86                    array,
87                    dynamic_word_index,
88                    constant_word_offset + element_size_words * index,
89                )
90                .def(self)
91            })
92            .collect::<Vec<_>>();
93        self.emit()
94            .composite_construct(result_type, None, args)
95            .unwrap()
96            .with_type(result_type)
97    }

99    fn recurse_load_type(
100        &mut self,
101        original_type: Word,
102        result_type: Word,
103        array: SpirvValue,
104        dynamic_word_index: SpirvValue,
105        constant_word_offset: u32,
106    ) -> SpirvValue {
107        match self.lookup_type(result_type) {
108            SpirvType::Integer(32, signed) => {
109                let val = self.load_u32(array, dynamic_word_index, constant_word_offset);
110                self.intcast(val, result_type, signed)
111            }
112            SpirvType::Float(32) => {
113                let val = self.load_u32(array, dynamic_word_index, constant_word_offset);
114                self.bitcast(val, result_type)
115            }
116            SpirvType::Vector { element, count } | SpirvType::Matrix { element, count } => self
117                .load_vec_mat_arr(
118                    original_type,
119                    result_type,
120                    array,
121                    dynamic_word_index,
122                    constant_word_offset,
123                    element,
124                    count,
125                ),
126            SpirvType::Array { element, count } => {
127                let count = match self.builder.lookup_const_scalar(count) {
128                    Some(count) => count as u32,
129                    None => return self.load_err(original_type, result_type),
130                };
131                self.load_vec_mat_arr(
132                    original_type,
133                    result_type,
134                    array,
135                    dynamic_word_index,
136                    constant_word_offset,
137                    element,
138                    count,
139                )
140            }
141            SpirvType::Adt {
142                size: Some(_),
143                field_types,
144                field_offsets,
145                ..
146            } => {
147                let args = field_types
148                    .iter()
149                    .zip(field_offsets)
150                    .map(|(&field_type, byte_offset)| {
151                        if byte_offset.bytes() % 4 != 0 {
152                            return None;
153                        }
154                        let word_offset = (byte_offset.bytes() / 4) as u32;
155                        Some(
156                            self.recurse_load_type(
157                                original_type,
158                                field_type,
159                                array,
160                                dynamic_word_index,
161                                constant_word_offset + word_offset,
162                            )
163                            .def(self),
164                        )
165                    })
166                    .collect::<Option<Vec<_>>>();
167                match args {
168                    None => self.load_err(original_type, result_type),
169                    Some(args) => self
170                        .emit()
171                        .composite_construct(result_type, None, args)
172                        .unwrap()
173                        .with_type(result_type),
174                }
175            }
176
177            _ => self.load_err(original_type, result_type),
178        }
179    }

181    /// Note: DOES NOT do bounds checking! Bounds checking is expected to be done in the caller.
182    pub fn codegen_buffer_load_intrinsic(
183        &mut self,
184        result_type: Word,
185        args: &[SpirvValue],
186        pass_mode: &PassMode,
187    ) -> SpirvValue {
188        match pass_mode {
189            PassMode::Ignore => {
190                return SpirvValue {
191                    kind: SpirvValueKind::IllegalTypeUsed(result_type),
192                    ty: result_type,
193                };
194            }
195            // PassMode::Pair is identical to PassMode::Direct - it's returned as a struct
196            PassMode::Direct(_) | PassMode::Pair(_, _) => (),
197            PassMode::Cast { .. } => {
198                self.fatal("PassMode::Cast not supported in codegen_buffer_load_intrinsic")
199            }
200            PassMode::Indirect { .. } => {
201                self.fatal("PassMode::Indirect not supported in codegen_buffer_load_intrinsic")
202            }
203        }
204
205        // Signature: fn load<T>(array: &[u32], index: u32) -> T;
206        if args.len() != 3 {
207            self.fatal(format!(
208                "buffer_load_intrinsic should have 3 args, it has {}",
209                args.len()
210            ));
211        }
212        // Note that the &[u32] gets split into two arguments - pointer, length
213        let array = args[0];
214        let byte_index = args[2];
215        let two = self.constant_u32(DUMMY_SP, 2);
216        let word_index = self.lshr(byte_index, two);
217        self.recurse_load_type(result_type, result_type, array, word_index, 0)
218    }

220    fn store_err(&mut self, original_type: Word, value: SpirvValue) -> Result<(), ErrorGuaranteed> {
221        let mut err = self.struct_err(format!(
222            "cannot store type {} in an untyped buffer store",
223            self.debug_type(original_type)
224        ));
225        if original_type != value.ty {
226            err.note(format!("due to containing type {}", value.ty));
227        }
228        Err(err.emit())
229    }

231    fn store_u32(
232        &mut self,
233        array: SpirvValue,
234        dynamic_index: SpirvValue,
235        constant_offset: u32,
236        value: SpirvValue,
237    ) -> Result<(), ErrorGuaranteed> {
238        let actual_index = if constant_offset != 0 {
239            let const_offset_val = self.constant_u32(DUMMY_SP, constant_offset);
240            self.add(dynamic_index, const_offset_val)
241        } else {
242            dynamic_index
243        };
244        let u32_ty = SpirvType::Integer(32, false).def(DUMMY_SP, self);
245        let u32_ptr = self.type_ptr_to(u32_ty);
246        let array = array.def(self);
247        let actual_index = actual_index.def(self);
248        let ptr = self
249            .emit()
250            .in_bounds_access_chain(u32_ptr, None, array, [actual_index])
251            .unwrap()
252            .with_type(u32_ptr);
253        if self.builder.has_capability(Capability::ShaderNonUniform) {
254            // apply NonUniform to the operation and the index
255            self.emit()
256                .decorate(ptr.def(self), Decoration::NonUniform, []);
257            self.emit()
258                .decorate(actual_index, Decoration::NonUniform, []);
259        }
260        self.store(value, ptr, Align::ONE);
261        Ok(())
262    }

264    #[allow(clippy::too_many_arguments)]
265    fn store_vec_mat_arr(
266        &mut self,
267        original_type: Word,
268        value: SpirvValue,
269        array: SpirvValue,
270        dynamic_word_index: SpirvValue,
271        constant_word_offset: u32,
272        element: Word,
273        count: u32,
274    ) -> Result<(), ErrorGuaranteed> {
275        let element_size_bytes = match self.lookup_type(element).sizeof(self) {
276            Some(size) => size,
277            None => return self.store_err(original_type, value),
278        };
279        if element_size_bytes.bytes() % 4 != 0 {
280            return self.store_err(original_type, value);
281        }
282        let element_size_words = (element_size_bytes.bytes() / 4) as u32;
283        for index in 0..count {
284            let element = self.extract_value(value, index as u64);
285            self.recurse_store_type(
286                original_type,
287                element,
288                array,
289                dynamic_word_index,
290                constant_word_offset + element_size_words * index,
291            )?;
292        }
293        Ok(())
294    }

296    fn recurse_store_type(
297        &mut self,
298        original_type: Word,
299        value: SpirvValue,
300        array: SpirvValue,
301        dynamic_word_index: SpirvValue,
302        constant_word_offset: u32,
303    ) -> Result<(), ErrorGuaranteed> {
304        match self.lookup_type(value.ty) {
305            SpirvType::Integer(32, signed) => {
306                let u32_ty = SpirvType::Integer(32, false).def(DUMMY_SP, self);
307                let value_u32 = self.intcast(value, u32_ty, signed);
308                self.store_u32(array, dynamic_word_index, constant_word_offset, value_u32)
309            }
310            SpirvType::Float(32) => {
311                let u32_ty = SpirvType::Integer(32, false).def(DUMMY_SP, self);
312                let value_u32 = self.bitcast(value, u32_ty);
313                self.store_u32(array, dynamic_word_index, constant_word_offset, value_u32)
314            }
315            SpirvType::Vector { element, count } | SpirvType::Matrix { element, count } => self
316                .store_vec_mat_arr(
317                    original_type,
318                    value,
319                    array,
320                    dynamic_word_index,
321                    constant_word_offset,
322                    element,
323                    count,
324                ),
325            SpirvType::Array { element, count } => {
326                let count = match self.builder.lookup_const_scalar(count) {
327                    Some(count) => count as u32,
328                    None => return self.store_err(original_type, value),
329                };
330                self.store_vec_mat_arr(
331                    original_type,
332                    value,
333                    array,
334                    dynamic_word_index,
335                    constant_word_offset,
336                    element,
337                    count,
338                )
339            }
340            SpirvType::Adt {
341                size: Some(_),
342                field_offsets,
343                ..
344            } => {
345                for (index, byte_offset) in field_offsets.iter().enumerate() {
346                    if byte_offset.bytes() % 4 != 0 {
347                        return self.store_err(original_type, value);
348                    }
349                    let word_offset = (byte_offset.bytes() / 4) as u32;
350                    let field = self.extract_value(value, index as u64);
351                    self.recurse_store_type(
352                        original_type,
353                        field,
354                        array,
355                        dynamic_word_index,
356                        constant_word_offset + word_offset,
357                    )?;
358                }
359                Ok(())
360            }
361
362            _ => self.store_err(original_type, value),
363        }
364    }

366    /// Note: DOES NOT do bounds checking! Bounds checking is expected to be done in the caller.
367    pub fn codegen_buffer_store_intrinsic(&mut self, args: &[SpirvValue], pass_mode: &PassMode) {
368        // Signature: fn store<T>(array: &[u32], index: u32, value: T);
369        let is_pair = match pass_mode {
370            // haha shrug
371            PassMode::Ignore => return,
372            PassMode::Direct(_) => false,
373            PassMode::Pair(_, _) => true,
374            PassMode::Cast { .. } => {
375                self.fatal("PassMode::Cast not supported in codegen_buffer_store_intrinsic")
376            }
377            PassMode::Indirect { .. } => {
378                self.fatal("PassMode::Indirect not supported in codegen_buffer_store_intrinsic")
379            }
380        };
381        let expected_args = if is_pair { 5 } else { 4 };
382        if args.len() != expected_args {
383            self.fatal(format!(
384                "buffer_store_intrinsic should have {} args, it has {}",
385                expected_args,
386                args.len()
387            ));
388        }
389        // Note that the &[u32] gets split into two arguments - pointer, length
390        let array = args[0];
391        let byte_index = args[2];
392        let two = self.constant_u32(DUMMY_SP, 2);
393        let word_index = self.lshr(byte_index, two);
394        if is_pair {
395            let value_one = args[3];
396            let value_two = args[4];
397            let one_result = self.recurse_store_type(value_one.ty, value_one, array, word_index, 0);
398
399            let size_of_one = self.lookup_type(value_one.ty).sizeof(self);
400            if one_result.is_ok() && size_of_one != Some(Size::from_bytes(4)) {
401                self.fatal("Expected PassMode::Pair first element to have size 4");
402            }
403
404            let _ = self.recurse_store_type(value_two.ty, value_two, array, word_index, 1);
405        } else {
406            let value = args[3];
407            let _ = self.recurse_store_type(value.ty, value, array, word_index, 0);
408        }
409    }
