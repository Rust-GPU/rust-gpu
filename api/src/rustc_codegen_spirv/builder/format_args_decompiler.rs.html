17// HACK(eddyb) Rust 2021 `panic!` always uses `format_args!`, even
18// in the simple case that used to pass a `&str` constant, which
19// would not remain reachable in the SPIR-V - but `format_args!` is
20// more complex and neither immediate (`fmt::Arguments` is too big)
21// nor simplified in MIR (e.g. promoted to a constant) in any way,
22// so we have to try and remove the `fmt::Arguments::new` call here.
23#[derive(Default)]
24pub struct DecodedFormatArgs<'tcx> {
25    /// If fully constant, the `pieces: &'a [&'static str]` input
26    /// of `fmt::Arguments<'a>` (i.e. the strings between args).
27    const_pieces: Option<SmallVec<[String; 2]>>,
28
29    /// Original references for `fmt::Arguments<'a>` dynamic arguments,
30    /// i.e. the `&'a T` passed to `fmt::rt::Argument::<'a>::new_*`,
31    /// tracking the type `T` and `char` formatting specifier.
32    ///
33    /// E.g. for `format_args!("{a} {b:x}")` they'll be:
34    /// * `&a` with `typeof a` and ' ',
35    /// * `&b` with `typeof b` and 'x'
36    ref_arg_ids_with_ty_and_spec: SmallVec<[(Word, Ty<'tcx>, char); 2]>,
37
38    /// If `fmt::Arguments::new_v1_formatted` was used, this holds
39    /// the length of the `&[fmt::rt::Placeholder]` slice, which
40    /// currently cannot be directly supported, and therefore even
41    /// if all of `ref_arg_ids_with_ty_and_spec` are printable,
42    /// a much jankier fallback still has to be used, as it were:
43    ///
44    /// `format!("a{{0}}b{{1}}c\n  with {{â€¦}} from: {}, {}", x, y)`
45    /// (w/ `const_pieces = ["a", "b", "c"]` & `ref_args = [&x, &y]`).
46    has_unknown_fmt_placeholder_to_args_mapping: Option<usize>,
47}
48
49pub struct FormatArgsNotRecognized(pub String);
50
51pub type FormatArgsResult<'tcx> = Result<DecodedFormatArgs<'tcx>, FormatArgsNotRecognized>;
52
53impl<'tcx> DecodedFormatArgs<'tcx> {
54    pub fn try_decode_and_remove_format_args<'a>(
55        builder: &mut Builder<'a, 'tcx>,
56        args: &[SpirvValue],
57    ) -> FormatArgsResult<'tcx> {
58        let mut decoded_format_args = DecodedFormatArgs::default();
59
60        // HACK(eddyb) work around mutable borrowing conflicts.
61        let cx = builder.cx;
62
63        let const_u32_as_usize = |ct_id| match cx.builder.lookup_const_by_id(ct_id)? {
64            SpirvConst::Scalar(x) => Some(u32::try_from(x).ok()? as usize),
65            _ => None,
66        };
67        let const_slice_as_elem_ids = |ptr_id: Word, len: usize| {
68            if let SpirvConst::PtrTo { pointee } = cx.builder.lookup_const_by_id(ptr_id)?
69                && let SpirvConst::Composite(elems) = cx.builder.lookup_const_by_id(pointee)?
70                && elems.len() == len
71            {
72                return Some(elems);
73            }
74            None
75        };
76        let const_str_as_utf8 = |&[str_ptr_id, str_len_id]: &[Word; 2]| {
77            let str_len = const_u32_as_usize(str_len_id)?;
78            let piece_str_bytes = const_slice_as_elem_ids(str_ptr_id, str_len)?
79                .iter()
80                .map(|&id| u8::try_from(const_u32_as_usize(id)?).ok())
81                .collect::<Option<Vec<u8>>>()?;
82            String::from_utf8(piece_str_bytes).ok()
83        };
84
85        // HACK(eddyb) `panic_explicit` doesn't take any regular arguments,
86        // only an (implicit) `&'static panic::Location<'static>`.
87        if args.len() == 1 {
88            decoded_format_args.const_pieces =
89                Some(["explicit panic".into()].into_iter().collect());
90            return Ok(decoded_format_args);
91        }
92
93        // HACK(eddyb) some entry-points only take a `&str`, not `fmt::Arguments`.
94        if let [
95            SpirvValue {
96                kind: SpirvValueKind::Def(a_id),
97                ..
98            },
99            SpirvValue {
100                kind: SpirvValueKind::Def(b_id),
101                ..
102            },
103            ref other_args @ ..,
104        ] = args[..]
105        {
106            // Optional `&'static panic::Location<'static>`.
107            if other_args.len() <= 1
108                && let Some(const_msg) = const_str_as_utf8(&[a_id, b_id])
109            {
110                decoded_format_args.const_pieces = Some([const_msg].into_iter().collect());
111                return Ok(decoded_format_args);
112            }
113        }
114
115        let format_args_id = match *args {
116            // HACK(eddyb) `panic_nounwind_fmt` takes an extra argument.
117            [
118                SpirvValue {
119                    kind: SpirvValueKind::Def(format_args_id),
120                    ..
121                },
122                _, // `&'static panic::Location<'static>`
123            ]
124            | [
125                SpirvValue {
126                    kind: SpirvValueKind::Def(format_args_id),
127                    ..
128                },
129                _, // `force_no_backtrace: bool`
130                _, // `&'static panic::Location<'static>`
131            ] => format_args_id,
132
133            _ => {
134                return Err(FormatArgsNotRecognized(
135                    "panic entry-point call args".into(),
136                ));
137            }
138        };
139
140        let custom_ext_inst_set_import = builder.ext_inst.borrow_mut().import_custom(builder);
141
142        // HACK(eddyb) we can remove SSA instructions even when they have
143        // side-effects, *as long as* they are "local" enough and cannot
144        // be observed from outside this current invocation - because the
145        // the abort, any SSA definitions or local variable writes can't
146        // be actually used anywhere else (other than *before* the abort).
147        let mut builder = builder.emit();
148        let func_idx = builder.selected_function().unwrap();
149        let block_idx = builder.selected_block().unwrap();
150        let func = &mut builder.module_mut().functions[func_idx];
151
152        // HACK(eddyb) this is used to check that all `Op{Store,Load}`s
153        // that may get removed, operate on local `OpVariable`s,
154        // i.e. are not externally observable.
155        let local_var_ids: FxHashSet<_> = func.blocks[0]
156            .instructions
157            .iter()
158            .take_while(|inst| inst.class.opcode == Op::Variable)
159            .map(|inst| inst.result_id.unwrap())
160            .collect();
161        let require_local_var = |ptr_id, var| {
162            Some(())
163                .filter(|()| local_var_ids.contains(&ptr_id))
164                .ok_or_else(|| FormatArgsNotRecognized(format!("{var} storage not local")))
165        };
166
167        let mut non_debug_insts = func.blocks[block_idx]
168            .instructions
169            .iter()
170            .enumerate()
171            .filter(|(_, inst)| {
172                let is_standard_debug = [Op::Line, Op::NoLine].contains(&inst.class.opcode);
173                let is_custom_debug = inst.class.opcode == Op::ExtInst
174                    && inst.operands[0].unwrap_id_ref() == custom_ext_inst_set_import
175                    && CustomOp::decode_from_ext_inst(inst).is_debuginfo();
176                !(is_standard_debug || is_custom_debug)
177            });
178
179        // HACK(eddyb) to aid in pattern-matching, relevant instructions
180        // are decoded to values of this `enum`. For instructions that
181        // produce results, the result ID is the first `ID` value.
182        #[derive(Debug)]
183        enum Inst<ID> {
184            Bitcast(ID, ID),
185            CompositeExtract(ID, ID, u32),
186            InBoundsAccessChain(ID, ID, u32),
187            InBoundsAccessChain2(ID, ID, u32, u32),
188            Store(ID, ID),
189            Load(ID, ID),
190            CopyMemory(ID, ID),
191            Call(ID, ID, SmallVec<[ID; 4]>),
192
193            // HACK(eddyb) this only exists for better error reporting,
194            // as `Result<Inst<...>, Op>` would only report one `Op`.
195            Unsupported(
196                // HACK(eddyb) only exists for `fmt::Debug` in case of error.
197                #[allow(dead_code)] Op,
198            ),
199        }
200
201        let taken_inst_idx_range = Cell::new(func.blocks[block_idx].instructions.len())..;
202
203        // Take `count` instructions, advancing backwards, but returning
204        // instructions in their original order (and decoded to `Inst`s).
205        let mut try_rev_take = |count: isize| {
206            // HACK(eddyb) this is extremely silly but it's easier to do
207            // this than to rely on `Iterator::peekable` or anything else,
208            // lower down this file, without messing up the state here.
209            let is_peek = count < 0;
210            let count = count.unsigned_abs();
211
212            let mut non_debug_insts_for_peek = is_peek.then(|| non_debug_insts.clone());
213            let non_debug_insts = non_debug_insts_for_peek
214                .as_mut()
215                .unwrap_or(&mut non_debug_insts);
216
217            // FIXME(eddyb) there might be an easier way to do this,
218            // e.g. maybe `map_while` + post-`collect` length check?
219            let maybe_rev_insts = (0..count).map(|_| {
220                let (i, inst) = non_debug_insts.next_back()?;
221                if !is_peek {
222                    taken_inst_idx_range.start.set(i);
223                }
224
225                // HACK(eddyb) avoid the logic below that assumes only ID operands
226                if inst.class.opcode == Op::CompositeExtract
227                    && let (Some(r), &[Operand::IdRef(x), Operand::LiteralBit32(i)]) =
228                        (inst.result_id, &inst.operands[..])
229                {
230                    return Some(Inst::CompositeExtract(r, x, i));
231                }
232
233                // HACK(eddyb) all instructions accepted below
234                // are expected to take no more than 4 operands,
235                // and this is easier to use than an iterator.
236                let id_operands = inst
237                    .operands
238                    .iter()
239                    .map(|operand| operand.id_ref_any())
240                    .collect::<Option<SmallVec<[_; 4]>>>()?;
241
242                let const_as_u32 = |id| match cx.builder.lookup_const_by_id(id)? {
243                    SpirvConst::Scalar(x) => u32::try_from(x).ok(),
244                    _ => None,
245                };
246
247                // Decode the instruction into one of our `Inst`s.
248                Some(
249                    match (inst.class.opcode, inst.result_id, &id_operands[..]) {
250                        (Op::Bitcast, Some(r), &[x]) => Inst::Bitcast(r, x),
251                        (Op::InBoundsAccessChain, Some(r), &[p, i]) => {
252                            if let Some(i) = const_as_u32(i) {
253                                Inst::InBoundsAccessChain(r, p, i)
254                            } else {
255                                Inst::Unsupported(inst.class.opcode)
256                            }
257                        }
258                        (Op::InBoundsAccessChain, Some(r), &[p, i, j]) => {
259                            if let [Some(i), Some(j)] = [i, j].map(const_as_u32) {
260                                Inst::InBoundsAccessChain2(r, p, i, j)
261                            } else {
262                                Inst::Unsupported(inst.class.opcode)
263                            }
264                        }
265                        (Op::Store, None, &[p, v]) => Inst::Store(p, v),
266                        (Op::Load, Some(r), &[p]) => Inst::Load(r, p),
267                        (Op::CopyMemory, None, &[a, b]) => Inst::CopyMemory(a, b),
268                        (Op::FunctionCall, Some(r), [f, args @ ..]) => {
269                            Inst::Call(r, *f, args.iter().copied().collect())
270                        }
271                        _ => Inst::Unsupported(inst.class.opcode),
272                    },
273                )
274            });
275            let mut insts = maybe_rev_insts.collect::<Option<SmallVec<[_; 4]>>>()?;
276            insts.reverse();
277            Some(insts)
278        };
279        let fmt_args_new_call_insts = try_rev_take(3).ok_or_else(|| {
280            FormatArgsNotRecognized("fmt::Arguments::new call: ran out of instructions".into())
281        })?;
282        let ((pieces_slice_ptr_id, pieces_len), (rt_args_slice_ptr_id, rt_args_count)) =
283            match fmt_args_new_call_insts[..] {
284                [
285                    Inst::Call(call_ret_id, callee_id, ref call_args),
286                    Inst::Store(st_dst_id, st_val_id),
287                    Inst::Load(ld_val_id, ld_src_id),
288                ] if call_ret_id == st_val_id
289                    && st_dst_id == ld_src_id
290                    && ld_val_id == format_args_id =>
291                {
292                    require_local_var(st_dst_id, "fmt::Arguments::new destination")?;
293
294                    let Some(&(pieces_len, rt_args_count)) =
295                        cx.fmt_args_new_fn_ids.borrow().get(&callee_id)
296                    else {
297                        return Err(FormatArgsNotRecognized(
298                            "fmt::Arguments::new callee not registered".into(),
299                        ));
300                    };
301
302                    match call_args[..] {
303                        // `<core::fmt::Arguments>::new_v1_formatted`
304                        //
305                        // HACK(eddyb) this isn't fully supported,
306                        // as that would require digging into unstable
307                        // internals of `core::fmt::rt::Placeholder`s,
308                        // but the whole call still needs to be removed,
309                        // and both const str pieces and runtime args
310                        // can still be printed (even if in jankier way).
311                        [
312                            pieces_slice_ptr_id,
313                            pieces_len_id,
314                            rt_args_slice_ptr_id,
315                            rt_args_len_id,
316                            fmt_placeholders_slice_ptr_id,
317                            fmt_placeholders_len_id,
318                        ] if (pieces_len, rt_args_count) == (!0, !0) => {
319                            let [pieces_len, rt_args_len, fmt_placeholders_len] =
320                                match [pieces_len_id, rt_args_len_id, fmt_placeholders_len_id]
321                                    .map(const_u32_as_usize)
322                                {
323                                    [Some(a), Some(b), Some(c)] => [a, b, c],
324                                    _ => {
325                                        return Err(FormatArgsNotRecognized(
326                                            "fmt::Arguments::new_v1_formatted \
327                                                     with dynamic lengths"
328                                                .into(),
329                                        ));
330                                    }
331                                };
332
333                            let prepare_args_insts = try_rev_take(2).ok_or_else(|| {
334                                FormatArgsNotRecognized(
335										"fmt::Arguments::new_v1_formatted call: ran out of instructions".into(),
336									)
337                            })?;
338                            let (rt_args_slice_ptr_id, _fmt_placeholders_slice_ptr_id) =
339                                match prepare_args_insts[..] {
340                                    [
341                                        Inst::Bitcast(rt_args_cast_out_id, rt_args_cast_in_id),
342                                        Inst::Bitcast(
343                                            placeholders_cast_out_id,
344                                            placeholders_cast_in_id,
345                                        ),
346                                    ] if rt_args_cast_out_id == rt_args_slice_ptr_id
347                                        && placeholders_cast_out_id
348                                            == fmt_placeholders_slice_ptr_id =>
349                                    {
350                                        (rt_args_cast_in_id, placeholders_cast_in_id)
351                                    }
352                                    _ => {
353                                        let mut insts = prepare_args_insts;
354                                        insts.extend(fmt_args_new_call_insts);
355                                        return Err(FormatArgsNotRecognized(format!(
356                                            "fmt::Arguments::new_v1_formatted call sequence ({insts:?})",
357                                        )));
358                                    }
359                                };
360
361                            decoded_format_args.has_unknown_fmt_placeholder_to_args_mapping =
362                                Some(fmt_placeholders_len);
363
364                            (
365                                (pieces_slice_ptr_id, pieces_len),
366                                (Some(rt_args_slice_ptr_id), rt_args_len),
367                            )
368                        }
369
370                        // `<core::fmt::Arguments>::new_v1`
371                        [pieces_slice_ptr_id, rt_args_slice_ptr_id] => (
372                            (pieces_slice_ptr_id, pieces_len),
373                            (Some(rt_args_slice_ptr_id), rt_args_count),
374                        ),
375
376                        // `<core::fmt::Arguments>::new_const`
377                        [pieces_slice_ptr_id] if rt_args_count == 0 => {
378                            ((pieces_slice_ptr_id, pieces_len), (None, rt_args_count))
379                        }
380
381                        _ => {
382                            return Err(FormatArgsNotRecognized(
383                                "fmt::Arguments::new call args".into(),
384                            ));
385                        }
386                    }
387                }
388                _ => {
389                    // HACK(eddyb) this gathers more context before reporting.
390                    let mut insts = fmt_args_new_call_insts;
391                    insts.reverse();
392                    while let Some(extra_inst) = try_rev_take(1) {
393                        insts.extend(extra_inst);
394                        if insts.len() >= 32 {
395                            break;
396                        }
397                    }
398                    insts.reverse();
399
400                    return Err(FormatArgsNotRecognized(format!(
401                        "fmt::Arguments::new call sequence ({insts:?})",
402                    )));
403                }
404            };
405
406        // HACK(eddyb) this is the worst part: if we do have runtime
407        // arguments (from e.g. new `assert!`s being added to `core`),
408        // we have to confirm their many instructions for removal.
409        if rt_args_count > 0 {
410            let rt_args_array_ptr_id = rt_args_slice_ptr_id.unwrap();
411
412            // Each runtime argument has A instructions to call one of
413            // the `fmt::rt::Argument::new_*` functions (and temporarily
414            // store its result), and B instructions to copy it into
415            // the appropriate slot in the array. The groups of A and B
416            // instructions, for all runtime args, are each separate,
417            // so the BÃ—N later instructions are all processed first,
418            // before moving (backwards) to the AÃ—N earlier instructions.
419
420            let rev_copies_to_rt_args_array_src_ptrs: SmallVec<[_; 4]> = (0..rt_args_count)
421                .rev()
422                .map(|rt_arg_idx| {
423                    let mut copy_to_rt_args_array_insts = try_rev_take(3).ok_or_else(|| {
424                        FormatArgsNotRecognized(
425                            "[fmt::rt::Argument; N] copy: ran out of instructions".into(),
426                        )
427                    })?;
428
429                    // HACK(eddyb) account for both the split and combined
430                    // access chain cases that `inbounds_gep` can now cause.
431                    if let Inst::InBoundsAccessChain(dst_field_ptr, dst_base_ptr, 0) =
432                        copy_to_rt_args_array_insts[0]
433                        && let Some(mut prev_insts) = try_rev_take(1)
434                    {
435                        assert_eq!(prev_insts.len(), 1);
436                        let prev_inst = prev_insts.pop().unwrap();
437
438                        match prev_inst {
439                            Inst::InBoundsAccessChain(array_elem_ptr, array_ptr, idx)
440                                if dst_base_ptr == array_elem_ptr =>
441                            {
442                                copy_to_rt_args_array_insts[0] =
443                                    Inst::InBoundsAccessChain2(dst_field_ptr, array_ptr, idx, 0);
444                            }
445                            _ => {
446                                // HACK(eddyb) don't lose the taken `prev_inst`.
447                                copy_to_rt_args_array_insts.insert(0, prev_inst);
448                            }
449                        }
450                    }
451
452                    match copy_to_rt_args_array_insts[..] {
453                        [
454                            Inst::InBoundsAccessChain2(
455                                dst_field_ptr,
456                                dst_array_base_ptr,
457                                array_idx,
458                                0,
459                            ),
460                            Inst::InBoundsAccessChain(src_field_ptr, src_base_ptr, 0),
461                            Inst::CopyMemory(copy_dst, copy_src),
462                        ] if dst_array_base_ptr == rt_args_array_ptr_id
463                            && array_idx as usize == rt_arg_idx
464                            && (copy_dst, copy_src) == (dst_field_ptr, src_field_ptr) =>
465                        {
466                            Ok(src_base_ptr)
467                        }
468                        _ => Err(FormatArgsNotRecognized(format!(
469                            "[fmt::rt::Argument; N] copy sequence ({copy_to_rt_args_array_insts:?})"
470                        ))),
471                    }
472                })
473                .collect::<Result<_, _>>()?;
474
475            // HACK(eddyb) sometimes there is an extra tuple of refs,
476            // nowadays, but MIR opts mean it's not always guaranteed,
477            // hopefully it's always uniform across all the arguments.
478            let mut maybe_ref_args_tmp_slot_ptr = None;
479
480            let rev_maybe_ref_arg_ids_with_ty_and_spec = ((0..rt_args_count)
481                .rev()
482                .zip_eq(rev_copies_to_rt_args_array_src_ptrs))
483            .map(|(rt_arg_idx, copy_to_rt_args_array_src_ptr)| {
484                let rt_arg_new_call_insts = try_rev_take(4).ok_or_else(|| {
485                    FormatArgsNotRecognized(
486                        "fmt::rt::Argument::new call: ran out of instructions".into(),
487                    )
488                })?;
489                let (ref_arg_id, ty, spec) = match rt_arg_new_call_insts[..] {
490                    [
491                        Inst::Call(call_ret_id, callee_id, ref call_args),
492                        Inst::InBoundsAccessChain(tmp_slot_field_ptr, tmp_slot_ptr, 0),
493                        Inst::CompositeExtract(field, wrapper_newtype, 0),
494                        Inst::Store(st_dst_ptr, st_val),
495                    ] if wrapper_newtype == call_ret_id
496                        && tmp_slot_ptr == copy_to_rt_args_array_src_ptr
497                        && (st_dst_ptr, st_val) == (tmp_slot_field_ptr, field) =>
498                    {
499                        cx.fmt_rt_arg_new_fn_ids_to_ty_and_spec
500                            .borrow()
501                            .get(&callee_id)
502                            .and_then(|&(ty, spec)| match call_args[..] {
503                                [x] => Some((x, ty, spec)),
504                                _ => None,
505                            })
506                    }
507                    _ => None,
508                }
509                .ok_or_else(|| {
510                    FormatArgsNotRecognized(format!(
511                        "fmt::rt::Argument::new call sequence ({rt_arg_new_call_insts:?})"
512                    ))
513                })?;
514
515                // HACK(eddyb) `0` (an invalid ID) is later used as a
516                // placeholder (see also `maybe_ref_args_tmp_slot_ptr`).
517                assert_ne!(ref_arg_id, 0);
518
519                // HACK(eddyb) `try_rev_take(-2)` is "peeking", not taking.
520                let maybe_ref_args_tuple_load_insts = try_rev_take(-2);
521                let maybe_ref_arg_id = match maybe_ref_args_tuple_load_insts.as_deref() {
522                    Some(
523                        &[
524                            Inst::InBoundsAccessChain(field_ptr, base_ptr, field_idx),
525                            Inst::Load(ld_val, ld_src_ptr),
526                        ],
527                    ) if maybe_ref_args_tmp_slot_ptr
528                        .is_none_or(|expected| base_ptr == expected)
529                        && field_idx as usize == rt_arg_idx
530                        && (ld_val, ld_src_ptr) == (ref_arg_id, field_ptr) =>
531                    {
532                        // HACK(eddyb) consume the peeked instructions.
533                        try_rev_take(2).unwrap();
534
535                        maybe_ref_args_tmp_slot_ptr = Some(base_ptr);
536
537                        // HACK(eddyb) using `0` (an invalid ID) as a
538                        // placeholder to require further processing.
539                        0
540                    }
541                    _ => ref_arg_id,
542                };
543
544                Ok((maybe_ref_arg_id, ty, spec))
545            })
546            .collect::<Result<_, _>>()?;
547
548            decoded_format_args.ref_arg_ids_with_ty_and_spec =
549                rev_maybe_ref_arg_ids_with_ty_and_spec;
550            decoded_format_args.ref_arg_ids_with_ty_and_spec.reverse();
551
552            // HACK(eddyb) see above for context regarding the use of
553            // `0` as placeholders and `maybe_ref_args_tmp_slot_ptr`.
554            if let Some(ref_args_tmp_slot_ptr) = maybe_ref_args_tmp_slot_ptr {
555                for (rt_arg_idx, (maybe_ref_arg_id, ..)) in decoded_format_args
556                    .ref_arg_ids_with_ty_and_spec
557                    .iter_mut()
558                    .enumerate()
559                    .rev()
560                {
561                    if *maybe_ref_arg_id == 0 {
562                        let ref_arg_store_insts = try_rev_take(2).ok_or_else(|| {
563                            FormatArgsNotRecognized(
564                                "fmt::rt::Argument::new argument store: ran out of instructions"
565                                    .into(),
566                            )
567                        })?;
568
569                        *maybe_ref_arg_id = match ref_arg_store_insts[..] {
570							[
571							Inst::InBoundsAccessChain(field_ptr, base_ptr, field_idx),
572							Inst::Store(st_dst_ptr, st_val),
573							] if base_ptr == ref_args_tmp_slot_ptr
574								&& field_idx as usize == rt_arg_idx
575								&& st_dst_ptr == field_ptr =>
576								{
577									Some(st_val)
578								}
579							_ => None,
580						}
581							.ok_or_else(|| {
582								FormatArgsNotRecognized(format!(
583									"fmt::rt::Argument::new argument store sequence ({ref_arg_store_insts:?})"
584								))
585							})?;
586                    }
587                }
588            }
589        }
590
591        // If the `pieces: &[&str]` slice needs a bitcast, it'll be here.
592        // HACK(eddyb) `try_rev_take(-1)` is "peeking", not taking.
593        let pieces_slice_ptr_id = match try_rev_take(-1).as_deref() {
594            Some(&[Inst::Bitcast(out_id, in_id)]) if out_id == pieces_slice_ptr_id => {
595                // HACK(eddyb) consume the peeked instructions.
596                try_rev_take(1).unwrap();
597
598                in_id
599            }
600            _ => pieces_slice_ptr_id,
601        };
602        decoded_format_args.const_pieces =
603            match const_slice_as_elem_ids(pieces_slice_ptr_id, pieces_len) {
604                Some(piece_ids) => piece_ids
605                    .iter()
606                    .map(|&piece_id| match cx.builder.lookup_const_by_id(piece_id)? {
607                        SpirvConst::Composite(piece) => const_str_as_utf8(piece.try_into().ok()?),
608                        _ => None,
609                    })
610                    .collect::<Option<_>>(),
611                // HACK(eddyb) minor upstream blunder results in at
612                // least one instance of a runtime `[&str; 1]` array,
613                // see also this comment left on the responsible PR:
614                // https://github.com/rust-lang/rust/pull/129658#discussion_r2181834781
615                // HACK(eddyb) `try_rev_take(-4)` is "peeking", not taking.
616                None if pieces_len == 1 => match try_rev_take(-4).as_deref() {
617                    Some(
618                        &[
619                            Inst::InBoundsAccessChain2(field0_ptr, array_ptr_0, 0, 0),
620                            Inst::Store(st0_dst_ptr, st0_val),
621                            Inst::InBoundsAccessChain2(field1_ptr, array_ptr_1, 0, 1),
622                            Inst::Store(st1_dst_ptr, st1_val),
623                        ],
624                    ) if [array_ptr_0, array_ptr_1] == [pieces_slice_ptr_id; 2]
625                        && st0_dst_ptr == field0_ptr
626                        && st1_dst_ptr == field1_ptr =>
627                    {
628                        // HACK(eddyb) consume the peeked instructions.
629                        try_rev_take(4).unwrap();
630
631                        const_str_as_utf8(&[st0_val, st1_val]).map(|s| [s].into_iter().collect())
632                    }
633                    _ => None,
634                },
635                None => None,
636            };
637
638        // Keep all instructions up to (but not including) the last one
639        // confirmed above to be the first instruction of `format_args!`.
640        func.blocks[block_idx]
641            .instructions
642            .truncate(taken_inst_idx_range.start.get());
643
644        Ok(decoded_format_args)
645    }
646}
647
648pub trait CodegenPanic<'a, 'tcx> {
649    fn codegen_panic(&self, builder: &mut Builder<'a, 'tcx>, result_type: Word) -> SpirvValue;
650}
651
652impl<'a, 'tcx> CodegenPanic<'a, 'tcx> for DecodedFormatArgs<'tcx> {
653    fn codegen_panic(&self, builder: &mut Builder<'a, 'tcx>, result_type: Word) -> SpirvValue {
654        match self {
655            DecodedFormatArgs {
656                const_pieces: None, ..
657            } => {
658                builder.abort_with_kind_and_message_debug_printf(
659                    "panic",
660                    "<unknown message>",
661                    std::iter::empty(),
662                );
663                builder.undef(result_type)
664            }
665
666            DecodedFormatArgs {
667                const_pieces: Some(const_pieces),
668                ref_arg_ids_with_ty_and_spec,
669                has_unknown_fmt_placeholder_to_args_mapping,
670            } => {
671                let mut debug_printf_args = SmallVec::<[_; 2]>::new();
672                let args = ref_arg_ids_with_ty_and_spec
673                    .iter()
674                    .map(|&(ref_id, ty, spec)| {
675                        use rustc_abi::{Float::*, Integer::*, Primitive::*};
676
677                        let layout = builder.layout_of(ty);
678
679                        let scalar = match layout.backend_repr {
680                            BackendRepr::Scalar(scalar) => Some(scalar.primitive()),
681                            _ => None,
682                        };
683                        let debug_printf_fmt = match (spec, scalar) {
684                            // FIXME(eddyb) support more of these,
685                            // potentially recursing to print ADTs.
686                            (' ' | '?', Some(Int(I32, false))) => "%u",
687                            ('x', Some(Int(I32, false))) => "%x",
688                            (' ' | '?', Some(Int(I32, true))) => "%i",
689                            (' ' | '?', Some(Float(F32))) => "%f",
690
691                            _ => "",
692                        };
693
694                        if debug_printf_fmt.is_empty() {
695                            return Cow::Owned(
696                                format!("{{/* unprintable {ty} */:{spec}}}").replace('%', "%%"),
697                            );
698                        }
699
700                        let spirv_type = layout.spirv_type(builder.span(), builder);
701                        debug_printf_args.push(
702                            builder
703                                .emit()
704                                .load(spirv_type, None, ref_id, None, [])
705                                .unwrap()
706                                .with_type(spirv_type),
707                        );
708                        Cow::Borrowed(debug_printf_fmt)
709                    });
710
711                // HACK(eddyb) due to `fmt::Arguments::new_v1_formatted`,
712                // we can't always assume that all the formatting arguments
713                // are used 1:1 as placeholders (i.e. between `const_pieces`).
714                let (placeholder_count, placeholders_are_args) =
715                    match has_unknown_fmt_placeholder_to_args_mapping {
716                        Some(count) => (*count, false),
717                        None => (args.len(), true),
718                    };
719
720                // HACK(eddyb) extra sanity check to avoid visual mishaps.
721                let valid_placeholder_count = placeholder_count
722                    .clamp(const_pieces.len().saturating_sub(1), const_pieces.len());
723                let placeholders_are_args =
724                    placeholders_are_args && placeholder_count == valid_placeholder_count;
725
726                // FIXME(eddyb) stop using `itertools`'s `intersperse`,
727                // when it gets stabilized on `Iterator` instead.
728                #[allow(unstable_name_collisions)]
729                let (placeholders, suffix) = if placeholders_are_args {
730                    (Either::Left(args), None)
731                } else {
732                    // See also `has_unknown_fmt_placeholder_to_args_mapping`
733                    // comment (which has an example for 3 pieces and 2 args).
734                    //
735                    // FIXME(eddyb) this could definitely be improved, but
736                    // so far this only really gets hit in esoteric `core`
737                    // internals (UB checks and `char::encode_utf{8,16}`).
738                    (
739                        Either::Right(
740                            (0..valid_placeholder_count).map(|i| format!("{{{i}}}").into()),
741                        ),
742                        Some(
743                            ["\n  with {â€¦} from: ".into()]
744                                .into_iter()
745                                .chain(args.intersperse(", ".into())),
746                        ),
747                    )
748                };
749
750                let message = const_pieces
751                    .into_iter()
752                    .map(|s| Cow::Owned(s.replace('%', "%%")))
753                    .interleave(placeholders)
754                    .chain(suffix.into_iter().flatten())
755                    .collect::<String>();
756
757                // HACK(eddyb) redirect any possible panic call to an abort, to avoid
758                // needing to materialize `&core::panic::Location` or `format_args!`.
759                builder.abort_with_kind_and_message_debug_printf(
760                    "panic",
761                    message,
762                    debug_printf_args,
763                );
764                builder.undef(result_type)
765            }
766        }
767    }
768}

770impl<'a, 'tcx> CodegenPanic<'a, 'tcx> for FormatArgsResult<'tcx> {
771    fn codegen_panic(&self, builder: &mut Builder<'a, 'tcx>, result_type: Word) -> SpirvValue {
772        match self {
773            Ok(e) => e.codegen_panic(builder, result_type),
774            Err(FormatArgsNotRecognized(step)) => {
775                if let Some(current_span) = builder.current_span {
776                    // HACK(eddyb) Cargo silences warnings in dependencies.
777                    let force_warn = |span, msg| -> rustc_errors::Diag<'_, ()> {
778                        rustc_errors::Diag::new(
779                            builder.tcx.dcx(),
780                            rustc_errors::Level::ForceWarning,
781                            msg,
782                        )
783                        .with_span(span)
784                    };
785                    let mut warn = force_warn(
786                        current_span,
787                        "failed to find and remove `format_args!` construction for this `panic!`",
788                    );
789
790                    warn.note(
791                        "compilation may later fail due to leftover `format_args!` internals",
792                    );
793
794                    if builder.tcx.sess.opts.unstable_opts.inline_mir != Some(false) {
795                        warn.note("missing `-Zinline-mir=off` flag (should've been set by `spirv-builder`)")
796                            .help("check `.cargo` and environment variables for potential overrides")
797                            .help("(or, if not using `spirv-builder` at all, add the flag manually)");
798                    } else {
799                        warn.note(format!("[RUST-GPU BUG] bailed from {step}"));
800                    }
801
802                    warn.emit();
803                }
804
805                let msg = "<unknown message> (failed to find/decode `format_args!` expansion)";
806                builder.abort_with_kind_and_message_debug_printf("panic", msg, std::iter::empty());
807                builder.undef(result_type)
808            }
809        }
810    }
811}
