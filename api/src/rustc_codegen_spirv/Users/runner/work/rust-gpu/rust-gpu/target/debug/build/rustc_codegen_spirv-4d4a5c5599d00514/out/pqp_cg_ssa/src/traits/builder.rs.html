256    /// Emits an `assume` that the integer value `imm` of type `ty` is contained in `range`.
257    ///
258    /// This *always* emits the assumption, so you probably want to check the
259    /// optimization level and `Scalar::is_always_valid` before calling it.
260    fn assume_integer_range(&mut self, imm: Self::Value, ty: Self::Type, range: WrappingRange) {
261        let WrappingRange { start, end } = range;
262
263        // Perhaps one day we'll be able to use assume operand bundles for this,
264        // but for now this encoding with a single icmp+assume is best per
265        // <https://github.com/llvm/llvm-project/issues/123278#issuecomment-2597440158>
266        let shifted = if start == 0 {
267            imm
268        } else {
269            let low = self.const_uint_big(ty, start);
270            self.sub(imm, low)
271        };
272        let width = self.const_uint_big(ty, u128::wrapping_sub(end, start));
273        let cmp = self.icmp(IntPredicate::IntULE, shifted, width);
274        self.assume(cmp);
275    }

277    /// Emits an `assume` that the `val` of pointer type is non-null.
278    ///
279    /// You may want to check the optimization level before bothering calling this.
280    fn assume_nonnull(&mut self, val: Self::Value) {
281        // Arguably in LLVM it'd be better to emit an assume operand bundle instead
282        // <https://llvm.org/docs/LangRef.html#assume-operand-bundles>
283        // but this works fine for all backends.
284
285        let null = self.const_null(self.type_ptr());
286        let is_null = self.icmp(IntPredicate::IntNE, val, null);
287        self.assume(is_null);

371    fn cast_float_to_int(
372        &mut self,
373        signed: bool,
374        x: Self::Value,
375        dest_ty: Self::Type,
376    ) -> Self::Value {
377        let in_ty = self.cx().val_ty(x);
378        let (float_ty, int_ty) = if self.cx().type_kind(dest_ty) == TypeKind::Vector
379            && self.cx().type_kind(in_ty) == TypeKind::Vector
380        {
381            (self.cx().element_type(in_ty), self.cx().element_type(dest_ty))
382        } else {
383            (in_ty, dest_ty)
384        };
385        assert_matches!(
386            self.cx().type_kind(float_ty),
387            TypeKind::Half | TypeKind::Float | TypeKind::Double | TypeKind::FP128
388        );
389        assert_eq!(self.cx().type_kind(int_ty), TypeKind::Integer);
390
391        if let Some(false) = self.cx().sess().opts.unstable_opts.saturating_float_casts {
392            return if signed { self.fptosi(x, dest_ty) } else { self.fptoui(x, dest_ty) };
393        }
394
395        if signed { self.fptosi_sat(x, dest_ty) } else { self.fptoui_sat(x, dest_ty) }
396    }

440    /// *Typed* copy for non-overlapping places.

442    /// Has a default implementation in terms of `memcpy`, but specific backends

444    ///
445    /// (For example, typed load-stores with alias metadata.)

447        &mut self,
448        dst: PlaceValue<Self::Value>,
449        src: PlaceValue<Self::Value>,
450        layout: TyAndLayout<'tcx>,
451    ) {
452        self.typed_place_copy_with_flags(dst, src, layout, MemFlags::empty());
453    }

455    fn typed_place_copy_with_flags(
456        &mut self,
457        dst: PlaceValue<Self::Value>,
458        src: PlaceValue<Self::Value>,
459        layout: TyAndLayout<'tcx>,
460        flags: MemFlags,
461    ) {
462        assert!(layout.is_sized(), "cannot typed-copy an unsigned type");
463        assert!(src.llextra.is_none(), "cannot directly copy from unsized values");
464        assert!(dst.llextra.is_none(), "cannot directly copy into unsized values");
465        if flags.contains(MemFlags::NONTEMPORAL) {
466            // HACK(nox): This is inefficient but there is no nontemporal memcpy.
467            let ty = self.backend_type(layout);
468            let val = self.load_from_place(ty, src);
469            self.store_to_place_with_flags(val, dst, flags);
470        } else if self.sess().opts.optimize == OptLevel::No && self.is_backend_immediate(layout) {
471            // If we're not optimizing, the aliasing information from `memcpy`
472            // isn't useful, so just load-store the value for smaller code.
473            let temp = self.load_operand(src.with_type(layout));
474            temp.val.store_with_flags(self, dst.with_type(layout), flags);
475        } else if !layout.is_zst() {
476            let bytes = self.const_usize(layout.size.bytes());
477            self.memcpy(dst.llval, dst.align, src.llval, src.align, bytes, flags);
478        }
479    }

481    /// *Typed* swap for non-overlapping places.
482    ///
483    /// Avoids `alloca`s for Immediates and ScalarPairs.
484    ///
485    /// FIXME: Maybe do something smarter for Ref types too?
486    /// For now, the `typed_swap_nonoverlapping` intrinsic just doesn't call this for those
487    /// cases (in non-debug), preferring the fallback body instead.
488    fn typed_place_swap(
489        &mut self,
490        left: PlaceValue<Self::Value>,
491        right: PlaceValue<Self::Value>,
492        layout: TyAndLayout<'tcx>,
493    ) {
494        let mut temp = self.load_operand(left.with_type(layout));
495        if let OperandValue::Ref(..) = temp.val {
496            // The SSA value isn't stand-alone, so we need to copy it elsewhere
497            let alloca = PlaceRef::alloca(self, layout);
498            self.typed_place_copy(alloca.val, left, layout);
499            temp = self.load_operand(alloca);
500        }
501        self.typed_place_copy(left, right, layout);
502        temp.val.store(self, right.with_type(layout));
503    }
