43fn load_metadata_with(
44    path: &Path,
45    f: impl for<'a> FnOnce(&'a [u8]) -> Result<&'a [u8], String>,
46) -> Result<OwnedSlice, String> {
47    let file =
48        File::open(path).map_err(|e| format!("failed to open file '{}': {}", path.display(), e))?;
49
50    unsafe { Mmap::map(file) }
51        .map_err(|e| format!("failed to mmap file '{}': {}", path.display(), e))
52        .and_then(|mmap| try_slice_owned(mmap, |mmap| f(mmap)))
53}

55impl MetadataLoader for DefaultMetadataLoader {
56    fn get_rlib_metadata(&self, target: &Target, path: &Path) -> Result<OwnedSlice, String> {
57        debug!("getting rlib metadata for {}", path.display());
58        load_metadata_with(path, |data| {
59            let archive = object::read::archive::ArchiveFile::parse(&*data)
60                .map_err(|e| format!("failed to parse rlib '{}': {}", path.display(), e))?;
61
62            for entry_result in archive.members() {
63                let entry = entry_result
64                    .map_err(|e| format!("failed to parse rlib '{}': {}", path.display(), e))?;
65                if entry.name() == METADATA_FILENAME.as_bytes() {
66                    let data = entry
67                        .data(data)
68                        .map_err(|e| format!("failed to parse rlib '{}': {}", path.display(), e))?;
69                    if target.is_like_aix {
70                        return get_metadata_xcoff(path, data);
71                    } else {
72                        return search_for_section(path, data, ".rmeta");
73                    }
74                }
75            }
76
77            Err(format!("metadata not found in rlib '{}'", path.display()))
78        })
79    }
80
81    fn get_dylib_metadata(&self, target: &Target, path: &Path) -> Result<OwnedSlice, String> {
82        debug!("getting dylib metadata for {}", path.display());
83        if target.is_like_aix {
84            load_metadata_with(path, |data| {
85                let archive = object::read::archive::ArchiveFile::parse(&*data).map_err(|e| {
86                    format!("failed to parse aix dylib '{}': {}", path.display(), e)
87                })?;
88
89                match archive.members().exactly_one() {
90                    Ok(lib) => {
91                        let lib = lib.map_err(|e| {
92                            format!("failed to parse aix dylib '{}': {}", path.display(), e)
93                        })?;
94                        let data = lib.data(data).map_err(|e| {
95                            format!("failed to parse aix dylib '{}': {}", path.display(), e)
96                        })?;
97                        get_metadata_xcoff(path, data)
98                    }
99                    Err(e) => Err(format!("failed to parse aix dylib '{}': {}", path.display(), e)),
100                }
101            })
102        } else {
103            load_metadata_with(path, |data| search_for_section(path, data, ".rustc"))
104        }
105    }
106}

108pub(super) fn search_for_section<'a>(
109    path: &Path,
110    bytes: &'a [u8],
111    section: &str,
112) -> Result<&'a [u8], String> {
113    let Ok(file) = object::File::parse(bytes) else {
114        // The parse above could fail for odd reasons like corruption, but for
115        // now we just interpret it as this target doesn't support metadata
116        // emission in object files so the entire byte slice itself is probably
117        // a metadata file. Ideally though if necessary we could at least check
118        // the prefix of bytes to see if it's an actual metadata object and if
119        // not forward the error along here.
120        return Ok(bytes);
121    };
122    file.section_by_name(section)
123        .ok_or_else(|| format!("no `{}` section in '{}'", section, path.display()))?
124        .data()
125        .map_err(|e| format!("failed to read {} section in '{}': {}", section, path.display(), e))
126}

128fn add_gnu_property_note(
129    file: &mut write::Object<'static>,
130    architecture: Architecture,
131    binary_format: BinaryFormat,
132    endianness: Endianness,
133) {
134    // check bti protection
135    if binary_format != BinaryFormat::Elf
136        || !matches!(architecture, Architecture::X86_64 | Architecture::Aarch64)
137    {
138        return;
139    }
140
141    let section = file.add_section(
142        file.segment_name(StandardSegment::Data).to_vec(),
143        b".note.gnu.property".to_vec(),
144        SectionKind::Note,
145    );
146    let mut data: Vec<u8> = Vec::new();
147    let n_namsz: u32 = 4; // Size of the n_name field
148    let n_descsz: u32 = 16; // Size of the n_desc field
149    let n_type: u32 = object::elf::NT_GNU_PROPERTY_TYPE_0; // Type of note descriptor
150    let header_values = [n_namsz, n_descsz, n_type];
151    header_values.iter().for_each(|v| {
152        data.extend_from_slice(&match endianness {
153            Endianness::Little => v.to_le_bytes(),
154            Endianness::Big => v.to_be_bytes(),
155        })
156    });
157    data.extend_from_slice(b"GNU\0"); // Owner of the program property note
158    let pr_type: u32 = match architecture {
159        Architecture::X86_64 => object::elf::GNU_PROPERTY_X86_FEATURE_1_AND,
160        Architecture::Aarch64 => object::elf::GNU_PROPERTY_AARCH64_FEATURE_1_AND,
161        _ => unreachable!(),
162    };
163    let pr_datasz: u32 = 4; //size of the pr_data field
164    let pr_data: u32 = 3; //program property descriptor
165    let pr_padding: u32 = 0;
166    let property_values = [pr_type, pr_datasz, pr_data, pr_padding];
167    property_values.iter().for_each(|v| {
168        data.extend_from_slice(&match endianness {
169            Endianness::Little => v.to_le_bytes(),
170            Endianness::Big => v.to_be_bytes(),
171        })
172    });
173    file.append_section_data(section, &data, 8);
174}

176pub(super) fn get_metadata_xcoff<'a>(path: &Path, data: &'a [u8]) -> Result<&'a [u8], String> {
177    let Ok(file) = object::File::parse(data) else {
178        return Ok(data);
179    };
180    let info_data = search_for_section(path, data, ".info")?;
181    if let Some(metadata_symbol) =
182        file.symbols().find(|sym| sym.name() == Ok(AIX_METADATA_SYMBOL_NAME))
183    {
184        let offset = metadata_symbol.address() as usize;
185        // The offset specifies the location of rustc metadata in the .info section of XCOFF.
186        // Each string stored in .info section of XCOFF is preceded by a 4-byte length field.
187        if offset < 4 {
188            return Err(format!("Invalid metadata symbol offset: {offset}"));
189        }
190        // XCOFF format uses big-endian byte order.
191        let len = u32::from_be_bytes(info_data[(offset - 4)..offset].try_into().unwrap()) as usize;
192        if offset + len > (info_data.len() as usize) {
193            return Err(format!(
194                "Metadata at offset {offset} with size {len} is beyond .info section"
195            ));
196        }
197        Ok(&info_data[offset..(offset + len)])
198    } else {
199        Err(format!("Unable to find symbol {AIX_METADATA_SYMBOL_NAME}"))
200    }
201}

206#[cfg(any())]
207pub(crate) fn create_object_file(sess: &Session) -> Option<write::Object<'static>> {
208    let endianness = match sess.target.options.endian {
209        Endian::Little => Endianness::Little,
210        Endian::Big => Endianness::Big,
211    };
212    let Some((architecture, sub_architecture)) =
213        sess.target.object_architecture(&sess.unstable_target_features)
214    else {
215        return None;
216    };
217    let binary_format = sess.target.binary_format.to_object();
218
219    let mut file = write::Object::new(binary_format, architecture, endianness);
220    file.set_sub_architecture(sub_architecture);
221    if sess.target.is_like_darwin {
222        if macho_is_arm64e(&sess.target) {
223            file.set_macho_cpu_subtype(object::macho::CPU_SUBTYPE_ARM64E);
224        }
225
226        file.set_macho_build_version(macho_object_build_version_for_target(sess))
227    }
228    if binary_format == BinaryFormat::Coff {
229        // Disable the default mangler to avoid mangling the special "@feat.00" symbol name.
230        let original_mangling = file.mangling();
231        file.set_mangling(object::write::Mangling::None);
232
233        let mut feature = 0;
234
235        if file.architecture() == object::Architecture::I386 {
236            // When linking with /SAFESEH on x86, lld requires that all linker inputs be marked as
237            // safe exception handling compatible. Metadata files masquerade as regular COFF
238            // objects and are treated as linker inputs, despite containing no actual code. Thus,
239            // they still need to be marked as safe exception handling compatible. See #96498.
240            // Reference: https://docs.microsoft.com/en-us/windows/win32/debug/pe-format
241            feature |= 1;
242        }
243
244        file.add_symbol(object::write::Symbol {
245            name: "@feat.00".into(),
246            value: feature,
247            size: 0,
248            kind: object::SymbolKind::Data,
249            scope: object::SymbolScope::Compilation,
250            weak: false,
251            section: object::write::SymbolSection::Absolute,
252            flags: object::SymbolFlags::None,
253        });
254
255        file.set_mangling(original_mangling);
256    }
257    let e_flags = elf_e_flags(architecture, sess);
258    // adapted from LLVM's `MCELFObjectTargetWriter::getOSABI`
259    let os_abi = elf_os_abi(sess);
260    let abi_version = 0;
261    add_gnu_property_note(&mut file, architecture, binary_format, endianness);
262    file.flags = FileFlags::Elf { os_abi, abi_version, e_flags };
263    Some(file)
264}

275#[cfg(any())]
276pub(super) fn elf_e_flags(architecture: Architecture, sess: &Session) -> u32 {
277    match architecture {
278        Architecture::Mips | Architecture::Mips64 | Architecture::Mips64_N32 => {
279            // "N32" indicates an "ILP32" data model on a 64-bit MIPS CPU
280            // like SPARC's "v8+", x86_64's "x32", or the watchOS "arm64_32".
281            let is_32bit = architecture == Architecture::Mips;
282            let mut e_flags = match sess.target.options.cpu.as_ref() {
283                "mips1" if is_32bit => elf::EF_MIPS_ARCH_1,
284                "mips2" if is_32bit => elf::EF_MIPS_ARCH_2,
285                "mips3" => elf::EF_MIPS_ARCH_3,
286                "mips4" => elf::EF_MIPS_ARCH_4,
287                "mips5" => elf::EF_MIPS_ARCH_5,
288                "mips32r2" if is_32bit => elf::EF_MIPS_ARCH_32R2,
289                "mips32r6" if is_32bit => elf::EF_MIPS_ARCH_32R6,
290                "mips64r2" if !is_32bit => elf::EF_MIPS_ARCH_64R2,
291                "mips64r6" if !is_32bit => elf::EF_MIPS_ARCH_64R6,
292                s if s.starts_with("mips32") && !is_32bit => {
293                    sess.dcx().fatal(format!("invalid CPU `{}` for 64-bit MIPS target", s))
294                }
295                s if s.starts_with("mips64") && is_32bit => {
296                    sess.dcx().fatal(format!("invalid CPU `{}` for 32-bit MIPS target", s))
297                }
298                _ if is_32bit => elf::EF_MIPS_ARCH_32R2,
299                _ => elf::EF_MIPS_ARCH_64R2,
300            };
301
302            // If the ABI is explicitly given, use it, or default to O32 on 32-bit MIPS,
303            // which is the only "true" 32-bit option that LLVM supports.
304            match sess.target.options.llvm_abiname.as_ref() {
305                "o32" if is_32bit => e_flags |= elf::EF_MIPS_ABI_O32,
306                "n32" if !is_32bit => e_flags |= elf::EF_MIPS_ABI2,
307                "n64" if !is_32bit => {}
308                "" if is_32bit => e_flags |= elf::EF_MIPS_ABI_O32,
309                "" => sess.dcx().fatal("LLVM ABI must be specifed for 64-bit MIPS targets"),
310                s if is_32bit => {
311                    sess.dcx().fatal(format!("invalid LLVM ABI `{}` for 32-bit MIPS target", s))
312                }
313                s => sess.dcx().fatal(format!("invalid LLVM ABI `{}` for 64-bit MIPS target", s)),
314            };
315
316            if sess.target.options.relocation_model != RelocModel::Static {
317                // PIC means position-independent code. CPIC means "calls PIC".
318                // CPIC was mutually exclusive with PIC according to
319                // the SVR4 MIPS ABI https://refspecs.linuxfoundation.org/elf/mipsabi.pdf
320                // and should have only appeared on static objects with dynamically calls.
321                // At some point someone (GCC?) decided to set CPIC even for PIC.
322                // Nowadays various things expect both set on the same object file
323                // and may even error if you mix CPIC and non-CPIC object files,
324                // despite that being the entire point of the CPIC ABI extension!
325                // As we are in Rome, we do as the Romans do.
326                e_flags |= elf::EF_MIPS_PIC | elf::EF_MIPS_CPIC;
327            }
328            if sess.target.options.cpu.contains("r6") {
329                e_flags |= elf::EF_MIPS_NAN2008;
330            }
331            e_flags
332        }
333        Architecture::Riscv32 | Architecture::Riscv64 => {
334            // Source: https://github.com/riscv-non-isa/riscv-elf-psabi-doc/blob/079772828bd10933d34121117a222b4cc0ee2200/riscv-elf.adoc
335            let mut e_flags: u32 = 0x0;
336
337            // Check if compressed is enabled
338            // `unstable_target_features` is used here because "c" is gated behind riscv_target_feature.
339            if sess.unstable_target_features.contains(&sym::c) {
340                e_flags |= elf::EF_RISCV_RVC;
341            }
342
343            // Set the appropriate flag based on ABI
344            // This needs to match LLVM `RISCVELFStreamer.cpp`
345            match &*sess.target.llvm_abiname {
346                "ilp32" | "lp64" => (),
347                "ilp32f" | "lp64f" => e_flags |= elf::EF_RISCV_FLOAT_ABI_SINGLE,
348                "ilp32d" | "lp64d" => e_flags |= elf::EF_RISCV_FLOAT_ABI_DOUBLE,
349                // Note that the `lp64e` is still unstable as it's not (yet) part of the ELF psABI.
350                "ilp32e" | "lp64e" => e_flags |= elf::EF_RISCV_RVE,
351                _ => bug!("unknown RISC-V ABI name"),
352            }
353
354            e_flags
355        }
356        Architecture::LoongArch32 | Architecture::LoongArch64 => {
357            // Source: https://github.com/loongson/la-abi-specs/blob/release/laelf.adoc#e_flags-identifies-abi-type-and-version
358            let mut e_flags: u32 = elf::EF_LARCH_OBJABI_V1;
359
360            // Set the appropriate flag based on ABI
361            // This needs to match LLVM `LoongArchELFStreamer.cpp`
362            match &*sess.target.llvm_abiname {
363                "ilp32s" | "lp64s" => e_flags |= elf::EF_LARCH_ABI_SOFT_FLOAT,
364                "ilp32f" | "lp64f" => e_flags |= elf::EF_LARCH_ABI_SINGLE_FLOAT,
365                "ilp32d" | "lp64d" => e_flags |= elf::EF_LARCH_ABI_DOUBLE_FLOAT,
366                _ => bug!("unknown LoongArch ABI name"),
367            }
368
369            e_flags
370        }
371        Architecture::Avr => {
372            // Resolve the ISA revision and set
373            // the appropriate EF_AVR_ARCH flag.
374            if let Some(ref cpu) = sess.opts.cg.target_cpu {
375                ef_avr_arch(cpu)
376            } else {
377                bug!("AVR CPU not explicitly specified")
378            }
379        }
380        Architecture::Csky => {
381            let e_flags = match sess.target.options.abi.as_ref() {
382                "abiv2" => elf::EF_CSKY_ABIV2,
383                _ => elf::EF_CSKY_ABIV1,
384            };
385            e_flags
386        }
387        Architecture::PowerPc64 => {
388            const EF_PPC64_ABI_UNKNOWN: u32 = 0;
389            const EF_PPC64_ABI_ELF_V1: u32 = 1;
390            const EF_PPC64_ABI_ELF_V2: u32 = 2;
391
392            match sess.target.options.llvm_abiname.as_ref() {
393                // If the flags do not correctly indicate the ABI,
394                // linkers such as ld.lld assume that the ppc64 object files are always ELFv2
395                // which leads to broken binaries if ELFv1 is used for the object files.
396                "elfv1" => EF_PPC64_ABI_ELF_V1,
397                "elfv2" => EF_PPC64_ABI_ELF_V2,
398                "" if sess.target.options.binary_format.to_object() == BinaryFormat::Elf => {
399                    bug!("No ABI specified for this PPC64 ELF target");
400                }
401                // Fall back
402                _ => EF_PPC64_ABI_UNKNOWN,
403            }
404        }
405        _ => 0,
406    }
407}

409/// Mach-O files contain information about:
410/// - The platform/OS they were built for (macOS/watchOS/Mac Catalyst/iOS simulator etc).
411/// - The minimum OS version / deployment target.
412/// - The version of the SDK they were targetting.
413///
414/// In the past, this was accomplished using the LC_VERSION_MIN_MACOSX, LC_VERSION_MIN_IPHONEOS,
415/// LC_VERSION_MIN_TVOS or LC_VERSION_MIN_WATCHOS load commands, which each contain information
416/// about the deployment target and SDK version, and implicitly, by their presence, which OS they
417/// target. Simulator targets were determined if the architecture was x86_64, but there was e.g. a
418/// LC_VERSION_MIN_IPHONEOS present.
419///
420/// This is of course brittle and limited, so modern tooling emit the LC_BUILD_VERSION load
421/// command (which contains all three pieces of information in one) when the deployment target is
422/// high enough, or the target is something that wouldn't be encodable with the old load commands
423/// (such as Mac Catalyst, or Aarch64 iOS simulator).
424///
425/// Since Xcode 15, Apple's LD apparently requires object files to use this load command, so this
426/// returns the `MachOBuildVersion` for the target to do so.
427fn macho_object_build_version_for_target(sess: &Session) -> object::write::MachOBuildVersion {
428    /// The `object` crate demands "X.Y.Z encoded in nibbles as xxxx.yy.zz"
429    /// e.g. minOS 14.0 = 0x000E0000, or SDK 16.2 = 0x00100200
430    fn pack_version(apple::OSVersion { major, minor, patch }: apple::OSVersion) -> u32 {
431        let (major, minor, patch) = (major as u32, minor as u32, patch as u32);
432        (major << 16) | (minor << 8) | patch
433    }
434
435    let platform = apple::macho_platform(&sess.target);
436    let min_os = sess.apple_deployment_target();
437
438    let mut build_version = object::write::MachOBuildVersion::default();
439    build_version.platform = platform;
440    build_version.minos = pack_version(min_os);
441    // The version here does not _really_ matter, since it is only used at runtime, and we specify
442    // it when linking the final binary, so we will omit the version. This is also what LLVM does,
443    // and the tooling also allows this (and shows the SDK version as `n/a`). Finally, it is the
444    // semantically correct choice, as the SDK has not influenced the binary generated by rustc at
445    // this point in time.
446    build_version.sdk = 0;
447
448    build_version
449}

461/// For rlibs we "pack" rustc metadata into a dummy object file.
462///
463/// Historically it was needed because rustc linked rlibs as whole-archive in some cases.
464/// In that case linkers try to include all files located in an archive, so if metadata is stored
465/// in an archive then it needs to be of a form that the linker is able to process.
466/// Now it's not clear whether metadata still needs to be wrapped into an object file or not.
467///
468/// Note, though, that we don't actually want this metadata to show up in any
469/// final output of the compiler. Instead this is purely for rustc's own
470/// metadata tracking purposes.
471///
472/// With the above in mind, each "flavor" of object format gets special
473/// handling here depending on the target:
474///
475/// * MachO - macos-like targets will insert the metadata into a section that
476///   is sort of fake dwarf debug info. Inspecting the source of the macos
477///   linker this causes these sections to be skipped automatically because
478///   it's not in an allowlist of otherwise well known dwarf section names to
479///   go into the final artifact.
480///
481/// * WebAssembly - this uses wasm files themselves as the object file format
482///   so an empty file with no linking metadata but a single custom section is
483///   created holding our metadata.
484///
485/// * COFF - Windows-like targets create an object with a section that has
486///   the `IMAGE_SCN_LNK_REMOVE` flag set which ensures that if the linker
487///   ever sees the section it doesn't process it and it's removed.
488///
489/// * ELF - All other targets are similar to Windows in that there's a
490///   `SHF_EXCLUDE` flag we can set on sections in an object file to get
491///   automatically removed from the final output.
492pub(crate) fn create_wrapper_file(
493    sess: &Session,
494    section_name: String,
495    data: &[u8],
496) -> (Vec<u8>, MetadataPosition) {
497    let Some(mut file) = create_object_file(sess) else {
498        if sess.target.is_like_wasm {
499            return (
500                create_metadata_file_for_wasm(sess, data, &section_name),
501                MetadataPosition::First,
502            );
503        }
504
505        // Targets using this branch don't have support implemented here yet or
506        // they're not yet implemented in the `object` crate and will likely
507        // fill out this module over time.
508        return (data.to_vec(), MetadataPosition::Last);
509    };
510    let section = if file.format() == BinaryFormat::Xcoff {
511        file.add_section(Vec::new(), b".info".to_vec(), SectionKind::Debug)
512    } else {
513        file.add_section(
514            file.segment_name(StandardSegment::Debug).to_vec(),
515            section_name.into_bytes(),
516            SectionKind::Debug,
517        )
518    };
519    match file.format() {
520        BinaryFormat::Coff => {
521            file.section_mut(section).flags =
522                SectionFlags::Coff { characteristics: pe::IMAGE_SCN_LNK_REMOVE };
523        }
524        BinaryFormat::Elf => {
525            file.section_mut(section).flags =
526                SectionFlags::Elf { sh_flags: elf::SHF_EXCLUDE as u64 };
527        }
528        BinaryFormat::Xcoff => {
529            // AIX system linker may aborts if it meets a valid XCOFF file in archive with no .text, no .data and no .bss.
530            file.add_section(Vec::new(), b".text".to_vec(), SectionKind::Text);
531            file.section_mut(section).flags =
532                SectionFlags::Xcoff { s_flags: xcoff::STYP_INFO as u32 };
533            // Encode string stored in .info section of XCOFF.
534            // FIXME: The length of data here is not guaranteed to fit in a u32.
535            // We may have to split the data into multiple pieces in order to
536            // store in .info section.
537            let len: u32 = data.len().try_into().unwrap();
538            let offset = file.append_section_data(section, &len.to_be_bytes(), 1);
539            // Add a symbol referring to the data in .info section.
540            file.add_symbol(Symbol {
541                name: AIX_METADATA_SYMBOL_NAME.into(),
542                value: offset + 4,
543                size: 0,
544                kind: SymbolKind::Unknown,
545                scope: SymbolScope::Compilation,
546                weak: false,
547                section: SymbolSection::Section(section),
548                flags: SymbolFlags::Xcoff {
549                    n_sclass: xcoff::C_INFO,
550                    x_smtyp: xcoff::C_HIDEXT,
551                    x_smclas: xcoff::C_HIDEXT,
552                    containing_csect: None,
553                },
554            });
555        }
556        _ => {}
557    };
558    file.append_section_data(section, data, 1);
559    (file.write().unwrap(), MetadataPosition::First)
560}

562// Historical note:
563//
564// When using link.exe it was seen that the section name `.note.rustc`
565// was getting shortened to `.note.ru`, and according to the PE and COFF
566// specification:
567//
568// > Executable images do not use a string table and do not support
569// > section names longer than 8characters
570//
571// https://docs.microsoft.com/en-us/windows/win32/debug/pe-format
572//
573// As a result, we choose a slightly shorter name! As to why
574// `.note.rustc` works on MinGW, see
575// https://github.com/llvm/llvm-project/blob/llvmorg-12.0.0/lld/COFF/Writer.cpp#L1190-L1197
576pub fn create_compressed_metadata_file(
577    sess: &Session,
578    metadata: &EncodedMetadata,
579    symbol_name: &str,
580) -> Vec<u8> {
581    let mut packed_metadata = rustc_metadata::METADATA_HEADER.to_vec();
582    packed_metadata.write_all(&(metadata.stub_or_full().len() as u64).to_le_bytes()).unwrap();
583    packed_metadata.extend(metadata.stub_or_full());
584
585    let Some(mut file) = create_object_file(sess) else {
586        if sess.target.is_like_wasm {
587            return create_metadata_file_for_wasm(sess, &packed_metadata, ".rustc");
588        }
589        return packed_metadata.to_vec();
590    };
591    if file.format() == BinaryFormat::Xcoff {
592        return create_compressed_metadata_file_for_xcoff(file, &packed_metadata, symbol_name);
593    }
594    let section = file.add_section(
595        file.segment_name(StandardSegment::Data).to_vec(),
596        b".rustc".to_vec(),
597        SectionKind::ReadOnlyData,
598    );
599    match file.format() {
600        BinaryFormat::Elf => {
601            // Explicitly set no flags to avoid SHF_ALLOC default for data section.
602            file.section_mut(section).flags = SectionFlags::Elf { sh_flags: 0 };
603        }
604        _ => {}
605    };
606    let offset = file.append_section_data(section, &packed_metadata, 1);
607
608    // For MachO and probably PE this is necessary to prevent the linker from throwing away the
609    // .rustc section. For ELF this isn't necessary, but it also doesn't harm.
610    file.add_symbol(Symbol {
611        name: symbol_name.as_bytes().to_vec(),
612        value: offset,
613        size: packed_metadata.len() as u64,
614        kind: SymbolKind::Data,
615        scope: SymbolScope::Dynamic,
616        weak: false,
617        section: SymbolSection::Section(section),
618        flags: SymbolFlags::None,
619    });
620
621    file.write().unwrap()
622}

624/// * Xcoff - On AIX, custom sections are merged into predefined sections,
625///   so custom .rustc section is not preserved during linking.
626///   For this reason, we store metadata in predefined .info section, and
627///   define a symbol to reference the metadata. To preserve metadata during
628///   linking on AIX, we have to
629///   1. Create an empty .text section, a empty .data section.
630///   2. Define an empty symbol named `symbol_name` inside .data section.
631///   3. Define an symbol named `AIX_METADATA_SYMBOL_NAME` referencing
632///      data inside .info section.
633///   From XCOFF's view, (2) creates a csect entry in the symbol table, the
634///   symbol created by (3) is a info symbol for the preceding csect. Thus
635///   two symbols are preserved during linking and we can use the second symbol
636///   to reference the metadata.
637pub fn create_compressed_metadata_file_for_xcoff(
638    mut file: write::Object<'_>,
639    data: &[u8],
640    symbol_name: &str,
641) -> Vec<u8> {
642    assert!(file.format() == BinaryFormat::Xcoff);
643    // AIX system linker may aborts if it meets a valid XCOFF file in archive with no .text, no .data and no .bss.
644    file.add_section(Vec::new(), b".text".to_vec(), SectionKind::Text);
645    let data_section = file.add_section(Vec::new(), b".data".to_vec(), SectionKind::Data);
646    let section = file.add_section(Vec::new(), b".info".to_vec(), SectionKind::Debug);
647    file.add_file_symbol("lib.rmeta".into());
648    file.section_mut(section).flags = SectionFlags::Xcoff { s_flags: xcoff::STYP_INFO as u32 };
649    // Add a global symbol to data_section.
650    file.add_symbol(Symbol {
651        name: symbol_name.as_bytes().into(),
652        value: 0,
653        size: 0,
654        kind: SymbolKind::Data,
655        scope: SymbolScope::Dynamic,
656        weak: true,
657        section: SymbolSection::Section(data_section),
658        flags: SymbolFlags::None,
659    });
660    let len: u32 = data.len().try_into().unwrap();
661    let offset = file.append_section_data(section, &len.to_be_bytes(), 1);
662    // Add a symbol referring to the rustc metadata.
663    file.add_symbol(Symbol {
664        name: AIX_METADATA_SYMBOL_NAME.into(),
665        value: offset + 4, // The metadata is preceded by a 4-byte length field.
666        size: 0,
667        kind: SymbolKind::Unknown,
668        scope: SymbolScope::Dynamic,
669        weak: false,
670        section: SymbolSection::Section(section),
671        flags: SymbolFlags::Xcoff {
672            n_sclass: xcoff::C_INFO,
673            x_smtyp: xcoff::C_HIDEXT,
674            x_smclas: xcoff::C_HIDEXT,
675            containing_csect: None,
676        },
677    });
678    file.append_section_data(section, data, 1);
679    file.write().unwrap()
680}

682/// Creates a simple WebAssembly object file, which is itself a wasm module,
683/// that contains a custom section of the name `section_name` with contents
684/// `data`.
685///
686/// NB: the `object` crate does not yet have support for writing the wasm
687/// object file format. In lieu of that the `wasm-encoder` crate is used to
688/// build a wasm file by hand.
689///
690/// The wasm object file format is defined at
691/// <https://github.com/WebAssembly/tool-conventions/blob/main/Linking.md>
692/// and mainly consists of a `linking` custom section. In this case the custom
693/// section there is empty except for a version marker indicating what format
694/// it's in.
695///
696/// The main purpose of this is to contain a custom section with `section_name`,
697/// which is then appended after `linking`.
698///
699/// As a further detail the object needs to have a 64-bit memory if `wasm64` is
700/// the target or otherwise it's interpreted as a 32-bit object which is
701/// incompatible with 64-bit ones.
702pub fn create_metadata_file_for_wasm(sess: &Session, data: &[u8], section_name: &str) -> Vec<u8> {
703    assert!(sess.target.is_like_wasm);
704    let mut module = wasm_encoder::Module::new();
705    let mut imports = wasm_encoder::ImportSection::new();
706
707    if sess.target.pointer_width == 64 {
708        imports.import(
709            "env",
710            "__linear_memory",
711            wasm_encoder::MemoryType {
712                minimum: 0,
713                maximum: None,
714                memory64: true,
715                shared: false,
716                page_size_log2: None,
717            },
718        );
719    }
720
721    if imports.len() > 0 {
722        module.section(&imports);
723    }
724    module.section(&wasm_encoder::CustomSection {
725        name: "linking".into(),
726        data: Cow::Borrowed(&[2]),
727    });
728    module.section(&wasm_encoder::CustomSection { name: section_name.into(), data: data.into() });
729    module.finish()
730}
