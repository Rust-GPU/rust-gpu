254    /// Emits an `assume` that the integer value `imm` of type `ty` is contained in `range`.
255    ///
256    /// This *always* emits the assumption, so you probably want to check the
257    /// optimization level and `Scalar::is_always_valid` before calling it.
258    fn assume_integer_range(&mut self, imm: Self::Value, ty: Self::Type, range: WrappingRange) {
259        let WrappingRange { start, end } = range;
260
261        // Perhaps one day we'll be able to use assume operand bundles for this,
262        // but for now this encoding with a single icmp+assume is best per
263        // <https://github.com/llvm/llvm-project/issues/123278#issuecomment-2597440158>
264        let shifted = if start == 0 {
265            imm
266        } else {
267            let low = self.const_uint_big(ty, start);
268            self.sub(imm, low)
269        };
270        let width = self.const_uint_big(ty, u128::wrapping_sub(end, start));
271        let cmp = self.icmp(IntPredicate::IntULE, shifted, width);
272        self.assume(cmp);
273    }

275    /// Emits an `assume` that the `val` of pointer type is non-null.
276    ///
277    /// You may want to check the optimization level before bothering calling this.
278    fn assume_nonnull(&mut self, val: Self::Value) {
279        // Arguably in LLVM it'd be better to emit an assume operand bundle instead
280        // <https://llvm.org/docs/LangRef.html#assume-operand-bundles>
281        // but this works fine for all backends.
282
283        let null = self.const_null(self.type_ptr());
284        let is_null = self.icmp(IntPredicate::IntNE, val, null);
285        self.assume(is_null);

369    fn cast_float_to_int(
370        &mut self,
371        signed: bool,
372        x: Self::Value,
373        dest_ty: Self::Type,
374    ) -> Self::Value {
375        let in_ty = self.cx().val_ty(x);
376        let (float_ty, int_ty) = if self.cx().type_kind(dest_ty) == TypeKind::Vector
377            && self.cx().type_kind(in_ty) == TypeKind::Vector
378        {
379            (self.cx().element_type(in_ty), self.cx().element_type(dest_ty))
380        } else {
381            (in_ty, dest_ty)
382        };
383        assert_matches!(
384            self.cx().type_kind(float_ty),
385            TypeKind::Half | TypeKind::Float | TypeKind::Double | TypeKind::FP128
386        );
387        assert_eq!(self.cx().type_kind(int_ty), TypeKind::Integer);
388
389        if let Some(false) = self.cx().sess().opts.unstable_opts.saturating_float_casts {
390            return if signed { self.fptosi(x, dest_ty) } else { self.fptoui(x, dest_ty) };
391        }
392
393        if signed { self.fptosi_sat(x, dest_ty) } else { self.fptoui_sat(x, dest_ty) }
394    }

438    /// *Typed* copy for non-overlapping places.

446        dst: PlaceValue<Self::Value>,
447        src: PlaceValue<Self::Value>,
448        layout: TyAndLayout<'tcx>,
449    ) {
450        self.typed_place_copy_with_flags(dst, src, layout, MemFlags::empty());
451    }

453    fn typed_place_copy_with_flags(
454        &mut self,
455        dst: PlaceValue<Self::Value>,
456        src: PlaceValue<Self::Value>,
457        layout: TyAndLayout<'tcx>,
458        flags: MemFlags,
459    ) {
460        assert!(layout.is_sized(), "cannot typed-copy an unsigned type");
461        assert!(src.llextra.is_none(), "cannot directly copy from unsized values");
462        assert!(dst.llextra.is_none(), "cannot directly copy into unsized values");
463        if flags.contains(MemFlags::NONTEMPORAL) {
464            // HACK(nox): This is inefficient but there is no nontemporal memcpy.
465            let ty = self.backend_type(layout);
466            let val = self.load_from_place(ty, src);
467            self.store_to_place_with_flags(val, dst, flags);
468        } else if self.sess().opts.optimize == OptLevel::No && self.is_backend_immediate(layout) {
469            // If we're not optimizing, the aliasing information from `memcpy`
470            // isn't useful, so just load-store the value for smaller code.
471            let temp = self.load_operand(src.with_type(layout));
472            temp.val.store_with_flags(self, dst.with_type(layout), flags);
473        } else if !layout.is_zst() {
474            let bytes = self.const_usize(layout.size.bytes());
475            self.memcpy(dst.llval, dst.align, src.llval, src.align, bytes, flags);
476        }
477    }

479    /// *Typed* swap for non-overlapping places.
480    ///
481    /// Avoids `alloca`s for Immediates and ScalarPairs.
482    ///
483    /// FIXME: Maybe do something smarter for Ref types too?
484    /// For now, the `typed_swap_nonoverlapping` intrinsic just doesn't call this for those
485    /// cases (in non-debug), preferring the fallback body instead.
486    fn typed_place_swap(
487        &mut self,
488        left: PlaceValue<Self::Value>,
489        right: PlaceValue<Self::Value>,
490        layout: TyAndLayout<'tcx>,
491    ) {
492        let mut temp = self.load_operand(left.with_type(layout));
493        if let OperandValue::Ref(..) = temp.val {
494            // The SSA value isn't stand-alone, so we need to copy it elsewhere
495            let alloca = PlaceRef::alloca(self, layout);
496            self.typed_place_copy(alloca.val, left, layout);
497            temp = self.load_operand(alloca);
498        }
499        self.typed_place_copy(left, right, layout);
500        temp.val.store(self, right.with_type(layout));
501    }
