16pub fn codegen_naked_asm<
17    'a,
18    'tcx,
19    Cx: LayoutOf<'tcx, LayoutOfResult = TyAndLayout<'tcx>>
20        + FnAbiOf<'tcx, FnAbiOfResult = &'tcx FnAbi<'tcx, Ty<'tcx>>>
21        + AsmCodegenMethods<'tcx>,
22>(
23    cx: &'a mut Cx,
24    instance: Instance<'tcx>,
25    item_data: MonoItemData,
26) {
27    assert!(!instance.args.has_infer());
28    let mir = cx.tcx().instance_mir(instance.def);
29
30    let rustc_middle::mir::TerminatorKind::InlineAsm {
31        asm_macro: _,
32        template,
33        ref operands,
34        options,
35        line_spans,
36        targets: _,
37        unwind: _,
38    } = mir.basic_blocks[START_BLOCK].terminator().kind
39    else {
40        bug!("#[naked] functions should always terminate with an asm! block")
41    };
42
43    let operands: Vec<_> =
44        operands.iter().map(|op| inline_to_global_operand::<Cx>(cx, instance, op)).collect();
45
46    let name = cx.mangled_name(instance);
47    let fn_abi = cx.fn_abi_of_instance(instance, ty::List::empty());
48    let (begin, end) = prefix_and_suffix(cx.tcx(), instance, &name, item_data, fn_abi);
49
50    let mut template_vec = Vec::new();
51    template_vec.push(rustc_ast::ast::InlineAsmTemplatePiece::String(begin.into()));
52    template_vec.extend(template.iter().cloned());
53    template_vec.push(rustc_ast::ast::InlineAsmTemplatePiece::String(end.into()));
54
55    cx.codegen_global_asm(&template_vec, &operands, options, line_spans);
56}

58fn inline_to_global_operand<'a, 'tcx, Cx: LayoutOf<'tcx, LayoutOfResult = TyAndLayout<'tcx>>>(
59    cx: &'a Cx,
60    instance: Instance<'tcx>,
61    op: &InlineAsmOperand<'tcx>,
62) -> GlobalAsmOperandRef<'tcx> {
63    match op {
64        InlineAsmOperand::Const { value } => {
65            let const_value = instance
66                .instantiate_mir_and_normalize_erasing_regions(
67                    cx.tcx(),
68                    cx.typing_env(),
69                    ty::EarlyBinder::bind(value.const_),
70                )
71                .eval(cx.tcx(), cx.typing_env(), value.span)
72                .expect("erroneous constant missed by mono item collection");
73
74            let mono_type = instance.instantiate_mir_and_normalize_erasing_regions(
75                cx.tcx(),
76                cx.typing_env(),
77                ty::EarlyBinder::bind(value.ty()),
78            );
79
80            let string = common::asm_const_to_str(
81                cx.tcx(),
82                value.span,
83                const_value,
84                cx.layout_of(mono_type),
85            );
86
87            GlobalAsmOperandRef::Const { string }
88        }
89        InlineAsmOperand::SymFn { value } => {
90            let mono_type = instance.instantiate_mir_and_normalize_erasing_regions(
91                cx.tcx(),
92                cx.typing_env(),
93                ty::EarlyBinder::bind(value.ty()),
94            );
95
96            let instance = match mono_type.kind() {
97                &ty::FnDef(def_id, args) => {
98                    Instance::expect_resolve(cx.tcx(), cx.typing_env(), def_id, args, value.span)
99                }
100                _ => bug!("asm sym is not a function"),
101            };
102
103            GlobalAsmOperandRef::SymFn { instance }
104        }
105        InlineAsmOperand::SymStatic { def_id } => {
106            GlobalAsmOperandRef::SymStatic { def_id: *def_id }
107        }
108        InlineAsmOperand::In { .. }
109        | InlineAsmOperand::Out { .. }
110        | InlineAsmOperand::InOut { .. }
111        | InlineAsmOperand::Label { .. } => {
112            bug!("invalid operand type for naked_asm!")
113        }
114    }
115}

117fn prefix_and_suffix<'tcx>(
118    tcx: TyCtxt<'tcx>,
119    instance: Instance<'tcx>,
120    asm_name: &str,
121    item_data: MonoItemData,
122    fn_abi: &FnAbi<'tcx, Ty<'tcx>>,
123) -> (String, String) {
124    use std::fmt::Write;
125
126    let asm_binary_format = &tcx.sess.target.binary_format;
127
128    let is_arm = tcx.sess.target.arch == "arm";
129    let is_thumb = tcx.sess.unstable_target_features.contains(&sym::thumb_mode);
130
131    let attrs = tcx.codegen_fn_attrs(instance.def_id());
132    let link_section = attrs.link_section.map(|symbol| symbol.as_str().to_string());
133
134    // function alignment can be set globally with the `-Zmin-function-alignment=<n>` flag;
135    // the alignment from a `#[repr(align(<n>))]` is used if it specifies a higher alignment.
136    // if no alignment is specified, an alignment of 4 bytes is used.
137    let min_function_alignment = tcx.sess.opts.unstable_opts.min_function_alignment;
138    let align_bytes =
139        Ord::max(min_function_alignment, attrs.alignment).map(|a| a.bytes()).unwrap_or(4);
140
141    // In particular, `.arm` can also be written `.code 32` and `.thumb` as `.code 16`.
142    let (arch_prefix, arch_suffix) = if is_arm {
143        (
144            match attrs.instruction_set {
145                None => match is_thumb {
146                    true => ".thumb\n.thumb_func",
147                    false => ".arm",
148                },
149                Some(InstructionSetAttr::ArmT32) => ".thumb\n.thumb_func",
150                Some(InstructionSetAttr::ArmA32) => ".arm",
151            },
152            match is_thumb {
153                true => ".thumb",
154                false => ".arm",
155            },
156        )
157    } else {
158        ("", "")
159    };
160
161    let emit_fatal = |msg| tcx.dcx().span_fatal(tcx.def_span(instance.def_id()), msg);
162
163    // see https://godbolt.org/z/cPK4sxKor.
164    let write_linkage = |w: &mut String| -> std::fmt::Result {
165        match item_data.linkage {
166            Linkage::External => {
167                writeln!(w, ".globl {asm_name}")?;
168            }
169            Linkage::LinkOnceAny | Linkage::LinkOnceODR | Linkage::WeakAny | Linkage::WeakODR => {
170                match asm_binary_format {
171                    BinaryFormat::Elf | BinaryFormat::Coff | BinaryFormat::Wasm => {
172                        writeln!(w, ".weak {asm_name}")?;
173                    }
174                    BinaryFormat::Xcoff => {
175                        // FIXME: there is currently no way of defining a weak symbol in inline assembly
176                        // for AIX. See https://github.com/llvm/llvm-project/issues/130269
177                        emit_fatal(
178                            "cannot create weak symbols from inline assembly for this target",
179                        )
180                    }
181                    BinaryFormat::MachO => {
182                        writeln!(w, ".globl {asm_name}")?;
183                        writeln!(w, ".weak_definition {asm_name}")?;
184                    }
185                }
186            }
187            Linkage::Internal => {
188                // write nothing
189            }
190            Linkage::Common => emit_fatal("Functions may not have common linkage"),
191            Linkage::AvailableExternally => {
192                // this would make the function equal an extern definition
193                emit_fatal("Functions may not have available_externally linkage")
194            }
195            Linkage::ExternalWeak => {
196                // FIXME: actually this causes a SIGILL in LLVM
197                emit_fatal("Functions may not have external weak linkage")
198            }
199        }
200
201        Ok(())
202    };
203
204    let mut begin = String::new();
205    let mut end = String::new();
206    match asm_binary_format {
207        BinaryFormat::Elf => {
208            let section = link_section.unwrap_or_else(|| format!(".text.{asm_name}"));
209
210            let progbits = match is_arm {
211                true => "%progbits",
212                false => "@progbits",
213            };
214
215            let function = match is_arm {
216                true => "%function",
217                false => "@function",
218            };
219
220            writeln!(begin, ".pushsection {section},\"ax\", {progbits}").unwrap();
221            writeln!(begin, ".balign {align_bytes}").unwrap();
222            write_linkage(&mut begin).unwrap();
223            match item_data.visibility {
224                Visibility::Default => {}
225                Visibility::Protected => writeln!(begin, ".protected {asm_name}").unwrap(),
226                Visibility::Hidden => writeln!(begin, ".hidden {asm_name}").unwrap(),
227            }
228            writeln!(begin, ".type {asm_name}, {function}").unwrap();
229            if !arch_prefix.is_empty() {
230                writeln!(begin, "{}", arch_prefix).unwrap();
231            }
232            writeln!(begin, "{asm_name}:").unwrap();
233
234            writeln!(end).unwrap();
235            writeln!(end, ".size {asm_name}, . - {asm_name}").unwrap();
236            writeln!(end, ".popsection").unwrap();
237            if !arch_suffix.is_empty() {
238                writeln!(end, "{}", arch_suffix).unwrap();
239            }
240        }
241        BinaryFormat::MachO => {
242            let section = link_section.unwrap_or_else(|| "__TEXT,__text".to_string());
243            writeln!(begin, ".pushsection {},regular,pure_instructions", section).unwrap();
244            writeln!(begin, ".balign {align_bytes}").unwrap();
245            write_linkage(&mut begin).unwrap();
246            match item_data.visibility {
247                Visibility::Default | Visibility::Protected => {}
248                Visibility::Hidden => writeln!(begin, ".private_extern {asm_name}").unwrap(),
249            }
250            writeln!(begin, "{asm_name}:").unwrap();
251
252            writeln!(end).unwrap();
253            writeln!(end, ".popsection").unwrap();
254            if !arch_suffix.is_empty() {
255                writeln!(end, "{}", arch_suffix).unwrap();
256            }
257        }
258        BinaryFormat::Coff => {
259            let section = link_section.unwrap_or_else(|| format!(".text.{asm_name}"));
260            writeln!(begin, ".pushsection {},\"xr\"", section).unwrap();
261            writeln!(begin, ".balign {align_bytes}").unwrap();
262            write_linkage(&mut begin).unwrap();
263            writeln!(begin, ".def {asm_name}").unwrap();
264            writeln!(begin, ".scl 2").unwrap();
265            writeln!(begin, ".type 32").unwrap();
266            writeln!(begin, ".endef").unwrap();
267            writeln!(begin, "{asm_name}:").unwrap();
268
269            writeln!(end).unwrap();
270            writeln!(end, ".popsection").unwrap();
271            if !arch_suffix.is_empty() {
272                writeln!(end, "{}", arch_suffix).unwrap();
273            }
274        }
275        BinaryFormat::Wasm => {
276            let section = link_section.unwrap_or_else(|| format!(".text.{asm_name}"));
277
278            writeln!(begin, ".section {section},\"\",@").unwrap();
279            // wasm functions cannot be aligned, so skip
280            write_linkage(&mut begin).unwrap();
281            if let Visibility::Hidden = item_data.visibility {
282                writeln!(begin, ".hidden {asm_name}").unwrap();
283            }
284            writeln!(begin, ".type {asm_name}, @function").unwrap();
285            if !arch_prefix.is_empty() {
286                writeln!(begin, "{}", arch_prefix).unwrap();
287            }
288            writeln!(begin, "{asm_name}:").unwrap();
289            writeln!(begin, ".functype {asm_name} {}", wasm_functype(tcx, fn_abi)).unwrap();
290
291            writeln!(end).unwrap();
292            // .size is ignored for function symbols, so we can skip it
293            writeln!(end, "end_function").unwrap();
294        }
295        BinaryFormat::Xcoff => {
296            // the LLVM XCOFFAsmParser is extremely incomplete and does not implement many of the
297            // documented directives.
298            //
299            // - https://github.com/llvm/llvm-project/blob/1b25c0c4da968fe78921ce77736e5baef4db75e3/llvm/lib/MC/MCParser/XCOFFAsmParser.cpp
300            // - https://www.ibm.com/docs/en/ssw_aix_71/assembler/assembler_pdf.pdf
301            //
302            // Consequently, we try our best here but cannot do as good a job as for other binary
303            // formats.
304
305            // FIXME: start a section. `.csect` is not currently implemented in LLVM
306
307            // fun fact: according to the assembler documentation, .align takes an exponent,
308            // but LLVM only accepts powers of 2 (but does emit the exponent)
309            // so when we hand `.align 32` to LLVM, the assembly output will contain `.align 5`
310            writeln!(begin, ".align {}", align_bytes).unwrap();
311
312            write_linkage(&mut begin).unwrap();
313            if let Visibility::Hidden = item_data.visibility {
314                // FIXME apparently `.globl {asm_name}, hidden` is valid
315                // but due to limitations with `.weak` (see above) we can't really use that in general yet
316            }
317            writeln!(begin, "{asm_name}:").unwrap();
318
319            writeln!(end).unwrap();
320            // FIXME: end the section?
321        }
322    }
323
324    (begin, end)
325}

327/// The webassembly type signature for the given function.
328///
329/// Used by the `.functype` directive on wasm targets.
330fn wasm_functype<'tcx>(tcx: TyCtxt<'tcx>, fn_abi: &FnAbi<'tcx, Ty<'tcx>>) -> String {
331    let mut signature = String::with_capacity(64);
332
333    let ptr_type = match tcx.data_layout.pointer_size.bits() {
334        32 => "i32",
335        64 => "i64",
336        other => bug!("wasm pointer size cannot be {other} bits"),
337    };
338
339    let hidden_return = matches!(fn_abi.ret.mode, PassMode::Indirect { .. });
340
341    signature.push('(');
342
343    if hidden_return {
344        signature.push_str(ptr_type);
345        if !fn_abi.args.is_empty() {
346            signature.push_str(", ");
347        }
348    }
349
350    let mut it = fn_abi.args.iter().peekable();
351    while let Some(arg_abi) = it.next() {
352        wasm_type(&mut signature, arg_abi, ptr_type);
353        if it.peek().is_some() {
354            signature.push_str(", ");
355        }
356    }
357
358    signature.push_str(") -> (");
359
360    if !hidden_return {
361        wasm_type(&mut signature, &fn_abi.ret, ptr_type);
362    }
363
364    signature.push(')');
365
366    signature
367}

369fn wasm_type<'tcx>(signature: &mut String, arg_abi: &ArgAbi<'_, Ty<'tcx>>, ptr_type: &'static str) {
370    match arg_abi.mode {
371        PassMode::Ignore => { /* do nothing */ }
372        PassMode::Direct(_) => {
373            let direct_type = match arg_abi.layout.backend_repr {
374                BackendRepr::Scalar(scalar) => wasm_primitive(scalar.primitive(), ptr_type),
375                BackendRepr::SimdVector { .. } => "v128",
376                other => unreachable!("unexpected BackendRepr: {:?}", other),
377            };
378
379            signature.push_str(direct_type);
380        }
381        PassMode::Pair(_, _) => match arg_abi.layout.backend_repr {
382            BackendRepr::ScalarPair(a, b) => {
383                signature.push_str(wasm_primitive(a.primitive(), ptr_type));
384                signature.push_str(", ");
385                signature.push_str(wasm_primitive(b.primitive(), ptr_type));
386            }
387            other => unreachable!("{other:?}"),
388        },
389        PassMode::Cast { pad_i32, ref cast } => {
390            // For wasm, Cast is used for single-field primitive wrappers like `struct Wrapper(i64);`
391            assert!(!pad_i32, "not currently used by wasm calling convention");
392            assert!(cast.prefix[0].is_none(), "no prefix");
393            assert_eq!(cast.rest.total, arg_abi.layout.size, "single item");
394
395            let wrapped_wasm_type = match cast.rest.unit.kind {
396                RegKind::Integer => match cast.rest.unit.size.bytes() {
397                    ..=4 => "i32",
398                    ..=8 => "i64",
399                    _ => ptr_type,
400                },
401                RegKind::Float => match cast.rest.unit.size.bytes() {
402                    ..=4 => "f32",
403                    ..=8 => "f64",
404                    _ => ptr_type,
405                },
406                RegKind::Vector => "v128",
407            };
408
409            signature.push_str(wrapped_wasm_type);
410        }
411        PassMode::Indirect { .. } => signature.push_str(ptr_type),
412    }
413}

415fn wasm_primitive(primitive: Primitive, ptr_type: &'static str) -> &'static str {
416    match primitive {
417        Primitive::Int(integer, _) => match integer {
418            Integer::I8 | Integer::I16 | Integer::I32 => "i32",
419            Integer::I64 => "i64",
420            Integer::I128 => "i64, i64",
421        },
422        Primitive::Float(float) => match float {
423            Float::F16 | Float::F32 => "f32",
424            Float::F64 => "f64",
425            Float::F128 => "i64, i64",
426        },
427        Primitive::Pointer(_) => ptr_type,
428    }
