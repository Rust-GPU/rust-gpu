22/// The representation of a Rust value. The enum variant is in fact
23/// uniquely determined by the value's type, but is kept as a
24/// safety check.
25#[derive(Copy, Clone, Debug)]
26pub enum OperandValue<V> {
27    /// A reference to the actual operand. The data is guaranteed
28    /// to be valid for the operand's lifetime.
29    /// The second value, if any, is the extra data (vtable or length)
30    /// which indicates that it refers to an unsized rvalue.
31    ///
32    /// An `OperandValue` *must* be this variant for any type for which
33    /// [`LayoutTypeCodegenMethods::is_backend_ref`] returns `true`.
34    /// (That basically amounts to "isn't one of the other variants".)
35    ///
36    /// This holds a [`PlaceValue`] (like a [`PlaceRef`] does) with a pointer
37    /// to the location holding the value. The type behind that pointer is the
38    /// one returned by [`LayoutTypeCodegenMethods::backend_type`].
39    Ref(PlaceValue<V>),
40    /// A single LLVM immediate value.
41    ///
42    /// An `OperandValue` *must* be this variant for any type for which
43    /// [`LayoutTypeCodegenMethods::is_backend_immediate`] returns `true`.
44    /// The backend value in this variant must be the *immediate* backend type,
45    /// as returned by [`LayoutTypeCodegenMethods::immediate_backend_type`].
46    Immediate(V),
47    /// A pair of immediate LLVM values. Used by wide pointers too.
48    ///
49    /// # Invariants
50    /// - For `Pair(a, b)`, `a` is always at offset 0, but may have `FieldIdx(1..)`
51    /// - `b` is not at offset 0, because `V` is not a 1ZST type.
52    /// - `a` and `b` will have a different FieldIdx, but otherwise `b`'s may be lower
53    ///   or they may not be adjacent, due to arbitrary numbers of 1ZST fields that
54    ///   will not affect the shape of the data which determines if `Pair` will be used.
55    /// - An `OperandValue` *must* be this variant for any type for which
56    /// [`LayoutTypeCodegenMethods::is_backend_scalar_pair`] returns `true`.
57    /// - The backend values in this variant must be the *immediate* backend types,
58    /// as returned by [`LayoutTypeCodegenMethods::scalar_pair_element_backend_type`]
59    /// with `immediate: true`.
60    Pair(V, V),
61    /// A value taking no bytes, and which therefore needs no LLVM value at all.
62    ///
63    /// If you ever need a `V` to pass to something, get a fresh poison value
64    /// from [`ConstCodegenMethods::const_poison`].
65    ///
66    /// An `OperandValue` *must* be this variant for any type for which
67    /// `is_zst` on its `Layout` returns `true`. Note however that
68    /// these values can still require alignment.
69    ZeroSized,
70}

73    /// Treat this value as a pointer and return the data pointer and
74    /// optional metadata as backend values.
75    ///
76    /// If you're making a place, use [`Self::deref`] instead.
77    pub(crate) fn pointer_parts(self) -> (V, Option<V>) {
78        match self {
79            OperandValue::Immediate(llptr) => (llptr, None),
80            OperandValue::Pair(llptr, llextra) => (llptr, Some(llextra)),
81            _ => bug!("OperandValue cannot be a pointer: {self:?}"),
82        }
83    }
84
85    /// Treat this value as a pointer and return the place to which it points.
86    ///
87    /// The pointer immediate doesn't inherently know its alignment,
88    /// so you need to pass it in. If you want to get it from a type's ABI
89    /// alignment, then maybe you want [`OperandRef::deref`] instead.
90    ///
91    /// This is the inverse of [`PlaceValue::address`].
92    pub(crate) fn deref(self, align: Align) -> PlaceValue<V> {
93        let (llval, llextra) = self.pointer_parts();
94        PlaceValue { llval, llextra, align }
95    }

97    pub(crate) fn is_expected_variant_for_type<'tcx, Cx: LayoutTypeCodegenMethods<'tcx>>(
98        &self,
99        cx: &Cx,
100        ty: TyAndLayout<'tcx>,
101    ) -> bool {
102        match self {
103            OperandValue::ZeroSized => ty.is_zst(),
104            OperandValue::Immediate(_) => cx.is_backend_immediate(ty),
105            OperandValue::Pair(_, _) => cx.is_backend_scalar_pair(ty),
106            OperandValue::Ref(_) => cx.is_backend_ref(ty),
107        }
108    }

111/// An `OperandRef` is an "SSA" reference to a Rust value, along with
112/// its type.
113///
114/// NOTE: unless you know a value's type exactly, you should not
115/// generate LLVM opcodes acting on it and instead act via methods,
116/// to avoid nasty edge cases. In particular, using `Builder::store`
117/// directly is sure to cause problems -- use `OperandRef::store`
118/// instead.
119#[derive(Copy, Clone)]
120pub struct OperandRef<'tcx, V> {
121    /// The value.
122    pub val: OperandValue<V>,
123
124    /// The layout of value, based on its Rust type.
125    pub layout: TyAndLayout<'tcx>,
126}

130        write!(f, "OperandRef({:?} @ {:?})", self.val, self.layout)
131    }

140    pub(crate) fn from_const<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
141        bx: &mut Bx,
142        val: mir::ConstValue<'tcx>,
143        ty: Ty<'tcx>,
144    ) -> Self {
145        let layout = bx.layout_of(ty);
146
147        let val = match val {
148            ConstValue::Scalar(x) => {
149                let BackendRepr::Scalar(scalar) = layout.backend_repr else {
150                    bug!("from_const: invalid ByVal layout: {:#?}", layout);
151                };
152                let llval = bx.scalar_to_backend(x, scalar, bx.immediate_backend_type(layout));
153                OperandValue::Immediate(llval)
154            }
155            ConstValue::ZeroSized => return OperandRef::zero_sized(layout),
156            ConstValue::Slice { data, meta } => {
157                let BackendRepr::ScalarPair(a_scalar, _) = layout.backend_repr else {
158                    bug!("from_const: invalid ScalarPair layout: {:#?}", layout);
159                };
160                let a = Scalar::from_pointer(
161                    Pointer::new(bx.tcx().reserve_and_set_memory_alloc(data).into(), Size::ZERO),
162                    &bx.tcx(),
163                );
164                let a_llval = bx.scalar_to_backend(
165                    a,
166                    a_scalar,
167                    bx.scalar_pair_element_backend_type(layout, 0, true),
168                );
169                let b_llval = bx.const_usize(meta);
170                OperandValue::Pair(a_llval, b_llval)
171            }
172            ConstValue::Indirect { alloc_id, offset } => {
173                let alloc = bx.tcx().global_alloc(alloc_id).unwrap_memory();
174                return Self::from_const_alloc(bx, layout, alloc, offset);
175            }
176        };
177
178        OperandRef { val, layout }
179    }

181    fn from_const_alloc<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
182        bx: &mut Bx,
183        layout: TyAndLayout<'tcx>,
184        alloc: rustc_middle::mir::interpret::ConstAllocation<'tcx>,
185        offset: Size,
186    ) -> Self {
187        let alloc_align = alloc.inner().align;
188        assert!(alloc_align >= layout.align.abi);
189
190        let read_scalar = |start, size, s: abi::Scalar, ty| {
191            match alloc.0.read_scalar(
192                bx,
193                alloc_range(start, size),
194                /*read_provenance*/ matches!(s.primitive(), abi::Primitive::Pointer(_)),
195            ) {
196                Ok(val) => bx.scalar_to_backend(val, s, ty),
197                Err(_) => bx.const_poison(ty),
198            }
199        };
200
201        // It may seem like all types with `Scalar` or `ScalarPair` ABI are fair game at this point.
202        // However, `MaybeUninit<u64>` is considered a `Scalar` as far as its layout is concerned --
203        // and yet cannot be represented by an interpreter `Scalar`, since we have to handle the
204        // case where some of the bytes are initialized and others are not. So, we need an extra
205        // check that walks over the type of `mplace` to make sure it is truly correct to treat this
206        // like a `Scalar` (or `ScalarPair`).
207        match layout.backend_repr {
208            BackendRepr::Scalar(s @ abi::Scalar::Initialized { .. }) => {
209                let size = s.size(bx);
210                assert_eq!(size, layout.size, "abi::Scalar size does not match layout size");
211                let val = read_scalar(offset, size, s, bx.immediate_backend_type(layout));
212                OperandRef { val: OperandValue::Immediate(val), layout }
213            }
214            BackendRepr::ScalarPair(
215                a @ abi::Scalar::Initialized { .. },
216                b @ abi::Scalar::Initialized { .. },
217            ) => {
218                let (a_size, b_size) = (a.size(bx), b.size(bx));
219                let b_offset = (offset + a_size).align_to(b.align(bx).abi);
220                assert!(b_offset.bytes() > 0);
221                let a_val = read_scalar(
222                    offset,
223                    a_size,
224                    a,
225                    bx.scalar_pair_element_backend_type(layout, 0, true),
226                );
227                let b_val = read_scalar(
228                    b_offset,
229                    b_size,
230                    b,
231                    bx.scalar_pair_element_backend_type(layout, 1, true),
232                );
233                OperandRef { val: OperandValue::Pair(a_val, b_val), layout }
234            }
235            _ if layout.is_zst() => OperandRef::zero_sized(layout),
236            _ => {
237                // Neither a scalar nor scalar pair. Load from a place
238                // FIXME: should we cache `const_data_from_alloc` to avoid repeating this for the
239                // same `ConstAllocation`?
240                let init = bx.const_data_from_alloc(alloc);
241                let base_addr = bx.static_addr_of(init, alloc_align, None);
242
243                let llval = bx.const_ptr_byte_offset(base_addr, offset);
244                bx.load_operand(PlaceRef::new_sized(llval, layout))
245            }
246        }
247    }

251    pub fn immediate(self) -> V {
252        match self.val {
253            OperandValue::Immediate(s) => s,
254            _ => bug!("not immediate: {:?}", self),
255        }
256    }

258    /// Asserts that this operand is a pointer (or reference) and returns
259    /// the place to which it points.  (This requires no code to be emitted
260    /// as we represent places using the pointer to the place.)
261    ///
262    /// This uses [`Ty::builtin_deref`] to include the type of the place and
263    /// assumes the place is aligned to the pointee's usual ABI alignment.
264    ///
265    /// If you don't need the type, see [`OperandValue::pointer_parts`]
266    /// or [`OperandValue::deref`].
267    pub fn deref<Cx: CodegenMethods<'tcx>>(self, cx: &Cx) -> PlaceRef<'tcx, V> {
268        if self.layout.ty.is_box() {
269            // Derefer should have removed all Box derefs
270            bug!("dereferencing {:?} in codegen", self.layout.ty);
271        }
272
273        let projected_ty = self
274            .layout
275            .ty
276            .builtin_deref(true)
277            .unwrap_or_else(|| bug!("deref of non-pointer {:?}", self));
278
279        let layout = cx.layout_of(projected_ty);
280        self.val.deref(layout.align.abi).with_type(layout)
281    }

283    /// If this operand is a `Pair`, we return an aggregate with the two values.
284    /// For other cases, see `immediate`.
285    pub fn immediate_or_packed_pair<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
286        self,
287        bx: &mut Bx,
288    ) -> V {
289        if let OperandValue::Pair(a, b) = self.val {
290            let llty = bx.cx().immediate_backend_type(self.layout);
291            debug!("Operand::immediate_or_packed_pair: packing {:?} into {:?}", self, llty);
292            // Reconstruct the immediate aggregate.
293            let mut llpair = bx.cx().const_poison(llty);
294            llpair = bx.insert_value(llpair, a, 0);
295            llpair = bx.insert_value(llpair, b, 1);
296            llpair
297        } else {
298            self.immediate()
299        }
300    }

302    /// If the type is a pair, we return a `Pair`, otherwise, an `Immediate`.
303    pub fn from_immediate_or_packed_pair<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
304        bx: &mut Bx,
305        llval: V,
306        layout: TyAndLayout<'tcx>,
307    ) -> Self {
308        let val = if let BackendRepr::ScalarPair(..) = layout.backend_repr {
309            debug!("Operand::from_immediate_or_packed_pair: unpacking {:?} @ {:?}", llval, layout);
310
311            // Deconstruct the immediate aggregate.
312            let a_llval = bx.extract_value(llval, 0);
313            let b_llval = bx.extract_value(llval, 1);
314            OperandValue::Pair(a_llval, b_llval)
315        } else {
316            OperandValue::Immediate(llval)
317        };
318        OperandRef { val, layout }
319    }

321    pub(crate) fn extract_field<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
322        &self,
323        fx: &mut FunctionCx<'a, 'tcx, Bx>,
324        bx: &mut Bx,
325        i: usize,
326    ) -> Self {
327        let field = self.layout.field(bx.cx(), i);
328        let offset = self.layout.fields.offset(i);
329
330        if !bx.is_backend_ref(self.layout) && bx.is_backend_ref(field) {
331            if let BackendRepr::SimdVector { count, .. } = self.layout.backend_repr
332                && let BackendRepr::Memory { sized: true } = field.backend_repr
333                && count.is_power_of_two()
334            {
335                assert_eq!(field.size, self.layout.size);
336                // This is being deprecated, but for now stdarch still needs it for
337                // Newtype vector of array, e.g. #[repr(simd)] struct S([i32; 4]);
338                let place = PlaceRef::alloca(bx, field);
339                self.val.store(bx, place.val.with_type(self.layout));
340                return bx.load_operand(place);
341            } else {
342                // Part of https://github.com/rust-lang/compiler-team/issues/838
343                bug!("Non-ref type {self:?} cannot project to ref field type {field:?}");
344            }
345        }
346
347        let val = if field.is_zst() {
348            OperandValue::ZeroSized
349        } else if field.size == self.layout.size {
350            assert_eq!(offset.bytes(), 0);
351            fx.codegen_transmute_operand(bx, *self, field).unwrap_or_else(|| {
352                bug!(
353                    "Expected `codegen_transmute_operand` to handle equal-size \
354                      field {i:?} projection from {self:?} to {field:?}"
355                )
356            })
357        } else {
358            let (in_scalar, imm) = match (self.val, self.layout.backend_repr) {
359                // `#[repr(simd)]` types are also immediate.
360                (OperandValue::Immediate(llval), BackendRepr::SimdVector { element, .. }) => {
361                    (Some(element), bx.extract_element(llval, bx.cx().const_usize(i as u64)))
362                }
363                // Extract a scalar component from a pair.
364                (OperandValue::Pair(a_llval, b_llval), BackendRepr::ScalarPair(a, b)) => {
365                    if offset.bytes() == 0 {
366                        assert_eq!(field.size, a.size(bx.cx()));
367                        (Some(a), a_llval)
368                    } else {
369                        assert_eq!(offset, a.size(bx.cx()).align_to(b.align(bx.cx()).abi));
370                        assert_eq!(field.size, b.size(bx.cx()));
371                        (Some(b), b_llval)
372                    }
373                }
374
375                _ => {
376                    span_bug!(fx.mir.span, "OperandRef::extract_field({:?}): not applicable", self)
377                }
378            };
379            OperandValue::Immediate(match field.backend_repr {
380                BackendRepr::SimdVector { .. } => imm,
381                BackendRepr::Scalar(out_scalar) => {
382                    let Some(in_scalar) = in_scalar else {
383                        span_bug!(
384                            fx.mir.span,
385                            "OperandRef::extract_field({:?}): missing input scalar for output scalar",
386                            self
387                        )
388                    };
389                    if in_scalar != out_scalar {
390                        // If the backend and backend_immediate types might differ,
391                        // flip back to the backend type then to the new immediate.
392                        // This avoids nop truncations, but still handles things like
393                        // Bools in union fields needs to be truncated.
394                        let backend = bx.from_immediate(imm);
395                        bx.to_immediate_scalar(backend, out_scalar)
396                    } else {
397                        imm
398                    }
399                }
400                BackendRepr::ScalarPair(_, _) | BackendRepr::Memory { .. } => bug!(),
401            })
402        };
403
404        OperandRef { val, layout: field }
405    }

407    /// Obtain the actual discriminant of a value.
408    #[instrument(level = "trace", skip(fx, bx))]
409    pub fn codegen_get_discr<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
410        self,
411        fx: &mut FunctionCx<'a, 'tcx, Bx>,
412        bx: &mut Bx,
413        cast_to: Ty<'tcx>,
414    ) -> V {
415        let dl = &bx.tcx().data_layout;
416        let cast_to_layout = bx.cx().layout_of(cast_to);
417        let cast_to = bx.cx().immediate_backend_type(cast_to_layout);
418
419        // We check uninhabitedness separately because a type like
420        // `enum Foo { Bar(i32, !) }` is still reported as `Variants::Single`,
421        // *not* as `Variants::Empty`.
422        if self.layout.is_uninhabited() {
423            return bx.cx().const_poison(cast_to);
424        }
425
426        let (tag_scalar, tag_encoding, tag_field) = match self.layout.variants {
427            Variants::Empty => unreachable!("we already handled uninhabited types"),
428            Variants::Single { index } => {
429                let discr_val =
430                    if let Some(discr) = self.layout.ty.discriminant_for_variant(bx.tcx(), index) {
431                        discr.val
432                    } else {
433                        // This arm is for types which are neither enums nor coroutines,
434                        // and thus for which the only possible "variant" should be the first one.
435                        assert_eq!(index, FIRST_VARIANT);
436                        // There's thus no actual discriminant to return, so we return
437                        // what it would have been if this was a single-variant enum.
438                        0
439                    };
440                return bx.cx().const_uint_big(cast_to, discr_val);
441            }
442            Variants::Multiple { tag, ref tag_encoding, tag_field, .. } => {
443                (tag, tag_encoding, tag_field)
444            }
445        };
446
447        // Read the tag/niche-encoded discriminant from memory.
448        let tag_op = match self.val {
449            OperandValue::ZeroSized => bug!(),
450            OperandValue::Immediate(_) | OperandValue::Pair(_, _) => {
451                self.extract_field(fx, bx, tag_field.as_usize())
452            }
453            OperandValue::Ref(place) => {
454                let tag = place.with_type(self.layout).project_field(bx, tag_field.as_usize());
455                bx.load_operand(tag)
456            }
457        };
458        let tag_imm = tag_op.immediate();
459
460        // Decode the discriminant (specifically if it's niche-encoded).
461        match *tag_encoding {
462            TagEncoding::Direct => {
463                let signed = match tag_scalar.primitive() {
464                    // We use `i1` for bytes that are always `0` or `1`,
465                    // e.g., `#[repr(i8)] enum E { A, B }`, but we can't
466                    // let LLVM interpret the `i1` as signed, because
467                    // then `i1 1` (i.e., `E::B`) is effectively `i8 -1`.
468                    Primitive::Int(_, signed) => !tag_scalar.is_bool() && signed,
469                    _ => false,
470                };
471                bx.intcast(tag_imm, cast_to, signed)
472            }
473            TagEncoding::Niche { untagged_variant, ref niche_variants, niche_start } => {
474                // Cast to an integer so we don't have to treat a pointer as a
475                // special case.
476                let (tag, tag_llty) = match tag_scalar.primitive() {
477                    // FIXME(erikdesjardins): handle non-default addrspace ptr sizes
478                    Primitive::Pointer(_) => {
479                        let t = bx.type_from_integer(dl.ptr_sized_integer());
480                        let tag = bx.ptrtoint(tag_imm, t);
481                        (tag, t)
482                    }
483                    _ => (tag_imm, bx.cx().immediate_backend_type(tag_op.layout)),
484                };
485
486                // Layout ensures that we only get here for cases where the discriminant
487                // value and the variant index match, since that's all `Niche` can encode.
488                // But for emphasis and debugging, let's double-check one anyway.
489                debug_assert_eq!(
490                    self.layout
491                        .ty
492                        .discriminant_for_variant(bx.tcx(), untagged_variant)
493                        .unwrap()
494                        .val,
495                    u128::from(untagged_variant.as_u32()),
496                );
497
498                let relative_max = niche_variants.end().as_u32() - niche_variants.start().as_u32();
499
500                // We have a subrange `niche_start..=niche_end` inside `range`.
501                // If the value of the tag is inside this subrange, it's a
502                // "niche value", an increment of the discriminant. Otherwise it
503                // indicates the untagged variant.
504                // A general algorithm to extract the discriminant from the tag
505                // is:
506                // relative_tag = tag - niche_start
507                // is_niche = relative_tag <= (ule) relative_max
508                // discr = if is_niche {
509                //     cast(relative_tag) + niche_variants.start()
510                // } else {
511                //     untagged_variant
512                // }
513                // However, we will likely be able to emit simpler code.
514                let (is_niche, tagged_discr, delta) = if relative_max == 0 {
515                    // Best case scenario: only one tagged variant. This will
516                    // likely become just a comparison and a jump.
517                    // The algorithm is:
518                    // is_niche = tag == niche_start
519                    // discr = if is_niche {
520                    //     niche_start
521                    // } else {
522                    //     untagged_variant
523                    // }
524                    let niche_start = bx.cx().const_uint_big(tag_llty, niche_start);
525                    let is_niche = bx.icmp(IntPredicate::IntEQ, tag, niche_start);
526                    let tagged_discr =
527                        bx.cx().const_uint(cast_to, niche_variants.start().as_u32() as u64);
528                    (is_niche, tagged_discr, 0)
529                } else {
530                    // The special cases don't apply, so we'll have to go with
531                    // the general algorithm.
532                    let relative_discr = bx.sub(tag, bx.cx().const_uint_big(tag_llty, niche_start));
533                    let cast_tag = bx.intcast(relative_discr, cast_to, false);
534                    let is_niche = bx.icmp(
535                        IntPredicate::IntULE,
536                        relative_discr,
537                        bx.cx().const_uint(tag_llty, relative_max as u64),
538                    );
539
540                    // Thanks to parameter attributes and load metadata, LLVM already knows
541                    // the general valid range of the tag. It's possible, though, for there
542                    // to be an impossible value *in the middle*, which those ranges don't
543                    // communicate, so it's worth an `assume` to let the optimizer know.
544                    if niche_variants.contains(&untagged_variant)
545                        && bx.cx().sess().opts.optimize != OptLevel::No
546                    {
547                        let impossible =
548                            u64::from(untagged_variant.as_u32() - niche_variants.start().as_u32());
549                        let impossible = bx.cx().const_uint(tag_llty, impossible);
550                        let ne = bx.icmp(IntPredicate::IntNE, relative_discr, impossible);
551                        bx.assume(ne);
552                    }
553
554                    (is_niche, cast_tag, niche_variants.start().as_u32() as u128)
555                };
556
557                let tagged_discr = if delta == 0 {
558                    tagged_discr
559                } else {
560                    bx.add(tagged_discr, bx.cx().const_uint_big(cast_to, delta))
561                };
562
563                let discr = bx.select(
564                    is_niche,
565                    tagged_discr,
566                    bx.cx().const_uint(cast_to, untagged_variant.as_u32() as u64),
567                );
568
569                // In principle we could insert assumes on the possible range of `discr`, but
570                // currently in LLVM this isn't worth it because the original `tag` will
571                // have either a `range` parameter attribute or `!range` metadata,
572                // or come from a `transmute` that already `assume`d it.
573
574                discr
575            }
576        }
577    }
578
579    /// Creates an incomplete operand containing the [`abi::Scalar`]s expected based
580    /// on the `layout` passed. This is for use with [`OperandRef::insert_field`]
581    /// later to set the necessary immediate(s).
582    ///
583    /// Returns `None` for `layout`s which cannot be built this way.
584    pub(crate) fn builder(
585        layout: TyAndLayout<'tcx>,
586    ) -> Option<OperandRef<'tcx, Result<V, abi::Scalar>>> {
587        let val = match layout.backend_repr {
588            BackendRepr::Memory { .. } if layout.is_zst() => OperandValue::ZeroSized,
589            BackendRepr::Scalar(s) => OperandValue::Immediate(Err(s)),
590            BackendRepr::ScalarPair(a, b) => OperandValue::Pair(Err(a), Err(b)),
591            BackendRepr::Memory { .. } | BackendRepr::SimdVector { .. } => return None,
592        };
593        Some(OperandRef { val, layout })
594    }
595}

597impl<'a, 'tcx, V: CodegenObject> OperandRef<'tcx, Result<V, abi::Scalar>> {
598    pub(crate) fn insert_field<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
599        &mut self,
600        bx: &mut Bx,
601        v: VariantIdx,
602        f: FieldIdx,
603        operand: OperandRef<'tcx, V>,
604    ) {
605        let (expect_zst, is_zero_offset) = if let abi::FieldsShape::Primitive = self.layout.fields {
606            // The other branch looking at field layouts ICEs for primitives,
607            // so we need to handle them separately.
608            // Multiple fields is possible for cases such as aggregating
609            // a thin pointer, where the second field is the unit.
610            assert!(!self.layout.is_zst());
611            assert_eq!(v, FIRST_VARIANT);
612            let first_field = f == FieldIdx::ZERO;
613            (!first_field, first_field)
614        } else {
615            let variant_layout = self.layout.for_variant(bx.cx(), v);
616            let field_layout = variant_layout.field(bx.cx(), f.as_usize());
617            let field_offset = variant_layout.fields.offset(f.as_usize());
618            (field_layout.is_zst(), field_offset == Size::ZERO)
619        };
620
621        let mut update = |tgt: &mut Result<V, abi::Scalar>, src, from_scalar| {
622            let from_bty = bx.cx().type_from_scalar(from_scalar);
623            let to_scalar = tgt.unwrap_err();
624            let to_bty = bx.cx().type_from_scalar(to_scalar);
625            let imm = transmute_immediate(bx, src, from_scalar, from_bty, to_scalar, to_bty);
626            *tgt = Ok(imm);
627        };
628
629        match (operand.val, operand.layout.backend_repr) {
630            (OperandValue::ZeroSized, _) if expect_zst => {}
631            (OperandValue::Immediate(v), BackendRepr::Scalar(from_scalar)) => match &mut self.val {
632                OperandValue::Immediate(val @ Err(_)) if is_zero_offset => {
633                    update(val, v, from_scalar);
634                }
635                OperandValue::Pair(fst @ Err(_), _) if is_zero_offset => {
636                    update(fst, v, from_scalar);
637                }
638                OperandValue::Pair(_, snd @ Err(_)) if !is_zero_offset => {
639                    update(snd, v, from_scalar);
640                }
641                _ => bug!("Tried to insert {operand:?} into {v:?}.{f:?} of {self:?}"),
642            },
643            (OperandValue::Pair(a, b), BackendRepr::ScalarPair(from_sa, from_sb)) => {
644                match &mut self.val {
645                    OperandValue::Pair(fst @ Err(_), snd @ Err(_)) => {
646                        update(fst, a, from_sa);
647                        update(snd, b, from_sb);
648                    }
649                    _ => bug!("Tried to insert {operand:?} into {v:?}.{f:?} of {self:?}"),
650                }
651            }
652            _ => bug!("Unsupported operand {operand:?} inserting into {v:?}.{f:?} of {self:?}"),
653        }
654    }
655
656    /// After having set all necessary fields, this converts the
657    /// `OperandValue<Result<V, _>>` (as obtained from [`OperandRef::builder`])
658    /// to the normal `OperandValue<V>`.
659    ///
660    /// ICEs if any required fields were not set.
661    pub fn build(&self) -> OperandRef<'tcx, V> {
662        let OperandRef { val, layout } = *self;
663
664        let unwrap = |r: Result<V, abi::Scalar>| match r {
665            Ok(v) => v,
666            Err(_) => bug!("OperandRef::build called while fields are missing {self:?}"),
667        };
668
669        let val = match val {
670            OperandValue::ZeroSized => OperandValue::ZeroSized,
671            OperandValue::Immediate(v) => OperandValue::Immediate(unwrap(v)),
672            OperandValue::Pair(a, b) => OperandValue::Pair(unwrap(a), unwrap(b)),
673            OperandValue::Ref(_) => bug!(),
674        };
675        OperandRef { val, layout }
676    }
677}

679impl<'a, 'tcx, V: CodegenObject> OperandValue<V> {
680    /// Returns an `OperandValue` that's generally UB to use in any way.
681    ///
682    /// Depending on the `layout`, returns `ZeroSized` for ZSTs, an `Immediate` or
683    /// `Pair` containing poison value(s), or a `Ref` containing a poison pointer.
684    ///
685    /// Supports sized types only.
686    pub fn poison<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
687        bx: &mut Bx,
688        layout: TyAndLayout<'tcx>,
689    ) -> OperandValue<V> {
690        assert!(layout.is_sized());
691        if layout.is_zst() {
692            OperandValue::ZeroSized
693        } else if bx.cx().is_backend_immediate(layout) {
694            let ibty = bx.cx().immediate_backend_type(layout);
695            OperandValue::Immediate(bx.const_poison(ibty))
696        } else if bx.cx().is_backend_scalar_pair(layout) {
697            let ibty0 = bx.cx().scalar_pair_element_backend_type(layout, 0, true);
698            let ibty1 = bx.cx().scalar_pair_element_backend_type(layout, 1, true);
699            OperandValue::Pair(bx.const_poison(ibty0), bx.const_poison(ibty1))
700        } else {
701            let ptr = bx.cx().type_ptr();
702            OperandValue::Ref(PlaceValue::new_sized(bx.const_poison(ptr), layout.align.abi))
703        }
704    }
705

737
738    pub(crate) fn store_with_flags<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
739        self,
740        bx: &mut Bx,
741        dest: PlaceRef<'tcx, V>,
742        flags: MemFlags,
743    ) {
744        debug!("OperandRef::store: operand={:?}, dest={:?}", self, dest);
745        match self {
746            OperandValue::ZeroSized => {
747                // Avoid generating stores of zero-sized values, because the only way to have a
748                // zero-sized value is through `undef`/`poison`, and the store itself is useless.
749            }
750            OperandValue::Ref(val) => {
751                assert!(dest.layout.is_sized(), "cannot directly store unsized values");
752                if val.llextra.is_some() {
753                    bug!("cannot directly store unsized values");
754                }
755                bx.typed_place_copy_with_flags(dest.val, val, dest.layout, flags);
756            }
757            OperandValue::Immediate(s) => {
758                let val = bx.from_immediate(s);
759                bx.store_with_flags(val, dest.val.llval, dest.val.align, flags);
760            }
761            OperandValue::Pair(a, b) => {
762                let BackendRepr::ScalarPair(a_scalar, b_scalar) = dest.layout.backend_repr else {
763                    bug!("store_with_flags: invalid ScalarPair layout: {:#?}", dest.layout);
764                };
765                let b_offset = a_scalar.size(bx).align_to(b_scalar.align(bx).abi);
766
767                let val = bx.from_immediate(a);
768                let align = dest.val.align;
769                bx.store_with_flags(val, dest.val.llval, align, flags);
770
771                let llptr = bx.inbounds_ptradd(dest.val.llval, bx.const_usize(b_offset.bytes()));
772                let val = bx.from_immediate(b);
773                let align = dest.val.align.restrict_for_offset(b_offset);
774                bx.store_with_flags(val, llptr, align, flags);
775            }
776        }
777    }
778
779    pub fn store_unsized<Bx: BuilderMethods<'a, 'tcx, Value = V>>(
780        self,
781        bx: &mut Bx,
782        indirect_dest: PlaceRef<'tcx, V>,
783    ) {
784        debug!("OperandRef::store_unsized: operand={:?}, indirect_dest={:?}", self, indirect_dest);
785        // `indirect_dest` must have `*mut T` type. We extract `T` out of it.
786        let unsized_ty = indirect_dest
787            .layout
788            .ty
789            .builtin_deref(true)
790            .unwrap_or_else(|| bug!("indirect_dest has non-pointer type: {:?}", indirect_dest));
791
792        let OperandValue::Ref(PlaceValue { llval: llptr, llextra: Some(llextra), .. }) = self
793        else {
794            bug!("store_unsized called with a sized value (or with an extern type)")
795        };
796
797        // Allocate an appropriate region on the stack, and copy the value into it. Since alloca
798        // doesn't support dynamic alignment, we allocate an extra align - 1 bytes, and align the
799        // pointer manually.
800        let (size, align) = size_of_val::size_and_align_of_dst(bx, unsized_ty, Some(llextra));
801        let one = bx.const_usize(1);
802        let align_minus_1 = bx.sub(align, one);
803        let size_extra = bx.add(size, align_minus_1);
804        let min_align = Align::ONE;
805        let alloca = bx.dynamic_alloca(size_extra, min_align);
806        let address = bx.ptrtoint(alloca, bx.type_isize());
807        let neg_address = bx.neg(address);
808        let offset = bx.and(neg_address, align_minus_1);
809        let dst = bx.inbounds_ptradd(alloca, offset);
810        bx.memcpy(dst, min_align, llptr, min_align, size, MemFlags::empty());
811
812        // Store the allocated region and the extra to the indirect place.
813        let indirect_operand = OperandValue::Pair(dst, llextra);
814        indirect_operand.store(bx, indirect_dest);
815    }
816}

818impl<'a, 'tcx, Bx: BuilderMethods<'a, 'tcx>> FunctionCx<'a, 'tcx, Bx> {
819    fn maybe_codegen_consume_direct(
820        &mut self,
821        bx: &mut Bx,
822        place_ref: mir::PlaceRef<'tcx>,
823    ) -> Option<OperandRef<'tcx, Bx::Value>> {
824        debug!("maybe_codegen_consume_direct(place_ref={:?})", place_ref);
825
826        match self.locals[place_ref.local] {
827            LocalRef::Operand(mut o) => {
828                // Moves out of scalar and scalar pair fields are trivial.
829                for elem in place_ref.projection.iter() {
830                    match elem {
831                        mir::ProjectionElem::Field(f, _) => {
832                            assert!(
833                                !o.layout.ty.is_any_ptr(),
834                                "Bad PlaceRef: destructing pointers should use cast/PtrMetadata, \
835                                 but tried to access field {f:?} of pointer {o:?}",
836                            );
837                            o = o.extract_field(self, bx, f.index());
838                        }
839                        mir::ProjectionElem::Index(_)
840                        | mir::ProjectionElem::ConstantIndex { .. } => {
841                            // ZSTs don't require any actual memory access.
842                            // FIXME(eddyb) deduplicate this with the identical
843                            // checks in `codegen_consume` and `extract_field`.
844                            let elem = o.layout.field(bx.cx(), 0);
845                            if elem.is_zst() {
846                                o = OperandRef::zero_sized(elem);
847                            } else {
848                                return None;
849                            }
850                        }
851                        _ => return None,
852                    }
853                }
854
855                Some(o)
856            }
857            LocalRef::PendingOperand => {
858                bug!("use of {:?} before def", place_ref);
859            }
860            LocalRef::Place(..) | LocalRef::UnsizedPlace(..) => {
861                // watch out for locals that do not have an
862                // alloca; they are handled somewhat differently
863                None
864            }
865        }
866    }
867
868    pub fn codegen_consume(
869        &mut self,
870        bx: &mut Bx,
871        place_ref: mir::PlaceRef<'tcx>,
872    ) -> OperandRef<'tcx, Bx::Value> {
873        debug!("codegen_consume(place_ref={:?})", place_ref);
874
875        let ty = self.monomorphized_place_ty(place_ref);
876        let layout = bx.cx().layout_of(ty);
877
878        // ZSTs don't require any actual memory access.
879        if layout.is_zst() {
880            return OperandRef::zero_sized(layout);
881        }
882
883        if let Some(o) = self.maybe_codegen_consume_direct(bx, place_ref) {
884            return o;
885        }
886
887        // for most places, to consume them we just load them
888        // out from their home
889        let place = self.codegen_place(bx, place_ref);
890        bx.load_operand(place)
891    }
892
893    pub fn codegen_operand(
894        &mut self,
895        bx: &mut Bx,
896        operand: &mir::Operand<'tcx>,
897    ) -> OperandRef<'tcx, Bx::Value> {
898        debug!("codegen_operand(operand={:?})", operand);
899
900        match *operand {
901            mir::Operand::Copy(ref place) | mir::Operand::Move(ref place) => {
902                self.codegen_consume(bx, place.as_ref())
903            }
904
905            mir::Operand::Constant(ref constant) => {
906                let constant_ty = self.monomorphize(constant.ty());
907                // Most SIMD vector constants should be passed as immediates.
908                // (In particular, some intrinsics really rely on this.)
909                if false {
910                    // However, some SIMD types do not actually use the vector ABI
911                    // (in particular, packed SIMD types do not). Ensure we exclude those.
912                    let layout = bx.layout_of(constant_ty);
913                    if let BackendRepr::SimdVector { .. } = layout.backend_repr {
914                        let (llval, ty) = self.immediate_const_vector(bx, constant);
915                        return OperandRef {
916                            val: OperandValue::Immediate(llval),
917                            layout: bx.layout_of(ty),
918                        };
919                    }
920                }
921                self.eval_mir_constant_to_operand(bx, constant)
922            }
923        }
924    }
