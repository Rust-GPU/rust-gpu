30fn linkage_by_name(tcx: TyCtxt<'_>, def_id: LocalDefId, name: &str) -> Linkage {
31    use rustc_middle::mir::mono::Linkage::*;
32
33    // Use the names from src/llvm/docs/LangRef.rst here. Most types are only
34    // applicable to variable declarations and may not really make sense for
35    // Rust code in the first place but allow them anyway and trust that the
36    // user knows what they're doing. Who knows, unanticipated use cases may pop
37    // up in the future.
38    //
39    // ghost, dllimport, dllexport and linkonce_odr_autohide are not supported
40    // and don't have to be, LLVM treats them as no-ops.
41    match name {
42        "available_externally" => AvailableExternally,
43        "common" => Common,
44        "extern_weak" => ExternalWeak,
45        "external" => External,
46        "internal" => Internal,
47        "linkonce" => LinkOnceAny,
48        "linkonce_odr" => LinkOnceODR,
49        "weak" => WeakAny,
50        "weak_odr" => WeakODR,
51        _ => tcx.dcx().span_fatal(tcx.def_span(def_id), "invalid linkage specified"),
52    }
53}

55fn codegen_fn_attrs(tcx: TyCtxt<'_>, did: LocalDefId) -> CodegenFnAttrs {
56    if cfg!(debug_assertions) {
57        let def_kind = tcx.def_kind(did);
58        assert!(
59            def_kind.has_codegen_attrs(),
60            "unexpected `def_kind` in `codegen_fn_attrs`: {def_kind:?}",
61        );
62    }
63
64    let attrs = tcx.hir_attrs(tcx.local_def_id_to_hir_id(did));
65    let mut codegen_fn_attrs = CodegenFnAttrs::new();
66    if tcx.should_inherit_track_caller(did) {
67        codegen_fn_attrs.flags |= CodegenFnAttrFlags::TRACK_CALLER;
68    }
69
70    // If our rustc version supports autodiff/enzyme, then we call our handler
71    // to check for any `#[rustc_autodiff(...)]` attributes.
72    if cfg!(llvm_enzyme) {
73        let ad = autodiff_attrs(tcx, did.into());
74        codegen_fn_attrs.autodiff_item = ad;
75    }
76
77    // When `no_builtins` is applied at the crate level, we should add the
78    // `no-builtins` attribute to each function to ensure it takes effect in LTO.
79    let crate_attrs = tcx.hir_attrs(rustc_hir::CRATE_HIR_ID);
80    let no_builtins = attr::contains_name(crate_attrs, sym::no_builtins);
81    if no_builtins {
82        codegen_fn_attrs.flags |= CodegenFnAttrFlags::NO_BUILTINS;
83    }
84
85    let rust_target_features = tcx.rust_target_features(LOCAL_CRATE);
86
87    let mut link_ordinal_span = None;
88    let mut no_sanitize_span = None;
89    let mut mixed_export_name_no_mangle_lint_state = MixedExportNameAndNoMangleState::default();
90    let mut no_mangle_span = None;
91
92    for attr in attrs.iter() {
93        // In some cases, attribute are only valid on functions, but it's the `check_attr`
94        // pass that check that they aren't used anywhere else, rather this module.
95        // In these cases, we bail from performing further checks that are only meaningful for
96        // functions (such as calling `fn_sig`, which ICEs if given a non-function). We also
97        // report a delayed bug, just in case `check_attr` isn't doing its job.
98        let fn_sig = || {
99            use DefKind::*;
100
101            let def_kind = tcx.def_kind(did);
102            if let Fn | AssocFn | Variant | Ctor(..) = def_kind {
103                Some(tcx.fn_sig(did))
104            } else {
105                tcx.dcx().span_delayed_bug(
106                    attr.span(),
107                    "this attribute can only be applied to functions",
108                );
109                None
110            }
111        };
112
113        if let hir::Attribute::Parsed(p) = attr {
114            match p {
115                AttributeKind::Repr(reprs) => {
116                    codegen_fn_attrs.alignment = reprs
117                        .iter()
118                        .filter_map(
119                            |(r, _)| if let ReprAttr::ReprAlign(x) = r { Some(*x) } else { None },
120                        )
121                        .max();
122                }
123                AttributeKind::Cold(_) => codegen_fn_attrs.flags |= CodegenFnAttrFlags::COLD,
124                AttributeKind::Align { align, .. } => codegen_fn_attrs.alignment = Some(*align),
125                _ => {}
126            }
127        }
128
129        let Some(Ident { name, .. }) = attr.ident() else {
130            continue;
131        };
132
133        match name {
134            sym::rustc_allocator => codegen_fn_attrs.flags |= CodegenFnAttrFlags::ALLOCATOR,
135            sym::ffi_pure => codegen_fn_attrs.flags |= CodegenFnAttrFlags::FFI_PURE,
136            sym::ffi_const => codegen_fn_attrs.flags |= CodegenFnAttrFlags::FFI_CONST,
137            sym::rustc_nounwind => codegen_fn_attrs.flags |= CodegenFnAttrFlags::NEVER_UNWIND,
138            sym::rustc_reallocator => codegen_fn_attrs.flags |= CodegenFnAttrFlags::REALLOCATOR,
139            sym::rustc_deallocator => codegen_fn_attrs.flags |= CodegenFnAttrFlags::DEALLOCATOR,
140            sym::rustc_allocator_zeroed => {
141                codegen_fn_attrs.flags |= CodegenFnAttrFlags::ALLOCATOR_ZEROED
142            }
143            sym::naked => codegen_fn_attrs.flags |= CodegenFnAttrFlags::NAKED,
144            sym::no_mangle => {
145                no_mangle_span = Some(attr.span());
146                if tcx.opt_item_name(did.to_def_id()).is_some() {
147                    codegen_fn_attrs.flags |= CodegenFnAttrFlags::NO_MANGLE;
148                    mixed_export_name_no_mangle_lint_state.track_no_mangle(
149                        attr.span(),
150                        tcx.local_def_id_to_hir_id(did),
151                        attr,
152                    );
153                } else {
154                    tcx.dcx()
155                        .struct_span_err(
156                            attr.span(),
157                            format!(
158                                "`#[no_mangle]` cannot be used on {} {} as it has no name",
159                                tcx.def_descr_article(did.to_def_id()),
160                                tcx.def_descr(did.to_def_id()),
161                            ),
162                        )
163                        .emit();
164                }
165            }
166            sym::rustc_std_internal_symbol => {
167                codegen_fn_attrs.flags |= CodegenFnAttrFlags::RUSTC_STD_INTERNAL_SYMBOL
168            }
169            sym::used => {
170                let inner = attr.meta_item_list();
171                match inner.as_deref() {
172                    Some([item]) if item.has_name(sym::linker) => {
173                        if !tcx.features().used_with_arg() {
174                            feature_err(
175                                &tcx.sess,
176                                sym::used_with_arg,
177                                attr.span(),
178                                "`#[used(linker)]` is currently unstable",
179                            )
180                            .emit();
181                        }
182                        codegen_fn_attrs.flags |= CodegenFnAttrFlags::USED_LINKER;
183                    }
184                    Some([item]) if item.has_name(sym::compiler) => {
185                        if !tcx.features().used_with_arg() {
186                            feature_err(
187                                &tcx.sess,
188                                sym::used_with_arg,
189                                attr.span(),
190                                "`#[used(compiler)]` is currently unstable",
191                            )
192                            .emit();
193                        }
194                        codegen_fn_attrs.flags |= CodegenFnAttrFlags::USED_COMPILER;
195                    }
196                    Some(_) => {
197                        tcx.dcx().emit_err(errors::ExpectedUsedSymbol { span: attr.span() });
198                    }
199                    None => {
200                        // Unconditionally using `llvm.used` causes issues in handling
201                        // `.init_array` with the gold linker. Luckily gold has been
202                        // deprecated with GCC 15 and rustc now warns about using gold.
203                        codegen_fn_attrs.flags |= CodegenFnAttrFlags::USED_LINKER
204                    }
205                }
206            }
207            sym::thread_local => codegen_fn_attrs.flags |= CodegenFnAttrFlags::THREAD_LOCAL,
208            sym::track_caller => {
209                let is_closure = tcx.is_closure_like(did.to_def_id());
210
211                if !is_closure
212                    && let Some(fn_sig) = fn_sig()
213                    && fn_sig.skip_binder().abi() != ExternAbi::Rust
214                {
215                    tcx.dcx().emit_err(errors::RequiresRustAbi { span: attr.span() });
216                }
217                if is_closure
218                    && !tcx.features().closure_track_caller()
219                    && !attr.span().allows_unstable(sym::closure_track_caller)
220                {
221                    feature_err(
222                        &tcx.sess,
223                        sym::closure_track_caller,
224                        attr.span(),
225                        "`#[track_caller]` on closures is currently unstable",
226                    )
227                    .emit();
228                }
229                codegen_fn_attrs.flags |= CodegenFnAttrFlags::TRACK_CALLER
230            }
231            sym::export_name => {
232                if let Some(s) = attr.value_str() {
233                    if s.as_str().contains('\0') {
234                        // `#[export_name = ...]` will be converted to a null-terminated string,
235                        // so it may not contain any null characters.
236                        tcx.dcx().emit_err(errors::NullOnExport { span: attr.span() });
237                    }
238                    codegen_fn_attrs.export_name = Some(s);
239                    mixed_export_name_no_mangle_lint_state.track_export_name(attr.span());
240                }
241            }
242            sym::target_feature => {
243                let Some(sig) = tcx.hir_node_by_def_id(did).fn_sig() else {
244                    tcx.dcx().span_delayed_bug(attr.span(), "target_feature applied to non-fn");
245                    continue;
246                };
247                let safe_target_features =
248                    matches!(sig.header.safety, hir::HeaderSafety::SafeTargetFeatures);
249                codegen_fn_attrs.safe_target_features = safe_target_features;
250                if safe_target_features {
251                    if tcx.sess.target.is_like_wasm || tcx.sess.opts.actually_rustdoc {
252                        // The `#[target_feature]` attribute is allowed on
253                        // WebAssembly targets on all functions. Prior to stabilizing
254                        // the `target_feature_11` feature, `#[target_feature]` was
255                        // only permitted on unsafe functions because on most targets
256                        // execution of instructions that are not supported is
257                        // considered undefined behavior. For WebAssembly which is a
258                        // 100% safe target at execution time it's not possible to
259                        // execute undefined instructions, and even if a future
260                        // feature was added in some form for this it would be a
261                        // deterministic trap. There is no undefined behavior when
262                        // executing WebAssembly so `#[target_feature]` is allowed
263                        // on safe functions (but again, only for WebAssembly)
264                        //
265                        // Note that this is also allowed if `actually_rustdoc` so
266                        // if a target is documenting some wasm-specific code then
267                        // it's not spuriously denied.
268                        //
269                        // Now that `#[target_feature]` is permitted on safe functions,
270                        // this exception must still exist for allowing the attribute on
271                        // `main`, `start`, and other functions that are not usually
272                        // allowed.
273                    } else {
274                        check_target_feature_trait_unsafe(tcx, did, attr.span());
275                    }
276                }
277                from_target_feature_attr(
278                    tcx,
279                    did,
280                    attr,
281                    rust_target_features,
282                    &mut codegen_fn_attrs.target_features,
283                );
284            }
285            sym::linkage => {
286                if let Some(val) = attr.value_str() {
287                    let linkage = Some(linkage_by_name(tcx, did, val.as_str()));
288                    if tcx.is_foreign_item(did) {
289                        codegen_fn_attrs.import_linkage = linkage;
290
291                        if tcx.is_mutable_static(did.into()) {
292                            let mut diag = tcx.dcx().struct_span_err(
293                                attr.span(),
294                                "extern mutable statics are not allowed with `#[linkage]`",
295                            );
296                            diag.note(
297                                "marking the extern static mutable would allow changing which \
298                                 symbol the static references rather than make the target of the \
299                                 symbol mutable",
300                            );
301                            diag.emit();
302                        }
303                    } else {
304                        codegen_fn_attrs.linkage = linkage;
305                    }
306                }
307            }
308            sym::link_section => {
309                if let Some(val) = attr.value_str() {
310                    if val.as_str().bytes().any(|b| b == 0) {
311                        let msg = format!("illegal null byte in link_section value: `{val}`");
312                        tcx.dcx().span_err(attr.span(), msg);
313                    } else {
314                        codegen_fn_attrs.link_section = Some(val);
315                    }
316                }
317            }
318            sym::link_name => codegen_fn_attrs.link_name = attr.value_str(),
319            sym::link_ordinal => {
320                link_ordinal_span = Some(attr.span());
321                if let ordinal @ Some(_) = check_link_ordinal(tcx, attr) {
322                    codegen_fn_attrs.link_ordinal = ordinal;
323                }
324            }
325            sym::no_sanitize => {
326                no_sanitize_span = Some(attr.span());
327                if let Some(list) = attr.meta_item_list() {
328                    for item in list.iter() {
329                        match item.name() {
330                            Some(sym::address) => {
331                                codegen_fn_attrs.no_sanitize |=
332                                    SanitizerSet::ADDRESS | SanitizerSet::KERNELADDRESS
333                            }
334                            Some(sym::cfi) => codegen_fn_attrs.no_sanitize |= SanitizerSet::CFI,
335                            Some(sym::kcfi) => codegen_fn_attrs.no_sanitize |= SanitizerSet::KCFI,
336                            Some(sym::memory) => {
337                                codegen_fn_attrs.no_sanitize |= SanitizerSet::MEMORY
338                            }
339                            Some(sym::memtag) => {
340                                codegen_fn_attrs.no_sanitize |= SanitizerSet::MEMTAG
341                            }
342                            Some(sym::shadow_call_stack) => {
343                                codegen_fn_attrs.no_sanitize |= SanitizerSet::SHADOWCALLSTACK
344                            }
345                            Some(sym::thread) => {
346                                codegen_fn_attrs.no_sanitize |= SanitizerSet::THREAD
347                            }
348                            Some(sym::hwaddress) => {
349                                codegen_fn_attrs.no_sanitize |= SanitizerSet::HWADDRESS
350                            }
351                            _ => {
352                                tcx.dcx().emit_err(errors::InvalidNoSanitize { span: item.span() });
353                            }
354                        }
355                    }
356                }
357            }
358            sym::instruction_set => {
359                codegen_fn_attrs.instruction_set =
360                    attr.meta_item_list().and_then(|l| match &l[..] {
361                        [MetaItemInner::MetaItem(set)] => {
362                            let segments =
363                                set.path.segments.iter().map(|x| x.ident.name).collect::<Vec<_>>();
364                            match segments.as_slice() {
365                                [sym::arm, sym::a32 | sym::t32]
366                                    if !tcx.sess.target.has_thumb_interworking =>
367                                {
368                                    tcx.dcx().emit_err(errors::UnsupportedInstructionSet {
369                                        span: attr.span(),
370                                    });
371                                    None
372                                }
373                                [sym::arm, sym::a32] => Some(InstructionSetAttr::ArmA32),
374                                [sym::arm, sym::t32] => Some(InstructionSetAttr::ArmT32),
375                                _ => {
376                                    tcx.dcx().emit_err(errors::InvalidInstructionSet {
377                                        span: attr.span(),
378                                    });
379                                    None
380                                }
381                            }
382                        }
383                        [] => {
384                            tcx.dcx().emit_err(errors::BareInstructionSet { span: attr.span() });
385                            None
386                        }
387                        _ => {
388                            tcx.dcx()
389                                .emit_err(errors::MultipleInstructionSet { span: attr.span() });
390                            None
391                        }
392                    })
393            }
394            sym::patchable_function_entry => {
395                codegen_fn_attrs.patchable_function_entry = attr.meta_item_list().and_then(|l| {
396                    let mut prefix = None;
397                    let mut entry = None;
398                    for item in l {
399                        let Some(meta_item) = item.meta_item() else {
400                            tcx.dcx().emit_err(errors::ExpectedNameValuePair { span: item.span() });
401                            continue;
402                        };
403
404                        let Some(name_value_lit) = meta_item.name_value_literal() else {
405                            tcx.dcx().emit_err(errors::ExpectedNameValuePair { span: item.span() });
406                            continue;
407                        };
408
409                        let attrib_to_write = match meta_item.name() {
410                            Some(sym::prefix_nops) => &mut prefix,
411                            Some(sym::entry_nops) => &mut entry,
412                            _ => {
413                                tcx.dcx().emit_err(errors::UnexpectedParameterName {
414                                    span: item.span(),
415                                    prefix_nops: sym::prefix_nops,
416                                    entry_nops: sym::entry_nops,
417                                });
418                                continue;
419                            }
420                        };
421
422                        let rustc_ast::LitKind::Int(val, _) = name_value_lit.kind else {
423                            tcx.dcx().emit_err(errors::InvalidLiteralValue {
424                                span: name_value_lit.span,
425                            });
426                            continue;
427                        };
428
429                        let Ok(val) = val.get().try_into() else {
430                            tcx.dcx()
431                                .emit_err(errors::OutOfRangeInteger { span: name_value_lit.span });
432                            continue;
433                        };
434
435                        *attrib_to_write = Some(val);
436                    }
437
438                    if let (None, None) = (prefix, entry) {
439                        tcx.dcx().span_err(attr.span(), "must specify at least one parameter");
440                    }
441
442                    Some(PatchableFunctionEntry::from_prefix_and_entry(
443                        prefix.unwrap_or(0),
444                        entry.unwrap_or(0),
445                    ))
446                })
447            }
448            _ => {}
449        }
450    }
451
452    mixed_export_name_no_mangle_lint_state.lint_if_mixed(tcx);
453
454    let inline_span;
455    (codegen_fn_attrs.inline, inline_span) = if let Some((inline_attr, span)) =
456        find_attr!(attrs, AttributeKind::Inline(i, span) => (*i, *span))
457    {
458        (inline_attr, Some(span))
459    } else {
460        (InlineAttr::None, None)
461    };
462
463    // naked function MUST NOT be inlined! This attribute is required for the rust compiler itself,
464    // but not for the code generation backend because at that point the naked function will just be
465    // a declaration, with a definition provided in global assembly.
466    if codegen_fn_attrs.flags.contains(CodegenFnAttrFlags::NAKED) {
467        codegen_fn_attrs.inline = InlineAttr::Never;
468    }
469
470    codegen_fn_attrs.optimize =
471        find_attr!(attrs, AttributeKind::Optimize(i, _) => *i).unwrap_or(OptimizeAttr::Default);
472
473    // #73631: closures inherit `#[target_feature]` annotations
474    //
475    // If this closure is marked `#[inline(always)]`, simply skip adding `#[target_feature]`.
476    //
477    // At this point, `unsafe` has already been checked and `#[target_feature]` only affects codegen.
478    // Due to LLVM limitations, emitting both `#[inline(always)]` and `#[target_feature]` is *unsound*:
479    // the function may be inlined into a caller with fewer target features. Also see
480    // <https://github.com/rust-lang/rust/issues/116573>.
481    //
482    // Using `#[inline(always)]` implies that this closure will most likely be inlined into
483    // its parent function, which effectively inherits the features anyway. Boxing this closure
484    // would result in this closure being compiled without the inherited target features, but this
485    // is probably a poor usage of `#[inline(always)]` and easily avoided by not using the attribute.
486    if tcx.is_closure_like(did.to_def_id()) && codegen_fn_attrs.inline != InlineAttr::Always {
487        let owner_id = tcx.parent(did.to_def_id());
488        if tcx.def_kind(owner_id).has_codegen_attrs() {
489            codegen_fn_attrs
490                .target_features
491                .extend(tcx.codegen_fn_attrs(owner_id).target_features.iter().copied());
492        }
493    }
494
495    // If a function uses `#[target_feature]` it can't be inlined into general
496    // purpose functions as they wouldn't have the right target features
497    // enabled. For that reason we also forbid `#[inline(always)]` as it can't be
498    // respected.
499    //
500    // `#[rustc_force_inline]` doesn't need to be prohibited here, only
501    // `#[inline(always)]`, as forced inlining is implemented entirely within
502    // rustc (and so the MIR inliner can do any necessary checks for compatible target
503    // features).
504    //
505    // This sidesteps the LLVM blockers in enabling `target_features` +
506    // `inline(always)` to be used together (see rust-lang/rust#116573 and
507    // llvm/llvm-project#70563).
508    if !codegen_fn_attrs.target_features.is_empty()
509        && matches!(codegen_fn_attrs.inline, InlineAttr::Always)
510        && let Some(span) = inline_span
511    {
512        tcx.dcx().span_err(span, "cannot use `#[inline(always)]` with `#[target_feature]`");
513    }
514
515    if !codegen_fn_attrs.no_sanitize.is_empty()
516        && codegen_fn_attrs.inline.always()
517        && let (Some(no_sanitize_span), Some(inline_span)) = (no_sanitize_span, inline_span)
518    {
519        let hir_id = tcx.local_def_id_to_hir_id(did);
520        tcx.node_span_lint(lint::builtin::INLINE_NO_SANITIZE, hir_id, no_sanitize_span, |lint| {
521            lint.primary_message("`no_sanitize` will have no effect after inlining");
522            lint.span_note(inline_span, "inlining requested here");
523        })
524    }
525
526    // Weak lang items have the same semantics as "std internal" symbols in the
527    // sense that they're preserved through all our LTO passes and only
528    // strippable by the linker.
529    //
530    // Additionally weak lang items have predetermined symbol names.
531    if let Some((name, _)) = lang_items::extract(attrs)
532        && let Some(lang_item) = LangItem::from_name(name)
533    {
534        if WEAK_LANG_ITEMS.contains(&lang_item) {
535            codegen_fn_attrs.flags |= CodegenFnAttrFlags::RUSTC_STD_INTERNAL_SYMBOL;
536        }
537        if let Some(link_name) = lang_item.link_name() {
538            codegen_fn_attrs.export_name = Some(link_name);
539            codegen_fn_attrs.link_name = Some(link_name);
540        }
541    }
542    check_link_name_xor_ordinal(tcx, &codegen_fn_attrs, link_ordinal_span);
543
544    if codegen_fn_attrs.flags.contains(CodegenFnAttrFlags::RUSTC_STD_INTERNAL_SYMBOL)
545        && codegen_fn_attrs.flags.contains(CodegenFnAttrFlags::NO_MANGLE)
546    {
547        let lang_item =
548            lang_items::extract(attrs).map_or(None, |(name, _span)| LangItem::from_name(name));
549        let mut err = tcx
550            .dcx()
551            .struct_span_err(
552                no_mangle_span.unwrap_or_default(),
553                "`#[no_mangle]` cannot be used on internal language items",
554            )
555            .with_note("Rustc requires this item to have a specific mangled name.")
556            .with_span_label(tcx.def_span(did), "should be the internal language item");
557        if let Some(lang_item) = lang_item {
558            if let Some(link_name) = lang_item.link_name() {
559                err = err
560                    .with_note("If you are trying to prevent mangling to ease debugging, many")
561                    .with_note(format!(
562                        "debuggers support a command such as `rbreak {link_name}` to"
563                    ))
564                    .with_note(format!(
565                        "match `.*{link_name}.*` instead of `break {link_name}` on a specific name"
566                    ))
567            }
568        }
569        err.emit();
570    }
571
572    // Any linkage to LLVM intrinsics for now forcibly marks them all as never
573    // unwinds since LLVM sometimes can't handle codegen which `invoke`s
574    // intrinsic functions.
575    if let Some(name) = &codegen_fn_attrs.link_name
576        && name.as_str().starts_with("llvm.")
577    {
578        codegen_fn_attrs.flags |= CodegenFnAttrFlags::NEVER_UNWIND;
579    }
580
581    if let Some(features) = check_tied_features(
582        tcx.sess,
583        &codegen_fn_attrs
584            .target_features
585            .iter()
586            .map(|features| (features.name.as_str(), true))
587            .collect(),
588    ) {
589        let span = tcx
590            .get_attrs(did, sym::target_feature)
591            .next()
592            .map_or_else(|| tcx.def_span(did), |a| a.span());
593        tcx.dcx()
594            .create_err(errors::TargetFeatureDisableOrEnable {
595                features,
596                span: Some(span),
597                missing_features: Some(errors::MissingFeatures),
598            })
599            .emit();
600    }
601
602    codegen_fn_attrs
603}

605/// Checks if the provided DefId is a method in a trait impl for a trait which has track_caller
606/// applied to the method prototype.
607fn should_inherit_track_caller(tcx: TyCtxt<'_>, def_id: DefId) -> bool {
608    if let Some(impl_item) = tcx.opt_associated_item(def_id)
609        && let ty::AssocItemContainer::Impl = impl_item.container
610        && let Some(trait_item) = impl_item.trait_item_def_id
611    {
612        return tcx.codegen_fn_attrs(trait_item).flags.intersects(CodegenFnAttrFlags::TRACK_CALLER);
613    }
614
615    false
616}

618fn check_link_ordinal(tcx: TyCtxt<'_>, attr: &hir::Attribute) -> Option<u16> {
619    use rustc_ast::{LitIntType, LitKind, MetaItemLit};
620    let meta_item_list = attr.meta_item_list()?;
621    let [sole_meta_list] = &meta_item_list[..] else {
622        tcx.dcx().emit_err(errors::InvalidLinkOrdinalNargs { span: attr.span() });
623        return None;
624    };
625    if let Some(MetaItemLit { kind: LitKind::Int(ordinal, LitIntType::Unsuffixed), .. }) =
626        sole_meta_list.lit()
627    {
628        // According to the table at
629        // https://docs.microsoft.com/en-us/windows/win32/debug/pe-format#import-header, the
630        // ordinal must fit into 16 bits. Similarly, the Ordinal field in COFFShortExport (defined
631        // in llvm/include/llvm/Object/COFFImportFile.h), which we use to communicate import
632        // information to LLVM for `#[link(kind = "raw-dylib"_])`, is also defined to be uint16_t.
633        //
634        // FIXME: should we allow an ordinal of 0?  The MSVC toolchain has inconsistent support for
635        // this: both LINK.EXE and LIB.EXE signal errors and abort when given a .DEF file that
636        // specifies a zero ordinal. However, llvm-dlltool is perfectly happy to generate an import
637        // library for such a .DEF file, and MSVC's LINK.EXE is also perfectly happy to consume an
638        // import library produced by LLVM with an ordinal of 0, and it generates an .EXE.  (I
639        // don't know yet if the resulting EXE runs, as I haven't yet built the necessary DLL --
640        // see earlier comment about LINK.EXE failing.)
641        if *ordinal <= u16::MAX as u128 {
642            Some(ordinal.get() as u16)
643        } else {
644            let msg = format!("ordinal value in `link_ordinal` is too large: `{ordinal}`");
645            tcx.dcx()
646                .struct_span_err(attr.span(), msg)
647                .with_note("the value may not exceed `u16::MAX`")
648                .emit();
649            None
650        }
651    } else {
652        tcx.dcx().emit_err(errors::InvalidLinkOrdinalFormat { span: attr.span() });
653        None
654    }
655}

657fn check_link_name_xor_ordinal(
658    tcx: TyCtxt<'_>,
659    codegen_fn_attrs: &CodegenFnAttrs,
660    inline_span: Option<Span>,
661) {
662    if codegen_fn_attrs.link_name.is_none() || codegen_fn_attrs.link_ordinal.is_none() {
663        return;
664    }
665    let msg = "cannot use `#[link_name]` with `#[link_ordinal]`";
666    if let Some(span) = inline_span {
667        tcx.dcx().span_err(span, msg);
668    } else {
669        tcx.dcx().err(msg);
670    }
671}

673#[derive(Default)]
674struct MixedExportNameAndNoMangleState<'a> {
675    export_name: Option<Span>,
676    hir_id: Option<HirId>,
677    no_mangle: Option<Span>,
678    no_mangle_attr: Option<&'a hir::Attribute>,
679}
680
681impl<'a> MixedExportNameAndNoMangleState<'a> {
682    fn track_export_name(&mut self, span: Span) {
683        self.export_name = Some(span);
684    }
685
686    fn track_no_mangle(&mut self, span: Span, hir_id: HirId, attr_name: &'a hir::Attribute) {
687        self.no_mangle = Some(span);
688        self.hir_id = Some(hir_id);
689        self.no_mangle_attr = Some(attr_name);
690    }
691
692    /// Emit diagnostics if the lint condition is met.
693    fn lint_if_mixed(self, tcx: TyCtxt<'_>) {
694        if let Self {
695            export_name: Some(export_name),
696            no_mangle: Some(no_mangle),
697            hir_id: Some(hir_id),
698            no_mangle_attr: Some(_),
699        } = self
700        {
701            tcx.emit_node_span_lint(
702                lint::builtin::UNUSED_ATTRIBUTES,
703                hir_id,
704                no_mangle,
705                errors::MixedExportNameAndNoMangle {
706                    no_mangle,
707                    no_mangle_attr: "#[unsafe(no_mangle)]".to_string(),
708                    export_name,
709                    removal_span: no_mangle,
710                },
711            );
712        }
713    }
714}

716/// We now check the #\[rustc_autodiff\] attributes which we generated from the #[autodiff(...)]
717/// macros. There are two forms. The pure one without args to mark primal functions (the functions
718/// being differentiated). The other form is #[rustc_autodiff(Mode, ActivityList)] on top of the
719/// placeholder functions. We wrote the rustc_autodiff attributes ourself, so this should never
720/// panic, unless we introduced a bug when parsing the autodiff macro.
721fn autodiff_attrs(tcx: TyCtxt<'_>, id: DefId) -> Option<AutoDiffAttrs> {
722    let attrs = tcx.get_attrs(id, sym::rustc_autodiff);
723
724    let attrs = attrs.filter(|attr| attr.has_name(sym::rustc_autodiff)).collect::<Vec<_>>();
725
726    // check for exactly one autodiff attribute on placeholder functions.
727    // There should only be one, since we generate a new placeholder per ad macro.
728    let attr = match &attrs[..] {
729        [] => return None,
730        [attr] => attr,
731        _ => {
732            span_bug!(attrs[1].span(), "cg_ssa: rustc_autodiff should only exist once per source");
733        }
734    };
735
736    let list = attr.meta_item_list().unwrap_or_default();
737
738    // empty autodiff attribute macros (i.e. `#[autodiff]`) are used to mark source functions
739    if list.is_empty() {
740        return Some(AutoDiffAttrs::source());
741    }
742
743    let [mode, width_meta, input_activities @ .., ret_activity] = &list[..] else {
744        span_bug!(attr.span(), "rustc_autodiff attribute must contain mode, width and activities");
745    };
746    let mode = if let MetaItemInner::MetaItem(MetaItem { path: p1, .. }) = mode {
747        p1.segments.first().unwrap().ident
748    } else {
749        span_bug!(attr.span(), "rustc_autodiff attribute must contain mode");
750    };
751
752    // parse mode
753    let mode = match mode.as_str() {
754        "Forward" => DiffMode::Forward,
755        "Reverse" => DiffMode::Reverse,
756        _ => {
757            span_bug!(mode.span, "rustc_autodiff attribute contains invalid mode");
758        }
759    };
760
761    let width: u32 = match width_meta {
762        MetaItemInner::MetaItem(MetaItem { path: p1, .. }) => {
763            let w = p1.segments.first().unwrap().ident;
764            match w.as_str().parse() {
765                Ok(val) => val,
766                Err(_) => {
767                    span_bug!(w.span, "rustc_autodiff width should fit u32");
768                }
769            }
770        }
771        MetaItemInner::Lit(lit) => {
772            if let LitKind::Int(val, _) = lit.kind {
773                match val.get().try_into() {
774                    Ok(val) => val,
775                    Err(_) => {
776                        span_bug!(lit.span, "rustc_autodiff width should fit u32");
777                    }
778                }
779            } else {
780                span_bug!(lit.span, "rustc_autodiff width should be an integer");
781            }
782        }
783    };
784
785    // First read the ret symbol from the attribute
786    let ret_symbol = if let MetaItemInner::MetaItem(MetaItem { path: p1, .. }) = ret_activity {
787        p1.segments.first().unwrap().ident
788    } else {
789        span_bug!(attr.span(), "rustc_autodiff attribute must contain the return activity");
790    };
791
792    // Then parse it into an actual DiffActivity
793    let Ok(ret_activity) = DiffActivity::from_str(ret_symbol.as_str()) else {
794        span_bug!(ret_symbol.span, "invalid return activity");
795    };
796
797    // Now parse all the intermediate (input) activities
798    let mut arg_activities: Vec<DiffActivity> = vec![];
799    for arg in input_activities {
800        let arg_symbol = if let MetaItemInner::MetaItem(MetaItem { path: p2, .. }) = arg {
801            match p2.segments.first() {
802                Some(x) => x.ident,
803                None => {
804                    span_bug!(
805                        arg.span(),
806                        "rustc_autodiff attribute must contain the input activity"
807                    );
808                }
809            }
810        } else {
811            span_bug!(arg.span(), "rustc_autodiff attribute must contain the input activity");
812        };
813
814        match DiffActivity::from_str(arg_symbol.as_str()) {
815            Ok(arg_activity) => arg_activities.push(arg_activity),
816            Err(_) => {
817                span_bug!(arg_symbol.span, "invalid input activity");
818            }
819        }
820    }
821
822    Some(AutoDiffAttrs { mode, width, ret_activity, input_activity: arg_activities })
823}
