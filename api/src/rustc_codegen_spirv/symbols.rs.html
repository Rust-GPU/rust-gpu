11/// Various places in the codebase (mostly attribute parsing) need to compare rustc Symbols to particular keywords.
12/// Symbols are interned, as in, they don't actually store the string itself inside them, but rather an index into a
13/// global table of strings. Then, whenever a new Symbol is created, the global table is checked to see if the string
14/// already exists, deduplicating it if so. This makes things like comparison and cloning really cheap. So, this struct
15/// is to allocate all our keywords up front and intern them all, so we can do comparisons really easily and fast.
16pub struct Symbols {
17    pub discriminant: Symbol,
18    pub rust_gpu: Symbol,
19    pub spirv: Symbol,
20    pub libm: Symbol,
21    pub entry_point_name: Symbol,
22    pub spv_khr_vulkan_memory_model: Symbol,
23
24    descriptor_set: Symbol,
25    binding: Symbol,
26    input_attachment_index: Symbol,
27
28    spec_constant: Symbol,
29    id: Symbol,
30    default: Symbol,
31
32    attributes: FxHashMap<Symbol, SpirvAttribute>,
33    execution_modes: FxHashMap<Symbol, (ExecutionMode, ExecutionModeExtraDim)>,
34    pub libm_intrinsics: FxHashMap<Symbol, libm_intrinsics::LibmIntrinsic>,
35}

359            ("per_primitive_ext", SpirvAttribute::PerPrimitiveExt),

366                SpirvAttribute::IntrinsicType(IntrinsicType::RuntimeArray),

370                SpirvAttribute::IntrinsicType(IntrinsicType::TypedBuffer),

384        let attributes_iter = builtins

386            .chain(execution_models)
387            .chain(custom_attributes)
388            .map(|(a, b)| (Symbol::intern(a), b));
389        let mut attributes = FxHashMap::default();
390        for (a, b) in attributes_iter {
391            let old = attributes.insert(a, b);
392            // `.collect()` into a FxHashMap does not error on duplicates, so manually write out the

396        let mut execution_modes = FxHashMap::default();
397        for &(key, mode, dim) in EXECUTION_MODES {
398            let old = execution_modes.insert(Symbol::intern(key), (mode, dim));
399            assert!(old.is_none());
400        }
401
402        let mut libm_intrinsics = FxHashMap::default();
403        for &(a, b) in libm_intrinsics::TABLE {
404            let old = libm_intrinsics.insert(Symbol::intern(a), b);
405            assert!(old.is_none());
406        }
407        Self {
408            discriminant: Symbol::intern("discriminant"),
409            rust_gpu: Symbol::intern("rust_gpu"),
410            spirv: Symbol::intern("spirv"),
411            libm: Symbol::intern("libm"),
412            entry_point_name: Symbol::intern("entry_point_name"),

414
415            descriptor_set: Symbol::intern("descriptor_set"),
416            binding: Symbol::intern("binding"),
417            input_attachment_index: Symbol::intern("input_attachment_index"),
418
419            spec_constant: Symbol::intern("spec_constant"),
420            id: Symbol::intern("id"),
421            default: Symbol::intern("default"),

423            attributes,
424            execution_modes,
425            libm_intrinsics,
426        }
427    }

429    /// Obtain an `Rc` handle to the current thread's `Symbols` instance, which
430    /// will be shared between all `Symbols::get()` calls on the same thread.
431    ///
432    /// While this is relatively cheap, prefer caching it in e.g. `CodegenCx`,

435        thread_local!(static SYMBOLS: Rc<Symbols> = Rc::new(Symbols::new()));

443// FIXME(eddyb) maybe move this to `attr`?
444pub(crate) fn parse_attrs_for_checking<'a>(
445    sym: &'a Symbols,
446    attrs: &'a [Attribute],
447) -> impl Iterator<Item = Result<(Span, SpirvAttribute), ParseAttrError>> + 'a {
448    attrs.iter().flat_map(move |attr| {
449        let (whole_attr_error, args) = match attr {
450            Attribute::Unparsed(item) => {
451                // #[...]
452                let s = &item.path.segments;
453                if s.len() > 1 && s[0].name == sym.rust_gpu {
454                    // #[rust_gpu ...]
455                    if s.len() != 2 || s[1].name != sym.spirv {
456                        // #[rust_gpu::...] but not #[rust_gpu::spirv]
457                        (
458                            Some(Err((
459                                attr.span(),
460                                "unknown `rust_gpu` attribute, expected `rust_gpu::spirv`"
461                                    .to_string(),
462                            ))),
463                            Default::default(),
464                        )
465                    } else if let Some(args) = attr.meta_item_list() {
466                        // #[rust_gpu::spirv(...)]
467                        (None, args)
468                    } else {
469                        // #[rust_gpu::spirv]
470                        (
471                            Some(Err((
472                                attr.span(),
473                                "#[rust_gpu::spirv(..)] attribute must have at least one argument"
474                                    .to_string(),
475                            ))),
476                            Default::default(),
477                        )
478                    }
479                } else {
480                    // #[...] but not #[rust_gpu ...]
481                    (None, Default::default())
482                }
483            }
484            Attribute::Parsed(_) => (None, Default::default()),
485        };
486
487        whole_attr_error
488            .into_iter()
489            .chain(args.into_iter().map(move |ref arg| {
490                let span = arg.span();
491                let parsed_attr = if arg.has_name(sym.descriptor_set) {
492                    SpirvAttribute::DescriptorSet(parse_attr_int_value(arg)?)
493                } else if arg.has_name(sym.binding) {
494                    SpirvAttribute::Binding(parse_attr_int_value(arg)?)
495                } else if arg.has_name(sym.input_attachment_index) {
496                    SpirvAttribute::InputAttachmentIndex(parse_attr_int_value(arg)?)
497                } else if arg.has_name(sym.spec_constant) {
498                    SpirvAttribute::SpecConstant(parse_spec_constant_attr(sym, arg)?)
499                } else {
500                    let name = match arg.ident() {
501                        Some(i) => i,
502                        None => {
503                            return Err((
504                                span,
505                                "#[spirv(..)] attribute argument must be single identifier"
506                                    .to_string(),
507                            ));
508                        }
509                    };
510                    sym.attributes.get(&name.name).map_or_else(
511                        || Err((name.span, "unknown argument to spirv attribute".to_string())),
512                        |a| {
513                            Ok(match a {
514                                SpirvAttribute::Entry(entry) => SpirvAttribute::Entry(
515                                    parse_entry_attrs(sym, arg, &name, entry.execution_model)?,
516                                ),
517                                _ => a.clone(),
518                            })
519                        },
520                    )?
521                };
522                Ok((span, parsed_attr))
523            }))
524    })
525}

527fn parse_spec_constant_attr(
528    sym: &Symbols,
529    arg: &MetaItemInner,
530) -> Result<SpecConstant, ParseAttrError> {
531    let mut id = None;
532    let mut default = None;
533
534    if let Some(attrs) = arg.meta_item_list() {
535        for attr in attrs {
536            if attr.has_name(sym.id) {
537                if id.is_none() {
538                    id = Some(parse_attr_int_value(attr)?);
539                } else {
540                    return Err((attr.span(), "`id` may only be specified once".into()));
541                }
542            } else if attr.has_name(sym.default) {
543                if default.is_none() {
544                    default = Some(parse_attr_int_value(attr)?);
545                } else {
546                    return Err((attr.span(), "`default` may only be specified once".into()));
547                }
548            } else {
549                return Err((attr.span(), "expected `id = ...` or `default = ...`".into()));
550            }
551        }
552    }
553    Ok(SpecConstant {
554        id: id.ok_or_else(|| (arg.span(), "expected `spec_constant(id = ...)`".into()))?,
555        default,
556    })
557}

559fn parse_attr_int_value(arg: &MetaItemInner) -> Result<u32, ParseAttrError> {
560    let arg = match arg.meta_item() {
561        Some(arg) => arg,
562        None => return Err((arg.span(), "attribute must have value".to_string())),
563    };
564    match arg.name_value_literal() {
565        Some(&MetaItemLit {
566            kind: LitKind::Int(x, LitIntType::Unsuffixed),
567            ..
568        }) if x <= u32::MAX as u128 => Ok(x.get() as u32),
569        _ => Err((arg.span, "attribute value must be integer".to_string())),
570    }
571}

573fn parse_local_size_attr(arg: &MetaItemInner) -> Result<[u32; 3], ParseAttrError> {
574    let arg = match arg.meta_item() {
575        Some(arg) => arg,
576        None => return Err((arg.span(), "attribute must have value".to_string())),
577    };
578    match arg.meta_item_list() {
579        Some(tuple) if !tuple.is_empty() && tuple.len() < 4 => {
580            let mut local_size = [1; 3];
581            for (idx, lit) in tuple.iter().enumerate() {
582                match lit {
583                    MetaItemInner::Lit(MetaItemLit {
584                        kind: LitKind::Int(x, LitIntType::Unsuffixed),
585                        ..
586                    }) if *x <= u32::MAX as u128 => local_size[idx] = x.get() as u32,
587                    _ => return Err((lit.span(), "must be a u32 literal".to_string())),
588                }
589            }
590            Ok(local_size)
591        }
592        Some([]) => Err((
593            arg.span,
594            "#[spirv(compute(threads(x, y, z)))] must have the x dimension specified, trailing ones may be elided".to_string(),
595        )),
596        Some(tuple) if tuple.len() > 3 => Err((
597            arg.span,
598            "#[spirv(compute(threads(x, y, z)))] is three dimensional".to_string(),
599        )),
600        _ => Err((
601            arg.span,
602            "#[spirv(compute(threads(x, y, z)))] must have 1 to 3 parameters, trailing ones may be elided".to_string(),
603        )),
604    }
605}
606
607// for a given entry, gather up the additional attributes
608// in this case ExecutionMode's, some have extra arguments
609// others are specified with x, y, or z components
610// ie #[spirv(fragment(origin_lower_left))] or #[spirv(gl_compute(local_size_x=64, local_size_y=8))]
611fn parse_entry_attrs(
612    sym: &Symbols,
613    arg: &MetaItemInner,
614    name: &Ident,
615    execution_model: ExecutionModel,
616) -> Result<Entry, ParseAttrError> {
617    use ExecutionMode::*;
618    use ExecutionModel::*;
619    let mut entry = Entry::from(execution_model);
620    let mut origin_mode: Option<ExecutionMode> = None;
621    let mut local_size: Option<[u32; 3]> = None;
622    let mut local_size_hint: Option<[u32; 3]> = None;
623    // Reserved
624    //let mut max_workgroup_size_intel: Option<[u32; 3]> = None;
625    if let Some(attrs) = arg.meta_item_list() {
626        for attr in attrs {
627            if let Some(attr_name) = attr.ident() {
628                if let Some((execution_mode, extra_dim)) = sym.execution_modes.get(&attr_name.name)
629                {
630                    use ExecutionModeExtraDim::*;
631                    let val = match extra_dim {
632                        None | Tuple => Option::None,
633                        _ => Some(parse_attr_int_value(attr)?),
634                    };
635                    match execution_mode {
636                        OriginUpperLeft | OriginLowerLeft => {
637                            origin_mode.replace(*execution_mode);
638                        }
639                        LocalSize => {
640                            if local_size.is_none() {
641                                local_size.replace(parse_local_size_attr(attr)?);
642                            } else {
643                                return Err((
644                                    attr_name.span,
645                                    String::from(
646                                        "`#[spirv(compute(threads))]` may only be specified once",
647                                    ),
648                                ));
649                            }
650                        }
651                        LocalSizeHint => {
652                            let val = val.unwrap();
653                            if local_size_hint.is_none() {
654                                local_size_hint.replace([1, 1, 1]);
655                            }
656                            let local_size_hint = local_size_hint.as_mut().unwrap();
657                            match extra_dim {
658                                X => {
659                                    local_size_hint[0] = val;
660                                }
661                                Y => {
662                                    local_size_hint[1] = val;
663                                }
664                                Z => {
665                                    local_size_hint[2] = val;
666                                }
667                                _ => unreachable!(),
668                            }
669                        }
670                        // Reserved
671                        /*MaxWorkgroupSizeINTEL => {
672                            let val = val.unwrap();
673                            if max_workgroup_size_intel.is_none() {
674                                max_workgroup_size_intel.replace([1, 1, 1]);
675                            }
676                            let max_workgroup_size_intel = max_workgroup_size_intel.as_mut()
677                                .unwrap();
678                            match extra_dim {
679                                X => {
680                                    max_workgroup_size_intel[0] = val;
681                                },
682                                Y => {
683                                    max_workgroup_size_intel[1] = val;
684                                },
685                                Z => {
686                                    max_workgroup_size_intel[2] = val;
687                                },
688                                _ => unreachable!(),
689                            }
690                        },*/
691                        _ => {
692                            if let Some(val) = val {
693                                entry
694                                    .execution_modes
695                                    .push((*execution_mode, ExecutionModeExtra::new([val])));
696                            } else {
697                                entry
698                                    .execution_modes
699                                    .push((*execution_mode, ExecutionModeExtra::new([])));
700                            }
701                        }
702                    }
703                } else if attr_name.name == sym.entry_point_name {
704                    match attr.value_str() {
705                        Some(sym) => {
706                            entry.name = Some(sym);
707                        }
708                        None => {
709                            return Err((
710                                attr_name.span,
711                                format!(
712                                    "#[spirv({name}(..))] unknown attribute argument {attr_name}"
713                                ),
714                            ));
715                        }
716                    }
717                } else {
718                    return Err((
719                        attr_name.span,
720                        format!("#[spirv({name}(..))] unknown attribute argument {attr_name}",),
721                    ));
722                }
723            } else {
724                return Err((
725                    arg.span(),
726                    format!("#[spirv({name}(..))] attribute argument must be single identifier"),
727                ));
728            }
729        }
730    }
731    match entry.execution_model {
732        Fragment => {
733            let origin_mode = origin_mode.unwrap_or(OriginUpperLeft);
734            entry
735                .execution_modes
736                .push((origin_mode, ExecutionModeExtra::new([])));
737        }
738        GLCompute | MeshNV | TaskNV | TaskEXT | MeshEXT => {
739            if let Some(local_size) = local_size {
740                entry
741                    .execution_modes
742                    .push((LocalSize, ExecutionModeExtra::new(local_size)));
743            } else {
744                return Err((
745                    arg.span(),
746                    String::from(
747                        "The `threads` argument must be specified when using `#[spirv(compute)]`, `#[spirv(mesh_nv)]`, `#[spirv(task_nv)]`, `#[spirv(task_ext)]` or `#[spirv(mesh_ext)]`",
748                    ),
749                ));
750            }
751        }
752        //TODO: Cover more defaults
753        _ => {}
754    }
755    Ok(entry)
756}
