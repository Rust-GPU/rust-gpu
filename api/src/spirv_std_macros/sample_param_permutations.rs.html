7const SAMPLE_PARAM_COUNT: usize = 4;
8const SAMPLE_PARAM_GENERICS: [&str; SAMPLE_PARAM_COUNT] = ["B", "L", "G", "S"];
9const SAMPLE_PARAM_TYPES: [&str; SAMPLE_PARAM_COUNT] = ["B", "L", "(G,G)", "S"];
10const SAMPLE_PARAM_OPERANDS: [&str; SAMPLE_PARAM_COUNT] = ["Bias", "Lod", "Grad", "Sample"];
11const SAMPLE_PARAM_NAMES: [&str; SAMPLE_PARAM_COUNT] = ["bias", "lod", "grad", "sample_index"];
12const SAMPLE_PARAM_GRAD_INDEX: usize = 2; // Grad requires some special handling because it uses 2 arguments
13const SAMPLE_PARAM_EXPLICIT_LOD_MASK: usize = 0b0110; // which params require the use of ExplicitLod rather than ImplicitLod
14
15fn is_grad(i: usize) -> bool {
16    i == SAMPLE_PARAM_GRAD_INDEX
17}
18
19struct SampleImplRewriter(usize, syn::Type);
20
21impl SampleImplRewriter {
22    pub fn rewrite(mask: usize, f: &syn::ItemImpl) -> syn::ItemImpl {
23        let mut new_impl = f.clone();
24        let mut ty_str = String::from("SampleParams<");
25
26        // based on the mask, form a `SampleParams` type string and add the generic parameters to the `impl<>` generics
27        // example type string: `"SampleParams<SomeTy<B>, NoneTy, NoneTy>"`
28        for i in 0..SAMPLE_PARAM_COUNT {
29            if mask & (1 << i) != 0 {
30                new_impl.generics.params.push(syn::GenericParam::Type(
31                    syn::Ident::new(SAMPLE_PARAM_GENERICS[i], Span::call_site()).into(),
32                ));
33                ty_str.push_str("SomeTy<");
34                ty_str.push_str(SAMPLE_PARAM_TYPES[i]);
35                ty_str.push('>');
36            } else {
37                ty_str.push_str("NoneTy");
38            }
39            ty_str.push(',');
40        }
41        ty_str.push('>');
42        let ty: syn::Type = syn::parse(ty_str.parse().unwrap()).unwrap();
43
44        // use the type to insert it into the generic argument of the trait we're implementing
45        // e.g., `ImageWithMethods<Dummy>` becomes `ImageWithMethods<SampleParams<SomeTy<B>, NoneTy, NoneTy>>`
46        if let Some(t) = &mut new_impl.trait_
47            && let syn::PathArguments::AngleBracketed(a) =
48                &mut t.1.segments.last_mut().unwrap().arguments
49            && let Some(syn::GenericArgument::Type(t)) = a.args.last_mut()
50        {
51            *t = ty.clone();
52        }
53
54        // rewrite the implemented functions
55        SampleImplRewriter(mask, ty).visit_item_impl_mut(&mut new_impl);
56        new_impl
57    }
58
59    // generates an operands string for use in the assembly, e.g. "Bias %bias Lod %lod", based on the mask
60    #[allow(clippy::needless_range_loop)]
61    fn get_operands(&self) -> String {
62        let mut op = String::new();
63        for i in 0..SAMPLE_PARAM_COUNT {
64            if self.0 & (1 << i) != 0 {
65                if is_grad(i) {
66                    op.push_str("Grad %grad_x %grad_y ");
67                } else {
68                    op.push_str(SAMPLE_PARAM_OPERANDS[i]);
69                    op.push_str(" %");
70                    op.push_str(SAMPLE_PARAM_NAMES[i]);
71                    op.push(' ');
72                }
73            }
74        }
75        op
76    }
77
78    // generates list of assembly loads for the data, e.g. "%bias = OpLoad _ {bias}", etc.
79    #[allow(clippy::needless_range_loop)]
80    fn add_loads(&self, t: &mut Vec<TokenTree>) {
81        for i in 0..SAMPLE_PARAM_COUNT {
82            if self.0 & (1 << i) != 0 {
83                if is_grad(i) {
84                    t.push(TokenTree::Literal(proc_macro2::Literal::string(
85                        "%grad_x = OpLoad _ {grad_x}",
86                    )));
87                    t.push(TokenTree::Punct(proc_macro2::Punct::new(
88                        ',',
89                        proc_macro2::Spacing::Alone,
90                    )));
91                    t.push(TokenTree::Literal(proc_macro2::Literal::string(
92                        "%grad_y = OpLoad _ {grad_y}",
93                    )));
94                    t.push(TokenTree::Punct(proc_macro2::Punct::new(
95                        ',',
96                        proc_macro2::Spacing::Alone,
97                    )));
98                } else {
99                    let s = format!("%{0} = OpLoad _ {{{0}}}", SAMPLE_PARAM_NAMES[i]);
100                    t.push(TokenTree::Literal(proc_macro2::Literal::string(s.as_str())));
101                    t.push(TokenTree::Punct(proc_macro2::Punct::new(
102                        ',',
103                        proc_macro2::Spacing::Alone,
104                    )));
105                }
106            }
107        }
108    }
109
110    // generates list of register specifications, e.g. `bias = in(reg) &params.bias.0, ...` as separate tokens
111    #[allow(clippy::needless_range_loop)]
112    fn add_regs(&self, t: &mut Vec<TokenTree>) {
113        for i in 0..SAMPLE_PARAM_COUNT {
114            if self.0 & (1 << i) != 0 {
115                // HACK(eddyb) the extra `{...}` force the pointers to be to
116                // fresh variables holding value copies, instead of the originals,
117                // allowing `OpLoad _` inference to pick the appropriate type.
118                let s = if is_grad(i) {
119                    "grad_x=in(reg) &{params.grad.0.0},grad_y=in(reg) &{params.grad.0.1},"
120                        .to_string()
121                } else {
122                    format!("{0} = in(reg) &{{params.{0}.0}},", SAMPLE_PARAM_NAMES[i])
123                };
124                let ts: proc_macro2::TokenStream = s.parse().unwrap();
125                t.extend(ts);
126            }
127        }
128    }
129}

131impl VisitMut for SampleImplRewriter {
132    fn visit_impl_item_fn_mut(&mut self, item: &mut ImplItemFn) {
133        // rewrite the last parameter of this method to be of type `SampleParams<...>` we generated earlier
134        if let Some(syn::FnArg::Typed(p)) = item.sig.inputs.last_mut() {
135            *p.ty.as_mut() = self.1.clone();
136        }
137        syn::visit_mut::visit_impl_item_fn_mut(self, item);
138    }
139
140    fn visit_macro_mut(&mut self, m: &mut syn::Macro) {
141        if m.path.is_ident("asm") {
142            // this is where the asm! block is manipulated
143            let t = m.tokens.clone();
144            let mut new_t = Vec::new();
145            let mut altered = false;
146
147            for tt in t {
148                match tt {
149                    TokenTree::Literal(l) => {
150                        if let Ok(l) = syn::parse::<syn::LitStr>(l.to_token_stream().into()) {
151                            // found a string literal
152                            let s = l.value();
153                            if s.contains("$PARAMS") {
154                                altered = true;
155                                // add load instructions before the sampling instruction
156                                self.add_loads(&mut new_t);
157                                // and insert image operands
158                                let s = s.replace("$PARAMS", &self.get_operands());
159                                let lod_type = if self.0 & SAMPLE_PARAM_EXPLICIT_LOD_MASK != 0 {
160                                    "ExplicitLod"
161                                } else {
162                                    "ImplicitLod"
163                                };
164                                let s = s.replace("$LOD", lod_type);
165
166                                new_t.push(TokenTree::Literal(proc_macro2::Literal::string(
167                                    s.as_str(),
168                                )));
169                            } else {
170                                new_t.push(TokenTree::Literal(l.token()));
171                            }
172                        } else {
173                            new_t.push(TokenTree::Literal(l));
174                        }
175                    }
176                    _ => {
177                        new_t.push(tt);
178                    }
179                }
180            }
181
182            if altered {
183                // finally, add register specs
184                self.add_regs(&mut new_t);
185            }
186
187            // replace all tokens within the asm! block with our new list
188            m.tokens = new_t.into_iter().collect();
189        }
190    }
191}

193pub fn gen_sample_param_permutations(item: TokenStream) -> TokenStream {
194    let item_impl = syn::parse_macro_input!(item as syn::ItemImpl);
195    let mut fns = Vec::new();
196
197    for m in 1..(1 << SAMPLE_PARAM_COUNT) {
198        fns.push(SampleImplRewriter::rewrite(m, &item_impl));
199    }
200
201    // uncomment to output generated tokenstream to stdout
202    //println!("{}", quote! { #(#fns)* }.to_string());
203    quote! { #(#fns)* }.into()
204}
