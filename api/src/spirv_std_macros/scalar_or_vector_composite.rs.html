9pub fn derive(item: TokenStream) -> syn::Result<TokenStream> {
10    // Whenever we'll properly resolve the crate symbol, replace this.
11    let spirv_std = quote!(spirv_std);
12
13    // Defer all validation to our codegen backend. Rather than erroring here, emit garbage.
14    let item = syn::parse2::<DeriveInput>(item)?;
15    let content = match &item.data {
16        Data::Enum(_) => derive_enum(&spirv_std, &item),
17        Data::Struct(data) => derive_struct(&spirv_std, data),
18        Data::Union(DataUnion { union_token, .. }) => {
19            Err(syn::Error::new_spanned(union_token, "Union not supported"))
20        }
21    }?;
22
23    let ident = &item.ident;
24    let gens = &item.generics.params;
25    let gen_refs = &item
26        .generics
27        .params
28        .iter()
29        .map(|p| match p {
30            GenericParam::Lifetime(p) => p.lifetime.to_token_stream(),
31            GenericParam::Type(p) => p.ident.to_token_stream(),
32            GenericParam::Const(p) => p.ident.to_token_stream(),
33        })
34        .collect::<Punctuated<_, Token![,]>>();
35    let where_clause = &item.generics.where_clause;
36
37    Ok(quote! {
38        impl<#gens> #spirv_std::ScalarComposite for #ident<#gen_refs> #where_clause {
39            #[inline]
40            fn transform<F: #spirv_std::ScalarOrVectorTransform>(self, f: &mut F) -> Self {
41                #content
42            }
43        }
44    })
45}

47pub fn derive_struct(spirv_std: &TokenStream, data: &DataStruct) -> syn::Result<TokenStream> {
48    Ok(match &data.fields {
49        Fields::Named(FieldsNamed { named, .. }) => {
50            let content = named
51                .iter()
52                .map(|f| {
53                    let ident = &f.ident;
54                    quote!(#ident: #spirv_std::ScalarComposite::transform(self.#ident, f))
55                })
56                .collect::<Punctuated<_, Token![,]>>();
57            quote!(Self { #content })
58        }
59        Fields::Unnamed(FieldsUnnamed { unnamed, .. }) => {
60            let content = (0..unnamed.len())
61                .map(|i| {
62                    let i = syn::Index::from(i);
63                    quote!(#spirv_std::ScalarComposite::transform(self.#i, f))
64                })
65                .collect::<Punctuated<_, Token![,]>>();
66            quote!(Self(#content))
67        }
68        Fields::Unit => quote!(Self),
69    })
70}

72pub fn derive_enum(spirv_std: &TokenStream, item: &DeriveInput) -> syn::Result<TokenStream> {
73    let mut attributes = item.attrs.iter().filter(|a| a.path().is_ident("repr"));
74    let repr = match (attributes.next(), attributes.next()) {
75        (None, _) => Err(syn::Error::new_spanned(
76            item,
77            "Missing #[repr(...)] attribute",
78        )),
79        (Some(repr), None) => Ok(repr),
80        (Some(_), Some(_)) => Err(syn::Error::new_spanned(
81            item,
82            "Multiple #[repr(...)] attributes found",
83        )),
84    }?;
85    let prim = &repr.meta.require_list()?.tokens;
86    Ok(quote! {
87        #spirv_std::assert_is_integer::<#prim>();
88        <Self as From<#prim>>::from(#spirv_std::ScalarComposite::transform(<Self as Into<#prim>>::into(self), f))
89    })
90}
