3//! Internal primitives for building view and projection matrices.
4//!
5//! Each function is parameterized over const generics that select the
6//! coordinate system convention:
7//!
8//! * `RH` - right-handed (`true`) vs left-handed (`false`) view space
9//! * `ZO` - NDC Z maps to `[0, 1]` (`true`) vs `[-1, 1]` (`false`)
10//! * `YFLIP` - NDC Y points down (`true`) vs up (`false`)
11//!
12//! These are called by the public wrappers in `lh::proj`, `rh::proj`,
13//! `lh::view`, and `rh::view` and should not be used directly.
14
15use crate::{f64::math, DAffine3, DMat3, DMat4, DQuat, DVec3, DVec4};

17/// Computes an orthonormal view basis from eye, direction, and up.
18#[inline(always)]
19#[must_use]
20fn look_to_axes4<const RH: bool>(eye: DVec3, dir: DVec3, up: DVec3) -> [DVec3; 4] {
21    glam_assert!(dir.is_normalized());
22    glam_assert!(up.is_normalized());
23    let f = if RH { dir } else { -dir };
24    let s = f.cross(up).normalize();
25    let u = s.cross(f);
26    [
27        DVec3::new(s.x, u.x, -f.x),
28        DVec3::new(s.y, u.y, -f.y),
29        DVec3::new(s.z, u.z, -f.z),
30        DVec3::new(-eye.dot(s), -eye.dot(u), eye.dot(f)),
31    ]
32}

34/// Same as [`look_to_axes4`] but without the translation row.
35#[inline(always)]
36#[must_use]
37fn look_to_axes3<const RH: bool>(dir: DVec3, up: DVec3) -> [DVec3; 3] {
38    glam_assert!(dir.is_normalized());
39    glam_assert!(up.is_normalized());
40    let f = if RH { dir } else { -dir };
41    let s = f.cross(up).normalize();
42    let u = s.cross(f);
43    [
44        DVec3::new(s.x, u.x, -f.x),
45        DVec3::new(s.y, u.y, -f.y),
46        DVec3::new(s.z, u.z, -f.z),
47    ]
48}

50/// Assembles a `DMat4` view matrix from eye, direction, and up.
51#[inline]
52#[must_use]
53pub(crate) fn look_to_mat4<const RH: bool>(eye: DVec3, dir: DVec3, up: DVec3) -> DMat4 {
54    let axes = look_to_axes4::<RH>(eye, dir, up);
55    DMat4::from_cols(
56        axes[0].extend(0.0),
57        axes[1].extend(0.0),
58        axes[2].extend(0.0),
59        axes[3].extend(1.0),
60    )
61}

90/// Builds a perspective projection matrix from FOV, aspect, and near/far planes.
91///
92/// # Panics
93///
94/// Will panic if `near` or `far` are <= 0 when `glam_assert` is enabled.
95#[inline]
96#[must_use]
97pub(crate) fn perspective<const RH: bool, const ZO: bool, const YFLIP: bool>(
98    vertical_fov: f64,
99    aspect_ratio: f64,
100    near: f64,
101    far: f64,
102) -> DMat4 {
103    glam_assert!(near > 0.0 && far > 0.0);
104    let (sin_fov, cos_fov) = math::sin_cos(0.5 * vertical_fov);
105    let h = cos_fov / sin_fov;
106    let xx = h / aspect_ratio;
107    let yy = if YFLIP { -h } else { h };
108    let z_range_inv = 1.0 / (far - near);
109
110    let (zz, tz, zw) = match (ZO, RH) {
111        (true, false) => (far * z_range_inv, -near * far * z_range_inv, 1.0),
112        (false, false) => (
113            (far + near) * z_range_inv,
114            -2.0 * far * near * z_range_inv,
115            1.0,
116        ),
117        (true, true) => (-far * z_range_inv, -near * far * z_range_inv, -1.0),
118        (false, true) => (
119            -(far + near) * z_range_inv,
120            -2.0 * far * near * z_range_inv,
121            -1.0,
122        ),
123    };
124
125    DMat4::from_cols(
126        DVec4::new(xx, 0.0, 0.0, 0.0),
127        DVec4::new(0.0, yy, 0.0, 0.0),
128        DVec4::new(0.0, 0.0, zz, zw),
129        DVec4::new(0.0, 0.0, tz, 0.0),
130    )
131}

133/// Like [`perspective`] but with an infinite far plane.
134///
135/// Depth approaches 1 as distance -> infinity.
136///
137/// # Panics
138///
139/// Will panic if `near` <= 0 when `glam_assert` is enabled.
140#[inline]
141#[must_use]
142pub(crate) fn perspective_infinite<const RH: bool, const ZO: bool, const YFLIP: bool>(
143    vertical_fov: f64,
144    aspect_ratio: f64,
145    near: f64,
146) -> DMat4 {
147    glam_assert!(near > 0.0);
148    let (sin_fov, cos_fov) = math::sin_cos(0.5 * vertical_fov);
149    let h = cos_fov / sin_fov;
150    let xx = h / aspect_ratio;
151    let yy = if YFLIP { -h } else { h };
152    let zz = if RH { -1.0 } else { 1.0 };
153    let zw = zz;
154    let tz = if ZO { -near } else { -2.0 * near };
155    DMat4::from_cols(
156        DVec4::new(xx, 0.0, 0.0, 0.0),
157        DVec4::new(0.0, yy, 0.0, 0.0),
158        DVec4::new(0.0, 0.0, zz, zw),
159        DVec4::new(0.0, 0.0, tz, 0.0),
160    )
161}

192/// Builds an orthographic projection from left, right, bottom, top, near, far bounds.
193#[inline]
194#[must_use]
195pub(crate) fn orthographic<const RH: bool, const ZO: bool, const YFLIP: bool>(
196    left: f64,
197    right: f64,
198    bottom: f64,
199    top: f64,
200    near: f64,
201    far: f64,
202) -> DMat4 {
203    let width_inv = 1.0 / (right - left);
204    let height_inv = 1.0 / (top - bottom);
205    let depth_inv = 1.0 / (far - near);
206    let xx = width_inv + width_inv;
207    let yy = if YFLIP {
208        -(height_inv + height_inv)
209    } else {
210        height_inv + height_inv
211    };
212    let zz = match (RH, ZO) {
213        (true, true) => -depth_inv,
214        (false, true) => depth_inv,
215        (true, false) => -2.0 * depth_inv,
216        (false, false) => 2.0 * depth_inv,
217    };
218    let tx = -(left + right) * width_inv;
219    let ty = -(top + bottom) * height_inv;
220    let tz = if ZO {
221        -near * depth_inv
222    } else {
223        -(far + near) * depth_inv
224    };
225    DMat4::from_cols(
226        DVec4::new(xx, 0.0, 0.0, 0.0),
227        DVec4::new(0.0, yy, 0.0, 0.0),
228        DVec4::new(0.0, 0.0, zz, 0.0),
229        DVec4::new(tx, ty, tz, 1.0),
230    )
231}

233/// Builds a perspective projection from a general frustum.
234///
235/// # Panics
236///
237/// Will panic if `near` or `far` are <= 0 when `glam_assert` is enabled.
238#[inline]
239#[must_use]
240pub(crate) fn frustum<const RH: bool, const ZO: bool, const YFLIP: bool>(
241    left: f64,
242    right: f64,
243    bottom: f64,
244    top: f64,
245    near: f64,
246    far: f64,
247) -> DMat4 {
248    glam_assert!(near > 0.0 && far > 0.0);
249    let inv_width = 1.0 / (right - left);
250    let inv_height = 1.0 / (top - bottom);
251    let inv_depth = 1.0 / (far - near);
252    let two_near = 2.0 * near;
253    let xx = two_near * inv_width;
254    let yy = if YFLIP {
255        -two_near * inv_height
256    } else {
257        two_near * inv_height
258    };
259    let zx = (right + left) * inv_width;
260    let zy = if YFLIP {
261        -(top + bottom) * inv_height
262    } else {
263        (top + bottom) * inv_height
264    };
265    let (zz, zw, tz) = match (RH, ZO) {
266        (true, true) => (-far * inv_depth, -1.0, -(far * near) * inv_depth),
267        (false, true) => (far * inv_depth, 1.0, -(far * near) * inv_depth),
268        (true, false) => (
269            -(far + near) * inv_depth,
270            -1.0,
271            -(2.0 * far * near) * inv_depth,
272        ),
273        (false, false) => (
274            (far + near) * inv_depth,
275            1.0,
276            -(2.0 * far * near) * inv_depth,
277        ),
278    };
279    DMat4::from_cols(
280        DVec4::new(xx, 0.0, 0.0, 0.0),
281        DVec4::new(0.0, yy, 0.0, 0.0),
282        DVec4::new(zx, zy, zz, zw),
283        DVec4::new(0.0, 0.0, tz, 0.0),
284    )
