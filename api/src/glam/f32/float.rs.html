    }

    #[inline]
    fn inverse_lerp(a: f32, b: f32, v: f32) -> f32 {
        (v - a) / (b - a)
    }

    fn remap(self, in_start: f32, in_end: f32, out_start: f32, out_end: f32) -> f32 {
