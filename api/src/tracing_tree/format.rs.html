20pub(crate) enum SpanMode {

22    PreOpen,
23    Open {

143    pub(crate) fn prefix(&self) -> String {
144        let mut buf = String::new();
145        if self.render_thread_ids {
146            write!(buf, "{:?}", std::thread::current().id()).unwrap();
147            if buf.ends_with(')') {
148                buf.truncate(buf.len() - 1);
149            }
150            if buf.starts_with("ThreadId(") {
151                buf.drain(0.."ThreadId(".len());
152            }
153        }
154        if self.render_thread_names {
155            if let Some(name) = std::thread::current().name() {
156                if self.render_thread_ids {
157                    buf.push(':');
158                }
159                buf.push_str(name);
160            }
161        }
162        buf
163    }

189    pub indent_buf: String,

192    ///
193    /// This serves to serialize spans as two events can be generated in different spans

208    pub fn flush_current_buf(&mut self, mut writer: impl io::Write) {
209        write!(writer, "{}", &self.current_buf).unwrap();
210        self.current_buf.clear();

214        self.current_buf.push_str(&self.indent_buf);

217
218    pub(crate) fn indent_current(&mut self, indent: usize, config: &Config, style: SpanMode) {
219        let prefix = config.prefix();
220
221        // Render something when wraparound occurs so the user is aware of it
222        if config.indent_lines {
223            self.current_buf.push('\n');
224
225            match style {
226                SpanMode::Close { .. } | SpanMode::PostClose => {
227                    if indent > 0 && (indent + 1) % config.wraparound == 0 {
228                        self.indent_buf.push_str(&prefix);
229                        for _ in 0..(indent % config.wraparound * config.indent_amount) {
230                            self.indent_buf.push_str(LINE_HORIZ);
231                        }
232                        self.indent_buf.push_str(LINE_OPEN);
233                        self.indent_buf.push('\n');
234                    }
235                }
236                _ => {}
237            }
238        }
239
240        indent_block(
241            &self.current_buf,
242            &mut self.indent_buf,
243            indent % config.wraparound,
244            config.indent_amount,
245            config.indent_lines,
246            &prefix,
247            style,
248        );
249
250        self.current_buf.clear();
251        self.flush_indent_buf();
252
253        // Render something when wraparound occurs so the user is aware of it
254        if config.indent_lines {
255            match style {
256                SpanMode::PreOpen { .. } | SpanMode::Open { .. } => {
257                    if indent > 0 && (indent + 1) % config.wraparound == 0 {
258                        self.current_buf.push_str(&prefix);
259                        for _ in 0..(indent % config.wraparound * config.indent_amount) {
260                            self.current_buf.push_str(LINE_HORIZ);
261                        }
262                        self.current_buf.push_str(LINE_CLOSE);
263                        self.current_buf.push('\n');
264                    }
265                }
266                _ => {}
267            }
268        }
269    }

274    pub comma: bool,

277impl<'a> Visit for FmtEvent<'a> {
278    fn record_debug(&mut self, field: &Field, value: &dyn fmt::Debug) {
279        let buf = &mut self.bufs.current_buf;
280        let comma = if self.comma { "," } else { "" };
281        match field.name() {
282            "message" => {
283                write!(buf, "{} {:?}", comma, value).unwrap();
284                self.comma = true;
285            }
286            // Skip fields that are actually log metadata that have already been handled
287            #[cfg(feature = "tracing-log")]
288            name if name.starts_with("log.") => {}
289            name => {
290                write!(buf, "{} {}={:?}", comma, name, value).unwrap();
291                self.comma = true;
292            }
293        }
294    }

312pub(crate) fn write_span_mode(buf: &mut String, style: SpanMode) {
313    match style {
314        SpanMode::Open { verbose: true } => buf.push_str("open(v)"),
315        SpanMode::Open { verbose: false } => buf.push_str("open"),
316        SpanMode::Retrace { verbose: false } => buf.push_str("retrace"),
317        SpanMode::Retrace { verbose: true } => buf.push_str("retrace(v)"),
318        SpanMode::Close { verbose: true } => buf.push_str("close(v)"),
319        SpanMode::Close { verbose: false } => buf.push_str("close"),
320        SpanMode::PreOpen => buf.push_str("pre_open"),
321        SpanMode::PostClose => buf.push_str("post_close"),
322        SpanMode::Event => buf.push_str("event"),
323    }
324
325    buf.push_str(": ")
326}

328fn indent_block_with_lines(
329    lines: &[&str],
330    buf: &mut String,
331    indent: usize,
332    // width of one level of indent
333    indent_amount: usize,
334    prefix: &str,
335    style: SpanMode,
336) {
337    let indent_spaces = indent * indent_amount;
338
339    if lines.is_empty() {
340        return;
341    } else if indent_spaces == 0 {
342        for line in lines {
343            buf.push_str(prefix);
344            // The first indent is special, we only need to print open/close and nothing else
345            if indent == 0 {
346                match style {
347                    SpanMode::Open { .. } => buf.push_str(LINE_OPEN),
348                    SpanMode::Retrace { .. } => buf.push_str(LINE_OPEN),
349                    SpanMode::Close { .. } => buf.push_str(LINE_CLOSE),
350                    SpanMode::PreOpen { .. } | SpanMode::PostClose => {}
351                    SpanMode::Event => {}
352                }
353            }
354            buf.push_str(line);
355            buf.push('\n');
356        }
357        return;
358    }
359
360    let mut s = String::with_capacity(indent_spaces + prefix.len());
361    s.push_str(prefix);
362
363    for _ in 0..(indent_spaces - indent_amount) {
364        s.push(' ');
365    }
366
367    // draw branch
368    buf.push_str(&s);
369
370    match style {
371        SpanMode::PreOpen => {
372            buf.push(LINE_OPEN2);
373            for _ in 1..(indent_amount / 2) {
374                buf.push_str(LINE_HORIZ);
375            }
376            buf.push_str(LINE_OPEN);
377        }
378        SpanMode::Open { verbose: false } | SpanMode::Retrace { verbose: false } => {
379            buf.push(LINE_OPEN2);
380            for _ in 1..indent_amount {
381                buf.push_str(LINE_HORIZ);
382            }
383            buf.push_str(LINE_OPEN);
384        }
385        SpanMode::Open { verbose: true } | SpanMode::Retrace { verbose: true } => {
386            buf.push(' ');
387            for _ in 1..(indent_amount / 2) {
388                buf.push(' ');
389            }
390            // We don't have the space for fancy rendering at single space indent.
391            if indent_amount > 1 {
392                buf.push(LINE_OPEN2);
393            }
394            for _ in (indent_amount / 2)..(indent_amount - 1) {
395                buf.push_str(LINE_HORIZ);
396            }
397            // We don't have the space for fancy rendering at single space indent.
398            if indent_amount > 1 {
399                buf.push_str(LINE_OPEN);
400            } else {
401                buf.push(' ');
402            }
403        }
404        SpanMode::Close { verbose: false } => {
405            buf.push(LINE_CLOSE2);
406            for _ in 1..indent_amount {
407                buf.push_str(LINE_HORIZ);
408            }
409            buf.push_str(LINE_CLOSE);
410        }
411        SpanMode::Close { verbose: true } => {
412            buf.push(' ');
413            for _ in 1..(indent_amount / 2) {
414                buf.push(' ');
415            }
416            // We don't have the space for fancy rendering at single space indent.
417            if indent_amount > 1 {
418                buf.push(LINE_CLOSE2);
419            }
420            for _ in (indent_amount / 2)..(indent_amount - 1) {
421                buf.push_str(LINE_HORIZ);
422            }
423            // We don't have the space for fancy rendering at single space indent.
424            if indent_amount > 1 {
425                buf.push_str(LINE_CLOSE);
426            } else {
427                buf.push(' ');
428            }
429        }
430        SpanMode::PostClose => {
431            buf.push(LINE_CLOSE2);
432            for _ in 1..(indent_amount / 2) {
433                buf.push_str(LINE_HORIZ);
434            }
435            buf.push_str(LINE_CLOSE);
436        }
437        SpanMode::Event => {
438            buf.push_str(LINE_BRANCH);
439
440            // add `indent_amount - 1` horizontal lines before the span/event
441            for _ in 0..(indent_amount - 1) {
442                buf.push_str(LINE_HORIZ);
443            }
444        }
445    }
446    buf.push_str(lines[0]);
447    buf.push('\n');
448
449    // add the rest of the indentation, since we don't want to draw horizontal lines
450    // for subsequent lines
451    for i in 0..indent_amount {
452        if i % indent_amount == 0 {
453            s.push_str(LINE_VERT);
454        } else {
455            s.push(' ');
456        }
457    }
458
459    // add all of the actual content, with each line preceded by the indent string
460    for line in &lines[1..] {
461        buf.push_str(&s);
462        buf.push_str(line);
463        buf.push('\n');
464    }
465}

467fn indent_block(
468    block: &str,
469    buf: &mut String,
470    mut indent: usize,
471    indent_amount: usize,
472    indent_lines: bool,
473    prefix: &str,
474    style: SpanMode,
475) {
476    let lines: Vec<&str> = block.lines().collect();
477    let indent_spaces = indent * indent_amount;
478    buf.reserve(block.len() + (lines.len() * indent_spaces));
479
480    // The PreOpen and PostClose need to match up with the indent of the entered child span one more indent
481    // deep
482    match style {
483        SpanMode::PreOpen | SpanMode::PostClose => {
484            indent += 1;
485        }
486        _ => (),
487    }
488
489    if indent_lines {
490        indent_block_with_lines(&lines, buf, indent, indent_amount, prefix, style);
491    } else {
492        let indent_str = String::from(" ").repeat(indent_spaces);
493        for line in lines {
494            buf.push_str(prefix);
495            buf.push(' ');
496            buf.push_str(&indent_str);
497            buf.push_str(line);
498            buf.push('\n');
499        }
500    }
