1use crate::SpirvTarget;
2use semver::Version;
3use std::ffi::OsString;
4use std::path::Path;

6/// Enum for different versions of target specs, to allow changing the target spec for different rust versions.
7/// The version listed in the enum is always the minimum version to require said target spec, with the newest version
8/// always at the top.
9#[allow(non_camel_case_types)]
10#[derive(Copy, Clone, Debug, Eq, PartialEq, Ord, PartialOrd)]
11pub enum TargetSpecVersion {
12    Older,
13    /// Introduced in `489c3ee6fd63da3ca7cf2b15e1ee709d8e078aab` in the old v2 target spec way, later ported to here.
14    /// remove `os: unknown`, add `crt-static-respected: true`
15    Rustc_1_85_0,
16    /// rustc 1.76 has been tested to correctly parse modern target spec jsons.
17    /// Some later version requires them.
18    /// Some earlier version fails with them (notably our 0.9.0 release).
19    Rustc_1_76_0,
20    /// rustc 1.93 requires that the value of "target-pointer-width" is no longer a string but u16
21    Rustc_1_93_0,
22    /// rustc 1.94.0 destabilised json target specs, requiring `-Ztarget-spec-json`
23    /// see <https://github.com/Rust-GPU/rust-gpu/pull/545>
24    /// see <https://github.com/rust-lang/rust/pull/150151>
25    Rustc_1_94_0,
26}
27
28#[derive(Clone, Debug, Default)]
29pub struct TargetSpec {
30    pub target: OsString,
31    pub extra_options: Vec<OsString>,
32}
33
34impl TargetSpec {
35    pub fn append_to_cmd(&self, cmd: &mut std::process::Command) {
36        cmd.args(self.extra_options.iter());
37        cmd.arg("--target");
38        cmd.arg(&self.target);
39    }
40}
41
42impl TargetSpecVersion {
43    /// Format the `--target` arg. On newer rustc versions, will create a compatible target spec json file and return
44    /// the absolute path to it, on older rustc versions may return the target name.
45    pub fn target_arg(
46        rustc_version: Version,
47        target: &SpirvTarget,
48        target_spec_folder: &Path,
49    ) -> std::io::Result<TargetSpec> {
50        let mut ret = TargetSpec::default();
51        let target_spec = Self::from_rustc_version(rustc_version);
52        if target_spec >= Self::Rustc_1_94_0 {
53            ret.extra_options.push("-Zjson-target-spec".into());
54        }
55
56        ret.target = if target_spec == Self::Older {
57            target.target().into()
58        } else {
59            std::fs::create_dir_all(target_spec_folder)?;
60            let spec_file = target_spec_folder.join(format!("{}.json", target.target()));
61            std::fs::write(&spec_file, target_spec.format_spec(target))?;
62            std::fs::canonicalize(spec_file)?.into_os_string()
63        };
64        Ok(ret)
65    }
66
67    /// Returns the version of the target spec required for a certain rustc version. May return `None` if the version
68    /// is old enough to not need target specs.
69    pub fn from_rustc_version(rustc_version: Version) -> Self {
70        if rustc_version >= Version::new(1, 94, 0) {
71            Self::Rustc_1_94_0
72        } else if rustc_version >= Version::new(1, 93, 0) {
73            Self::Rustc_1_93_0
74        } else if rustc_version >= Version::new(1, 85, 0) {
75            Self::Rustc_1_85_0
76        } else if rustc_version >= Version::new(1, 76, 0) {
77            Self::Rustc_1_76_0
78        } else {
79            Self::Older
80        }
81    }
82
83    /// format the target spec json
84    pub fn format_spec(&self, target: &SpirvTarget) -> String {
85        let target_env = target.env();
86        let (extra, target_pointer_width) = match self {
87            TargetSpecVersion::Older => panic!("no target specs for older rustc versions"),
88            TargetSpecVersion::Rustc_1_76_0 => (r#""os": "unknown","#, "\"32\""),
89            TargetSpecVersion::Rustc_1_85_0 => (r#""crt-static-respected": true,"#, "\"32\""),
90            TargetSpecVersion::Rustc_1_93_0 | TargetSpecVersion::Rustc_1_94_0 => {
91                (r#""crt-static-respected": true,"#, "32")
92            }
93        };
94        format!(
95            r#"{{
96  "allows-weak-linkage": false,
97  "arch": "spirv",
98  "crt-objects-fallback": "false",
99  "crt-static-allows-dylibs": true,
100  "data-layout": "e-m:e-p:32:32:32-i64:64-n8:16:32:64",
101  "dll-prefix": "",
102  "dll-suffix": ".spv.json",
103  "dynamic-linking": true,
104  "emit-debug-gdb-scripts": false,
105  "env": "{target_env}",
106  "linker-flavor": "unix",
107  "linker-is-gnu": false,
108  "llvm-target": "spirv-unknown-{target_env}",
109  "main-needs-argc-argv": false,
110  "metadata": {{
111    "description": null,
112    "host_tools": null,
113    "std": null,
114    "tier": null
115  }},
116  {extra}
117  "panic-strategy": "abort",
118  "simd-types-indirect": false,
119  "target-pointer-width": {target_pointer_width}
120}}"#
121        )
122    }
123}
